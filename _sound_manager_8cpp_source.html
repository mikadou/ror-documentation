219 void SoundManager::recomputeSource(int source_index, int reason, float vfl, Vector3* vvec)
220 {
221  if (!audio_device)
222  return;
223  audio_sources[source_index]->computeAudibility(camera_position);
224 
225  if (audio_sources[source_index]->audibility == 0.0f)
226  {
227  if (audio_sources[source_index]->hardware_index != -1)
228  {
229  // retire the source if it is currently assigned
230  retire(source_index);
231  }
232  }
233  else
234  {
235  // this is a potentially audible m_audio_sources[source_index]
236  if (audio_sources[source_index]->hardware_index != -1)
237  {
238  ALuint hw_source = hardware_sources[audio_sources[source_index]->hardware_index];
239  // m_audio_sources[source_index] already playing
240  // update the AL settings
241  switch (reason)
242  {
243  case Sound::REASON_PLAY: alSourcePlay(hw_source);
244  break;
245  case Sound::REASON_STOP: alSourceStop(hw_source);
246  break;
247  case Sound::REASON_GAIN: alSourcef(hw_source, AL_GAIN, vfl * App::GetAudioMasterVolume());
248  break;
249  case Sound::REASON_LOOP: alSourcei(hw_source, AL_LOOPING, (vfl > 0.5) ? AL_TRUE : AL_FALSE);
250  break;
251  case Sound::REASON_PTCH: alSourcef(hw_source, AL_PITCH, vfl);
252  break;
253  case Sound::REASON_POSN: alSource3f(hw_source, AL_POSITION, vvec->x, vvec->y, vvec->z);
254  break;
255  case Sound::REASON_VLCT: alSource3f(hw_source, AL_VELOCITY, vvec->x, vvec->y, vvec->z);
256  break;
257  default: break;
258  }
259  }
260  else
261  {
262  // try to make it play by the hardware
263  // check if there is one free m_audio_sources[source_index] in the pool
264  if (hardware_sources_in_use_count < hardware_sources_num)
265  {
266  for (int i = 0; i < hardware_sources_num; i++)
267  {
268  if (hardware_sources_map[i] == -1)
269  {
270  assign(source_index, i);
271  break;
272  }
273  }
274  }
275  else
276  {
277  // now, compute who is the faintest
278  // note: we know the table m_hardware_sources_map is full!
279  float fv = 1.0f;
280  int al_faintest = 0;
281  for (int i = 0; i < hardware_sources_num; i++)
282  {
283  if (hardware_sources_map[i] >= 0 && audio_sources[hardware_sources_map[i]]->audibility < fv)
284  {
285  fv = audio_sources[hardware_sources_map[i]]->audibility;
286  al_faintest = i;
287  }
288  }
289  // check to ensure that the sound is louder than the faintest sound currently playing
290  if (fv < audio_sources[source_index]->audibility)
291  {
292  // this new m_audio_sources[source_index] is louder than the faintest!
293  retire(hardware_sources_map[al_faintest]);
294  assign(source_index, al_faintest);
295  }
296  // else this m_audio_sources[source_index] is too faint, we don't play it!
297  }
298  }
299  }
300 }

302 void SoundManager::assign(int source_index, int hardware_index)
303 {
304  if (!audio_device)
305  return;
306  audio_sources[source_index]->hardware_index = hardware_index;
307  hardware_sources_map[hardware_index] = source_index;
308 
309  ALuint hw_source = hardware_sources[hardware_index];
310  Sound* audio_source = audio_sources[source_index];
311 
312  // the hardware source is supposed to be stopped!
313  alSourcei(hw_source, AL_BUFFER, audio_source->buffer);
314  alSourcef(hw_source, AL_GAIN, audio_source->gain * App::GetAudioMasterVolume());
315  alSourcei(hw_source, AL_LOOPING, (audio_source->loop) ? AL_TRUE : AL_FALSE);
316  alSourcef(hw_source, AL_PITCH, audio_source->pitch);
317  alSource3f(hw_source, AL_POSITION, audio_source->position.x, audio_source->position.y, audio_source->position.z);
318  alSource3f(hw_source, AL_VELOCITY, audio_source->velocity.x, audio_source->velocity.y, audio_source->velocity.z);
319 
320  if (audio_source->should_play)
321  {
322  alSourcePlay(hw_source);
323  }
324 
325  hardware_sources_in_use_count++;
326 }

365 Sound* SoundManager::createSound(String filename)
366 {
367  if (!audio_device)
368  return NULL;
369 
370  if (audio_buffers_in_use_count >= MAX_AUDIO_BUFFERS)
371  {
372  LOG("SoundManager: Reached MAX_AUDIO_BUFFERS limit (" + TOSTRING(MAX_AUDIO_BUFFERS) + ")");
373  return NULL;
374  }
375 
376  ALuint buffer = 0;
377 
378  // is the file already loaded?
379  for (int i = 0; i < audio_buffers_in_use_count; i++)
380  {
381  if (filename == audio_buffer_file_name[i])
382  {
383  buffer = audio_buffers[i];
384  break;
385  }
386  }
387 
388  if (!buffer)
389  {
390  // load the file
391  alGenBuffers(1, &audio_buffers[audio_buffers_in_use_count]);
392  if (loadWAVFile(filename, audio_buffers[audio_buffers_in_use_count]))
393  {
394  // there was an error!
395  alDeleteBuffers(1, &audio_buffers[audio_buffers_in_use_count]);
396  audio_buffer_file_name[audio_buffers_in_use_count] = "";
397  return NULL;
398  }
399  buffer = audio_buffers[audio_buffers_in_use_count];
400  audio_buffer_file_name[audio_buffers_in_use_count] = filename;
401  }
402 
403  audio_sources[audio_buffers_in_use_count] = new Sound(buffer, this, audio_buffers_in_use_count);
404 
405  return audio_sources[audio_buffers_in_use_count++];
406 }

408 bool SoundManager::loadWAVFile(String filename, ALuint buffer)
409 {
410  if (!audio_device)
411  return true;
412  LOG("Loading WAV file "+filename);
413 
414  // create the Stream
415  ResourceGroupManager* rgm = ResourceGroupManager::getSingletonPtr();
416  String group = rgm->findGroupContainingResource(filename);
417  DataStreamPtr stream = rgm->openResource(filename, group);
418 
419  // load RIFF/WAVE
420  char magic[5];
421  magic[4] = 0;
422  unsigned int lbuf; // uint32_t
423  unsigned short sbuf; // uint16_t
424 
425  // check magic
426  if (stream->read(magic, 4) != 4)
427  {
428  LOG("Could not read file "+filename);
429  return true;
430  }
431  if (String(magic) != String("RIFF"))
432  {
433  LOG("Invalid WAV file (no RIFF): "+filename);
434  return true;
435  }
436  // skip 4 bytes (magic)
437  stream->skip(4);
438  // check file format
439  if (stream->read(magic, 4) != 4)
440  {
441  LOG("Could not read file "+filename);
442  return true;
443  }
444  if (String(magic) != String("WAVE"))
445  {
446  LOG("Invalid WAV file (no WAVE): "+filename);
447  return true;
448  }
449  // check 'fmt ' sub chunk (1)
450  if (stream->read(magic, 4) != 4)
451  {
452  LOG("Could not read file "+filename);
453  return true;
454  }
455  if (String(magic) != String("fmt "))
456  {
457  LOG("Invalid WAV file (no fmt): "+filename);
458  return true;
459  }
460  // read (1)'s size
461  if (stream->read(&lbuf, 4) != 4)
462  {
463  LOG("Could not read file "+filename);
464  return true;
465  }
466  unsigned long subChunk1Size = lbuf;
467  if (subChunk1Size < 16)
468  {
469  LOG("Invalid WAV file (invalid subChunk1Size): "+filename);
470  return true;
471  }
472  // check PCM audio format
473  if (stream->read(&sbuf, 2) != 2)
474  {
475  LOG("Could not read file "+filename);
476  return true;
477  }
478  unsigned short audioFormat = sbuf;
479  if (audioFormat != 1)
480  {
481  LOG("Invalid WAV file (invalid audioformat "+TOSTRING(audioFormat)+"): "+filename);
482  return true;
483  }
484  // read number of channels
485  if (stream->read(&sbuf, 2) != 2)
486  {
487  LOG("Could not read file "+filename);
488  return true;
489  }
490  unsigned short channels = sbuf;
491  // read frequency (sample rate)
492  if (stream->read(&lbuf, 4) != 4)
493  {
494  LOG("Could not read file "+filename);
495  return true;
496  }
497  unsigned long freq = lbuf;
498  // skip 6 bytes (Byte rate (4), Block align (2))
499  stream->skip(6);
500  // read bits per sample
501  if (stream->read(&sbuf, 2) != 2)
502  {
503  LOG("Could not read file "+filename);
504  return true;
505  }
506  unsigned short bps = sbuf;
507  // check 'data' sub chunk (2)
508  if (stream->read(magic, 4) != 4)
509  {
510  LOG("Could not read file "+filename);
511  return true;
512  }
513  if (String(magic) != String("data") && String(magic) != String("fact"))
514  {
515  LOG("Invalid WAV file (no data/fact): "+filename);
516  return true;
517  }
518  // fact is an option section we don't need to worry about
519  if (String(magic) == String("fact"))
520  {
521  stream->skip(8);
522  // now we should hit the data chunk
523  if (stream->read(magic, 4) != 4)
524  {
525  LOG("Could not read file "+filename);
526  return true;
527  }
528  if (String(magic) != String("data"))
529  {
530  LOG("Invalid WAV file (no data): "+filename);
531  return true;
532  }
533  }
534  // the next four bytes are the remaining size of the file
535  if (stream->read(&lbuf, 4) != 4)
536  {
537  LOG("Could not read file "+filename);
538  return true;
539  }
540 
541  unsigned long dataSize = lbuf;
542  int format = 0;
543 
544  if (channels == 1 && bps == 8)
545  format = AL_FORMAT_MONO8;
546  else if (channels == 1 && bps == 16)
547  format = AL_FORMAT_MONO16;
548  else if (channels == 2 && bps == 8)
549  format = AL_FORMAT_STEREO16;
550  else if (channels == 2 && bps == 16)
551  format = AL_FORMAT_STEREO16;
552  else
553  {
554  LOG("Invalid WAV file (wrong channels/bps): "+filename);
555  return true;
556  }
557 
558  if (channels != 1) LOG("Invalid WAV file: the file needs to be mono, and nothing else. Will try to continue anyways ...");
559 
560  // ok, creating buffer
561  void* bdata = malloc(dataSize);
562  if (!bdata)
563  {
564  LOG("Memory error reading file "+filename);
565  return true;
566  }
567  if (stream->read(bdata, dataSize) != dataSize)
568  {
569  LOG("Could not read file "+filename);
570  free(bdata);
571  return true;
572  }
573 
574  //LOG("alBufferData: format "+TOSTRING(format)+" size "+TOSTRING(dataSize)+" freq "+TOSTRING(freq));
575  alGetError(); // Reset errors
576  ALint error;
577  alBufferData(buffer, format, bdata, dataSize, freq);
578  error = alGetError();
579 
580  free(bdata);
581  // stream will be closed by itself
582 
583  if (error != AL_NO_ERROR)
584  {
585  LOG("OpenAL error while loading buffer for "+filename+" : "+TOSTRING(error));
586  return true;
587  }
588 
589  return false;
590 }
