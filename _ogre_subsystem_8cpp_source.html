59 bool OgreSubsystem::Configure()
60 {
61  // Show the configuration dialog and initialise the system
62  // You can skip this and use root.restoreConfig() to load configuration
63  // settings if you were sure there are valid ones saved in ogre.cfg
64  bool use_ogre_config = BSETTING("USE_OGRE_CONFIG", false);
65 
66  bool ok = false;
67  if (use_ogre_config)
68  {
69  ok = m_ogre_root->showConfigDialog();
70  }
71  else
72  {
73  ok = m_ogre_root->restoreConfig();
74  }
75  if (ok)
76  {
77  // If returned true, user clicked OK so initialise
78  // Here we choose to let the system create a default rendering window by passing 'true'
79  m_render_window = m_ogre_root->initialise(true, "Rigs of Rods version " + Ogre::String(ROR_VERSION_STRING));
80 
81  // set window icon correctly
82  fixRenderWindowIcon(m_render_window);
83 
84  return true;
85  }
86  else
87  {
88  ErrorUtils::ShowError(_L("Configuration error"), _L("Run the RoRconfig program first."));
89  exit(1);
90  }
91  return true;
92 }

94 bool OgreSubsystem::LoadOgrePlugins(Ogre::String const & pluginsfile)
95 {
96  Ogre::ConfigFile cfg;
97 
98  try
99  {
100  cfg.load( pluginsfile );
101  }
102  catch (Ogre::Exception e)
103  {
104  Ogre::LogManager::getSingleton().logMessage(pluginsfile + " not found, automatic plugin loading disabled. Message: " + e.getFullDescription());
105  return false;
106  }
107 
108  Ogre::String pluginDir = cfg.getSetting("PluginFolder");
109  Ogre::StringVector pluginList = cfg.getMultiSetting("Plugin");
110 
111  if (pluginDir.empty())
112  {
113  pluginDir = RoR::App::GetSysProcessDir();
114  }
115 
116  for ( Ogre::StringVector::iterator it = pluginList.begin(); it != pluginList.end(); ++it )
117  {
118  Ogre::String pluginFilename = pluginDir + PATH_SLASH + (*it);
119  try
120  {
121  m_ogre_root->loadPlugin(pluginFilename);
122  }
123  catch (Ogre::Exception &e)
124  {
125  LOG("failed to load plugin: " + pluginFilename + ": " + e.getFullDescription());
126  }
127  }
128  return true;
129 }

131 bool OgreSubsystem::StartOgre(Ogre::String const & hwnd, Ogre::String const & mainhwnd)
132 {
133  m_hwnd = hwnd;
134  m_main_hwnd = mainhwnd;
135 
136  std::string log_filepath = RoR::App::GetSysLogsDir() + PATH_SLASH + "RoR.log";
137  std::string cfg_filepath = RoR::App::GetSysConfigDir() + PATH_SLASH + "ogre.cfg";
138  m_ogre_root = new Ogre::Root("", cfg_filepath, log_filepath);
139 
140  // load plugins manually
141  std::string plugins_filepath = RoR::App::GetSysProcessDir() + PATH_SLASH + "plugins.cfg";
142  this->LoadOgrePlugins(plugins_filepath);
143 
144  // configure RoR
145  this->Configure();
146 
147  m_viewport = m_render_window->addViewport(nullptr);
148 
149  m_viewport->setBackgroundColour(Ogre::ColourValue(0.5f, 0.5f, 0.5f, 1.0f));
150 
151  m_viewport->setCamera(nullptr);
152  m_viewport->setBackgroundColour(Ogre::ColourValue::Black);
153 
154  Ogre::TextureManager::getSingleton().setDefaultNumMipmaps(5);
155 
156  m_timer = new Ogre::Timer();
157  m_timer->reset();
158 
159  m_render_window->setActive(true);
160 
161  return true;
162 }
