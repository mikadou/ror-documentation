41 bool intersectsTerrain(Vector3 a, Vector3 b)
42 {
43  int steps = std::max(10.0f, a.distance(b));
44  for (int i = 0; i < steps; i++)
45  {
46  Vector3 pos = a + (b - a) * (float)i / steps;
47  float y = a.y + (b.y - a.y) * (float)i / steps;
48  float h = gEnv->terrainManager->getHeightFinder()->getHeightAt(pos.x, pos.z);
49  if (h > y)
50  {
51  return true;
52  }
53  }
54  return false;
55 }
