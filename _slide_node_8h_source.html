50 static inline Ogre::Vector3 nearestPoint(const Ogre::Vector3& pt1,
52  const Ogre::Vector3& pt2,
53  const Ogre::Vector3& tp)
54 {
55  const Ogre::Vector3 a = tp - pt1;
56  const Ogre::Vector3 b = fast_normalise(pt2-pt1);
57 
58  return pt1 + (a.dotProduct(b)) * b;
59 }

61 static inline Ogre::Vector3 nearestPointOnLine(const Ogre::Vector3& pt1,
62  const Ogre::Vector3& pt2,
63  const Ogre::Vector3& tp)
64 {
65  Ogre::Vector3 a = tp - pt1;
66  Ogre::Vector3 b = pt2 - pt1;
67  Ogre::Real len = fast_length(b);
68 
69  b = fast_normalise(b);
70  len = std::max(0.0f, std::min(a.dotProduct(b), len));
71  b *= len;
72 
73  a = pt1;
74  a += b;
75  return a;
76 }
