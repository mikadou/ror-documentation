269  return;
270  }
271  else if (args[0] == "/gravity")
272  {
273  float gValue;
274 
275  if (args.size() > 1)
276  {
277  if (args[1] == "earth")
278  gValue = -9.81;

290  }
291

294  return;
295  }
296  else if (args[0] == "/setwaterlevel")
297  {

299  {

305  }

309  }
310  return;
311  }

313  {

319  }

321  {
322  Vector3 pos = b->getPosition();

324  }
325 
326  return;
327  }
328  else if (args[0] == "/goto" && (is_appstate_sim && !is_sim_select))
329  {
330  if (args.size() != 4)
331  {
332  putMessage(CONSOLE_MSGTYPE_INFO, CONSOLE_HELP, RoR::Color::CommandColour + _L("usage: /goto x y z"), "information.png");
