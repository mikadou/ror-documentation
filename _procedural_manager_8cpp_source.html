57 int ProceduralManager::updateObject(ProceduralObject& po)
58 {
59  if (po.loadingState == 1 && po.road)
60  deleteObject(po);
61  // create new road2 object
62  po.road = new Road2(objectcounter++);;
63 
64  std::vector<ProceduralPoint>::iterator it;
65  for (it = po.points.begin(); it != po.points.end(); it++)
66  {
67  ProceduralPoint pp = *it;
68  po.road->addBlock(pp.position, pp.rotation, pp.type, pp.width, pp.bwidth, pp.bheight, pp.pillartype);
69  }
70  po.road->finish();
71 
72  po.loadingState = 1;
73  return 0;
74 }
