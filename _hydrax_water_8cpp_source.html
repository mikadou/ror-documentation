53 void HydraxWater::InitHydrax()
54 {
55  mHydrax = new Hydrax::Hydrax(gEnv->sceneManager, mRenderCamera, RoR::App::GetOgreSubsystem()->GetViewport());
56 
57  waternoise = new Hydrax::Noise::Perlin();
58  mModule = new Hydrax::Module::ProjectedGrid(// Hydrax parent pointer
59  mHydrax,
60  // Noise module
61  waternoise,
62  // Base plane
63  Ogre::Plane(Ogre::Vector3::UNIT_Y, 0),
64  // Normal mode
65  Hydrax::MaterialManager::NM_VERTEX,
66  // Projected grid options
67  Hydrax::Module::ProjectedGrid::Options());
68 
69  mHydrax->setModule(static_cast<Hydrax::Module::Module*>(mModule));
70 
71  mHydrax->loadCfg(CurrentConfigFile);
72 
73  // Choose shader language based on renderer (HLSL=0, CG=1, GLSL=2)
74  if (Root::getSingleton().getRenderSystem()->getName() == "Direct3D9 Rendering Subsystem" || Root::getSingleton().getRenderSystem()->getName() == "Direct3D11 Rendering Subsystem")
75  {
76  mHydrax->setShaderMode(static_cast<Hydrax::MaterialManager::ShaderMode>(0));
77  }
78  else
79  {
80  mHydrax->setShaderMode(static_cast<Hydrax::MaterialManager::ShaderMode>(2));
81  }
82 
83  mHydrax->create();
84  mHydrax->setPosition(Ogre::Vector3(0, waterHeight, 0));
85 }
