80 void TextureToolWindow::fillCombo()
81 {
82  bool dynamicOnly = mChkDynamic->getStateSelected();
83  mCBo->removeAllItems();
84 
85  ResourceManager::ResourceMapIterator it = TextureManager::getSingleton().getResourceIterator();
86 
87  while (it.hasMoreElements())
88  {
89 #ifdef ROR_USE_OGRE_1_9
90  TexturePtr txt(it.getNext().staticCast<Texture>());
91 #else
92  TexturePtr txt(it.getNext());
93 #endif
94 
95  if (dynamicOnly && ((txt->getUsage() & TU_STATIC) != 0))
96  continue;
97 
98  mCBo->addItem(txt->getName());
99  }
100 
101  if (mCBo->getItemCount() > 0)
102  {
103  mCBo->setIndexSelected(0);
104  mCBo->beginToItemSelected();
105  updateControls(mCBo->getItemNameAt(0));
106  }
107 }

114 void TextureToolWindow::saveTexture(String texName, bool usePNG)
115 {
116  try
117  {
118  TexturePtr tex = TextureManager::getSingleton().getByName(texName);
119  if (tex.isNull())
120  return;
121 
122  Image img;
123  tex->convertToImage(img);
124 
125  // Save to disk!
126  String outname = App::GetSysUserDir() + PATH_SLASH + texName;
127  if (usePNG)
128  outname += ".png";
129 
130  img.save(outname);
131 
132  UTFString msg = _L("saved texture as ") + outname;
133  RoR::App::GetConsole()->putMessage(Console::CONSOLE_MSGTYPE_INFO, Console::CONSOLE_MSGTYPE_INFO, msg, "information.png");
134  }
135  catch (Exception& e)
136  {
137  UTFString str = "Exception while saving image: " + e.getFullDescription();
138  RoR::App::GetConsole()->putMessage(Console::CONSOLE_MSGTYPE_INFO, Console::CONSOLE_MSGTYPE_INFO, str, "error.png");
139  }
140 }

142 void TextureToolWindow::updateControls(String texName)
143 {
144  try
145  {
146  bool exists = TextureManager::getSingleton().resourceExists(texName);
147  if (!exists)
148  {
149  mTxt->setCaption(convertToMyGUIString("Texture not found:\n" + texName));
150  mBtnSavePNG->setEnabled(false);
151  return;
152  }
153 
154  TexturePtr tex = TextureManager::getSingleton().getByName(texName);
155  if (tex.isNull())
156  {
157  mTxt->setCaption(convertToMyGUIString("Error loading texture:\n" + texName));
158  mBtnSavePNG->setEnabled(false);
159  return;
160  }
161 
162  String str = "#aa0000" + texName + "#000000\n";
163  str += "#00aa00res: #000000" + TOSTRING(tex->getWidth()) + " x " + TOSTRING(tex->getHeight()) + " pixels\n";
164  str += "#00aa00size: #000000" + formatBytes(tex->getSize()) + "\n";
165  str += "#00aa00format: #000000" + PixelUtil::getFormatName(tex->getFormat()) + "\n";
166  if (tex->getNumFaces() > 1)
167  str += "#00aa00faces: #000000" + TOSTRING(tex->getNumFaces()) + "\n";
168  if (tex->getFSAA() > 0)
169  str += "#00aa00FSAA: #000000" + TOSTRING(tex->getFSAA()) + "\n";
170  if (tex->getNumMipmaps() > 0)
171  str += "#00aa00mipmaps: #000000" + TOSTRING(tex->getNumMipmaps()) + "\n";
172 
173  String typeStr = "";
174  switch (tex->getTextureType())
175  {
176  case TEX_TYPE_1D: typeStr = "1D";
177  break;
178  case TEX_TYPE_2D: typeStr = "2D";
179  break;
180  case TEX_TYPE_3D: typeStr = "3D";
181  break;
182  case TEX_TYPE_CUBE_MAP: typeStr = "Cube Map";
183  break;
184  }
185  str += "#00aa00type: #000000" + typeStr + "\n";
186 
187  String usageStr = "";
188  if (tex->getUsage() & TU_STATIC)
189  usageStr += "static,\n";
190  if (tex->getUsage() & TU_DYNAMIC)
191  usageStr += "dynamic,\n";
192  if (tex->getUsage() & TU_WRITE_ONLY)
193  usageStr += "write only,\n";
194  if (tex->getUsage() & TU_STATIC_WRITE_ONLY)
195  usageStr += "static write only,\n";
196  if (tex->getUsage() & TU_DYNAMIC_WRITE_ONLY)
197  usageStr += "dynamic write only,\n";
198  if (tex->getUsage() & TU_DYNAMIC_WRITE_ONLY_DISCARDABLE)
199  usageStr += "dynamic write only discardable,\n";
200  if (tex->getUsage() & TU_AUTOMIPMAP)
201  usageStr += "automipmap,\n";
202  if (tex->getUsage() & TU_RENDERTARGET)
203  usageStr += "rendertarget,\n";
204  if (tex->getUsage() & TU_DEFAULT)
205  usageStr += "default\n";
206 
207  str += "#00aa00usage: #000000" + usageStr + "\n";
208  if (tex->getDepth() > 1)
209  str += "#00aa00depth: #000000" + TOSTRING(tex->getDepth()) + "\n";
210 
211  mTxt->setCaption(convertToMyGUIString(str));
212  mImage->setImageTexture(texName);
213  mBtnSavePNG->setEnabled(true);
214  }
215  catch (Exception& e)
216  {
217  UTFString str = "Exception while opening texture:" + e.getFullDescription();
218  RoR::App::GetConsole()->putMessage(Console::CONSOLE_MSGTYPE_INFO, Console::CONSOLE_MSGTYPE_INFO, str, "error.png");
219  }
