30 MyGUI::Char translateWin32Text(MyGUI::KeyCode kc)
31 {
32  static WCHAR deadKey = 0;
33 
34  BYTE keyState[256];
35  HKL layout = GetKeyboardLayout(0);
36  if (GetKeyboardState(keyState) == 0)
37  return 0;
38 
39  int code = *((int*)&kc);
40  unsigned int vk = MapVirtualKeyEx((UINT)code, 3, layout);
41  if (vk == 0)
42  return 0;
43 
44  WCHAR buff[3] = {0, 0, 0};
45  int ascii = ToUnicodeEx(vk, (UINT)code, keyState, buff, 3, 0, layout);
46  if (ascii == 1 && deadKey != '\0')
47  {
48  // A dead key is stored and we have just converted a character key
49  // Combine the two into a single character
50  WCHAR wcBuff[3] = {buff[0], deadKey, '\0'};
51  WCHAR out[3];
52 
53  deadKey = '\0';
54  if (FoldStringW(MAP_PRECOMPOSED, (LPWSTR)wcBuff, 3, (LPWSTR)out, 3))
55  return out[0];
56  }
57  else if (ascii == 1)
58  {
59  // We have a single character
60  deadKey = '\0';
61  return buff[0];
62  }
63  else if (ascii == 2)
64  {
65  // Convert a non-combining diacritical mark into a combining diacritical mark
66  // Combining versions range from 0x300 to 0x36F; only 5 (for French) have been mapped below
67  // http://www.fileformat.info/info/unicode/block/combining_diacritical_marks/images.htm
68  switch (buff[0])
69  {
70  case 0x5E: // Circumflex accent: �
71  deadKey = 0x302;
72  break;
73  case 0x60: // Grave accent: �
74  deadKey = 0x300;
75  break;
76  case 0xA8: // Diaeresis: �
77  deadKey = 0x308;
78  break;
79  case 0xB4: // Acute accent: �
80  deadKey = 0x301;
81  break;
82  case 0xB8: // Cedilla: �
83  deadKey = 0x327;
84  break;
85  default:
86  deadKey = buff[0];
87  break;
88  }
89  }
90 
91  return 0;
92 }
93 #endif

238 bool GUIInputManager::keyPressed(const OIS::KeyEvent& _arg)
239 {
240  MyGUI::Char text = (MyGUI::Char)_arg.text;
241  MyGUI::KeyCode key = MyGUI::KeyCode::Enum(_arg.key);
242  int scan_code = MYGUI_GET_SCANCODE(key);
243 
244  if (scan_code > 70 && scan_code < 84)
245  {
246  static MyGUI::Char nums[13] = {55, 56, 57, 45, 52, 53, 54, 43, 49, 50, 51, 48, 46};
247  text = nums[scan_code - 71];
248  }
249  else if (key == MyGUI::KeyCode::Divide)
250  {
251  text = '/';
252  }
253  else
254  {
255 #if MYGUI_PLATFORM == MYGUI_PLATFORM_WIN32
256  text = translateWin32Text(key);
257 #endif
258  }
259 
260  // fallback, handle by GUI, then by RoR::SceneMouse
261  bool handled = MyGUI::InputManager::getInstance().injectKeyPress(key, text);
262 
263  if (handled)
264  {
265  MyGUI::Widget* w = MyGUI::InputManager::getInstance().getKeyFocusWidget();
266  // hack for console, we want to use the mouse through that control
267  if (w && w->getName().substr(0, 7) == "Console" && w->getName() != "ConsoleInput")
268  handled = false;
269  }
270 
271  if (!handled)
272  {
273  RoR::SceneMouse* sm = RoR::App::GetSceneMouse();
274  if (sm)
