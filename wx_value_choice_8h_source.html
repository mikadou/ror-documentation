91  void sort(bool descending = true)
92  {
93  if(!sortable) return;
94  // remove all items from the control and re-add them in the order of the map
95  wxChoice::Clear(); // clear items from control
96  id_id_map.clear();
97 
98  // use the caption for the control (2. in the pair)
99 
100  if(descending)
101  {
102  valMap::reverse_iterator it;
103  for(it = id_value_map.rbegin(); it != id_value_map.rend(); it++)
104  {
105  int id = this->Append(it->second.second);
106  id_id_map[id] = it->first;
107  }
108  } else
109  {
110  valMap::iterator it;
111  for(it = id_value_map.begin(); it != id_value_map.end(); it++)
112  {
113  int id = this->Append(it->second.second);
114  id_id_map[id] = it->first;
115  }
116  }
117  }
