33 #define PROCESS_SECTION_IN_ANY_MODULE(_KEYWORD_, _FIELD_, _FUNCTION_) \
34 { \
35  SetCurrentKeyword(_KEYWORD_); \
36  auto module_itor = m_selected_modules.begin(); \
37  auto module_end = m_selected_modules.end(); \
38  for (; module_itor != module_end; ++module_itor) \
39  { \
40  auto* def = module_itor->get()->_FIELD_.get(); \
41  if (def != nullptr) \
42  { \
43  try { \
44  _FUNCTION_(*def); \
45  } \
46  catch (Exception ex) \
47  { \
48  AddMessage(Message::TYPE_ERROR, ex.what()); \
49  } \
50  catch (...) \
51  { \
52  AddMessage(Message::TYPE_ERROR, "An unknown exception has occured"); \
53  } \
54  break; \
55  } \
56  } \
57  SetCurrentKeyword(RigDef::File::KEYWORD_INVALID); \
58 }

82 #define PROCESS_SECTION_IN_ALL_MODULES(_KEYWORD_, _FIELD_, _FUNCTION_) \
83 { \
84  SetCurrentKeyword(_KEYWORD_); \
85  auto module_itor = m_selected_modules.begin(); \
86  auto module_end = m_selected_modules.end(); \
87  for (; module_itor != module_end; ++module_itor) \
88  { \
89  auto& field = module_itor->get()->_FIELD_; \
90  auto section_itor = field.begin(); \
91  auto section_end = field.end(); \
92  for (; section_itor != section_end; ++section_itor) \
93  { \
94  try { \
95  _FUNCTION_(*section_itor); \
96  } \
97  catch (RigSpawner::Exception & ex) \
