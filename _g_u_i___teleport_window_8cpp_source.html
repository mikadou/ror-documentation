203 void TeleportWindow::MinimapPanelResized(MyGUI::Widget* panel)
204 {
205  float panel_width = static_cast<float>(panel->getWidth());
206  float panel_height = static_cast<float>(panel->getHeight());
207  // Fit the minimap into panel, preserving aspect ratio
208  float mini_width = panel_width;
209  float mini_height = (mini_width / m_map_size.x) * m_map_size.z;
210  if (mini_height >= panel_height)
211  {
212  mini_height = panel_height;
213  mini_width = (mini_height / m_map_size.z) * m_map_size.x;
214  }
215 
216  // Update minimap widget
217  MyGUI::IntCoord coord;
218  coord.left = static_cast<int>((panel_width / 2.f) - (mini_width / 2));
219  coord.top = static_cast<int>((panel_height / 2.f) - (mini_height / 2));
220  coord.width = static_cast<int>(mini_width);
221  coord.height = static_cast<int>(mini_height);
222  m_minimap_image->setCoord(coord);
223 
224  // Update icon positions
225  for (MyGUI::ImageBox* tp_icon: m_telepoint_icons)
226  {
227  Terrn2Telepoint* def = GetTeleIconUserdata(tp_icon);
228  int pos_x = static_cast<int>(((def->position.x / m_map_size.x) * mini_width ) - (TELEICON_SIZE/2));
229  int pos_y = static_cast<int>(((def->position.z / m_map_size.z) * mini_height) - (TELEICON_SIZE/2));
230  tp_icon->setPosition(pos_x, pos_y);
231  }
232 }
