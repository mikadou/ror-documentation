122 void ContentManager::AddResourcePack(ResourcePack const& resource_pack)
123 {
124  std::stringstream log_msg;
125  if (BITMASK_64_IS_1(m_loaded_resource_packs, resource_pack.mask)) // Already loaded?
126  {
127  log_msg << "[RoR|ContentManager] Resource pack \"" << resource_pack.name << "\" already loaded.";
128  LOG(log_msg.str());
129  return;
130  }
131  log_msg << "[RoR|ContentManager] Loading resource pack \"" << resource_pack.name << "\" from group \"" << resource_pack.resource_group_name << "\"";
132  Ogre::String resources_dir = App::GetSysResourcesDir() + PATH_SLASH;
133  Ogre::String zip_path = resources_dir + resource_pack.name + Ogre::String(".zip");
134  if (PlatformUtils::FileExists(zip_path))
135  {
136  log_msg << " (ZIP archive)";
137  LOG(log_msg.str());
138  ResourceGroupManager::getSingleton().addResourceLocation(zip_path, "Zip", resource_pack.resource_group_name);
139  BITMASK_64_SET_1(m_loaded_resource_packs, resource_pack.mask);
140  }
141  else
142  {
143  Ogre::String dir_path = resources_dir + resource_pack.name;
144  if (PlatformUtils::FolderExists(dir_path))
145  {
146  log_msg << " (directory)";
147  LOG(log_msg.str());
148  ResourceGroupManager::getSingleton().addResourceLocation(dir_path, "FileSystem", resource_pack.resource_group_name);
149  BITMASK_64_SET_1(m_loaded_resource_packs, resource_pack.mask);
150  }
151  else
152  {
153  log_msg << " failed, data not found.";
154  throw std::runtime_error(log_msg.str());
155  }
156  }

346 void ContentManager::exploreFolders(Ogre::String rg)
347 {
348  ResourceGroupManager& rgm = ResourceGroupManager::getSingleton();
349 
350  FileInfoListPtr files = rgm.findResourceFileInfo(rg, "*", true); // searching for dirs
351  FileInfoList::iterator iterFiles = files->begin();
352  for (; iterFiles != files->end(); ++iterFiles)
353  {
354  if (!iterFiles->archive)
355  continue;
356  if (iterFiles->filename == String(".svn"))
357  continue;
358  // trying to get the full path
359  String fullpath = iterFiles->archive->getName() + PATH_SLASH;
360  rgm.addResourceLocation(fullpath + iterFiles->filename, "FileSystem", rg);
361  }
362  LOG("initialiseResourceGroups: "+rg);
363  try
364  {
365  if (BSETTING("Background Loading", false))
366  ResourceBackgroundQueue::getSingleton().initialiseResourceGroup(rg);
367  else
368  ResourceGroupManager::getSingleton().initialiseResourceGroup(rg);
369  }
370  catch (Ogre::Exception& e)
371  {
372  LOG("catched error while initializing Resource group '" + rg + "' : " + e.getFullDescription());
373  }

417 void ContentManager::RegenCache()
418 {
419 
420  RoR::App::GetCacheSystem()->Startup(true); // true = force regeneration
421 
422  // Get stats
423  int num_new = RoR::App::GetCacheSystem()->newFiles;
424  int num_changed = RoR::App::GetCacheSystem()->changedFiles;
425  int num_deleted = RoR::App::GetCacheSystem()->deletedFiles;
426 
427  // Report
428  Ogre::UTFString str = _L("Cache regeneration done.\n");
429  if (num_new > 0)
430  {
431  str = str + TOUTFSTRING(num_new) + _L(" new files\n");
432  }
433  if (num_changed > 0)
434  {
435  str = str + TOUTFSTRING(num_changed) + _L(" changed files\n");
436  }
437  if (num_deleted > 0)
438  {
439  str = str + TOUTFSTRING(num_deleted) + _L(" deleted files\n");
440  }
441  if (num_new + num_changed + num_deleted == 0)
442  {
443  str = str + _L("no changes");
444  }
445  str = str + _L("\n(These stats can be imprecise)");
446 
447  ErrorUtils::ShowError(_L("Cache regeneration done"), str);
448 }
