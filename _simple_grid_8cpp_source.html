375  void SimpleGrid::_calculeNormals()
376  {
377  if (getNormalMode() != MaterialManager::NM_VERTEX)
378  {
379  return;
380  }
381 
382  int v, u;
383  Ogre::Vector3 vec1, vec2, normal;
384 
385  Mesh::POS_NORM_VERTEX* Vertices = static_cast<Mesh::POS_NORM_VERTEX*>(mVertices);
386 
387  for(v=1; v<(mOptions.Complexity-1); v++)
388  {
389  for(u=1; u<(mOptions.Complexity-1); u++)
390  {
391  vec1 = Ogre::Vector3(
392  Vertices[v*mOptions.Complexity + u + 1].x-Vertices[v*mOptions.Complexity + u - 1].x,
393  Vertices[v*mOptions.Complexity + u + 1].y-Vertices[v*mOptions.Complexity + u - 1].y,
394  Vertices[v*mOptions.Complexity + u + 1].z-Vertices[v*mOptions.Complexity + u - 1].z);
395 
396  vec2 = Ogre::Vector3(
397  Vertices[(v+1)*mOptions.Complexity + u].x - Vertices[(v-1)*mOptions.Complexity + u].x,
398  Vertices[(v+1)*mOptions.Complexity + u].y - Vertices[(v-1)*mOptions.Complexity + u].y,
399  Vertices[(v+1)*mOptions.Complexity + u].z - Vertices[(v-1)*mOptions.Complexity + u].z);
400 
401  normal = vec2.crossProduct(vec1);
402 
403  Vertices[v*mOptions.Complexity + u].nx = normal.x;
404  Vertices[v*mOptions.Complexity + u].ny = normal.y;
405  Vertices[v*mOptions.Complexity + u].nz = normal.z;
406  }
407  }
408  }
