95

98  Quaternion rotX = dir.getRotationTo(camDir, Vector3::UNIT_Y);
99  camRotX = rotX.getYaw();
100 
101  // Corner case handling
102  Radian angle = dir.angleBetween(camDir);
103  if ( angle > Radian(Math::HALF_PI) )
104  {
105  if ( std::abs(Radian(camRotX).valueRadians()) < Math::HALF_PI )
106  {
107  if ( camRotX < Radian(0.0f) )
108  camRotX -= Radian(Math::HALF_PI);
109  else
110  camRotX += Radian(Math::HALF_PI);
