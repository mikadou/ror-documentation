47 int ShadowManager::updateShadowTechnique()
48 {
49  float scoef = 0.5;
50  gEnv->sceneManager->setShadowColour(Ogre::ColourValue(0.563 + scoef, 0.578 + scoef, 0.625 + scoef));
51  gEnv->sceneManager->setShowDebugShadows(false);
52 
53  RoR::App::GfxShadowType type = RoR::App::GetGfxShadowType();
54 
55  if (type == RoR::App::GFX_SHADOW_TYPE_TEXTURE)
56  {
57  gEnv->sceneManager->setShadowFarDistance(RoR::App::GetGfxSightRange());
58  processTextureShadows();
59  }
60  else if (type == RoR::App::GFX_SHADOW_TYPE_PSSM)
61  {
62  processPSSM();
63  if (gEnv->sceneManager->getShowDebugShadows())
64  {
65  // add the overlay elements to show the shadow maps:
66  // init overlay elements
67  OverlayManager& mgr = Ogre::OverlayManager::getSingleton();
68  Overlay* overlay = mgr.create("DebugOverlay");
69 
70  for (int i = 0; i < PSSM_Shadows.ShadowsTextureNum; ++i)
71  {
72  TexturePtr tex = gEnv->sceneManager->getShadowTexture(i);
73 
74  // Set up a debug panel to display the shadow
75  MaterialPtr debugMat = MaterialManager::getSingleton().create("Ogre/DebugTexture" + StringConverter::toString(i), ResourceGroupManager::DEFAULT_RESOURCE_GROUP_NAME);
76  debugMat->getTechnique(0)->getPass(0)->setLightingEnabled(false);
77  TextureUnitState* t = debugMat->getTechnique(0)->getPass(0)->createTextureUnitState(tex->getName());
78  t->setTextureAddressingMode(TextureUnitState::TAM_CLAMP);
79 
80  OverlayContainer* debugPanel = (OverlayContainer*)(OverlayManager::getSingleton().createOverlayElement("Panel", "Ogre/DebugTexPanel" + StringConverter::toString(i)));
81  debugPanel->_setPosition(0.8, i * 0.25);
82  debugPanel->_setDimensions(0.2, 0.24);
83  debugPanel->setMaterialName(debugMat->getName());
84  debugPanel->setEnabled(true);
85  overlay->add2D(debugPanel);
86  overlay->show();
87  }
88  }
89  }
90  return 0;
91 }

99 void ShadowManager::processPSSM()
100 {
101  gEnv->sceneManager->setShadowTechnique(Ogre::SHADOWTYPE_TEXTURE_ADDITIVE_INTEGRATED);
102 
103  gEnv->sceneManager->setShadowDirectionalLightExtrusionDistance(299.0f);
104  gEnv->sceneManager->setShadowFarDistance(350.0f);
105  gEnv->sceneManager->setShadowTextureCountPerLightType(Ogre::Light::LT_DIRECTIONAL, PSSM_Shadows.ShadowsTextureNum);
106  gEnv->sceneManager->setShadowTextureCount(PSSM_Shadows.ShadowsTextureNum);
107 
108  gEnv->sceneManager->setShadowTextureSelfShadow(true);
109  gEnv->sceneManager->setShadowCasterRenderBackFaces(true);
110 
111  //Caster is set via materials
112  gEnv->sceneManager->setShadowTextureCasterMaterial("Ogre/shadow/depth/caster");
113 
114  if (PSSM_Shadows.Quality == 3)
115  {
116  gEnv->sceneManager->setShadowTextureConfig(0, 4096, 4096, PF_FLOAT32_R);
117  gEnv->sceneManager->setShadowTextureConfig(1, 3072, 3072, PF_FLOAT32_R);
118  gEnv->sceneManager->setShadowTextureConfig(2, 2048, 2048, PF_FLOAT32_R);
119  PSSM_Shadows.lambda = 0.965f;
120  }
121  else if (PSSM_Shadows.Quality == 2)
122  {
123  gEnv->sceneManager->setShadowTextureConfig(0, 3072, 3072, PF_FLOAT32_R);
124  gEnv->sceneManager->setShadowTextureConfig(1, 2048, 2048, PF_FLOAT32_R);
125  gEnv->sceneManager->setShadowTextureConfig(2, 2048, 2048, PF_FLOAT32_R);
126  PSSM_Shadows.lambda = 0.97f;
127  }
128  else if (PSSM_Shadows.Quality == 1)
129  {
130  gEnv->sceneManager->setShadowTextureConfig(0, 2048, 2048, PF_FLOAT32_R);
131  gEnv->sceneManager->setShadowTextureConfig(1, 1024, 1024, PF_FLOAT32_R);
132  gEnv->sceneManager->setShadowTextureConfig(2, 1024, 1024, PF_FLOAT32_R);
133  PSSM_Shadows.lambda = 0.975f;
134  }
135  else
136  {
137  gEnv->sceneManager->setShadowTextureConfig(0, 1024, 1024, PF_FLOAT32_R);
138  gEnv->sceneManager->setShadowTextureConfig(1, 1024, 1024, PF_FLOAT32_R);
139  gEnv->sceneManager->setShadowTextureConfig(2, 512, 512, PF_FLOAT32_R);
140  PSSM_Shadows.lambda = 0.98f;
141  }
142 
143  if (PSSM_Shadows.mPSSMSetup.isNull())
144  {
145  // shadow camera setup
146  Ogre::PSSMShadowCameraSetup* pssmSetup = new Ogre::PSSMShadowCameraSetup();
147 
148  pssmSetup->calculateSplitPoints(3, gEnv->mainCamera->getNearClipDistance(), gEnv->sceneManager->getShadowFarDistance(), PSSM_Shadows.lambda);
149  pssmSetup->setSplitPadding(gEnv->mainCamera->getNearClipDistance());
150 
151  pssmSetup->setOptimalAdjustFactor(0, -1);
152  pssmSetup->setOptimalAdjustFactor(1, -1);
153  pssmSetup->setOptimalAdjustFactor(2, -1);
154 
155  PSSM_Shadows.mPSSMSetup.bind(pssmSetup);
156 
157  //Send split info to managed materials
158  setManagedMaterialSplitPoints(pssmSetup->getSplitPoints());
159  }
160  gEnv->sceneManager->setShadowCameraSetup(PSSM_Shadows.mPSSMSetup);
161 }
