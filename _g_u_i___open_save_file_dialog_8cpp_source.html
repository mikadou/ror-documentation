177 void OpenSaveFileDialog::update()
178 {
179  if (mCurrentFolder.empty())
180  mCurrentFolder = "/";
181  mCurrentFolderField->setCaption(mCurrentFolder);
182 
183  mListFiles->removeAllItems();
184 
185  // add all folders first
186  FileSystem::VectorFileInfo infos;
187  getSystemFileList(infos, mCurrentFolder, L"*.*");
188 
189  for (FileSystem::VectorFileInfo::iterator item = infos.begin(); item != infos.end(); ++item)
190  {
191  if ((*item).folder)
192  mListFiles->addItem(L"[" + (*item).name + L"]", *item);
193  }
194 
195  if (!mFolderMode)
196  {
197  // add files by given mask
198  infos.clear();
199  getSystemFileList(infos, mCurrentFolder, mFileMask);
200 
201  for (FileSystem::VectorFileInfo::iterator item = infos.begin(); item != infos.end(); ++item)
202  {
203  if (!(*item).folder)
204  mListFiles->addItem((*item).name, *item);
205  }
206  }
207 }
