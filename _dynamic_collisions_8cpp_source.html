47 static bool BackfaceCollisionTest(const float distance,
48  const Vector3 &normal,
49  const node_t &surface_point,
50  const std::vector<int> &neighbour_node_ids,
51  const node_t nodes[])
52 {
53  auto sign = [](float x){ return (x >= 0) ? 1 : -1; };
54 
55  // Summarize over the collision node and its connected neighbour nodes to infer on which side
56  // of the collision plane most of these nodes are located.
57  // Nodes in front contribute positively, nodes on the backface contribute negatively.
58 
59  // the contribution of the collision node itself has triple weight in this heuristic
60  const int weight = 3;
61  int face_indicator = weight * sign(distance);
62 
63  // calculate the contribution of neighbouring nodes (if it can still change the final outcome)
64  if (neighbour_node_ids.size() > weight) {
65  for (auto id : neighbour_node_ids) {
66  const auto neighbour_distance = normal.dotProduct(nodes[id].AbsPosition - surface_point.AbsPosition);
67  face_indicator += sign(neighbour_distance);
68  }
69  }
70 
71  // a negative sum indicates that the collision is occuring on the backface
72  return (face_indicator < 0);
73 }

86 static bool InsideTriangleTest(const CartesianToTriangleTransform::TriangleCoord &local, const float margin)
87 {

95 void ResolveCollisionForces(const float penetration_depth,
96  node_t &hitnode, node_t &na, node_t &nb, node_t &no,
97  const float alpha, const float beta, const float gamma,
98  const Vector3 &normal,
99  const float dt,
100  ground_model_t &submesh_ground_model)
101 {
102 
103  //Find the point's velocity relative to the triangle
104  const auto vecrelVel = (hitnode.Velocity - (na.Velocity * alpha + nb.Velocity * beta + no.Velocity * gamma));
105 
106  //Find the velocity perpendicular to the triangle
107  //if it points away from the triangle the ignore it (set it to 0)
108  const float velForce = std::max(0.0f, -vecrelVel.dotProduct(normal));
109 
110  //Velocity impulse
111  const auto inv_dt = 1.0 / dt;
112  const float vi = hitnode.mass * inv_dt * (velForce + inv_dt * penetration_depth) * 0.5f;
113 
114  //The force that the triangle puts on the point
115  //(applied only when it is towards the point)
116  const auto triangle_force = na.Forces * alpha + nb.Forces * beta + no.Forces * gamma;
117  const float trfnormal = std::max(0.0f, triangle_force.dotProduct(normal));
118 
119  //The force that the point puts on the triangle
120  //(applied only when it is towards the triangle)
121  const float pfnormal = std::min(0.0f, hitnode.Forces.dotProduct(normal));
122 
123  const float fl = (vi + trfnormal - pfnormal) * 0.5f;
124 
125  auto forcevec = Vector3::ZERO;
126  float nso; // TODO unused
127 
128  //Calculate the collision forces
129  primitiveCollision(&hitnode, forcevec, vecrelVel, normal, ((float) dt), &submesh_ground_model, &nso, penetration_depth, fl);
130 
131  // apply resulting collision force
132  hitnode.Forces += forcevec;
133  na.Forces -= alpha * forcevec;
134  nb.Forces -= beta * forcevec;
135  no.Forces -= gamma * forcevec;
136 }
