142 void VehicleAI::update(float dt, int doUpdate)
143 {
144  if (is_waiting)
145  {
146  wait_time -= dt;
147  if (wait_time < 0)
148  {
149  is_waiting = false;
150  }
151  return;
152  }
153 
154  Vector3 mAgentPosition = beam->getPosition();
155 
156  if (current_waypoint.distance(mAgentPosition) < 5)
157  {
158  updateWaypoint();
159  return;
160  }
161 
162  Vector3 TargetPosition = current_waypoint;
163  TargetPosition.y = 0; //Vector3 > Vector2
164  Quaternion TargetOrientation = Quaternion::ZERO;
165 
166  mAgentPosition.y = 0; //Vector3 > Vector2
167  Quaternion mAgentOrientation = Quaternion(Radian(beam->getHeadingDirectionAngle()), Vector3::NEGATIVE_UNIT_Y);
168  mAgentOrientation.normalise();
169 
170  Vector3 mVectorToTarget = TargetPosition - mAgentPosition; // A-B = B->A
171  mAgentPosition.normalise();
172 
173  Vector3 mAgentHeading = mAgentOrientation * mAgentPosition;
174  Vector3 mTargetHeading = TargetOrientation * TargetPosition;
175  mAgentHeading.normalise();
176  mTargetHeading.normalise();
177 
178  // Compute new torque scalar (-1.0 to 1.0) based on heading vector to target.
179  Vector3 mSteeringForce = mAgentOrientation.Inverse() * mVectorToTarget;
180  mSteeringForce.normalise();
181 
182  float mYaw = mSteeringForce.x;
183  float mPitch = mSteeringForce.z;
184  //float mRoll = mTargetVO.getRotationTo( mAgentVO ).getRoll().valueRadians();
185 
186  if (mPitch > 0)
187  {
188  if (mYaw > 0)
189  mYaw = 1;
190  else
191  mYaw = -1;
192  }
193 
194  // actually steer
195  beam->hydrodircommand = mYaw;//mYaw
196 
197  if (beam->engine)
198  {
199  // start engine if not running
200  if (!beam->engine->isRunning())
201  beam->engine->start();
202 
203  float kmh_wheel_speed = beam->getWheelSpeed() * 3.6;
204 
205  if (abs(mYaw) < 0.5f)
206  {
207  if (kmh_wheel_speed < maxspeed - 1)
208  {
209  beam->brake = 0;
210  beam->engine->autoSetAcc(acc_power);
211  }
212  else if (kmh_wheel_speed > maxspeed + 1)
213  {
214  beam->brake = beam->brakeforce / 3;
215  beam->engine->autoSetAcc(0);
216  }
217  else
218  {
219  beam->brake = 0;
220  beam->engine->autoSetAcc(0);
221  }
222  }
223  else
224  {
225  if (kmh_wheel_speed < maxspeed - 1)
226  {
227  beam->brake = 0;
228  beam->engine->autoSetAcc(acc_power / 3);
229  }
230  else if (kmh_wheel_speed > maxspeed + 1)
231  {
232  beam->brake = beam->brakeforce / 2;
233  beam->engine->autoSetAcc(0);
234  }
235  else
236  {
237  beam->brake = 0;
238  beam->engine->autoSetAcc(0);
239  }
240  }
241  }
242 }
