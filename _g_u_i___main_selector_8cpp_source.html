473 size_t CLASS::SearchCompare(Ogre::String searchString, CacheEntry* ce)
474 {
475  if (searchString.find(":") == Ogre::String::npos)
476  {
477  // normal search
478 
479  // the name
480  Ogre::String dname_lower = ce->dname;
481  Ogre::StringUtil::toLowerCase(dname_lower);
482  if (dname_lower.find(searchString) != Ogre::String::npos)
483  return dname_lower.find(searchString);
484 
485  // the filename
486  Ogre::String fname_lower = ce->fname;
487  Ogre::StringUtil::toLowerCase(fname_lower);
488  if (fname_lower.find(searchString) != Ogre::String::npos)
489  return 100 + fname_lower.find(searchString);
490 
491  // the description
492  Ogre::String desc = ce->description;
493  Ogre::StringUtil::toLowerCase(desc);
494  if (desc.find(searchString) != Ogre::String::npos)
495  return 200 + desc.find(searchString);
496 
497  // the authors
498  if (!ce->authors.empty())
499  {
500  std::vector<AuthorInfo>::const_iterator it;
501  for (it = ce->authors.begin(); it != ce->authors.end(); it++)
502  {
503  // author name
504  Ogre::String aname = it->name;
505  Ogre::StringUtil::toLowerCase(aname);
506  if (aname.find(searchString) != Ogre::String::npos)
507  return 300 + aname.find(searchString);
508 
509  // author email
510  Ogre::String aemail = it->email;
511  Ogre::StringUtil::toLowerCase(aemail);
512  if (aemail.find(searchString) != Ogre::String::npos)
513  return 400 + aemail.find(searchString);
514  }
515  }
516  return Ogre::String::npos;
517  }
518  else
519  {
520  Ogre::StringVector v = Ogre::StringUtil::split(searchString, ":");
521  if (v.size() < 2)
522  return Ogre::String::npos; //invalid syntax
523 
524  if (v[0] == "hash")
525  {
526  Ogre::String hash = ce->hash;
527  Ogre::StringUtil::toLowerCase(hash);
528  return hash.find(v[1]);
529  }
530  else if (v[0] == "guid")
531  {
532  Ogre::String guid = ce->guid;
533  Ogre::StringUtil::toLowerCase(guid);
534  return guid.find(v[1]);
535  }
536  else if (v[0] == "author")
537  {
538  // the authors
539  if (!ce->authors.empty())
540  {
541  std::vector<AuthorInfo>::const_iterator it;
542  for (it = ce->authors.begin(); it != ce->authors.end(); it++)
543  {
544  // author name
545  Ogre::String aname = it->name;
546  Ogre::StringUtil::toLowerCase(aname);
547  if (aname.find(v[1]) != Ogre::String::npos)
548  return aname.find(v[1]);
549 
550  // author email
551  Ogre::String aemail = it->email;
552  Ogre::StringUtil::toLowerCase(aemail);
553  if (aemail.find(v[1]) != Ogre::String::npos)
554  return aemail.find(v[1]);
555  }
556  }
557  return Ogre::String::npos;
558  }
559  else if (v[0] == "wheels")
560  {
561  Ogre::String wheelsStr = TOUTFSTRING(ce->wheelcount) + "x" + TOUTFSTRING(ce->propwheelcount);
562  return wheelsStr.find(v[1]);
563  }
564  else if (v[0] == "file")
565  {
566  Ogre::String fn = ce->fname;
567  Ogre::StringUtil::toLowerCase(fn);
568  return fn.find(v[1]);
569  }
570  }
571  return Ogre::String::npos;
572 }
