57 void CameraBehaviorOrbit::update(const CameraManager::CameraContext& ctx)
58 {
59  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_CAMERA_LOOKBACK))
60  {
61  if (camRotX > Degree(0))
62  {
63  camRotX = Degree(0);
64  }
65  else
66  {
67  camRotX = Degree(180);
68  }
69  }
70 
71  camRotX += (RoR::App::GetInputEngine()->getEventValue(EV_CAMERA_ROTATE_RIGHT) - RoR::App::GetInputEngine()->getEventValue(EV_CAMERA_ROTATE_LEFT)) * ctx.mRotScale;
72  camRotY += (RoR::App::GetInputEngine()->getEventValue(EV_CAMERA_ROTATE_UP) - RoR::App::GetInputEngine()->getEventValue(EV_CAMERA_ROTATE_DOWN)) * ctx.mRotScale;
73 
74  camRotY = std::max((Radian)Degree(-80), camRotY);
75  camRotY = std::min(camRotY, (Radian)Degree(88));
76 
77  camRotXSwivel = (RoR::App::GetInputEngine()->getEventValue(EV_CAMERA_SWIVEL_RIGHT) - RoR::App::GetInputEngine()->getEventValue(EV_CAMERA_SWIVEL_LEFT)) * Degree(90);
78  camRotYSwivel = (RoR::App::GetInputEngine()->getEventValue(EV_CAMERA_SWIVEL_UP) - RoR::App::GetInputEngine()->getEventValue(EV_CAMERA_SWIVEL_DOWN)) * Degree(60);
79 
80  camRotYSwivel = std::max((Radian)Degree(-80) - camRotY, camRotYSwivel);
81  camRotYSwivel = std::min(camRotYSwivel, (Radian)Degree(88) - camRotY);
82 
83  if (RoR::App::GetInputEngine()->getEventBoolValue(EV_CAMERA_ZOOM_IN) && camDist > 1)
84  {
85  camDist -= ctx.mTransScale;
86  }
87  if (RoR::App::GetInputEngine()->getEventBoolValue(EV_CAMERA_ZOOM_IN_FAST) && camDist > 1)
88  {
89  camDist -= ctx.mTransScale * 10;
90  }
91  if (RoR::App::GetInputEngine()->getEventBoolValue(EV_CAMERA_ZOOM_OUT))
92  {
93  camDist += ctx.mTransScale;
94  }
95  if (RoR::App::GetInputEngine()->getEventBoolValue(EV_CAMERA_ZOOM_OUT_FAST))
96  {
97  camDist += ctx.mTransScale * 10;
98  }
99 
100  if (RoR::App::GetInputEngine()->getEventBoolValue(EV_CAMERA_RESET))
101  {
102  reset(ctx);
103  }
104 
105  if (RoR::App::GetInputEngine()->isKeyDown(OIS::KC_RSHIFT) && RoR::App::GetInputEngine()->isKeyDownValueBounce(OIS::KC_SPACE))
106  {
107  limitCamMovement = !limitCamMovement;
108  if (limitCamMovement)
109  {
110  RoR::App::GetConsole()->putMessage(Console::CONSOLE_MSGTYPE_INFO, Console::CONSOLE_SYSTEM_NOTICE, _L("Limited camera movement enabled"), "camera_go.png", 3000);
111  RoR::App::GetGuiManager()->PushNotification("Notice:", _L("Limited camera movement enabled"));
112  }
113  else
114  {
115  RoR::App::GetConsole()->putMessage(Console::CONSOLE_MSGTYPE_INFO, Console::CONSOLE_SYSTEM_NOTICE, _L("Limited camera movement disabled"), "camera_go.png", 3000);
116  RoR::App::GetGuiManager()->PushNotification("Notice:", _L("Limited camera movement disabled"));
117  }
118  }
119 
120  if (limitCamMovement && camDistMin > 0.0f)
121  {
122  camDist = std::max(camDistMin, camDist);
123  }
124  if (limitCamMovement && camDistMax > 0.0f)
125  {
126  camDist = std::min(camDist, camDistMax);
127  }
128 
129  camDist = std::max(0.0f, camDist);
130 
131  Vector3 desiredPosition = camLookAt + camDist * 0.5f * Vector3(
132  sin(targetDirection.valueRadians() + (camRotX - camRotXSwivel).valueRadians()) * cos(targetPitch.valueRadians() + (camRotY - camRotYSwivel).valueRadians())
133  , sin(targetPitch.valueRadians() + (camRotY - camRotYSwivel).valueRadians())
134  , cos(targetDirection.valueRadians() + (camRotX - camRotXSwivel).valueRadians()) * cos(targetPitch.valueRadians() + (camRotY - camRotYSwivel).valueRadians())
135  );
136 
137  if (limitCamMovement && gEnv->terrainManager && gEnv->terrainManager->getHeightFinder())
138  {
139  float h = gEnv->terrainManager->getHeightFinder()->getHeightAt(desiredPosition.x, desiredPosition.z) + 1.0f;
140 
141  desiredPosition.y = std::max(h, desiredPosition.y);
142  }
143 
144  if (camLookAtLast == Vector3::ZERO)
145  {
146  camLookAtLast = camLookAt;
147  }
148  if (camLookAtSmooth == Vector3::ZERO)
149  {
150  camLookAtSmooth = camLookAt;
151  }
152  if (camLookAtSmoothLast == Vector3::ZERO)
153  {
154  camLookAtSmoothLast = camLookAtSmooth;
155  }
156 
157  Vector3 camDisplacement = camLookAt - camLookAtLast;
158  Vector3 precedingLookAt = camLookAtSmoothLast + camDisplacement;
159  Vector3 precedingPosition = gEnv->mainCamera->getPosition() + camDisplacement;
160 
161  Vector3 camPosition = (1.0f / (camRatio + 1.0f)) * desiredPosition + (camRatio / (camRatio + 1.0f)) * precedingPosition;
162 
163  if (gEnv->collisions && gEnv->collisions->forcecam)
164  {
165  gEnv->mainCamera->setPosition(gEnv->collisions->forcecampos);
166  gEnv->collisions->forcecam = false;
167  }
168  else
169  {
170  if (ctx.mCurrTruck && ctx.mCurrTruck->replaymode && camDisplacement != Vector3::ZERO)
171  gEnv->mainCamera->setPosition(desiredPosition);
172  else
173  gEnv->mainCamera->setPosition(camPosition);
174  }
175 
176  camLookAtSmooth = (1.0f / (camRatio + 1.0f)) * camLookAt + (camRatio / (camRatio + 1.0f)) * precedingLookAt;
177 
178  camLookAtLast = camLookAt;
179  camLookAtSmoothLast = camLookAtSmooth;
180  gEnv->mainCamera->lookAt(camLookAtSmooth);
181 }

183 bool CameraBehaviorOrbit::mouseMoved(const CameraManager::CameraContext& ctx, const OIS::MouseEvent& _arg)
184 {
185  const OIS::MouseState ms = _arg.state;
186 
187  if (ms.buttonDown(OIS::MB_Right))
188  {
189  float scale = RoR::App::GetInputEngine()->isKeyDown(OIS::KC_LMENU) ? 0.002f : 0.02f;
190  camRotX += Degree(ms.X.rel * 0.13f);
191  camRotY += Degree(-ms.Y.rel * 0.13f);
192  camDist += -ms.Z.rel * scale;

194  }
