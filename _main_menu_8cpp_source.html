88 void MainMenu::EnterMainMenuLoop()
89 {
90  RoRWindowEventUtilities::addWindowEventListener(App::GetOgreSubsystem()->GetRenderWindow(), this);
91 
92  // ==== FPS-limiter ====
93  // TODO: Is this necessary in menu?
94 
95  unsigned long timeSinceLastFrame = 1;
96  unsigned long startTime = 0;
97  unsigned long minTimePerFrame = 0;
98  unsigned long fpsLimit = App::GetGfxFpsLimit();
99 
100  if (fpsLimit < 10 || fpsLimit >= 200)
101  {
102  fpsLimit = 0;
103  }
104 
105  if (fpsLimit)
106  {
107  minTimePerFrame = 1000 / fpsLimit;
108  }
109 
110  while (App::GetPendingAppState() == App::APP_STATE_NONE)
111  {
112  startTime = App::GetOgreSubsystem()->GetTimer()->getMilliseconds();
113 
114 
115  MainMenuLoopUpdate(timeSinceLastFrame);
116 
117  if (RoR::App::GetGuiManager()->GetMainSelector()->IsFinishedSelecting())
118  {
119  CacheEntry* selected_map = RoR::App::GetGuiManager()->GetMainSelector()->GetSelectedEntry();
120  if (selected_map != nullptr)
121  {
122  App::GetGuiManager()->GetMainSelector()->Reset(); // TODO: Eliminate this mechanism ~ only_a_ptr 09/2016
123  App::SetPendingAppState(App::APP_STATE_SIMULATION);
124  App::SetSimNextTerrain(selected_map->fname);
125  }
126  }
127 
128 #if OGRE_PLATFORM == OGRE_PLATFORM_WIN32 || OGRE_PLATFORM == OGRE_PLATFORM_LINUX
129  RoRWindowEventUtilities::messagePump();
130 #endif
131  Ogre::RenderWindow* rw = RoR::App::GetOgreSubsystem()->GetRenderWindow();
132  if (rw->isClosed())
133  {
134  App::SetPendingAppState(App::APP_STATE_SHUTDOWN);
135  continue;
136  }
137 
138  RoR::App::GetOgreSubsystem()->GetOgreRoot()->renderOneFrame();
139 
140 #ifdef USE_SOCKETW
141  if ((App::GetActiveMpState() == App::MP_STATE_CONNECTED) && RoR::Networking::CheckError())
142  {
143  Ogre::String title = Ogre::UTFString(_L("Network fatal error: ")).asUTF8();
144  Ogre::String msg = RoR::Networking::GetErrorMessage().asUTF8();
145  App::GetGuiManager()->ShowMessageBox(title, msg, true, "OK", true, false, "");
146  App::SetPendingAppState(App::APP_STATE_MAIN_MENU);
147 
148  RoR::App::GetGuiManager()->GetMainSelector()->Hide();
149  RoR::App::GetGuiManager()->GetMainSelector()->Show(LT_Terrain);
150  }
151 #endif
152 
153  if (!rw->isActive() && rw->isVisible())
154  rw->update(); // update even when in background !
155 
156  // FPS-limiter. TODO: Is this necessary in menu?
157  if (fpsLimit && timeSinceLastFrame < minTimePerFrame)
158  {
159  // Sleep twice as long as we were too fast.
160  int ms = static_cast<int>((minTimePerFrame - timeSinceLastFrame) << 1);
161  std::this_thread::sleep_for(std::chrono::milliseconds(ms));
162  }
163 
164  timeSinceLastFrame = RoR::App::GetOgreSubsystem()->GetTimer()->getMilliseconds() - startTime;
165  }
166 
167  RoRWindowEventUtilities::removeWindowEventListener(App::GetOgreSubsystem()->GetRenderWindow(), this);
168 }

170 void MainMenu::MainMenuLoopUpdate(float seconds_since_last_frame)
171 {
172  if (seconds_since_last_frame == 0)
173  {
174  return;
175  }
176  if (seconds_since_last_frame > 1.0 / 20.0) // TODO: Does this have any sense for menu?
177  {
178  seconds_since_last_frame = 1.0 / 20.0;
179  }
180 
181  if (RoR::App::GetOgreSubsystem()->GetRenderWindow()->isClosed())
182  {
183  App::SetPendingAppState(App::APP_STATE_SHUTDOWN);
184  return;
185  }
186 
187 #ifdef USE_SOCKETW
188  if (App::GetActiveMpState() == App::MP_STATE_CONNECTED)
189  {
190  App::GetGuiManager()->GetMpClientList()->update();
191  }
192  else if (App::GetGuiManager()->GetMpSelector()->IsRefreshThreadRunning())
193  {
194  App::GetGuiManager()->GetMpSelector()->CheckAndProcessRefreshResult();
195  }
196 #endif // USE_SOCKETW
197 
198  RoR::App::GetInputEngine()->Capture();
199 
200  MainMenuLoopUpdateEvents(seconds_since_last_frame);
201 
202 #ifdef USE_ANGELSCRIPT
203  ScriptEngine::getSingleton().framestep(seconds_since_last_frame);
204 #endif
205 }

207 void MainMenu::MainMenuLoopUpdateEvents(float seconds_since_last_frame)
208 {
209  if (seconds_since_last_frame == 0.0f)
210  {
211  return;
212  }
213 
214  RoR::App::GetInputEngine()->updateKeyBounces(seconds_since_last_frame);
215 
216  if (! RoR::App::GetInputEngine()->getInputsChanged())
217  {
218  return;
219  }
220 
221  if (RoR::App::GetOverlayWrapper() != nullptr)
222  {
223  RoR::App::GetOverlayWrapper()->update(seconds_since_last_frame); // TODO: What's the meaning of this? It only updates some internal timer. ~ only_a_ptr
224  }
225 
226  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_COMMON_QUIT_GAME))
227  {
228  //TODO: Go back to menu
229  App::SetPendingAppState(App::APP_STATE_SHUTDOWN);
230  return;
231  }
232 
233  auto gui_man = App::GetGuiManager();
234  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_COMMON_CONSOLE_TOGGLE, 5.f))
235  {
236  gui_man->SetVisible_Console(!gui_man->IsVisible_Console());
237  }
238 
239  // TODO: screenshot
240 
241  // FOV settings disabled in menu
242 
243  // FIXME: full screen/windowed screen switching
244  //if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_COMMON_FULLSCREEN_TOGGLE, 2.0f)) {}
245 }

247 void MainMenu::JoinMultiplayerServer()
248 {
249 #ifdef USE_SOCKETW
250 
251  auto gui = App::GetGuiManager();
252  gui->SetVisible_MultiplayerSelector(false);
253  gui->SetVisible_GameMainMenu(false);
254 
255  gui->GetLoadingWindow()->setAutotrack(_L("Connecting to server ..."));
256 
257  if (!Networking::Connect())
258  {
259  LOG("connection failed. server down?");
260  gui->SetVisible_LoadingWindow(false);
261  gui->SetVisible_GameMainMenu(true);
262 
263  gui->ShowMessageBox("Connection failed", Networking::GetErrorMessage().asUTF8_c_str(), true, "OK", true, false, "");
264  return;
265  }
266 
267  gui->SetVisible_LoadingWindow(false);
268  gui->SetVisible_MpClientList(true);
269  gui->GetMpClientList()->update();
270 
271  ChatSystem::SendStreamSetup();
272 
273 #ifdef USE_MUMBLE
274  SoundScriptManager::getSingleton().CheckAndCreateMumble();
275 #endif // USE_MUMBLE
276 
277  String terrain_name = Networking::GetTerrainName();
278  if (terrain_name != "any")
279  {
280  App::SetSimNextTerrain(terrain_name);
281  App::SetPendingAppState(App::APP_STATE_SIMULATION);
282  }
283  else
284  {
285  // Connected -> go directly to map selector
286  if (App::GetDiagPreselectedTerrain() == "")
287  {
288  gui->GetMainSelector()->Reset();
289  gui->GetMainSelector()->Show(LT_Terrain);
290  }
291  else
292  {
293  App::SetPendingAppState(App::APP_STATE_SIMULATION);
294  }
295  }
296 #endif //SOCKETW
297 }
