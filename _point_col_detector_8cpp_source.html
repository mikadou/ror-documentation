289 void PointColDetector::partintwo(const int start, const int median, const int end, const int axis, float &minex, float &maxex)
290 {
291  int i, j, l, m;
292  int k = median;
293  l = start;
294  m = end - 1;
295 
296  float x = ref_list[k].point[axis];
297  while (l < m) {
298  i = l;
299  j = m;
300  while (!(j < k || k < i)) {
301  while (ref_list[i].point[axis] < x) {
302  i++;
303  }
304  while (x < ref_list[j].point[axis]) {
305  j--;
306  }
307 
308  std::swap(ref_list[i], ref_list[j]);
309  i++;
310  j--;
311  }
312  if (j < k) {
313  l = i;
314  }
315  if (k < i) {
316  m = j;
317  }
318  x = ref_list[k].point[axis];
319  }
320 
321  minex = x;

323  for (int i = start; i < median; ++i) {
324  minex = std::min(ref_list[i].point[axis], minex);
