33 Real CmdKeyInertia::calcCmdKeyDelay(Real cmdInput, int cmdKey, Real dt)
34 {
35  if (!cmdKeyInertia[cmdKey].startSpline || !cmdKeyInertia[cmdKey].stopSpline)
36  {
37  return cmdInput;
38  }
39 
40  Real calculatedOutput = cmdKeyInertia[cmdKey].lastOutput;
41  Real lastOutput = cmdKeyInertia[cmdKey].lastOutput;
42  // rel difference to calculate if we have to use start values(accelerating) or stop values
43  Real relDiff = fabs(cmdInput) - fabs(lastOutput);
44  // difference to calculate if were are on the negative side
45  Real absDiff = cmdInput - lastOutput;
46  // if the value is close to our input, reset the timer
47  if (fabs(absDiff) < 0.002)
48  cmdKeyInertia[cmdKey].time = 0;
49  // +dt after the timer had been set to zero prevents the motion to stop at 0.002
50  cmdKeyInertia[cmdKey].time += dt;
51 
52  Real startFactor = cmdKeyInertia[cmdKey].startDelay * cmdKeyInertia[cmdKey].time;
53  Real stopFactor = cmdKeyInertia[cmdKey].stopDelay * cmdKeyInertia[cmdKey].time;
54  // positive values between 0 and 1
55  if (absDiff > 0)
56  { // we have to accelerate our last outout to the new commanded input
57  if (relDiff > 0)
58  calculatedOutput = lastOutput + calculateCmdOutput(startFactor, cmdKeyInertia[cmdKey].startSpline);
59  if (relDiff < 0)
60  // we have to deccelerate our last outout to the new commanded input
61  calculatedOutput = lastOutput + calculateCmdOutput(stopFactor, cmdKeyInertia[cmdKey].stopSpline);
62  if (calculatedOutput > cmdInput)
63  // if the calculated value is bigger than input set to input to avoid overshooting
64  calculatedOutput = cmdInput;
65  }
66  // negative values, mainly needed for hydros, between 0 and -1
67  if (absDiff < 0)
68  {
69  if (relDiff > 0)
70  calculatedOutput = lastOutput - calculateCmdOutput(startFactor, cmdKeyInertia[cmdKey].startSpline);
71  if (relDiff < 0)
72  calculatedOutput = lastOutput - calculateCmdOutput(stopFactor, cmdKeyInertia[cmdKey].stopSpline);
73  if (calculatedOutput < cmdInput)
74  calculatedOutput = cmdInput;
75  }
76  cmdKeyInertia[cmdKey].lastOutput = calculatedOutput;
77  return calculatedOutput;
78 }

120 int CmdKeyInertia::loadDefaultInertiaModels()
121 {
122  // check if we have a config file
123  String group = "";
124  try
125  {
126  group = ResourceGroupManager::getSingleton().findGroupContainingResource("inertia_models.cfg");
127  }
128  catch (...)
129  {
130  }
131  // emit a warning if we did not found the file
132  if (group.empty())
133  {
134  LOG("Inertia| inertia_models.cfg not found");
135  return 1;
136  }
137 
138  // open the file for reading
139  DataStreamPtr ds = ResourceGroupManager::getSingleton().openResource("inertia_models.cfg", group);
140  String line = "";
141  String currentModel = "";
142 
143  while (!ds->eof())
144  {
145  line = RoR::Utils::SanitizeUtf8String(ds->getLine());
146  StringUtil::trim(line);
147 
148  if (line.empty() || line[0] == ';')
149  continue;
150 
151  Ogre::StringVector args = StringUtil::split(line, ",");
152 
153  if (args.size() == 1)
154  {
155  currentModel = line;
156  continue;
157  }
158 
159  // process the line if we got a model
160  if (!currentModel.empty())
161  processLine(args, currentModel);
162  }
163  return 0;
164 }

166 int CmdKeyInertia::processLine(Ogre::StringVector args, String model)
167 {
168  // we only accept 2 arguments
169  if (args.size() != 2)
170  return 1;
171  // parse the data
172  float pointx = StringConverter::parseReal(args[0]);
173  float pointy = StringConverter::parseReal(args[1]);
174  Vector3 point = Vector3(pointx, pointy, 0.0f);
175 
176  // find the spline to attach the points
177  if (splines.find(model) == splines.end())
178  {
179  splines[model] = SimpleSpline();
180  }
181 
182  // attach the points to the spline
183  splines[model].addPoint(point);
184 
185  return 0;
186 }
