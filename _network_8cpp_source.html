235 {

267 int ReceiveMessage(RoRnet::Header *head, char* content, int bufferlen)
268 {
269  SWBaseSocket::SWBaseError error;
270 
271  char buffer[RORNET_MAX_MESSAGE_LENGTH] = {0};
272 
273 #ifdef DEBUG
274  LOG_THREAD("[RoR|Networking] ReceiveMessage() waiting...");
275 #endif //DEBUG
276 
277  int hlen = 0;
278  while (hlen < (int)sizeof(RoRnet::Header))
279  {
280  int recvnum = socket.recv(buffer + hlen, sizeof(RoRnet::Header) - hlen, &error);
281  if (recvnum < 0 && !m_shutdown)
282  {
283  LOG("NET receive error 1: " + error.get_error());
284  return -1;
285  }
286  else if (m_shutdown)
287  {
288  break;
289  }
290  hlen += recvnum;
291  }
292 
293  if (m_shutdown)
294  {
295  return RECVMESSAGE_RETVAL_SHUTDOWN;
296  }
297 
298  memcpy(head, buffer, sizeof(RoRnet::Header));
299 
300 #ifdef DEBUG
301  LOG_THREAD("[RoR|Networking] ReceiveMessage() header received");
302 #endif //DEBUG
303 
304  if (head->size >= RORNET_MAX_MESSAGE_LENGTH)
305  {
306  return -3;
307  }
308 
309  if (head->size > 0)
310  {
311  // Read the packet content
312  while (hlen < (int)sizeof(RoRnet::Header) + (int)head->size)
313  {
314  int recvnum = socket.recv(buffer + hlen, (head->size + sizeof(RoRnet::Header)) - hlen, &error);
315  if (recvnum < 0 && !m_shutdown)
316  {
317  LOG_THREAD("NET receive error 2: "+ error.get_error());
318  return -1;
319  }
320  else if (m_shutdown)
321  {
322  break;
323  }
324  hlen += recvnum;
325  }
326  }
327 
328  memcpy(content, buffer + sizeof(RoRnet::Header), bufferlen);
329 
330 #ifdef DEBUG
331  LOG_THREAD("[RoR|Networking] ReceiveMessage() body received");
332 #endif //DEBUG
333 
334  return 0;
335 }

337 void SendThread()
338 {
339  LOG("[RoR|Networking] SendThread started");
340  while (!m_shutdown)
341  {
342  std::unique_lock<std::mutex> queue_lock(m_send_packetqueue_mutex);
343  while (!m_shutdown && m_send_packet_buffer.empty())
344  {
345  m_send_packet_available_cv.wait(queue_lock);
346  }
347 
348  while (!m_shutdown && !m_send_packet_buffer.empty())
349  {
350  send_packet_t packet = m_send_packet_buffer.front();
351  m_send_packet_buffer.pop_front();
352 
353  queue_lock.unlock();
354  SendMessageRaw(packet.buffer, packet.size);
355  queue_lock.lock();
356  }
357  queue_lock.unlock();
358  std::this_thread::sleep_for(std::chrono::milliseconds(200));
359  queue_lock.lock();
360  }
361  LOG("[RoR|Networking] SendThread stopped");
362 }

364 void RecvThread()
365 {
366  LOG_THREAD("[RoR|Networking] RecvThread starting...");
367 
368  RoRnet::Header header;
369 
370  char *buffer = (char*)malloc(RORNET_MAX_MESSAGE_LENGTH);
371 
372  while (!m_shutdown)
373  {
374  int err = ReceiveMessage(&header, buffer, RORNET_MAX_MESSAGE_LENGTH);
375  //LOG("Received data: " + TOSTRING(header.command) + ", source: " + TOSTRING(header.source) + ":" + TOSTRING(header.streamid) + ", size: " + TOSTRING(header.size));
376  if (err != 0)
377  {
378  if (err != RECVMESSAGE_RETVAL_SHUTDOWN)
379  {
380  std::stringstream s;
381  s << "[RoR|Networking] RecvThread: Error while receiving data: " << err << ", tid: " <<std::this_thread::get_id();
382  NetFatalError(s.str());
383  }
384  break;
385  }
386 
387  if (header.command == MSG2_STREAM_REGISTER)
388  {
389  if (header.source == m_uid)
390  continue;
391 
392  RoRnet::StreamRegister *reg = (RoRnet::StreamRegister *)buffer;
393 
394  LOG(" * received stream registration: " + TOSTRING(header.source) + ": " + TOSTRING(header.streamid) + ", type: " + TOSTRING(reg->type));
395  }
396  else if (header.command == MSG2_STREAM_REGISTER_RESULT)
397  {
398  RoRnet::StreamRegister *reg = (RoRnet::StreamRegister *)buffer;
399  LOG(" * received stream registration result: " + TOSTRING(header.source) + ": " + TOSTRING(header.streamid) + ", status: " + TOSTRING(reg->status));
400  }
401  else if (header.command == MSG2_STREAM_UNREGISTER)
402  {
403  LOG(" * received stream deregistration: " + TOSTRING(header.source) + ": " + TOSTRING(header.streamid));
404  }
405  else if (header.command == MSG2_UTF8_CHAT || header.command == MSG2_UTF8_PRIVCHAT)
406  {
407  // Chat message
408  }
409  else if (header.command == MSG2_NETQUALITY && header.source == -1)
410  {
411  if (header.size != sizeof(int))
412  {
413  continue;
414  }
415  int quality = *(int *)buffer;
416  SetNetQuality(quality);
417  continue;
418  }
419  else if (header.command == MSG2_USER_LEAVE)
420  {
421  if (header.source == m_uid)
422  {
423  NetFatalError(_L("disconnected: remote side closed the connection"));
424  return;
425  }
426 
427  { // Lock scope
428  std::lock_guard<std::mutex> lock(m_users_mutex);
429  auto user = std::find_if(m_users.begin(), m_users.end(), [header](const RoRnet::UserInfo u) { return static_cast<int>(u.uniqueid) == header.source; });
430  if (user != m_users.end())
431  {
432  Ogre::UTFString msg = RoR::ChatSystem::GetColouredName(user->username, user->colournum) + RoR::Color::CommandColour + _L(" left the game");
433  const char *utf8_line = msg.asUTF8_c_str();
434  RoRnet::Header head;
435  head.command = MSG2_UTF8_CHAT;
436  head.source = -1;
437  head.size = (int)strlen(utf8_line);
438  QueueStreamData(head, (char *)utf8_line, strlen(utf8_line) + 1);
439  LOG_THREAD(Ogre::UTFString(user->username) + _L(" left the game"));
440  m_users.erase(user);
441  }
442  }
443  RoR::App::GetGuiManager()->GetTopMenubar()->triggerUpdateVehicleList();
444  }
445  else if (header.command == MSG2_USER_INFO || header.command == MSG2_USER_JOIN)
446  {
447  if (header.source == m_uid)
448  {
449  std::lock_guard<std::mutex> lock(m_userdata_mutex);
450  memcpy(&m_userdata, buffer, sizeof(RoRnet::UserInfo));
451  m_authlevel = m_userdata.authstatus;
452  m_username = Ogre::UTFString(m_userdata.username);
453  // TODO: Update the global variable 'mp_player_name' in a threadsafe way.
454  }
455  else
456  {
457  RoRnet::UserInfo user_info;
458  memcpy(&user_info, buffer, sizeof(RoRnet::UserInfo));
459 
460  bool user_exists = false;
461  {
462  std::lock_guard<std::mutex> lock(m_users_mutex);
463  for (RoRnet::UserInfo &user : m_users)
464  {
465  if ((int)user.uniqueid == header.source)
466  {
467  user = user_info;
468  user_exists = true;
469  break;
470  }
471  }
472  if (!user_exists)
473  {
474  m_users.push_back(user_info);
475  Ogre::UTFString msg = RoR::ChatSystem::GetColouredName(user_info.username, user_info.colournum) + RoR::Color::CommandColour + _L(" joined the game");
476  const char *utf8_line = msg.asUTF8_c_str();
477  RoRnet::Header head;
478  head.command = MSG2_UTF8_CHAT;
479  head.source = -1;
480  head.size = (int)strlen(utf8_line);
481  QueueStreamData(head, (char *)utf8_line, strlen(utf8_line) + 1);
482  LOG(Ogre::UTFString(user_info.username) + _L(" joined the game"));
483  }
484  }
485  RoR::App::GetGuiManager()->GetTopMenubar()->triggerUpdateVehicleList();
486  }
487  continue;
488  }
489  else if (header.command == MSG2_GAME_CMD)
490  {
491 #ifdef USE_ANGELSCRIPT
492  ScriptEngine::getSingleton().queueStringForExecution(Ogre::String(buffer));
493 #endif // USE_ANGELSCRIPT
494  continue;
495  }
496  //DebugPacket("receive-1", &header, buffer);
497 
498  QueueStreamData(header, buffer, RORNET_MAX_MESSAGE_LENGTH);
499  }
500 
501  m_recv_stopped = true;
502 
503  LOG_THREAD("[RoR|Networking] RecvThread stopped");
504 }

522 bool Connect()
523 {
524  // Temporary workaround for unrecoverable error
525  if (m_socket_broken)
526  {
527  RoR::App::SetActiveMpState(App::MP_STATE_DISABLED);
528  RoR::App::SetPendingMpState(App::MP_STATE_NONE);
529  Ogre::UTFString msg = "Connection failed."
530  "\n\nNetworking is unable to recover from previous error (abrupt disconnect)."
531  "\n\nPlease restart Rigs of Rods.";
532  SetErrorMessage(msg);
533  return false;
534  }
535 
536  // Reset errors
537  SetErrorMessage("");
538  m_net_fatal_error = false;
539 
540  m_username = App::GetMpPlayerName();
541 
542  LOG("[RoR|Networking] Trying to join server '" + App::GetMpServerHost() + "' on port " + TOSTRING(App::GetMpServerPort()) + "'...");
543 
544  SWBaseSocket::SWBaseError error;
545 
546  socket.set_timeout(10, 10000);
547  socket.connect(App::GetMpServerPort(), App::GetMpServerHost(), &error);
548  if (error != SWBaseSocket::ok)
549  {
550  CouldNotConnect("Could not create connection", false);
551  return false;
552  }
553  if (!SendNetMessage(MSG2_HELLO, 0, (int)strlen(RORNET_VERSION), (char *)RORNET_VERSION))
554  {
555  CouldNotConnect(_L("Establishing network session: error sending hello"));
556  return false;
557  }
558 
559  RoRnet::Header header;
560  char buffer[RORNET_MAX_MESSAGE_LENGTH] = {0};
561 
562  // Receive server (rornet protocol) version
563  if (ReceiveMessage(&header, buffer, 255))
564  {
565  CouldNotConnect(_L("Establishing network session: error getting server version"));
566  return false;
567  }
568  if (header.command == MSG2_WRONG_VER_LEGACY) // Pre-RoRnet_2.40 server
569  {
570  RoRnet::LegacyServerInfo info;
571  memcpy(&info, buffer, sizeof(RoRnet::LegacyServerInfo));
572  Ogre::UTFString format_wstr = _L("Establishing network session: wrong server version, you are using version '%s' and the server is using '%s'");
573  const char* server_ver = (info.protocolversion[0] != 0) ? info.protocolversion : "~ RoRnet_2.38 or earlier (not detected) ~";
574  char msg_buf[500];
575  snprintf(msg_buf, 500, format_wstr.asUTF8_c_str(), RORNET_VERSION, server_ver);
576  CouldNotConnect(msg_buf);
577  return false;
578  }
579  if (header.command == MSG2_WRONG_VER)
580  {
581  CouldNotConnect(_L("server uses a different protocol version"));
582  return false;
583  }
584  if (header.command != MSG2_HELLO)
585  {
586  CouldNotConnect(_L("Establishing network session: error getting server hello"));
587  return false;
588  }
589 
590  // Save server settings
591  memcpy(&m_server_settings, buffer, sizeof(RoRnet::ServerInfo));
592 
593  if (strncmp(m_server_settings.protocolversion, RORNET_VERSION, strlen(RORNET_VERSION)))
594  {
595  wchar_t tmp[512] = L"";
596  Ogre::UTFString tmp2 = _L("Establishing network session: wrong server version, you are using version '%s' and the server is using '%s'");
597  swprintf(tmp, 512, tmp2.asWStr_c_str(), RORNET_VERSION, m_server_settings.protocolversion);
598  CouldNotConnect(MyGUI::UString(tmp).asUTF8_c_str());
599  return false;
600  }
601 
602  // First handshake done, increase the timeout, important!
603  socket.set_timeout(0, 0);
604 
605  // Send credentials
606  char pwbuffer[250] = {0};
607  strncpy(pwbuffer, App::GetMpServerPassword().c_str(), 250);
608 
609  char sha1pwresult[250] = {0};
610  if (strnlen(pwbuffer, 250) > 0)
611  {
612  RoR::CSHA1 sha1;
613  sha1.UpdateHash((uint8_t *)pwbuffer, (uint32_t)strnlen(pwbuffer, 250));
614  sha1.Final();
615  sha1.ReportHash(sha1pwresult, RoR::CSHA1::REPORT_HEX_SHORT);
616  }
617 
618  Ogre::String usertokenhash = SSETTING("User Token Hash", "");
619 
620  // Construct user credentials
621  // Beware of the wchar_t converted to UTF8 for networking
622  RoRnet::UserInfo c;

736 void AddPacket(int streamid, int type, int len, char *content)
737 {
738  if (len > RORNET_MAX_MESSAGE_LENGTH)
739  {
740  LOGSTREAM << "[RoR|Networking] Discarding network packet (StreamID: "
741  <<streamid<<", Type: "<<type<<"), length is " << len << ", max is " << RORNET_MAX_MESSAGE_LENGTH;
742  return;
743  }
744 
745  send_packet_t packet;
746  memset(&packet, 0, sizeof(send_packet_t));
747 
748  char *buffer = (char*)(packet.buffer);
749 
750  RoRnet::Header *head = (RoRnet::Header *)buffer;
751  head->command = type;
752  head->source = m_uid;
753  head->size = len;
754  head->streamid = streamid;
755 
756  // then copy the contents
757  char *bufferContent = (char *)(buffer + sizeof(RoRnet::Header));
758  memcpy(bufferContent, content, len);
759 
760  // record the packet size
761  packet.size = len + sizeof(RoRnet::Header);
762 
763  { // Lock scope
764  std::lock_guard<std::mutex> lock(m_send_packetqueue_mutex);
765  if (type == MSG2_STREAM_DATA)
766  {
767  if (m_send_packet_buffer.size() > m_packet_buffer_size)
768  {
769  // buffer full, discard unimportant data packets
770  return;
771  }
772  auto search = std::find_if(m_send_packet_buffer.begin(), m_send_packet_buffer.end(), [packet](const send_packet_t& p) { return memcmp(packet.buffer, p.buffer, sizeof(RoRnet::Header)) == 0; });
773  if (search != m_send_packet_buffer.end())
774  {
775  // Found an older packet with the same header -> replace it
776  (*search) = packet;
777  return;
778  }
779  }
780  m_send_packet_buffer.push_back(packet);
781  }
782 
783  m_send_packet_available_cv.notify_one();
784 }
