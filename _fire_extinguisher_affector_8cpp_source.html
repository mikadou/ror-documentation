75 //-----------------------------------------------------------------------
76 void FireExtinguisherAffector::_affectParticles(ParticleSystem* pSystem, Real timeElapsed)
77 {
78  ExtinguishableFireAffectorFactory::affectorIterator affIt = mEfaf->getAffectorIterator();
79  ExtinguishableFireAffector* fire;
80 
81  while(affIt.hasMoreElements())
82  {
83  fire = (ExtinguishableFireAffector*)affIt.getNext();
84 
85  if (fire->isTemplate())
86  continue;
87 
88  Real squaredRadius = Math::Pow(fire->getRadius(), 2);
89  Vector3 middlePoint = fire->getAbsoluteMiddlePoint();
90 
91  ParticleIterator pi = pSystem->_getIterator();
92  Particle *p;
93  int fireHits = 0;
94  while (!pi.end())
95  {
96  p = pi.getNext();
97 
98  if ( middlePoint.squaredDistance(p->position) < squaredRadius )
99  {
100  // This particle is inside the fire, dispose of it in the next update
101  p->timeToLive = 0;
102  ++fireHits;
103  }
104  }
105  if (fireHits>0)
106  {
107  Real intensity = fire->reduceIntensity(fireHits*mEffectiveness);
108  if (intensity<0) delete fire->getParticleSystem();
109  }
110  }
111 }
