92 bool CameraManager::Update(float dt, Beam* cur_truck, float sim_speed) // Called every frame
93 {
94  if (RoR::App::GetActiveSimState() == RoR::App::SIM_STATE_PAUSED) { return true; } // Do nothing when paused
95 
96  if ( dt == 0 ) return false;
97 
98  mTransScale = mTransSpeed * dt;
99  mRotScale = mRotateSpeed * dt;
100 
101  ctx.mCurrTruck = cur_truck;
102  ctx.mSimSpeed = sim_speed;
103  ctx.mDt = dt;
104  ctx.mRotScale = Degree(mRotScale);
105  ctx.mTransScale = mTransScale;
106  ctx.fovInternal = Degree(App::GetGfxFovInternal());
107  ctx.fovExternal = Degree(App::GetGfxFovExternal());
108 
109  if ( currentBehaviorID < CAMERA_BEHAVIOR_END && RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_CAMERA_CHANGE) )
110  {
111  switchToNextBehavior(false);
112  }
113 
114  if ( RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_CAMERA_FREE_MODE_FIX) )
115  {
116  toggleBehavior(CAMERA_BEHAVIOR_FIXED);
117  }
118 
119  if ( RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_CAMERA_FREE_MODE) )
120  {
121  toggleBehavior(CAMERA_BEHAVIOR_FREE);
122  }
123 
124  if ( currentBehavior )
125  {
126  currentBehavior->update(ctx);
127  } else
128  {
129  switchBehavior(CAMERA_BEHAVIOR_CHARACTER);
130  }
131 
132  return true;
133 }

144 void CameraManager::switchBehavior(int newBehaviorID, bool reset)
145 {
146  if (newBehaviorID == currentBehaviorID)
147  {
148  return;
149  }
150 
151  if ( globalBehaviors.find(newBehaviorID) == globalBehaviors.end() )
152  {
153  return;
154  }
155 
156  // deactivate old
157  if ( currentBehavior )
158  {
159  currentBehavior->deactivate(ctx);
160  }
161 
162  // set new
163  currentBehavior = globalBehaviors[newBehaviorID];
164  currentBehaviorID = newBehaviorID;
165 
166  // activate new
167  if (ctx.mCurrTruck != nullptr)
168  {
169  ctx.mCurrTruck->GetCameraContext()->behavior = RoR::PerVehicleCameraContext::CAMCTX_BEHAVIOR_EXTERNAL;
170  }
171  currentBehavior->activate(ctx, reset);
172 }

174 void CameraManager::SwitchBehaviorOnVehicleChange(int newBehaviorID, bool reset, Beam* old_vehicle, Beam* new_vehicle)
175 {

200 
201  // activate new
202  ctx.mCurrTruck = new_vehicle;
203  currentBehavior->activate(ctx, reset);
204 }

206 void CameraManager::toggleBehavior(int behavior)
207 {
208  static std::stack<int> precedingBehaviors;
209 
210  if ( behavior != currentBehaviorID && (precedingBehaviors.empty() || precedingBehaviors.top() != behavior))
211  {
212  if ( currentBehaviorID >= 0 )
213  {
214  precedingBehaviors.push(currentBehaviorID);
215  }
216  switchBehavior(behavior);
217  } else if ( !precedingBehaviors.empty() )
218  {
219  switchBehavior(precedingBehaviors.top(), false);
220  precedingBehaviors.pop();
221  } else
222  {
223  switchToNextBehavior();
224  }
225 }

296 void CameraManager::NotifyVehicleChanged(Beam* old_vehicle, Beam* new_vehicle)
297 {
298  // Getting out of vehicle
299  if (new_vehicle == nullptr)
300  {
301  ctx.mCurrTruck = nullptr;
302  if ( !(this->currentBehaviorID == CAMERA_BEHAVIOR_STATIC && m_config_exit_vehicle_keep_fixedfreecam) &&
303  !(this->currentBehaviorID == CAMERA_BEHAVIOR_FIXED && m_config_exit_vehicle_keep_fixedfreecam) &&
304  !(this->currentBehaviorID == CAMERA_BEHAVIOR_FREE && m_config_exit_vehicle_keep_fixedfreecam) )
305  {
306  this->switchBehavior(CAMERA_BEHAVIOR_CHARACTER);
307  }
308  return;
309  }
310 
311  // Getting in vehicle
312  if ( !(this->currentBehaviorID == CAMERA_BEHAVIOR_STATIC && m_config_enter_vehicle_keep_fixedfreecam) &&
313  !(this->currentBehaviorID == CAMERA_BEHAVIOR_FIXED && m_config_enter_vehicle_keep_fixedfreecam) &&
314  !(this->currentBehaviorID == CAMERA_BEHAVIOR_FREE && m_config_enter_vehicle_keep_fixedfreecam) )
315  {
316  // Change camera
317  switch (new_vehicle->GetCameraContext()->behavior)
318  {
319  case RoR::PerVehicleCameraContext::CAMCTX_BEHAVIOR_VEHICLE_3rdPERSON:
320  this->SwitchBehaviorOnVehicleChange(CAMERA_BEHAVIOR_VEHICLE, true, old_vehicle, new_vehicle);
321  break;
322 
323  case RoR::PerVehicleCameraContext::CAMCTX_BEHAVIOR_VEHICLE_SPLINE:
324  this->SwitchBehaviorOnVehicleChange(CAMERA_BEHAVIOR_VEHICLE_SPLINE, true, old_vehicle, new_vehicle);
325  break;
326 
327  case RoR::PerVehicleCameraContext::CAMCTX_BEHAVIOR_VEHICLE_CINECAM:
328  this->SwitchBehaviorOnVehicleChange(CAMERA_BEHAVIOR_VEHICLE_CINECAM, true, old_vehicle, new_vehicle);
329  break;
330 
331  default:
332  this->SwitchBehaviorOnVehicleChange(CAMERA_BEHAVIOR_VEHICLE, true, old_vehicle, new_vehicle);
333  }
334  }
