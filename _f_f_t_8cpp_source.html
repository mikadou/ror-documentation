433  {

436  angularFrequencies = new float[resolution*resolution];

438  re = new float[resolution*resolution];

442 
443  std::complex<float>* pInitialWavesData = initialWaves;
444  float* pAngularFrequenciesData = angularFrequencies;
445 
446  float u, v,
447  temp;
448 
449  for (u = 0; u < resolution; u++)
450  {
451  wave.x = (-0.5f * resolution + u) * (2.0f* Ogre::Math::PI / mOptions.PhysicalResolution);
452 
453  for (v = 0; v < resolution; v++)
454  {
455  wave.y = (-0.5f * resolution + v) * (2.0f* Ogre::Math::PI / mOptions.PhysicalResolution);
456 
457  temp = Ogre::Math::Sqrt(0.5f * _getPhillipsSpectrum(wave, mOptions.WindDirection, mOptions.KwPower));
458  *pInitialWavesData++ = std::complex<float>(_getGaussianRandomFloat() * temp, _getGaussianRandomFloat() * temp);
459 
460  temp=9.81f * wave.length();
461  *pAngularFrequenciesData++ = Ogre::Math::Sqrt(temp);
462  }

468  void FFT::_calculeNoise(const float &delta)
469  {
470  time += delta*mOptions.AnimationSpeed;
471 
472  std::complex<float>* pData = currentWaves;
473 
474  int u, v;
475 
476  float wt,
477  coswt, sinwt,
478  realVal, imagVal;
479 
480  for (u = 0; u < resolution; u++)
481  {
482  for (v = 0; v< resolution ; v++)
483  {
484  const std::complex<float>& positive_h0 = initialWaves[u * (resolution)+v];
485  const std::complex<float>& negative_h0 = initialWaves[(resolution-1 - u) * (resolution) + (resolution-1- v)];
486 
487  wt = angularFrequencies[u * (resolution) + v] * time;
488 
489  coswt = Ogre::Math::Cos(wt);
490  sinwt = Ogre::Math::Sin(wt);
491 
492  realVal =
493  positive_h0.real() * coswt - positive_h0.imag() * sinwt + negative_h0.real() * coswt - (-negative_h0.imag()) * (-sinwt),
494  imagVal =
495  positive_h0.real() * sinwt + positive_h0.imag() * coswt + negative_h0.real() * (-sinwt) + (-negative_h0.imag()) * coswt;
496 
497  *pData++ = std::complex<float>(realVal, imagVal);
498  }
499  }
500 
501  _executeInverseFFT();
502  _normalizeFFTData(0);

505  const float FFT::_getGaussianRandomFloat() const
506  {
507  float x1, x2, w, y1;
508 
509  do
510  {
511  x1 = 2.0f * uniform_deviate() - 1.0f;
512  x2 = 2.0f * uniform_deviate() - 1.0f;
513 
514  w = x1 * x1 + x2 * x2;
515 
516  } while ( w >= 1.0f );
517 
518  w = Ogre::Math::Sqrt( (-2.0f * Ogre::Math::Log( w ) ) / w );
519  y1 = x1 * w;
520 
521  return y1;
522  }

524  const float FFT::_getPhillipsSpectrum(const Ogre::Vector2& waveVector, const Ogre::Vector2& wind, const float& kwPower_) const
525  {
526  // Compute the length of the vector
527  float k = waveVector.length();
528 
529  // To avoid division by 0
530  if (k < 0.0000001f)
531  {
532  return 0;
533  }
534  else
535  {
536  float windVelocity = wind.length(),
537  l = pow(windVelocity,2.0f)/9.81f,
538  dot=waveVector.dotProduct(wind);
539 
540  return mOptions.Amplitude*
541  (Ogre::Math::Exp(-1 / pow(k * l,2)) / (Ogre::Math::Pow(k,2) *
542  Ogre::Math::Pow(k,2))) * Ogre::Math::Pow(-dot/ (k * windVelocity), kwPower_);
543  }

546  void FFT::_executeInverseFFT()
547  {
548  int l2n = 0, p = 1;
549  while (p < resolution)
550  {
551  p *= 2; l2n++;
552  }
553  int l2m = 0; p = 1;
554  while (p < resolution)
555  {
556  p *= 2; l2m++;
557  }
558 
559  resolution = 1<<l2m;
560  resolution = 1<<l2n;
561 
562  int x, y, i;
563 
564  for(x = 0; x <resolution; x++)
565  {
566  for(y = 0; y <resolution; y++)
567  {
568  re[resolution * x + y] = currentWaves[resolution * x + y].real();
569  img[resolution * x + y] = currentWaves[resolution * x + y].imag();
570  }
571  }
572 
573  //Bit reversal of each row
574  int j, k;
575  for(y = 0; y < resolution; y++) //for each row
576  {
577  j = 0;
578  for(i = 0; i < resolution - 1; i++)
579  {
580  re[resolution * i + y] = currentWaves[resolution * j + y].real();
581  img[resolution * i + y] = currentWaves[resolution * j + y].imag();
582 
583  k = resolution / 2;
584  while (k <= j)
585  {
586  j -= k;
587  k/= 2;
588  }
589 
590  j += k;
591  }
592  }
593 
594  //Bit reversal of each column
595  float tx = 0, ty = 0;
596  for(x = 0; x < resolution; x++) //for each column
597  {
598  j = 0;
599  for(i = 0; i < resolution - 1; i++)
600  {
601  if(i < j)
602  {
603  tx = re[resolution * x + i];
604  ty = img[resolution * x + i];
605  re[resolution * x + i] = re[resolution * x + j];
606  img[resolution * x + i] = img[resolution * x + j];
607  re[resolution * x + j] = tx;
608  img[resolution * x + j] = ty;
609  }
610  k = resolution / 2;
611  while (k <= j)
612  {
613  j -= k;
614  k/= 2;
615  }
616  j += k;
617  }
618  }
619 
620  //Calculate the FFT of the columns
621  float ca, sa,
622  u1, u2,
623  t1, t2,
624  z;
625 
626  int l1, l2,
627  l, i1;
628 
629  for(x = 0; x < resolution; x++) //for each column
630  {
631  //This is the 1D FFT:
632  ca = -1.0;
633  sa = 0.0;
634  l1 = 1, l2 = 1;
635 
636  for(l=0;l<l2n;l++)
637  {
638  l1 = l2;
639  l2 *= 2;
640  u1 = 1.0;
641  u2 = 0.0;
642  for(j = 0; j < l1; j++)
643  {
644  for(i = j; i < resolution; i += l2)
645  {
646  i1 = i + l1;
647  t1 = u1 * re[resolution * x + i1] - u2 * img[resolution * x + i1];
648  t2 = u1 * img[resolution * x + i1] + u2 * re[resolution * x + i1];
649  re[resolution * x + i1] = re[resolution * x + i] - t1;
650  img[resolution * x + i1] = img[resolution * x + i] - t2;
651  re[resolution * x + i] += t1;
652  img[resolution * x + i] += t2;
653  }
654  z = u1 * ca - u2 * sa;
655  u2 = u1 * sa + u2 * ca;
656  u1 = z;
657  }
658  sa = Ogre::Math::Sqrt((1.0f - ca) / 2.0f);
659  ca = Ogre::Math::Sqrt((1.0f+ca) / 2.0f);
660  }
661  }
662  //Calculate the FFT of the rows
663  for(y = 0; y < resolution; y++) //for each row
664  {
665  //This is the 1D FFT:
666  ca = -1.0;
667  sa = 0.0;
668  l1= 1, l2 = 1;
669 
670  for(l = 0; l < l2m; l++)
671  {
672  l1 = l2;
673  l2 *= 2;
674  u1 = 1.0;
675  u2 = 0.0;
676  for(j = 0; j < l1; j++)
677  {
678  for(i = j; i < resolution; i += l2)
679  {
680  i1 = i + l1;
681  t1 = u1 * re[resolution * i1 + y] - u2 * img[resolution * i1 + y];
