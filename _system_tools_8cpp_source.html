116 bool ExtractZipFiles(const wxString& aZipFile, const wxString& aTargetDir)
117 {
118  // from http://wiki.wxwidgets.org/WxZipInputStream
119  bool ret = true;
120  //wxFileSystem fs;
121  std::unique_ptr<wxZipEntry> entry(new wxZipEntry());
122  do
123  {
124  wxFileInputStream in(aZipFile);
125  if (!in)
126  {
127  wxLogError(_T("Can not open file '")+aZipFile+wxT("'."));
128  ret = false;
129  break;
130  }
131  wxZipInputStream zip(in);
132 
133  while (entry.reset(zip.GetNextEntry()), entry.get() != NULL)
134  {
135  // access meta-data
136  wxString name = entry->GetName();
137  name = aTargetDir + wxFileName::GetPathSeparator() + name;
138 
139  // read 'zip' to access the entry's data
140  if (entry->IsDir())
141  {
142  int perm = entry->GetMode();
143  wxFileName::Mkdir(name, perm, wxPATH_MKDIR_FULL);
144  } else
145  {
146  zip.OpenEntry(*entry.get());
147  if (!zip.CanRead())
148  {
149  wxLogError(_T("Can not read zip entry '") + entry->GetName() + wxT("'."));
150  ret = false;
151  break;
152  }
153 
154  wxFileOutputStream file(name);
155 
156  if (!file)
157  {
158  wxLogError(_T("Can not create file '")+name+wxT("'."));
159  ret = false;
160  break;
161  }
162  zip.Read(file);
163  }
164  }
165  } while(false);
166  return ret;
167 }

171 std::string ComposeVendorMapFilename(std::string vendor)
172 {
173  std::string repl = "\\/ #@?!$%^&*()+=-><.:'|\";";
174  std::string vendorstr = std::string(vendor);
175  for(unsigned int c1 = 0; c1 < repl.size(); c1++)
176  for(unsigned int c2 = 0; c2 < vendorstr.size(); c2++)
177  if(vendorstr[c2] == repl[c1]) vendorstr[c2] = '_';
178  vendorstr += ".map";
179  return vendorstr;
180 }
