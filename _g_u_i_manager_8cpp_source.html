143 GUIManager::GUIManager() :
144  m_renderwindow_closed(false),
145  m_impl(nullptr)
146 {
147  RoR::App::GetOgreSubsystem()->GetOgreRoot()->addFrameListener(this);
148  RoRWindowEventUtilities::addWindowEventListener(RoR::App::GetOgreSubsystem()->GetRenderWindow(), this);
149 
150  std::string gui_logfilename = App::GetSysLogsDir() + PATH_SLASH + "MyGUI.log";
151  auto mygui_platform = new MyGUI::OgrePlatform();
152  mygui_platform->initialise(
153  RoR::App::GetOgreSubsystem()->GetRenderWindow(),
154  gEnv->sceneManager,
155  Ogre::ResourceGroupManager::AUTODETECT_RESOURCE_GROUP_NAME,
156  gui_logfilename); // use cache resource group so preview images are working
157  auto mygui = new MyGUI::Gui();
158 
159  // empty init
160  mygui->initialise("");
161 
162  // add layer factory
163  MyGUI::FactoryManager::getInstance().registerFactory<MyGUI::RTTLayer>("Layer");
164 
165  // then load the actual config
166  MyGUI::ResourceManager::getInstance().load(RESOURCE_FILENAME);
167 
168  MyGUI::ResourceManager::getInstance().load(LanguageEngine::getSingleton().getMyGUIFontConfigFilename());
169 
170  m_impl = new GuiManagerImpl();
171  m_impl->mygui_platform = mygui_platform;
172  m_impl->mygui = mygui;
173 
174  // move the mouse into the middle of the screen, assuming we start at the top left corner (0,0)
175  MyGUI::InputManager::getInstance().injectMouseMove(RoR::App::GetOgreSubsystem()->GetRenderWindow()->getWidth()*0.5f, RoR::App::GetOgreSubsystem()->GetRenderWindow()->getHeight()*0.5f, 0);
176  MyGUI::PointerManager::getInstance().setVisible(true);
177 #ifdef _WIN32
178  MyGUI::LanguageManager::getInstance().eventRequestTag = MyGUI::newDelegate(this, &GUIManager::eventRequestTag);
179 #endif // _WIN32
180  windowResized(RoR::App::GetOgreSubsystem()->GetRenderWindow());
181 }

276 Ogre::String GUIManager::getRandomWallpaperImage()
277 {
278  using namespace Ogre;
279  FileInfoListPtr files = ResourceGroupManager::getSingleton().findResourceFileInfo("Wallpapers", "*.jpg", false);
280  if (files.isNull() || files->empty())
281  {
282  files = ResourceGroupManager::getSingleton().findResourceFileInfo("Wallpapers", "*.png", false);
283  if (files.isNull() || files->empty())
284  return "";
285  }
286  srand ( time(NULL) );
287 
288  int num = 0;
289  for (int i = 0; i<Math::RangeRandom(0, 10); i++)
290  num = Math::RangeRandom(0, files->size());
291 
292  return files->at(num).filename;
293 }

394  m_impl->panel_FrictionSettings .SetVisible(false);
395  m_impl->panel_GamePauseMenu .SetVisible(false);

397  m_impl->panel_TeleportWindow .SetVisible(false);
398  m_impl->panel_VehicleDescription .SetVisible(false);
399  m_impl->panel_SpawnerReport .SetVisible(false);
400  m_impl->panel_SimUtils .SetBaseVisible(false);
401  m_impl->panel_MpClientList .SetVisible(mp_state == App::MP_STATE_CONNECTED);
