166 std::string GetParentDirectory(const char* src_buff)
167 {
168  const char* start = src_buff;
169  int count = strlen(src_buff);
170  // Trim trailing separator(s)
171  for (;;)
172  {
173  if (count == 0) { return ""; }
174  if (!IsSeparator(start[count - 1])) { break; }
175  --count;
176  }
177  // Remove last path entry
178  for (;;)
179  {
180  if (count == 0) { return ""; }
181  if (IsSeparator(start[count - 1])) {break; }
182  --count;
183  }
184  // Trim rear separator(s)
185  for (;;)
186  {
187  if (count == 0) { return ""; }
188  if (!IsSeparator(start[count - 1])) { break; }
189  --count;
190  }
191  return std::string(start, count);
192 }

194 int DetectBasePaths()
195 {
196  char buf[1000] = "";
197 
198  // Process dir (system)
199 #if OGRE_PLATFORM == OGRE_PLATFORM_WIN32 // NOTE: We use non-UNICODE interfaces for simplicity
200  // Process dir
201  if (!GetModuleFileNameA(nullptr, buf, 1000))
202  {
203  return -1;
204  }
205 
206 #elif OGRE_PLATFORM == OGRE_PLATFORM_LINUX // http://stackoverflow.com/a/625523
207  // Process dir
208  if (readlink("/proc/self/exe", buf, 1000) == -1)
209  {
210  return -1;
211  }
212 
213 #elif OGRE_PLATFORM == OGRE_PLATFORM_APPLE
214  // Process dir
215  uint32_t length = 1000;
216  if (_NSGetExecutablePath(procpath, &lenb) == -1) // Returns absolute path to binary
217  {
218  return -1;
219  }
220 #endif
221  App::SetSysProcessDir(RoR::System::GetParentDirectory(buf));
222 
223  // User directory (local override)
224  std::string local_userdir = App::GetSysProcessDir() + PATH_SLASH + "config";
225  if (FolderExists(local_userdir.c_str()))
226  {
227  App::SetSysUserDir(local_userdir);
228  return 0; // Done!
229  }
230 
231  // User directory (system)
232 #if OGRE_PLATFORM == OGRE_PLATFORM_WIN32 // NOTE: We use non-UNICODE interfaces for simplicity
233  if (SHGetFolderPathA(nullptr, CSIDL_PERSONAL, nullptr, SHGFP_TYPE_CURRENT, buf) != S_OK)
234  {
235  return -2;
236  }
237  sprintf(buf, "%s\\Rigs of Rods %s", buf, ROR_VERSION_STRING_SHORT);
238 
239 #elif OGRE_PLATFORM == OGRE_PLATFORM_LINUX
240  snprintf(buf, 1000, "%s/.rigsofrods", getenv("HOME"));
241 
242 #elif OGRE_PLATFORM == OGRE_PLATFORM_APPLE
243  snprintf(buf, 1000, "%s/RigsOfRods", getenv("HOME"));
244 
245 #endif
246 
247  App::SetSysUserDir(buf);
248  return 0;
249 }

256 void Settings::ProcessCommandLine(int argc, char *argv[])
257 {
258  CSimpleOpt args(argc, argv, cmdline_options);
259 
260  while (args.Next())
261  {
262  if (args.LastError() != SO_SUCCESS)
263  {
264  RoR::App::SetPendingAppState(RoR::App::APP_STATE_PRINT_HELP_EXIT);
265  return;
266  }
267  else if (args.OptionId() == OPT_HELP)
268  {
269  RoR::App::SetPendingAppState(RoR::App::APP_STATE_PRINT_HELP_EXIT);
270  return;
271  }
272  else if (args.OptionId() == OPT_VER)
273  {
274  RoR::App::SetPendingAppState(RoR::App::APP_STATE_PRINT_VERSION_EXIT);
275  return;
276  }
277  else if (args.OptionId() == OPT_TRUCK)
278  {
279  App::SetDiagPreselectedVehicle(args.OptionArg());
280  }
281  else if (args.OptionId() == OPT_TRUCKCONFIG)
282  {
283  App::SetDiagPreselectedVehConfig(args.OptionArg());
284  }
285  else if (args.OptionId() == OPT_MAP)
286  {
287  RoR::App::SetDiagPreselectedTerrain(args.OptionArg());
288  }
289  else if (args.OptionId() == OPT_NOCRASHCRPT)
290  {
291  SETTINGS.setSetting("NoCrashRpt", "Yes");
292  }
293  else if (args.OptionId() == OPT_USERPATH)
294  {
295  SETTINGS.setSetting("userpath", String(args.OptionArg()));
296  }
297  else if (args.OptionId() == OPT_WDIR)
298  {
299 #if OGRE_PLATFORM == OGRE_PLATFORM_WIN32
300  SetCurrentDirectory(args.OptionArg());
301 #endif
302  }
303  else if (args.OptionId() == OPT_STATE)
304  {
305  SETTINGS.setSetting("StartState", args.OptionArg());
306  }
307  else if (args.OptionId() == OPT_NOCACHE)
308  {
309  SETTINGS.setSetting("NOCACHE", "Yes");
310  }
311  else if (args.OptionId() == OPT_ADVLOG)
312  {
313  SETTINGS.setSetting("Advanced Logging", "Yes");
314  }
315  else if (args.OptionId() == OPT_INCLUDEPATH)
316  {
317  SETTINGS.setSetting("resourceIncludePath", args.OptionArg());
318  }
319  else if (args.OptionId() == OPT_CHECKCACHE)
320  {
321  // just regen cache and exit
322  SETTINGS.setSetting("regen-cache-only", "Yes");
323  }
324  else if (args.OptionId() == OPT_ENTERTRUCK)
325  {
326  App::SetDiagPreselectedVehEnter(true);
327  }
328  else if (args.OptionId() == OPT_SETUP)
329  {
330  SETTINGS.setSetting("USE_OGRE_CONFIG", "Yes");
331  }
332  else if (args.OptionId() == OPT_JOINMPSERVER)
333  {
334  std::string server_args = args.OptionArg();
335  const int colon = server_args.rfind(":");
336  if (colon != std::string::npos)
337  {
338  RoR::App::SetPendingMpState(RoR::App::MP_STATE_CONNECTED);
339 
340  std::string host_str;
341  std::string port_str;
342  if (server_args.find("rorserver://") != String::npos) // Windows URI Scheme retuns rorserver://server:port/
343  {
344  host_str = server_args.substr(12, colon - 12);
345  port_str = server_args.substr(colon + 1, server_args.length() - colon - 2);
346  }
347  else
348  {
349  host_str = server_args.substr(0, colon);
350  port_str = server_args.substr(colon + 1, server_args.length());
351  }
352  RoR::App::SetMpServerHost(host_str);
353  RoR::App::SetMpServerPort(Ogre::StringConverter::parseInt(port_str));
354  }
355  }
356  }
357 }

749 void Settings::LoadSettings(std::string filepath)
750 {
751  ConfigFile cfg;
752  try
753  {
754  cfg.load(filepath, "=:\t", false);
755 
756  // load all settings into a map!
757  ConfigFile::SettingsIterator i = cfg.getSettingsIterator();
758  String s_value, s_name;
759  while (i.hasMoreElements())
760  {
761  s_name = RoR::Utils::SanitizeUtf8String(i.peekNextKey());
762  s_value = RoR::Utils::SanitizeUtf8String(i.getNext());
763 
764  // Purge unwanted entries
765  if (s_name == "Program Path") { continue; }
766 
767  // Process and clear GVar values
768  if (this->ParseGlobalVarSetting(s_name, s_value))
769  {
770  continue;
771  }
772 
773  settings[s_name] = s_value;
774  }
775  }
776  catch (Ogre::FileNotFoundException e) {} // Just continue with defaults...
777 
778  // generate hash of the token
779  String usertoken = SSETTING("User Token", "");
780  char usertokensha1result[250];
781  memset(usertokensha1result, 0, 250);
782  if (usertoken.size()>0)
783  {
784  RoR::CSHA1 sha1;
785  sha1.UpdateHash((uint8_t *)usertoken.c_str(), (uint32_t)usertoken.size());
786  sha1.Final();
787  sha1.ReportHash(usertokensha1result, RoR::CSHA1::REPORT_HEX_SHORT);
788  }
789 
790  setSetting("User Token Hash", String(usertokensha1result));
791 }
