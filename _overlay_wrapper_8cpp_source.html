121 void OverlayWrapper::resizeOverlay(LoadedOverlay& lo)
122 {
123  // enforce 4:3 for overlays
124  float w = win->getWidth();
125  float h = win->getHeight();
126  float s = (4.0f / 3.0f) / (w / h);
127 
128  // window is higher than wide
129  if (s > 1)
130  s = (3.0f / 4.0f) / (h / w);
131 
132  // originals
133  lo.o->setScale(lo.orgScaleX, lo.orgScaleY);
134  lo.o->setScroll(0, 0);
135 
136  // now the new values
137  lo.o->setScale(s, s);
138  lo.o->scroll(1 - s, s - 1);
139 }
