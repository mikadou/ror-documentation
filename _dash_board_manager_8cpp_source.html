251 void DashBoard::update(float& dt)
252 {
253  // walk all controls and animate them
254  for (int i = 0; i < free_controls; i++)
255  {
256  // get its value from its linkage
257  if (controls[i].animationType == ANIM_ROTATE)
258  {
259  // get the value
260  float val = manager->getNumeric(controls[i].linkID);
261  // calculate the angle
262  float angle = (val - controls[i].vmin) * (controls[i].wmax - controls[i].wmin) / (controls[i].vmax - controls[i].vmin) + controls[i].wmin;
263 
264  if (fabs(val - controls[i].last) < 0.02f)
265  continue;
266 
267  controls[i].last = val;
268 
269  // enforce limits
270  if (angle < controls[i].wmin)
271  angle = controls[i].wmin;
272  else if (angle > controls[i].wmax)
273  angle = controls[i].wmax;
274  // rotate finally
275  controls[i].rotImg->setAngle(Ogre::Degree(angle).valueRadians());
276  }
277  else if (controls[i].animationType == ANIM_LAMP)
278  {
279  // or a lamp?
280  bool state = false;
281  // conditional
282  if (controls[i].condition == CONDITION_GREATER)
283  {
284  float val = manager->getNumeric(controls[i].linkID);
285  state = (val > controls[i].conditionArgument);
286  }
287  else if (controls[i].condition == CONDITION_LESSER)
288  {
289  float val = manager->getNumeric(controls[i].linkID);
290  state = (val < controls[i].conditionArgument);
291  }
292  else
293  {
294  state = (manager->getNumeric(controls[i].linkID) > 0);
295  }
296 
297  if (state == controls[i].lastState)
298  continue;
299  controls[i].lastState = state;
300 
301  // switch states
302  if (state)
303  {
304  controls[i].img->setImageTexture(String(controls[i].texture) + "-on.png");
305  }
306  else
307  {
308  controls[i].img->setImageTexture(String(controls[i].texture) + "-off.png");
309  }
310  }
311  else if (controls[i].animationType == ANIM_SERIES)
312  {
313  float val = manager->getNumeric(controls[i].linkID);
314 
315  String fn = String(controls[i].texture) + String("-") + TOSTRING((int)val) + String(".png");
316 
317  if (fabs(val - controls[i].last) < 0.2f)
318  continue;
319  controls[i].last = val;
320 
321  controls[i].img->setImageTexture(fn);
322  }
323  else if (controls[i].animationType == ANIM_SCALE)
324  {
325  float val = manager->getNumeric(controls[i].linkID);
326 
327  if (fabs(val - controls[i].last) < 0.2f)
328  continue;
329  controls[i].last = val;
330 
331  float scale = (val - controls[i].vmin) * (controls[i].wmax - controls[i].wmin) / (controls[i].vmax - controls[i].vmin) + controls[i].wmin;
332  if (controls[i].direction == DIRECTION_UP)
333  {
334  controls[i].widget->setPosition(controls[i].initialPosition.left, controls[i].initialPosition.top - scale);
335  controls[i].widget->setSize(controls[i].initialSize.width, controls[i].initialSize.height + scale);
336  }
337  else if (controls[i].direction == DIRECTION_DOWN)
338  {
339  controls[i].widget->setPosition(controls[i].initialPosition.left, controls[i].initialPosition.top);
340  controls[i].widget->setSize(controls[i].initialSize.width, controls[i].initialSize.height + scale);
341  }
342  else if (controls[i].direction == DIRECTION_LEFT)
343  {
344  controls[i].widget->setPosition(controls[i].initialPosition.left - scale, controls[i].initialPosition.top);
345  controls[i].widget->setSize(controls[i].initialSize.width + scale, controls[i].initialSize.height);
346  }
347  else if (controls[i].direction == DIRECTION_RIGHT)
348  {
349  controls[i].widget->setPosition(controls[i].initialPosition.left, controls[i].initialPosition.top);
350  controls[i].widget->setSize(controls[i].initialSize.width + scale, controls[i].initialSize.height);
351  }
352  }
353  else if (controls[i].animationType == ANIM_TRANSLATE)
354  {
355  float val = manager->getNumeric(controls[i].linkID);
356 
357  if (fabs(val - controls[i].last) < 0.2f)
358  continue;
359  controls[i].last = val;
360 
361  float translation = (val - controls[i].vmin) * (controls[i].wmax - controls[i].wmin) / (controls[i].vmax - controls[i].vmin) + controls[i].wmin;
362  if (controls[i].direction == DIRECTION_UP)
363  controls[i].widget->setPosition(controls[i].initialPosition.left, controls[i].initialPosition.top - translation);
364  else if (controls[i].direction == DIRECTION_DOWN)
365  controls[i].widget->setPosition(controls[i].initialPosition.left, controls[i].initialPosition.top + translation);
366  else if (controls[i].direction == DIRECTION_LEFT)
367  controls[i].widget->setPosition(controls[i].initialPosition.left - translation, controls[i].initialPosition.top);
368  else if (controls[i].direction == DIRECTION_RIGHT)
369  controls[i].widget->setPosition(controls[i].initialPosition.left + translation, controls[i].initialPosition.top);
370  }
371  else if (controls[i].animationType == ANIM_TEXTFORMAT)
372  {
373  float val = manager->getNumeric(controls[i].linkID);
374 
375  if (fabs(val - controls[i].last) < 0.2f)
376  continue;
377  controls[i].last = val;
378 
379  MyGUI::UString s;
380  if (strlen(controls[i].format) == 0)
381  {
382  s = Ogre::StringConverter::toString(val);
383  }
384  else
385  {
386  char tmp[1024] = "";
387  sprintf(tmp, controls[i].format, val);
388  s = MyGUI::UString(tmp);
389  }
390 
391  controls[i].txt->setCaption(s);
392  }
393  else if (controls[i].animationType == ANIM_TEXTSTRING)
394  {
395  char* val = manager->getChar(controls[i].linkID);
396  controls[i].txt->setCaption(MyGUI::UString(val));
397  }
398  }

554  return;
555  }
556  // then specializations
