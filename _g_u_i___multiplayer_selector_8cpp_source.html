76 Json::Value FetchServerlist(std::string portal_url)
78 {
79  std::string serverlist_url = portal_url + "/server-list?json=true";
80  std::string response_payload;
81  std::string response_header;
82  long response_code = 0;
83 
84  CURL *curl = curl_easy_init();
85  curl_easy_setopt(curl, CURLOPT_URL, serverlist_url.c_str());
86  curl_easy_setopt(curl, CURLOPT_NOPROGRESS, 1L);
87  curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, CurlWriteFunc);
88  curl_easy_setopt(curl, CURLOPT_WRITEDATA, &response_payload);
89  curl_easy_setopt(curl, CURLOPT_HEADERDATA, &response_header);
90 
91  curl_easy_perform(curl);
92  curl_easy_getinfo(curl, CURLINFO_RESPONSE_CODE, &response_code);
93  curl_easy_cleanup(curl);
94  curl = nullptr;
95 
96  Json::Value result(Json::objectValue);
97 
98  if (response_code != 200)
99  {
100  Ogre::LogManager::getSingleton().stream()
101  << "[RoR|Multiplayer] Failed to retrieve serverlist; HTTP status code: " << response_code;
102  result["success"] = false;
103  result["message"] = "Error connecting to server :(";
104  return result;
105  }
106 
107  Json::CharReaderBuilder b;
108  Json::CharReader* reader = b.newCharReader(); // Java, anyone?
109  const char* payload = response_payload.c_str();
110  const char* payload_end = payload + response_payload.size();
111  std::string parse_errors;
112  Json::Value parse_output;
113  if (!reader->parse(payload, payload_end, &parse_output, &parse_errors))
114  {
115  Ogre::LogManager::getSingleton().stream()
116  << "[RoR|Multiplayer] Error parsing serverlist JSON, messages: " << parse_errors;
117  result["success"] = false;
118  result["message"] = "Server returned invalid data :(";
119  return result;
120  }
121 
122  result["success"] = true;
123  result["data"] = parse_output;
124  return result;
125 }
126 #endif // defined(USE_CURL) && defined(USE_JSONCPP)

210 void CLASS::CheckAndProcessRefreshResult()
211 {
212 #ifdef USE_JSONCPP
213  std::future_status status = m_serverlist_data->future_json.wait_for(std::chrono::seconds(0));
214  if (status != std::future_status::ready)
215  {
216  return;
217  }
218 
219  auto json = m_serverlist_data->future_json.get();
220  if (!json.isObject() || json["success"] != true || !json["data"].isArray())
221  {
222  std::string message = (json["message"].isString()) ? json["message"].asString() : "Could not connect to server :(";
223  m_status_label->setTextColour(status_failure_color);
224  m_status_label->setCaption(message);
225  m_refresh_button->setEnabled(true);
226  m_is_refreshing = false;
227  return;
228  }
229 
230  int num_rows = json["data"].size();
231  if (num_rows == 0)
232  {
233  m_status_label->setTextColour(status_emptylist_color);
234  m_status_label->setCaption("There are no available servers :/");
235  m_refresh_button->setEnabled(true);
236  m_is_refreshing = false;
237  return;
238  }
239 
240  for (int i = 0; i < num_rows; ++i)
241  {
242  auto row = json["data"][i];
243  m_servers_list->addItem(row["name"].asString(), row); // Only fills 1st column. Userdata = JSON
244  m_servers_list->setSubItemNameAt(1, i, row["terrain-name"].asString()); // Fill other columns
245  m_servers_list->setSubItemNameAt(2, i, row["current-users"].asString() + " / " +row["max-clients"].asString());
246  m_servers_list->setSubItemNameAt(3, i, row["ip"].asString());
247  }
248 
249  m_status_label->setVisible(false);
250  m_refresh_button->setEnabled(true);
251  m_is_refreshing = false;
252 #endif // USE_JSONCPP
253 }
