73 {

77

79 
80  String group = "";
81  try
82  {
83  group = ResourceGroupManager::getSingleton().findGroupContainingResource(filename);
84  }
85  catch (...)

87  // we wont catch anything, since the path could be absolute as well, then the group is not found

91  try
92  {
93  // try to load directly otherwise via resource group
94  if (group == "")
95  cfg.loadDirect(filename);
96  else
97  cfg.loadFromResourceSystem(filename, group, "\x09:=", true);
98  }
99  catch (Ogre::Exception& e)
100  {
101  ErrorUtils::ShowError(_L("Error while loading landuse config"), e.getFullDescription());
102  return 1;
103  }
104 
105  Ogre::ConfigFile::SectionIterator seci = cfg.getSectionIterator();
106  Ogre::String secName, kname, kvalue;
107  while (seci.hasMoreElements())
108  {
109  secName = seci.peekNextKey();
110  Ogre::ConfigFile::SettingsMultiMap* settings = seci.getNext();
111  Ogre::ConfigFile::SettingsMultiMap::iterator i;
112  for (i = settings->begin(); i != settings->end(); ++i)
113  {
114  kname = i->first;
115  kvalue = i->second;
116  // we got all the data available now, processing now
117  if (secName == "general" || secName == "config")
118  {
119  // set some class properties accoring to the information in this section
120  if (kname == "texture")
121  textureFilename = kvalue;
122  else if (kname == "frictionconfig" || kname == "loadGroundModelsConfig")
123  gEnv->collisions->loadGroundModelsConfigFile(kvalue);
124  else if (kname == "defaultuse")
125  default_ground_model = gEnv->collisions->getGroundModelByString(kvalue);
126  }
127  else if (secName == "use-map")
128  {
129  if (kname.size() != 10)
130  {
131  LOG("invalid color in landuse line in " + filename);
132  continue;
133  }
134  char* ptr; //not used
135  unsigned int color = strtoul(kname.c_str(), &ptr, 16);
136  usemap[color] = kvalue;
137  }
138  }
139  }
140 
141 #ifdef USE_PAGED
142  // process the config data and load the buffers finally
143  try
144  {
145  Forests::ColorMap* colourMap = Forests::ColorMap::load(textureFilename, Forests::CHANNEL_COLOR);
146  colourMap->setFilter(Forests::MAPFILTER_NONE);
147 
148  /*
149  // debug things below
150  printf("found ground use definitions:\n");
151  for (std::map < uint32, String >::iterator it=usemap.begin(); it!=usemap.end(); it++)
152  {
153  printf(" 0x%Lx : %s\n", it->first, it->second.c_str());
154  }
155  */
156 
157  bool bgr = colourMap->getPixelBox().format == PF_A8B8G8R8;
158 
159  Ogre::TRect<Ogre::Real> bounds = Forests::TBounds(0, 0, mapsize.x, mapsize.z);
160 
161  // now allocate the data buffer to hold pointers to ground models
162  data = new ground_model_t*[(int)(mapsize.x * mapsize.z)];
163  ground_model_t** ptr = data;
164  //std::map < String, int > counters;
165  for (int z = 0; z < mapsize.z; z++)
166  {
167  for (int x = 0; x < mapsize.x; x++)
168  {
169  unsigned int col = colourMap->getColorAt(x, z, bounds);
170  if (bgr)
171  {
172  // Swap red and blue values
173  unsigned int cols = col & 0xFF00FF00;
174  cols |= (col & 0xFF) << 16;
175  cols |= (col & 0xFF0000) >> 16;
176  col = cols;
177  }
178  String use = usemap[col];
179  //if (use!="")
180  // counters[use]++;
181 
182  // store the pointer to the ground model in the data slot
183  *ptr = gEnv->collisions->getGroundModelByString(use);
184  ptr++;
185  }
186  }
187  }
188  catch (...)
189  {
190  Log("Landuse: Failed to load texture: " + textureFilename);
191  }
192 #endif // USE_PAGED
193 
194  return 0;
195 }
