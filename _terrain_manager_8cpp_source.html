133 void TerrainManager::loadTerrain(String filename)
134 {
135  DataStreamPtr ds;
136 
137  try
138  {
139  String group = ResourceGroupManager::getSingleton().findGroupContainingResource(filename);
140  ds = ResourceGroupManager::getSingleton().openResource(filename, group);
141  }
142  catch (...)
143  {
144  LOG("[RoR|Terrain] File not found: " + filename);
145  return;
146  }
147 
148  PROGRESS_WINDOW(10, _L("Loading Terrain Configuration"));
149 
150  LOG(" ===== LOADING TERRAIN " + filename);
151 
152  Terrn2Parser parser;
153  if (! parser.LoadTerrn2(m_def, ds))
154  {
155  LOG("[RoR|Terrain] Failed to parse: " + filename);
156  for (std::string msg : parser.GetMessages())
157  {
158  LOG("[RoR|Terrain] \tMessage: " + msg);
159  }
160  return;
161  }
162  gravity = m_def.gravity;
163 
164  // then, init the subsystems, order is important :)
165  initSubSystems();
166 
167  fixCompositorClearColor();
168 
169  LOG(" ===== LOADING TERRAIN GEOMETRY " + filename);
170 
171  // load the terrain geometry
172  PROGRESS_WINDOW(80, _L("Loading Terrain Geometry"));
173  geometry_manager->loadOgreTerrainConfig(m_def.ogre_ter_conf_filename);
174 
175  LOG(" ===== LOADING TERRAIN WATER " + filename);
176  // must happen here
177  initWater();
178 
179  LOG(" ===== LOADING TERRAIN OBJECTS " + filename);
180 
181  PROGRESS_WINDOW(90, _L("Loading Terrain Objects"));
182  loadTerrainObjects();
183 
184  collisions->printStats();
185 
186  // bake the decals
187  //finishTerrainDecal();
188 
189  // init things after loading the terrain
190  initTerrainCollisions();
191 
192  // init the survey map
193  if (!RoR::App::GetGfxMinimapDisabled())
194  {
195  PROGRESS_WINDOW(45, _L("Initializing Overview Map Subsystem"));
196  m_survey_map = new SurveyMapManager();
197  }
198 
199  collisions->finishLoadingTerrain();
200  LOG(" ===== TERRAIN LOADING DONE " + filename);
201 }

520 void TerrainManager::initWater()
521 {
522  // disabled in global config
523  if (App::GetGfxWaterMode() == App::GFX_WATER_NONE)
524  return;
525 
526  // disabled in map config
527  if (!m_def.has_water)
528  {
529  return;
530  }
531 
532  if (App::GetGfxWaterMode() == App::GFX_WATER_HYDRAX)
533  {
534  // try to load hydrax config
535  if (!m_def.hydrax_conf_file.empty() && ResourceGroupManager::getSingleton().resourceExistsInAnyGroup(m_def.hydrax_conf_file))
536  {
537  hw = new HydraxWater(m_def.water_height, m_def.hydrax_conf_file);
538  }
539  else
540  {
541  // no config provided, fall back to the default one
542  hw = new HydraxWater(m_def.water_height);
543  }
544 
545  water = hw;
546 
547  //Apply depth technique to the terrain
548  TerrainGroup::TerrainIterator ti = geometry_manager->getTerrainGroup()->getTerrainIterator();
549  while (ti.hasMoreElements())
550  {
551  Terrain* t = ti.getNext()->instance;
552  MaterialPtr ptr = t->getMaterial();
553  hw->GetHydrax()->getMaterialManager()->addDepthTechnique(ptr->createTechnique());
554  }
555  }
556  else
557  {
558  if (water == nullptr)
559  water = new Water();
560  else if (water != nullptr)
561  {
562  delete(water);
563  water = new Water();
564  }
565 
566  water->setHeight (m_def.water_height);
567  }
568 }
