30 LocalStorage::LocalStorage(AngelScript::asIScriptEngine *engine_in, std::string fileName_in, const std::string &sectionName_in)
31 {
32  refCount++;
33  cgflag=false;
34  this->engine = engine_in;
35  engine->NotifyGarbageCollectorOfNewObject(this, engine->GetTypeInfoByDecl("LocalStorage"));
36 
37  // inversed logic, better use a whitelist instead of a blacklist, so you are on the safe side ;) - tdev
38  std::string allowedChars = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789_-";
39  for (std::string::iterator it = fileName_in.begin() ; it < fileName_in.end() ; ++it){
40  if ( allowedChars.find(*it) == std::string::npos )
41  *it = '_';
42  }
43 
44  sectionName = sectionName_in.substr(0, sectionName_in.find(".", 0));
45 
46  filename = RoR::App::GetSysCacheDir() + PATH_SLASH + fileName_in + ".asdata";
47  separators = "=";
48  loadDict();
49 
50  saved = true;
51 }

274 void LocalStorage::parseKey(std::string &key, std::string &section)
275 {
276  size_t dot = key.find(".", 0);
277  if ( dot != std::string::npos )
278  {
279  section = key.substr(0, dot);
280 
281  if ( !section.length() )
282  section = sectionName;
283 
284  key.erase(0, dot+1);
285  }
286  else
287  section = sectionName;
288 }
