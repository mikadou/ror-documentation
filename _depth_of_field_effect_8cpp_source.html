159 void DepthOfFieldEffect::notifyMaterialSetup(uint32 passId, MaterialPtr& material)
160 {
161  switch (passId)
162  {
163  case BlurPass:
164  {
165  // Adjust fragment program parameters
166  Ogre::Vector3 ps = Ogre::Vector3(1.0f / (mWidth / BLUR_DIVISOR), 1.0f / (mHeight / BLUR_DIVISOR), 1.0f);
167  float pixelSize[3] = {ps.x, ps.y, ps.z};
168  GpuProgramParametersSharedPtr fragParams = material->getBestTechnique()->getPass(0)->getFragmentProgramParameters();
169  if ((!fragParams.isNull()) && (fragParams->_findNamedConstantDefinition("pixelSize")))
170  fragParams->setNamedConstant("pixelSize", pixelSize, 1, 3);
171 
172  break;
173  }
174 
175  case OutputPass:
176  {
177  float pixelSizeScene[3] = {
178  1.0f / mWidth,
179  1.0f / mHeight,
180  0};
181 
182  float pixelSizeBlur[3] = {
183  1.0f / (mWidth / BLUR_DIVISOR),
184  1.0f / (mHeight / BLUR_DIVISOR),
185  0};
186 
187  // Adjust fragment program parameters
188  GpuProgramParametersSharedPtr fragParams =
189  material->getBestTechnique()->getPass(0)->getFragmentProgramParameters();
190  if ((!fragParams.isNull()) && (fragParams->_findNamedConstantDefinition("pixelSizeScene")))
191  fragParams->setNamedConstant("pixelSizeScene", pixelSizeScene, 1, 3);
192  if ((!fragParams.isNull()) && (fragParams->_findNamedConstantDefinition("pixelSizeBlur")))
193  fragParams->setNamedConstant("pixelSizeBlur", pixelSizeBlur, 1, 3);
194 
195  break;
196  }
197  }
198 }

200 void DepthOfFieldEffect::preViewportUpdate(const Ogre::RenderTargetViewportEvent& evt)
201 {
202  float dofParams[4] = {mNearDepth, mFocalDepth, mFarDepth, mFarBlurCutoff};
203 
204  // Adjust fragment program parameters for depth pass
205  GpuProgramParametersSharedPtr fragParams =
206  mDepthTechnique->getPass(0)->getFragmentProgramParameters();
207  if ((!fragParams.isNull()) && (fragParams->_findNamedConstantDefinition("dofParams")))
208  fragParams->setNamedConstant("dofParams", dofParams, 1, 4);
209 
210  evt.source->setVisibilityMask(~DEPTHMAP_DISABLED);
211 
212  // Add 'this' as a RenderableListener to replace the technique for all renderables
213  RenderQueue* queue = evt.source->getCamera()->getSceneManager()->getRenderQueue();
214  queue->setRenderableListener(this);

353 bool DOFManager::frameStarted(const FrameEvent& evt)
354 {
355  // Focusing
356  switch (mFocusMode)
357  {
358  case Auto:
359  {
360  // TODO: Replace with accurate ray/triangle collision detection
361  Real currentFocalDistance = mLens->getFocalDistance();
362 
363  // TODO: Continuous AF / triggered
364  mAutoTime -= evt.timeSinceLastFrame;
365  if (mAutoTime <= 0.0f)
366  {
367  mAutoTime = 0.5f;
368 
369  Vector3 lookAt(Vector3::ZERO);
370 
371  Beam* currTruck = m_sim_controller->GetBeamFactory()->getCurrentTruck();
372  if (currTruck)
373  {
374  lookAt = currTruck->getPosition();
375  }
376  else if (gEnv->player)
377  {
378  lookAt = gEnv->player->getPosition();
379  }
380 
381  targetFocalDistance = gEnv->mainCamera->getPosition().distance(lookAt) / 2.0f; // Needs further investigation
382 
383  setLensFOV(Radian(gEnv->mainCamera->getFOVy()));
384  }
385 
386  // Slowly adjust the focal distance (emulate auto focus motor)
387  if (currentFocalDistance < targetFocalDistance)
388  {
389  mLens->setFocalDistance(std::min(currentFocalDistance + mAutoSpeed * evt.timeSinceLastFrame, targetFocalDistance));
390  }
391  else if (currentFocalDistance > targetFocalDistance)
392  {
393  mLens->setFocalDistance(std::max(currentFocalDistance - mAutoSpeed * evt.timeSinceLastFrame, targetFocalDistance));
394  }
395 
396  break;
397  }
398 
399  case Manual:
400  //we set the values elsewhere
401  break;
402  }
403 
404  // Update Depth of Field effect
405  if (mFocusMode != Pinhole)
406  {
407  mDepthOfFieldEffect->setEnabled(true);
408 
409  // Calculate and set depth of field using lens
410  float nearDepth, focalDepth, farDepth;
411  mLens->recalculateDepthOfField(nearDepth, focalDepth, farDepth);
412  mDepthOfFieldEffect->setFocalDepths(nearDepth, focalDepth, farDepth);
413  }
414  else
415  {
416  mDepthOfFieldEffect->setEnabled(false);
417  }
418 
419  return true;
420 }
