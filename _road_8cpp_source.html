46 void Road::preparePending()
47 {
48  //setup rotation points
49  lastpturn = pturn;
50  protl = ppos + Quaternion(Degree(pturn), Vector3::UNIT_Y) * Vector3(0, 0, 4.5);
51  protr = ppos + Quaternion(Degree(pturn), Vector3::UNIT_Y) * Vector3(0, 0, -4.5);
52  tenode->setPosition(ppos);
53  tenode->setOrientation(Quaternion(Degree(pturn), Vector3::UNIT_Y) * Quaternion(Degree(ppitch), Vector3::UNIT_Z));
54  tenode->pitch(Degree(-90));
55 }

57 void Road::updatePending()
58 {
59  if (pturn - lastpturn > 0)
60  {
61  tenode->setPosition(Quaternion(Degree(pturn - lastpturn), Vector3::UNIT_Y) * (ppos - protl) + protl);

63  else

69  }
70  tenode->setOrientation(Quaternion(Degree(pturn), Vector3::UNIT_Y) * Quaternion(Degree(ppitch), Vector3::UNIT_Z));

135 void Road::append()
