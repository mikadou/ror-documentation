188 int Collisions::loadGroundModelsConfigFile(Ogre::String filename)
189 {
190  String group = "";
191  try
192  {
193  group = ResourceGroupManager::getSingleton().findGroupContainingResource(filename);
194  }
195  catch (...)
196  {
197  // we wont catch anything, since the path could be absolute as well, then the group is not found
198  }
199 
200  Ogre::ConfigFile cfg;
201  try
202  {
203  // try to load directly otherwise via resource group
204  if (group == "")
205  cfg.loadDirect(filename);
206  else
207  cfg.loadFromResourceSystem(filename, group, "\x09:=", true);
208  }
209  catch (Ogre::Exception& e)
210  {
211  ErrorUtils::ShowError("Error while loading ground model", e.getFullDescription());
212  return 1;
213  }
214 
215  // parse the whole config
216  parseGroundConfig(&cfg);
217 
218  // after it was parsed, resolve the dependencies
219  std::map<Ogre::String, ground_model_t>::iterator it;
220  for (it=ground_models.begin(); it!=ground_models.end(); it++)
221  {
222  if (!it->second.basename) continue; // no base, normal material
223  String bname = String(it->second.basename);
224  if (ground_models.find(bname) == ground_models.end()) continue; // base not found!
225  // copy the values from the base if not set otherwise
226  ground_model_t *thisgm = &(it->second);
227  ground_model_t *basegm = &ground_models[bname];
228  memcpy(thisgm, basegm, sizeof(ground_model_t));
229  // re-set the name
230  strncpy(thisgm->name, it->first.c_str(), 255);
231  // after that we need to reload the config to overwrite settings of the base
232  parseGroundConfig(&cfg, it->first);
233  }
234  // check the version
235  if (this->collision_version != LATEST_GROUND_MODEL_VERSION)
236  {
237  ErrorUtils::ShowError(_L("Configuration error"), _L("Your ground configuration is too old, please copy skeleton/config/ground_models.cfg to My Documents/Rigs of Rods/config"));
238  exit(124);
239  }
240  return 0;

400 int Collisions::removeCollisionTri(int number)
401 {
402  if (number > free_collision_tri) return -1;
403 
404  Vector3 p1 = collision_tris[number].a;
405  Vector3 p2 = collision_tris[number].b;
406  Vector3 p3 = collision_tris[number].c;
407 
408  // compute tri AAB
409  AxisAlignedBox aab;
410  aab.merge(p1);
411  aab.merge(p2);
412  aab.merge(p3);
413  // register this collision tri in the index
414  int ilox, ihix, iloz, ihiz;
415  ilox=(int)(aab.getMinimum().x/(float)CELL_SIZE);
416  if (ilox<0) ilox=0; if (ilox>MAXIMUM_CELL) ilox=MAXIMUM_CELL;
417  ihix=(int)(aab.getMaximum().x/(float)CELL_SIZE);
418  if (ihix<0) ihix=0; if (ihix>MAXIMUM_CELL) ihix=MAXIMUM_CELL;
419  iloz=(int)(aab.getMinimum().z/(float)CELL_SIZE);

421  ihiz=(int)(aab.getMaximum().z/(float)CELL_SIZE);
422  if (ihiz<0) ihiz=0; if (ihiz>MAXIMUM_CELL) ihiz=MAXIMUM_CELL;

456 {

1376 void primitiveCollision(node_t *node, Vector3 &force, const Vector3 &velocity, const Vector3 &normal, float dt, ground_model_t* gm, float* nso, float penetration, float reaction)
1377 {
1378  // normal velocity
1379 
1380  float Vnormal = velocity.dotProduct(normal);
1381 
1382  // if we are inside the fluid (solid ground is below us)
1383  if (gm->solid_ground_level != 0.0f && penetration >= 0)
1384  {
1385  if (nso) *nso = 0.0f;
1386 
1387  float Vsquared = velocity.squaredLength();
1388  // First of all calculate power law fluid viscosity
1389  float m = gm->flow_consistency_index * approx_pow(Vsquared, (gm->flow_behavior_index - 1.0f)*0.5f);
1390 
1391  // Then calculate drag based on above. We'are using a simplified Stokes' drag.
1392  // Per node fluid drag surface coefficient set by node property applies here
1393  Vector3 Fdrag = velocity * (-m * node->surface_coef);
1394 
1395  // If we have anisotropic drag
1396  if (gm->drag_anisotropy < 1.0f && Vnormal > 0)
1397  {
1398  float da_factor;
1399  if (Vsquared > gm->va * gm->va)
1400  da_factor = 1.0;
1401  else
1402  da_factor = Vsquared / (gm->va * gm->va);
1403  Fdrag += (Vnormal * m * (1.0f - gm->drag_anisotropy) * da_factor) * normal;
1404  }
1405  force += Fdrag;
1406 
1407  // Now calculate upwards force based on a simplified boyancy equation;
1408  // If the fluid is pseudoplastic then boyancy is constrained to only "stopping" a node from going downwards
1409  // Buoyancy per node volume coefficient set by node property applies here
1410  float Fboyancy = gm->fluid_density * penetration * (-DEFAULT_GRAVITY) * node->volume_coef;
1411  if (gm->flow_behavior_index < 1.0f && Vnormal >= 0.0f)
1412  {
1413  float Fnormal = force.dotProduct(normal);
1414  if (Fnormal < 0 && Fboyancy>-Fnormal)
1415  {
1416  Fboyancy = -Fnormal;
1417  }
1418  }
1419  force += Fboyancy*normal;
1420  }
1421 
1422  // if we are inside or touching the solid ground
1423  if (penetration >= gm->solid_ground_level)
1424  {
1425  Vector3 slip = velocity - Vnormal*normal;
1426  float slipv = slip.squaredLength();
1427  if (fabs(slipv) > 1e-08f)
1428  {
1429  float invslipv = fast_invSqrt(slipv);
1430  slip = slip*invslipv;
1431  slipv = slipv*invslipv;
1432  } else
1433  {
1434  slipv = sqrt(slipv);
1435  }
1436 
1437  if (nso && gm->solid_ground_level == 0.0f) *nso = slipv;
1438 
1439  float Fnormal = 0.0f;
1440  float Fdnormal = 0.0f;
1441  float Freaction;
1442  float Greaction;
1443 
1444  Fnormal = force.dotProduct(normal);
1445  Fdnormal = Fnormal;
1446 
1447  // steady force
1448  if (reaction < 0)
1449  {
1450  Freaction = -Fnormal;
1451  // impact force
1452  if (Vnormal < 0)
1453  {
1454  Freaction += -Vnormal * node->mass / dt; // Newton's second law
1455  }
1456  if (Freaction < 0) Freaction = 0.0f;
1457  } else
1458  {
1459  Freaction = reaction;
1460  Fnormal = 0.0f;
1461  }
1462  float ff;
1463  // If the velocity that we slip is lower than adhesion velocity and
1464  // we have a downforce and the slip forces are lower than static friction
1465  // forces then it's time to go into static friction physics mode.
1466  // This code is a direct translation of textbook static friction physics
1467  Greaction = (Freaction * gm->strength * node->friction_coef); //General moderated reaction, node property sets friction_coef as a pernodefriction setting
1468  float msGreaction = (gm->ms) * Greaction;
1469  if (slipv < (gm->va) && Greaction > 0.0f && (force - Fdnormal * normal).squaredLength() <= msGreaction * msGreaction)
1470  {
1471  // Static friction model (with a little smoothing to help the integrator deal with it)
1472  ff = -msGreaction * (1.0f - approx_exp(-slipv / gm->va));
1473  force = (Fnormal + Freaction) * normal + ff*slip;
1474  } else
1475  {
1476  // Stribek model. It also comes directly from textbooks.
1477  float g = gm->mc + (gm->ms - gm->mc) * std::min(1.0f, approx_exp(-approx_pow(slipv / gm->vs, gm->alpha)));
1478  ff = -(g + gm->t2 * slipv) * Greaction;
1479  force += Freaction * normal + ff*slip;
1480  }
1481  }
