98 inline void getSystemFileList(VectorFileInfo& _result, const std::wstring& _folder, const std::wstring& _mask)
99 {
100 #if MYGUI_PLATFORM == MYGUI_PLATFORM_WIN32
101  //FIXME add optional parameter?
102  bool ms_IgnoreHidden = true;
103 
104  long lHandle, res;
105  struct _wfinddata_t tagData;
106 
107  // pattern can contain a directory name, separate it from mask
108  size_t pos = _mask.find_last_of(L"/\\");
109  std::wstring directory;
110  if (pos != _mask.npos)
111  directory = _mask.substr(0, pos);
112 
113  std::wstring full_mask = concatenatePath(_folder, _mask);
114 
115  lHandle = _wfindfirst(full_mask.c_str(), &tagData);
116  res = 0;
117  while (lHandle != -1 && res != -1)
118  {
119  if ((!ms_IgnoreHidden || (tagData.attrib & _A_HIDDEN) == 0) &&
120  (!isReservedDir(tagData.name)))
121  {
122  _result.push_back(FileInfo(concatenatePath(directory, tagData.name), (tagData.attrib & _A_SUBDIR) != 0));
123  }
124  res = _wfindnext(lHandle, &tagData);
125  }
126  // Close if we found any files
127  if (lHandle != -1)
128  _findclose(lHandle);
129 #else
130  DIR* dir = opendir(MyGUI::UString(_folder).asUTF8_c_str());
131  struct dirent* dp;
132 
133  if (dir == NULL)
134  {
135  /* opendir() failed */
136  }
137 
138  rewinddir (dir);
139 
140  while ((dp = readdir (dir)) != NULL)
141  {
142  if (!isReservedDir (MyGUI::UString(dp->d_name).asWStr_c_str()))
143  _result.push_back(FileInfo(MyGUI::UString(dp->d_name).asWStr(), (dp->d_type == DT_DIR)));
144  }
145 
146  closedir(dir);
147 #endif
148 }

164 inline void scanFolder(VectorWString& _result, const std::wstring& _folder, bool _recursive, const std::wstring& _mask, bool _fullpath)
165 {
166  std::wstring folder = _folder;
167  if (!folder.empty())
168  folder += L"/";
169 
170  VectorFileInfo result;
171  getSystemFileList(result, folder, _mask);
172 
173  for (VectorFileInfo::const_iterator item = result.begin(); item != result.end(); ++item)
174  {
175  if (item->folder)
176  continue;
177 
178  if (_fullpath)
179  _result.push_back(folder + item->name);
180  else
181  _result.push_back(item->name);
182  }
183 
184  if (_recursive)
185  {
186  getSystemFileList(result, folder, L"*");
187 
188  for (VectorFileInfo::const_iterator item = result.begin(); item != result.end(); ++item)
189  {
190  if (!item->folder
191  || item->name == L".."
192  || item->name == L".")
193  continue;
194  scanFolder(_result, folder + item->name, _recursive, _mask, _fullpath);
195  }
196  }
197 }
