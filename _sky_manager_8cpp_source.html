86  TerrainGeometryManager* gm = gEnv->terrainManager->getGeometryManager();

88  gm->updateLightMap();
89  }
90

94 void SkyManager::loadScript(String script, int fogStart, int fogEnd)
95 {
96  // load the caelum config
97  try
98  {
99  Caelum::CaelumPlugin::getSingleton().loadCaelumSystemFromScript(mCaelumSystem, script);
100 
101  // overwrite some settings
102 #ifdef CAELUM_VERSION_SEC
103  // important: overwrite fog settings if not using infinite farclip
104  if (fogStart != -1 && fogEnd != -1 && fogStart < fogEnd)
105  {
106  // setting farclip (hacky)
107  gEnv->mainCamera->setFarClipDistance(fogEnd / 0.8);
108  // custom boundaries
109  mCaelumSystem->setManageSceneFog(FOG_LINEAR);
110  mCaelumSystem->setManageSceneFogStart(fogStart);
111  mCaelumSystem->setManageSceneFogEnd(fogEnd);
112  }
113  else if (gEnv->mainCamera->getFarClipDistance() > 0)
114  {
115  if (fogStart != -1 && fogEnd != -1)
116  {
117  LOG("CaelumFogStart must be smaller then CaelumFogEnd. Ignoring boundaries.");
118  }
119  else if (fogStart != -1 || fogEnd != -1)
120  {
121  LOG("You always need to define both boundaries (CaelumFogStart AND CaelumFogEnd). Ignoring boundaries.");
122  }
123  // non infinite farclip
124  Real farclip = gEnv->mainCamera->getFarClipDistance();
125  mCaelumSystem->setManageSceneFog(FOG_LINEAR);
126  mCaelumSystem->setManageSceneFogStart(farclip * 0.7);
127  mCaelumSystem->setManageSceneFogEnd(farclip * 0.9);
128  }
129  else
130  {
131  // no fog in infinite farclip
132  mCaelumSystem->setManageSceneFog(FOG_NONE);
133  }
134 #else
135 #error please use a recent Caelum version, see http://www.rigsofrods.org/wiki/pages/Compiling_3rd_party_libraries#Caelum
136 #endif // CAELUM_VERSION
137  // now optimize the moon a bit
138  if (mCaelumSystem->getMoon())
139  {
140  mCaelumSystem->getMoon()->setAutoDisable(true);
141  //mCaelumSystem->getMoon()->setAutoDisableThreshold(1);
142  mCaelumSystem->getMoon()->setForceDisable(true);
143  mCaelumSystem->getMoon()->getMainLight()->setCastShadows(false);
144  }
145 
146  mCaelumSystem->setEnsureSingleShadowSource(true);
147  mCaelumSystem->setEnsureSingleLightSource(true);
148 
149  // enforcing update, so shadows are set correctly before creating the terrain
150  forceUpdate(0.1);
151  }
152  catch (Exception& e)
153  {
154  LOG("exception upon loading sky script: " + e.getFullDescription());
155  }
156  Ogre::Vector3 lightsrc = mCaelumSystem->getSun()->getMainLight()->getDirection();
157  mCaelumSystem->getSun()->getMainLight()->setDirection(lightsrc.normalisedCopy());
158 }
