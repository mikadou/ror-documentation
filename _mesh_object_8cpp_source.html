95 void MeshObject::postProcess()
96 {
97  static const float sightrange = App::GetGfxSightRange();
98 
99  loaded = true;
100  if (!sceneNode)
101  return;
102 
103  // important: you need to add the LODs before creating the entity
104  // now find possible LODs, needs to be done before calling createEntity()
105  if (!mesh.isNull())
106  {
107  String basename, ext;
108  StringUtil::splitBaseFilename(meshName, basename, ext);
109 
110  String group = ResourceGroupManager::getSingleton().findGroupContainingResource(meshName);
111 
112  // the classic LODs
113  FileInfoListPtr files = ResourceGroupManager::getSingleton().findResourceFileInfo(group, basename + "_lod*.mesh");
114  for (FileInfoList::iterator iterFiles = files->begin(); iterFiles != files->end(); ++iterFiles)
115  {
116  String format = basename + "_lod%d.mesh";
117  int i = -1;
118  int r = sscanf(iterFiles->filename.c_str(), format.c_str(), &i);
119 
120  if (r <= 0 || i < 0)
121  continue;
122 
123  float distance = 3;
124 
125  // we need to tune this according to our sightrange
126  if (sightrange > gEnv->terrainManager->UNLIMITED_SIGHTRANGE)
127  {
128  // unlimited
129  if (i == 1)
130  distance = 200;
131  else if (i == 2)
132  distance = 600;
133  else if (i == 3)
134  distance = 2000;
135  else if (i == 4)
136  distance = 5000;
137  }
138  else
139  {
140  // limited
141  if (i == 1)
142  distance = std::max(20.0f, sightrange * 0.1f);
143  else if (i == 2)
144  distance = std::max(20.0f, sightrange * 0.2f);
145  else if (i == 3)
146  distance = std::max(20.0f, sightrange * 0.3f);
147  else if (i == 4)
148  distance = std::max(20.0f, sightrange * 0.4f);
149  }
150 
151  Ogre::MeshManager::getSingleton().load(iterFiles->filename, mesh->getGroup());
152  mesh->createManualLodLevel(distance, iterFiles->filename);
153  }
154 
155  // the custom LODs
156  FileInfoListPtr files2 = ResourceGroupManager::getSingleton().findResourceFileInfo(group, basename + "_clod_*.mesh");
157  for (FileInfoList::iterator iterFiles = files2->begin(); iterFiles != files2->end(); ++iterFiles)
158  {
159  // and custom LODs
160  String format = basename + "_clod_%d.mesh";
161  int i = -1;
162  int r = sscanf(iterFiles->filename.c_str(), format.c_str(), &i);
163  if (r <= 0 || i < 0)
164  continue;
165 
166  Ogre::MeshManager::getSingleton().load(iterFiles->filename, mesh->getGroup());
167  mesh->createManualLodLevel(i, iterFiles->filename);
168  }
169  }
170 
171  // now create an entity around the mesh and attach it to the scene graph
172  try
173  {
174  if (entityName.empty())
175  ent = gEnv->sceneManager->createEntity(meshName);
176  else
177  ent = gEnv->sceneManager->createEntity(entityName, meshName);
178  if (ent)
179  sceneNode->attachObject(ent);
180  }
181  catch (Ogre::Exception& e)
182  {
183  LOG("error loading mesh: " + meshName + ": " + e.getFullDescription());
184  return;
185  }
186 
187  // only set it if different from default (true)
188  if (!castshadows && sceneNode && sceneNode->numAttachedObjects() > 0)
189  sceneNode->getAttachedObject(0)->setCastShadows(castshadows);
190 
191  sceneNode->setVisible(visible);
192 }

194 void MeshObject::loadMesh()
195 {
196  try
197  {
198  Ogre::String resourceGroup = Ogre::ResourceGroupManager::DEFAULT_RESOURCE_GROUP_NAME;
199  mesh = static_cast<Ogre::MeshPtr>(Ogre::MeshManager::getSingleton().create(meshName, resourceGroup));
200  if (backgroundLoading)
201  {
202  mesh->setBackgroundLoaded(true);
203  mesh->addListener(this);
204  ticket = Ogre::ResourceBackgroundQueue::getSingleton().load(
205  Ogre::MeshManager::getSingletonPtr()->getResourceType(),
206  mesh->getName(),
207  resourceGroup,
208  false, 0, 0, 0);
209 
210  // try to load its textures in the background
211  for (int i = 0; i < mesh->getNumSubMeshes(); i++)
212  {
213  SubMesh* sm = mesh->getSubMesh(i);
214  String materialName = sm->getMaterialName();
215  Ogre::MaterialPtr mat = RoR::OgreSubsystem::GetMaterialByName(materialName);
216  if (mat.isNull())
217  continue;
218  for (int tn = 0; tn < mat->getNumTechniques(); tn++)
219  {
220  Technique* t = mat->getTechnique(tn);
221  for (int pn = 0; pn < t->getNumPasses(); pn++)
222  {
223  Pass* p = t->getPass(pn);
224  for (int tun = 0; tun < p->getNumTextureUnitStates(); tun++)
225  {
226  TextureUnitState* tu = p->getTextureUnitState(tun);
227  String textureName = tu->getTextureName();
228  // now add this texture to the background loading queue
229  Ogre::TexturePtr tex = static_cast<Ogre::TexturePtr>(Ogre::TextureManager::getSingleton().create(textureName, resourceGroup));
230  tex->setBackgroundLoaded(true);
231  tex->addListener(this);
232  ticket = Ogre::ResourceBackgroundQueue::getSingleton().load(
233  Ogre::TextureManager::getSingletonPtr()->getResourceType(),
234  tex->getName(),
235  resourceGroup,
236  false, 0, 0, 0);
237  }
238  }
239  }
240  }
241  }
242 
243  if (!backgroundLoading)
244  postProcess();
245  }
246  catch (Ogre::Exception* e)
247  {
248  LOG("exception while loading mesh: " + e->getFullDescription());
249  }
250 }
