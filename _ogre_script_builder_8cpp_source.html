38  // (reconstructed wrong by CScriptBuilder because it doesn't know about OGRE's ZIP files).

44  size_t slash_pos = full_path.rfind('/'); // AngelScript always uses forward slashes in paths.
45  if (slash_pos != std::string::npos)

48  }
49  else
