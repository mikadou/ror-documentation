76 void LanguageEngine::setup()
77 {
78 #ifdef USE_MOFILEREADER
79  // load language, must happen after initializing Settings class and Ogre Root!
80  // also it must happen after loading all basic resources!
81  reader = new moFileLib::moFileReader();
82 
83  String language = App::GetAppLanguage();
84  String language_short = App::GetAppLocale().substr(0, 2); // only first two characters are important
85 
86  // Load a .mo-File.
87  LOG("*** Loading Language ***");
88  std::string lang_dir = App::GetSysProcessDir() + PATH_SLASH + "languages" + PATH_SLASH;
89  String langfile = lang_dir + language_short + String("/LC_MESSAGES/ror.mo");
90  if (reader->ReadFile(langfile.c_str()) != moFileLib::moFileReader::EC_SUCCESS)
91  {
92  LOG("* error loading language file " + langfile);
93  return;
94  }
95  working = true;
96 
97  // add resource path
98  ResourceGroupManager::getSingleton().addResourceLocation(lang_dir + language_short + String("/LC_MESSAGES"), "FileSystem", "LanguageFolder");
99 
100  ResourceGroupManager::getSingleton().initialiseResourceGroup("LanguageFolder");
101 
102  // now load the code ranges
103  // be aware, that this approach only works if we load just one language, and not multiple
104  setupCodeRanges("code_range.txt", "LanguageFolder");
105 #else
106  // init gettext
107  bindtextdomain("ror","languages");
108  textdomain("ror");
109 
110  char *curr_locale = setlocale(LC_ALL,NULL);
111  if (curr_locale)
112  LOG("system locale is: " + String(curr_locale));
113  else
114  LOG("unable to read system locale!");
115 
116  String language_short = App::GetAppLocale().substr(0, 2); // only first two characters are important
117  if (!language_short.empty())
118  {
119  LOG("setting new locale to " + language_short);
120  char *newlocale = setlocale(LC_ALL, language_short.c_str());
121  if (newlocale)
122  LOG("new locale is: " + String(newlocale));
123  else
124  LOG("error setting new locale");
125  } else
126  {
127  LOG("not changing locale, using system locale");
128  }
129 
130 #endif // USE_MOFILEREADER
131  LOG("* Language successfully loaded");
132 }
