166 void SurveyMapManager::setMapZoom(Real zoomValue, bool update /*= true*/, bool permanent /*= true*/)
167 {

223 void SurveyMapManager::setWindowPosition(int x, int y, float size)
224 {
225  int realx = 0;
226  int realy = 0;
227 
228  updateRenderMetrics();
229 
230  realw = realh = size * std::min(rWinWidth, rWinHeight);
231

233  {
234  realx = 0;

236  else if (x == 0)

243  }
244 
245  if (y == -1)

250  {

255  realy = rWinHeight - realh;

280 void SurveyMapManager::Update(Ogre::Real dt, Beam* curr_truck)
281 {
282  if (dt == 0)
283  return;
284 
285  mVelocity = 0.0f;
286 
287  if (curr_truck)
288  {
289  mVelocity = curr_truck->nodes[0].Velocity.length();
290  }
291 
292  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_SURVEY_MAP_TOGGLE_VIEW))
293  {
294  toggleMapView();
295  }
296 
297  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_SURVEY_MAP_TOGGLE_ICONS))
298  {
299  mMapEntitiesVisible = !mMapEntitiesVisible;
300  }
301 
302  if (mMapMode == SURVEY_MAP_NONE)
303  return;
304 
305  switch (mMapMode)
306  {
307  case SURVEY_MAP_SMALL:
308  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_SURVEY_MAP_ALPHA))
309  {
310  toggleMapAlpha();
311  }
312  if (RoR::App::GetInputEngine()->getEventBoolValue(EV_SURVEY_MAP_ZOOM_IN))
313  {
314  setMapZoomRelative(1.0f);
315  }
316  if (RoR::App::GetInputEngine()->getEventBoolValue(EV_SURVEY_MAP_ZOOM_OUT))
317  {
318  setMapZoomRelative(-1.0f);
319  }
320  if (getMapZoom() > 0.0f)
321  {
322  if (curr_truck)
323  {
324  setMapCenter(curr_truck->getPosition(), mMapCenterThreshold * (1 - mMapZoom));
325  }
326  else
327  {
328  setMapCenter(gEnv->player->getPosition(), mMapCenterThreshold * (1 - mMapZoom));
329  }
330  }
331  else
332  {
333  setMapCenter(Vector3(mMapSize.x / 2.0f, 0.0f, mMapSize.z / 2.0f));
334  }
335  break;
336 
337  case SURVEY_MAP_BIG:
338  setMapCenter(Vector3(mMapSize.x / 2.0f, 0.0f, mMapSize.z / 2.0f));
339 
340  if (getMapZoom() > 0.0f)
341  {
342  setMapZoom(0.0f, mMapTextureNeedsUpdate, false);
343  mMapTextureNeedsUpdate = false;
344  }
345 
346  if (curr_truck &&
347  gEnv->cameraManager &&
348  gEnv->cameraManager->hasActiveBehavior() &&
349  !gEnv->cameraManager->gameControlsLocked())
350  {
351  if (mVelocity > 5.0f || gEnv->cameraManager->getCurrentBehavior() == CameraManager::CAMERA_BEHAVIOR_VEHICLE_CINECAM)
352  {
353  setWindowPosition(1, -1, 0.3f);
354  //setAlpha(mAlpha);
355  mMapMode = SURVEY_MAP_SMALL;
356  }
357  else
358  {
359  // Soft cross-fade into small map
360  //setAlpha(1.0f / sqrt(std::max(1.0f, mVelocity - 1.0f)), false);
361  }
362  }
363  else
364  {
365  if (getAlpha() != mAlpha)
366  {
367  setAlpha(mAlpha);
368  }
369 
370  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_SURVEY_MAP_ALPHA))
371  {
372  toggleMapAlpha();
373  }
374  }
375  break;
376 
377  default:
378  break;
379  }
380 
381  updateMapEntityPositions();
382 }

384 void SurveyMapManager::toggleMapView()
385 {
386  mMapMode = (mMapMode + 1) % SURVEY_MAP_END;
387 
388  if (mMapMode == SURVEY_MAP_BIG && (mVelocity > 5.0f ||
389  (gEnv->cameraManager &&
390  gEnv->cameraManager->hasActiveBehavior() &&
391  gEnv->cameraManager->getCurrentBehavior() == CameraManager::CAMERA_BEHAVIOR_VEHICLE_CINECAM)))
392  {
393  mMapMode = (mMapMode + 1) % SURVEY_MAP_END;
394  }
395 
396  if (mMapMode == SURVEY_MAP_NONE)
397  {
398  setVisibility(false);
399  }
400  else
401  {
402  if (mMapMode == SURVEY_MAP_SMALL)
403  {
404  setWindowPosition(1, -1, 0.3f);
405  }
406  else if (mMapMode == SURVEY_MAP_BIG)
407  {
408  setWindowPosition(0, 0, 0.98f);
409  }
410  setAlpha(mAlpha);
411  setVisibility(true);
412  }
413 }
