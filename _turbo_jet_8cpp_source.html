135 void Turbojet::updateVisuals()
136 {
137  //nozzle
138  nzsnode->setPosition(nodes[nodeback].AbsPosition);
139  //build a local system
140  Vector3 laxis = nodes[nodefront].RelPosition - nodes[nodeback].RelPosition;
141  laxis.normalise();
142  Vector3 paxis = Plane(laxis, 0).projectVector(nodes[noderef].RelPosition - nodes[nodeback].RelPosition);
143  paxis.normalise();
144  Vector3 taxis = laxis.crossProduct(paxis);
145  Quaternion dir = Quaternion(laxis, paxis, taxis);
146  nzsnode->setOrientation(dir);
147  //afterburner
148  if (afterburner)
149  {
150  absnode->setVisible(true);
151  float flamelength = (afterburnthrust / 15.0) * (rpm / 100.0);
152  flamelength = flamelength * (1.0 + (((Real)rand() / (Real)RAND_MAX) - 0.5) / 10.0);
153  absnode->setScale(flamelength, radius * 2.0, radius * 2.0);
154  absnode->setPosition(nodes[nodeback].AbsPosition + dir * Vector3(-0.2, 0.0, 0.0));
155  absnode->setOrientation(dir);
156  }
157  else
158  absnode->setVisible(false);
159  //smoke
160  if (smokeNode)
161  {
162  smokeNode->setPosition(nodes[nodeback].AbsPosition);
163  ParticleEmitter* emit = smokePS->getEmitter(0);
164  ParticleEmitter* hemit = 0;
165  if (heathazePS)
166  hemit = heathazePS->getEmitter(0);
167  emit->setDirection(-axis);
168  emit->setParticleVelocity(exhaust_velocity);

173  }
174  if (!failed)
175  {
176  if (ignition)
177  {
178  emit->setEnabled(true);
179  emit->setColour(ColourValue(0.0, 0.0, 0.0, 0.02 + throtle * 0.03));

213 void Turbojet::updateForces(float dt, int doUpdate)
214 {
215  if (doUpdate)
216  {
217 #ifdef USE_OPENAL
218  //sound update
219  SoundScriptManager::getSingleton().modulate(trucknum, mod_id, rpm);
220 #endif //OPENAL
221  }
222  timer += dt;
223  axis = nodes[nodefront].RelPosition - nodes[nodeback].RelPosition;
224  float axlen = axis.length();
225  axis = axis / axlen; //normalize
226  if (fabs(reflen - axlen) > 0.1)
227  {
228  rpm = 0;
229  failed = true;
230  }; //check for broken
231 
232  float warmupfactor = 1.0;
233  if (warmup)
234  {
235  warmupfactor = (timer - warmupstart) / warmuptime;
236  if (warmupfactor >= 1.0)
237  warmup = false;
238  }
239 
240  //turbine RPM
241  //braking (compression)
242  if (rpm < 0)
243  rpm = 0;
244  float torque = -rpm / 100.0;
245  //powering with limiter
246  if (rpm < 100.0 && !failed && ignition)
247  torque += ((0.2 + throtle * 0.8) * warmupfactor);
248  //integration
249  rpm += (double)dt * torque * 30.0;
250 
251  float enginethrust = 0;
252  if (!failed && ignition)
253  {
254  enginethrust = maxdrythrust * rpm / 100.0;
255  afterburner = (afterburnable && throtle > 0.95 && rpm > 80);
256  if (afterburner)
257  enginethrust += (afterburnthrust - maxdrythrust);
258  }
259  else
260  afterburner = false;
261 #ifdef USE_OPENAL
262  if (afterburner)
263  SoundScriptManager::getSingleton().trigStart(trucknum, ab_id);
264  else
265  SoundScriptManager::getSingleton().trigStop(trucknum, ab_id);
266 #endif //OPENAL
267 
268  nodes[nodeback].Forces += (enginethrust * 1000.0) * axis;
269  exhaust_velocity = enginethrust * 5.6 / area;
270 }
