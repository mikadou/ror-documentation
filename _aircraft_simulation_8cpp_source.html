40 void AircraftSimulation::UpdateVehicle(Beam* vehicle, float seconds_since_last_frame)
41 {
42  //autopilot
43  if (vehicle->autopilot && vehicle->autopilot->wantsdisconnect)
44  {
45  vehicle->disconnectAutopilot();
46  }
47  //AIRPLANE KEYS
48  float commandrate = 4.0;
49  //turning
50  if (vehicle->replaymode)
51  {
52  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_COMMON_REPLAY_FORWARD, 0.1f) && vehicle->replaypos <= 0)
53  {
54  vehicle->replaypos++;
55  }
56  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_COMMON_REPLAY_BACKWARD, 0.1f) && vehicle->replaypos > -vehicle->replaylen)
57  {
58  vehicle->replaypos--;
59  }
60  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_COMMON_REPLAY_FAST_FORWARD, 0.1f) && vehicle->replaypos + 10 <= 0)
61  {
62  vehicle->replaypos += 10;
63  }
64  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_COMMON_REPLAY_FAST_BACKWARD, 0.1f) && vehicle->replaypos - 10 > -vehicle->replaylen)
65  {
66  vehicle->replaypos -= 10;
67  }
68  }
69  else
70  {
71  float tmp_left = RoR::App::GetInputEngine()->getEventValue(EV_AIRPLANE_STEER_LEFT);
72  float tmp_right = RoR::App::GetInputEngine()->getEventValue(EV_AIRPLANE_STEER_RIGHT);
73  float sum_steer = -tmp_left + tmp_right;
74  RoR::App::GetInputEngine()->smoothValue(vehicle->aileron, sum_steer, seconds_since_last_frame * commandrate);
75  vehicle->hydrodircommand = vehicle->aileron;
76  vehicle->hydroSpeedCoupling = !(RoR::App::GetInputEngine()->isEventAnalog(EV_AIRPLANE_STEER_LEFT) && RoR::App::GetInputEngine()->isEventAnalog(EV_AIRPLANE_STEER_RIGHT));
77  }
78 
79  //pitch
80  float tmp_pitch_up = RoR::App::GetInputEngine()->getEventValue(EV_AIRPLANE_ELEVATOR_UP);
81  float tmp_pitch_down = RoR::App::GetInputEngine()->getEventValue(EV_AIRPLANE_ELEVATOR_DOWN);
82  float sum_pitch = tmp_pitch_down - tmp_pitch_up;
83  RoR::App::GetInputEngine()->smoothValue(vehicle->elevator, sum_pitch, seconds_since_last_frame * commandrate);
84 
85  //rudder
86  float tmp_rudder_left = RoR::App::GetInputEngine()->getEventValue(EV_AIRPLANE_RUDDER_LEFT);
87  float tmp_rudder_right = RoR::App::GetInputEngine()->getEventValue(EV_AIRPLANE_RUDDER_RIGHT);
88  float sum_rudder = tmp_rudder_left - tmp_rudder_right;
89  RoR::App::GetInputEngine()->smoothValue(vehicle->rudder, sum_rudder, seconds_since_last_frame * commandrate);
90 
91  //brake
92  if (!vehicle->replaymode && !vehicle->parkingbrake)
93  {
94  vehicle->brake = 0.0;
95  float brakevalue = RoR::App::GetInputEngine()->getEventValue(EV_AIRPLANE_BRAKE);
96  vehicle->brake = vehicle->brakeforce * 0.66 * brakevalue;
97  }
98  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_AIRPLANE_PARKING_BRAKE))
99  {
100  vehicle->parkingbrakeToggle();
101  if (RoR::App::GetOverlayWrapper())
102  {
103  if (vehicle->parkingbrake)
104  {
105  Ogre::OverlayManager::getSingleton().getOverlayElement("tracks/ap_brks_but")->setMaterialName("tracks/brks-on");
106  }
107  else
108  {
109  Ogre::OverlayManager::getSingleton().getOverlayElement("tracks/ap_brks_but")->setMaterialName("tracks/brks-off");
110  }
111  }
112  }
113  //reverse
114  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_AIRPLANE_REVERSE))
115  {
116  for (int i = 0; i < vehicle->free_aeroengine; i++)
117  {
118  vehicle->aeroengines[i]->toggleReverse();
119  }
120  }
121 
122  // toggle engines
123  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_AIRPLANE_TOGGLE_ENGINES))
124  {
125  for (int i = 0; i < vehicle->free_aeroengine; i++)
126  {
127  vehicle->aeroengines[i]->flipStart();
128  }
129  }
130 
131  //flaps
132  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_AIRPLANE_FLAPS_NONE))
133  {
134  if (vehicle->flap > 0)
135  {
136  vehicle->flap = 0;
137  }
138  }
139  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_AIRPLANE_FLAPS_FULL))
140  {
141  if (vehicle->flap < 5)
142  {
143  vehicle->flap = 5;
144  }
145  }
146  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_AIRPLANE_FLAPS_LESS))
147  {
148  if (vehicle->flap > 0)
149  {
150  vehicle->flap = (vehicle->flap) - 1;
151  }
152  }
153  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_AIRPLANE_FLAPS_MORE))
154  {
155  if (vehicle->flap < 5)
156  {
157  vehicle->flap = (vehicle->flap) + 1;
158  }
159  }
160 
161  //airbrakes
162  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_AIRPLANE_AIRBRAKES_NONE))
163  {
164  if (vehicle->airbrakeval > 0)
165  {
166  vehicle->airbrakeval = 0;
167  }
168  }
169  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_AIRPLANE_AIRBRAKES_FULL))
170  {
171  if (vehicle->airbrakeval < 5)
172  {
173  vehicle->airbrakeval = 5;
174  }
175  }
176  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_AIRPLANE_AIRBRAKES_LESS))
177  {
178  if (vehicle->airbrakeval > 0)
179  {
180  vehicle->airbrakeval = (vehicle->airbrakeval) - 1;
181  }
182  }
183  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_AIRPLANE_AIRBRAKES_MORE))
184  {
185  if (vehicle->airbrakeval < 5)
186  {
187  vehicle->airbrakeval = (vehicle->airbrakeval) + 1;
188  }
189  }
190 
191  //throttle
192  float tmp_throttle = RoR::App::GetInputEngine()->getEventBoolValue(EV_AIRPLANE_THROTTLE);
193  if (tmp_throttle > 0)
194  {
195  for (int i = 0; i < vehicle->free_aeroengine; i++)
196  {
197  vehicle->aeroengines[i]->setThrottle(tmp_throttle);
198  }
199  }
200  if (RoR::App::GetInputEngine()->isEventDefined(EV_AIRPLANE_THROTTLE_AXIS))
201  {
202  float f = RoR::App::GetInputEngine()->getEventValue(EV_AIRPLANE_THROTTLE_AXIS);
203  for (int i = 0; i < vehicle->free_aeroengine; i++)
204  {
205  vehicle->aeroengines[i]->setThrottle(f);
206  }
207  }
208  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_AIRPLANE_THROTTLE_DOWN, 0.1f))
209  {
210  //throttle down
211  for (int i = 0; i < vehicle->free_aeroengine; i++)
212  {
213  vehicle->aeroengines[i]->setThrottle(vehicle->aeroengines[i]->getThrottle() - 0.05);
214  }
215  }
216  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_AIRPLANE_THROTTLE_UP, 0.1f))
217  {
218  //throttle up
219  for (int i = 0; i < vehicle->free_aeroengine; i++)
220  {
221  vehicle->aeroengines[i]->setThrottle(vehicle->aeroengines[i]->getThrottle() + 0.05);
222  }
223  }
224  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_AIRPLANE_THROTTLE_NO, 0.1f))
225  {
226  // no throttle
227  for (int i = 0; i < vehicle->free_aeroengine; i++)
228  {
229  vehicle->aeroengines[i]->setThrottle(0);
230  }
231  }
232  if (RoR::App::GetInputEngine()->getEventBoolValueBounce(EV_AIRPLANE_THROTTLE_FULL, 0.1f))
233  {
234  // full throttle
235  for (int i = 0; i < vehicle->free_aeroengine; i++)
236  {
237  vehicle->aeroengines[i]->setThrottle(1);
238  }
239  }
240  if (vehicle->autopilot)
241  {
242  for (int i = 0; i < vehicle->free_aeroengine; i++)
243  {
244  vehicle->aeroengines[i]->setThrottle(vehicle->autopilot->getThrottle(vehicle->aeroengines[i]->getThrottle(), seconds_since_last_frame));
245  }
246  }
247 }
