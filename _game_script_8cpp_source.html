364 void GameScript::spawnObject(const String& objectName, const String& instanceName, const Vector3& pos, const Vector3& rot, const String& eventhandler, bool uniquifyMaterials)
365 {
366  if ((gEnv->terrainManager == nullptr) || (gEnv->terrainManager->getObjectManager() == nullptr))
367  {
368  this->logFormat("spawnObject(): Cannot spawn object, no terrain loaded!");
369  return;
370  }
371 
372  try
373  {
374  AngelScript::asIScriptModule* module = mse->getEngine()->GetModule(mse->moduleName, AngelScript::asGM_ONLY_IF_EXISTS);
375  if (module == nullptr)
376  {
377  this->logFormat("spawnObject(): Failed to fetch/create script module '%s'", mse->moduleName);
378  return;
379  }
380 
381  int handler_func_id = -1; // no function
382  if (!eventhandler.empty())
383  {
384  AngelScript::asIScriptFunction* handler_func = module->GetFunctionByName(eventhandler.c_str());
385  if (handler_func != nullptr)
386  {
387  handler_func_id = handler_func->GetId();
388  }
389  else
390  {
391  this->logFormat("spawnObject(): Warning; Failed to find handler function '%s' in script module '%s'",
392  eventhandler.c_str(), mse->moduleName);
393  }
394  }
395 
396  SceneNode* bakeNode = gEnv->sceneManager->getRootSceneNode()->createChildSceneNode();
397  const String type = "";
398  gEnv->terrainManager->getObjectManager()->loadObject(objectName, pos, rot, bakeNode, instanceName, type, true, handler_func_id, uniquifyMaterials);
399  }
400  catch (std::exception e)
401  {
402  this->logFormat("spawnObject(): An exception occurred, message: %s", e.what());
403  return;
404  }
405 }

480 int GameScript::getSafeTextureUnitState(TextureUnitState** tu, const String materialName, int techniqueNum, int passNum, int textureUnitNum)
481 {
482  try
483  {
484  MaterialPtr m = MaterialManager::getSingleton().getByName(materialName);
485  if (m.isNull())
486  return 1;
487 
488  // verify technique
489  if (techniqueNum < 0 || techniqueNum > m->getNumTechniques())
490  return 2;
491  Technique* t = m->getTechnique(techniqueNum);
492  if (!t)
493  return 2;
494 
495  //verify pass
496  if (passNum < 0 || passNum > t->getNumPasses())
497  return 3;
498  Pass* p = t->getPass(passNum);
499  if (!p)
500  return 3;
501 
502  //verify texture unit
503  if (textureUnitNum < 0 || textureUnitNum > p->getNumTextureUnitStates())
504  return 4;
505  TextureUnitState* tut = p->getTextureUnitState(textureUnitNum);
506  if (!tut)
507  return 4;
508 
509  *tu = tut;
510  return 0;
511  }
512  catch (Exception e)
513  {
514  this->log("Exception in getSafeTextureUnitState(): " + e.getFullDescription());
515  }
516  return 1;
517 }
