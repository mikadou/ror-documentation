164 UTFString TopMenubar::getUserString(RoRnet::UserInfo &user, int num_vehicles)
165 {
166  UTFString tmp = RoR::ChatSystem::GetColouredName(user.username, user.colournum);
167 
168  tmp = tmp + U(": ");
169 
170  // some more info
171  if (user.authstatus & RoRnet::AUTH_BOT)
172  tmp = tmp + _L("#0000c9 Bot, ");
173  else if (user.authstatus & RoRnet::AUTH_BANNED)
174  tmp = tmp + _L("banned, ");
175  else if (user.authstatus & RoRnet::AUTH_RANKED)
176  tmp = tmp + _L("#00c900 Ranked, ");
177  else if (user.authstatus & RoRnet::AUTH_MOD)
178  tmp = tmp + _L("#c90000 Moderator, ");
179  else if (user.authstatus & RoRnet::AUTH_ADMIN)
180  tmp = tmp + _L("#c97100 Admin, ");
181 
182  tmp = tmp + _L("#ff8d00 version: #3eff20 ");
183  tmp = tmp + ANSI_TO_UTF(user.clientversion);
184  tmp = tmp + U(", ");
185 
186  tmp = tmp + _L("#ff8d00 language: #46b1f9 ");
187  tmp = tmp + ANSI_TO_UTF(user.language);
188  tmp = tmp + U(", ");
189 
190  if (num_vehicles == 0)
191  tmp = tmp + _L("no vehicles");
192  else
193  tmp = tmp + TOUTFSTRING(num_vehicles) + _L(" vehicles");
194 
195  return tmp;

198 void TopMenubar::addUserToMenu(RoRnet::UserInfo &user)
199 {
200  int numTrucks = m_sim_controller->GetBeamFactory()->getTruckCount();
201  Beam **trucks = m_sim_controller->GetBeamFactory()->getTrucks();
202 
203  // now search the vehicles of that user together
204  std::vector<int> matches;
205  for (int j = 0; j < numTrucks; j++)
206  {
207  if (!trucks[j]) continue;
208 
209  if (trucks[j]->m_source_id == user.uniqueid)
210  {
211  // match, found truck :)
212  matches.push_back(j);
213  }
214  }
215 
216  // now add this user to the list
217  {
218  MyGUI::UString userStr = "- " + convertToMyGUIString(getUserString(user, (int)matches.size()));
219  // finally add the user line
220  m_vehicles_menu_widget->addItem(userStr, MyGUI::MenuItemType::Normal, "USER_"+TOSTRING(user.uniqueid));
221 
222  // and add the vehicles below the user name
223  if (!matches.empty())
224  {
225  for (unsigned int j = 0; j < matches.size(); j++)
226  {
227  char tmp[512] = "";
228  sprintf(tmp, " + %s (%s)", trucks[matches[j]]->realtruckname.c_str(), trucks[matches[j]]->realtruckfilename.c_str());
229  MyGUI::UString vehName = convertToMyGUIString(ANSI_TO_UTF(tmp));
230  m_vehicles_menu_widget->addItem(vehName, MyGUI::MenuItemType::Normal, "TRUCK_"+TOSTRING(matches[j]));
231  }
232  }
233  }
234 }

236 void TopMenubar::vehiclesListUpdate()
237 {
238  m_vehicles_menu_widget->removeAllItems();
239 
240  if (!(App::GetActiveMpState() == App::MP_STATE_CONNECTED))
241  {
242  // single player mode: add vehicles simply, no users
243  int numTrucks = m_sim_controller->GetBeamFactory()->getTruckCount();
244  Beam **trucks = m_sim_controller->GetBeamFactory()->getTrucks();
245 
246  // simple iterate through :)
247  for (int i = 0; i < numTrucks; i++)
248  {
249  if (!trucks[i]) continue;
250 
251  if (trucks[i]->hideInChooser) continue;
252 
253  char tmp[255] = {};
254  sprintf(tmp, "[%d] %s", i, trucks[i]->realtruckname.c_str());
255 
256  m_vehicles_menu_widget->addItem(String(tmp), MyGUI::MenuItemType::Normal, "TRUCK_"+TOSTRING(i));
257  }
258  } else
259  {
260 #ifdef USE_SOCKETW
261  // sort the list according to the network users
262 
263  RoRnet::UserInfo local_user = RoR::Networking::GetLocalUserData();
264  addUserToMenu(local_user);
265 
266  auto users = RoR::Networking::GetUserInfos();
267  for (auto user : users)
268  {
269  addUserToMenu(user);
270  }
271 #endif // USE_SOCKETW
272  }

275 void TopMenubar::onMenuBtn(MyGUI::MenuCtrlPtr _sender, MyGUI::MenuItemPtr _item)
276 {
277  UTFString miname = UTFString(_item->getCaption().asWStr());
278  String id = _item->getItemId();
279  auto* gui_man = RoR::App::GetGuiManager();
280 
281  if (id.substr(0,6) == "TRUCK_")
282  {
283  int truck = PARSEINT(id.substr(6));
284  if (truck >= 0 && truck < m_sim_controller->GetBeamFactory()->getTruckCount())
285  {
286  m_sim_controller->GetBeamFactory()->setCurrentTruck(truck);
287  }
288  }
289 
290  if (id.substr(0,5) == "USER_")
291  {
292  int user_uid = PARSEINT(id.substr(5));
293 
294  // cannot whisper with self...
295 #ifdef USE_SOCKETW
296  if (user_uid == RoR::Networking::GetUID()) return;
297 #endif // USE_SOCKETW
298 
299  //RoR::App::GetConsole()->startPrivateChat(user_uid);
300  //TODO: Separate Chat and console
301  }
302 
303  const auto app_state = App::GetActiveAppState();
304  if (app_state == App::APP_STATE_BOOTSTRAP)
305  {
306  return;
307  }
308 
309  if (_item == m_item_spawner_log)
310  {
311  App::GetGuiManager()->SetVisible_SpawnerReport(true);
312  }
313  else if (miname == _L("Get new vehicle") && gEnv->player)
314  {
315  if (app_state != App::APP_STATE_SIMULATION)
316  {
317  return;
318  }
319  App::SetActiveSimState(App::SIM_STATE_SELECTING); // TODO: use 'pending' mechanism
320  gui_man->GetMainSelector()->Show(LT_AllBeam);
321 
322  } else if (miname == _L("Reload current vehicle") && gEnv->player)
323  {
324  if ((m_sim_controller->GetBeamFactory()->getCurrentTruckNumber() != -1) && (m_sim_controller != nullptr))
325  {
326  m_sim_controller->ReloadCurrentTruck(); // TODO: Use SIM_STATE + 'pending' mechanisms
327  gui_man->UnfocusGui();
328  }
329  }
330  else if (miname == _L("Back to menu"))
331  {
332  App::SetPendingAppState(App::APP_STATE_MAIN_MENU);
333  }
334  else if (miname == _L("Remove current vehicle"))
335  {
336  m_sim_controller->GetBeamFactory()->removeCurrentTruck();
337 
338  } else if (miname == _L("Activate all vehicles"))
339  {
340  m_sim_controller->GetBeamFactory()->activateAllTrucks();
341 
342  } else if (miname == _L("Activated vehicles never sleep"))
343  {
344  m_sim_controller->GetBeamFactory()->setTrucksForcedActive(true);
345  _item->setCaption(_L("Activated vehicles can sleep"));
346 
347  } else if (miname == _L("Activated vehicles can sleep"))
348  {
349  m_sim_controller->GetBeamFactory()->setTrucksForcedActive(false);
350  _item->setCaption(_L("Activated vehicles never sleep"));
351 
352  } else if (miname == _L("Send all vehicles to sleep"))
353  {
354  // get out first
355  if (m_sim_controller->GetBeamFactory()->getCurrentTruckNumber() != -1)
356  m_sim_controller->GetBeamFactory()->setCurrentTruck(-1);
357  m_sim_controller->GetBeamFactory()->sendAllTrucksSleeping();
358 
359  } else if (miname == _L("Friction Settings"))
360  {
361  App::GetGuiManager()->SetVisible_FrictionSettings(true);
362  } else if (miname == _L("Exit"))
363  {
364  App::SetPendingAppState(App::APP_STATE_SHUTDOWN);
365  } else if (miname == _L("Show Console"))
366  {
367  gui_man->SetVisible_Console(! gui_man->IsVisible_Console());
368  }
369  // the debug menu
370  else if (miname == _L("no visual debug"))
371  {
372  Beam *b = m_sim_controller->GetBeamFactory()->getCurrentTruck();
373  if (b) b->setDebugOverlayState(0);
374  } else if (miname == _L("show Node numbers"))
375  {
376  Beam *b = m_sim_controller->GetBeamFactory()->getCurrentTruck();
377  if (b) b->setDebugOverlayState(1);
378  } else if (miname == _L("show Beam numbers"))
379  {

381  if (b) b->setDebugOverlayState(2);
382  } else if (miname == _L("show Node&Beam numbers"))
383  {

385  if (b) b->setDebugOverlayState(3);
386  } else if (miname == _L("show Node mass"))
