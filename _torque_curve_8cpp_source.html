96 int TorqueCurve::processLine(Ogre::StringVector args, String model)
97 {
98  // if its just one arguments, it must be a known model
99  if (args.size() == 1)
100  return setTorqueModel(args[0]);
101 
102  // we only accept 2 arguments
103  if (args.size() != 2)
104  return 1;
105  // parse the data
106  float pointx = StringConverter::parseReal(args[0]);
107  float pointy = StringConverter::parseReal(args[1]);
108  Vector3 point = Vector3(pointx, pointy, 0);
109 
110  // find the spline to attach the points
111  if (splines.find(model) == splines.end())
112  splines[model] = SimpleSpline();
113 
114  // attach the points to the spline
115  // LOG("curve "+model+" : " + TOSTRING(point));
116  splines[model].addPoint(point);
117 
118  // special case for custom model:
119  // we set it as active curve as well!
120  if (model == TorqueCurve::customModel)
121  setTorqueModel(TorqueCurve::customModel);
122 
123  return 0;
124 }

163 int TorqueCurve::spaceCurveEvenly(Ogre::SimpleSpline* spline)
164 {
165  if (!spline)
166  return 2;
167 
168  SimpleSpline tmpSpline = *spline;
169  Real points = tmpSpline.getNumPoints();
170 
171  if (points > 1)
172  {
173  // clear the original spline, so it's prepared for the new corrected numbers
174  spline->clear();
175  Real minDistance = tmpSpline.getPoint(1).x - tmpSpline.getPoint(0).x;
176  // looking for the minimum distance (spacing) in the current spline
177  for (int i = 2; i < points; i++)
178  {
179  Real distance = tmpSpline.getPoint(i).x - tmpSpline.getPoint(i - 1).x;
180  minDistance = std::min(distance, minDistance);
181  }
182  // the rpm points must be in an ascending order, as the points should be added at the end of the spline
183  if (minDistance < 0)
184  return 1;
185  // first(smallest)- and last(greatest) rpm
186  Vector3 minPoint = tmpSpline.getPoint(0);
187  Vector3 maxPoint = tmpSpline.getPoint(points - 1);
188 
189  Real rpmPoint = minPoint.x;
190  int pointIndex = 1; // this is the index used to interpolate between the rpm points
191  while (rpmPoint <= maxPoint.x && pointIndex < points)
192  {
193  // if actual rpm is higher than point of the spline, proceed to the next point
194  if (rpmPoint > tmpSpline.getPoint(pointIndex).x)
195  pointIndex++;
196  // interpolate(linear)
197  Real newPoint = tmpSpline.getPoint(pointIndex - 1).y + (tmpSpline.getPoint(pointIndex).y - tmpSpline.getPoint(pointIndex - 1).y) /
198  (tmpSpline.getPoint(pointIndex).x - tmpSpline.getPoint(pointIndex - 1).x) * (rpmPoint - tmpSpline.getPoint(pointIndex - 1).x);
199  spline->addPoint(Vector3(rpmPoint, newPoint, 0));
200  rpmPoint += minDistance;
201  }
202  // if the last point is missing due the even spacing, we add the last point manually
203  // criterion is that it must be smaller than 1% of the maximum rpm.
204  if (spline->getPoint(spline->getNumPoints() - 1).x < maxPoint.x && (rpmPoint - maxPoint.x) < 0.01 * maxPoint.x)
205  {
206  spline->addPoint(Vector3(rpmPoint, maxPoint.y, 0));
207  }
208  }
209 
210  return 0;
211 }
