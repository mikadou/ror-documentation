85 float Autopilot::getAilerons()
86 {
87  float val = 0;
88  if (ref_l && ref_r)
89  {
90  float rat = (ref_r->RelPosition.y - ref_l->RelPosition.y) / ref_span;
91  float bank = 90.0;
92  if (rat >= 1.0)
93  bank = 90.0;
94  else if (rat <= -1.0)
95  bank = -90.0;
96  else
97  bank = 57.3 * asin(rat);
98  if (mode_heading == HEADING_WLV)
99  {
100  val = bank / 100.0;
101  if (val > 0.5)
102  val = 0.5;
103  if (val < -0.5)
104  val = -0.5;
105  }
106  if (mode_heading == HEADING_FIXED)
107  {
108  Vector3 vel = (ref_l->Velocity + ref_r->Velocity) / 2.0;
109  float curdir = atan2(vel.x, -vel.z) * 57.295779513082;
110  float want_bank = curdir - (float)heading;
111  if (want_bank < -180.0)
112  want_bank += 360.0;
113  want_bank = want_bank * 2.0;
114  if (want_bank > 45.0)
115  want_bank = 45.0;
116  if (want_bank < -45.0)
117  want_bank = -45.0;
118  val = (bank - want_bank) / 100.0;
119  if (val > 0.5)
120  val = 0.5;
121  if (val < -0.5)
122  val = -0.5;
123  }
124  if (mode_heading == HEADING_NAV)
125  {
126  //compute intercept heading
127  float error_heading = lastradioh / 10.0;
128  if (error_heading > 1.0)
129  error_heading = 1.0;
130  if (error_heading < -1.0)
131  error_heading = -1.0;
132  float offcourse_tolerance = lastradiorwd / 30.0;
133  if (offcourse_tolerance > 60.0)
134  offcourse_tolerance = 60.0;
135  float intercept_heading = lastradiorwh + error_heading * offcourse_tolerance;
136  Vector3 vel = (ref_l->Velocity + ref_r->Velocity) / 2.0;
137  float curdir = atan2(vel.x, -vel.z) * 57.295779513082;
138  float want_bank = curdir - intercept_heading;
139  if (want_bank < -180.0)
140  want_bank += 360.0;
141  want_bank = want_bank * 2.0;
142  if (want_bank > 45.0)
143  want_bank = 45.0;
144  if (want_bank < -45.0)
145  want_bank = -45.0;
146  val = (bank - want_bank) / 100.0;
147  if (val > 0.5)
148  val = 0.5;
149  if (val < -0.5)
150  val = -0.5;
151  }
152  }
153  last_aileron = (last_aileron + val) / 2.0;
154  return last_aileron;
155 }

157 float Autopilot::getElevator()
158 {
159  float val = 0;
160  if (ref_l && ref_r && ref_b)
161  {
162  float wanted_vs = (float)vs / 196.87;
163  float current_vs = (ref_l->Velocity.y + ref_r->Velocity.y) / 2.0;
164  float pitch_var = current_vs - ref_b->Velocity.y;
165  if (mode_alt == ALT_VS)
166  {
167  if (mode_heading == HEADING_NAV)
168  {
169  //this is cheating a bit
170  float ch = lastradiorwd * sin((lastradiov + 4.0) / 57.295779513082);
171  float oh = lastradiorwd * sin((4.0) / 57.295779513082);
172  wanted_vs = 5000.0 / 196.87;
173  float wanted_vs2 = (-ch + oh) / 5.0;
174  if (wanted_vs2 < -wanted_vs)
175  wanted_vs2 = -wanted_vs;
176  if (wanted_vs2 > wanted_vs)
177  wanted_vs2 = wanted_vs;
178  val = (wanted_vs2 - current_vs) / 40.0 + pitch_var / 40.0;
179  if (val > 0.75)
180  val = 0.75;
181  if (val < -0.75)
182  val = -0.75;
183  }
184  else
185  {
186  val = (wanted_vs - current_vs) / 40.0 + pitch_var / 40.0;
187  if (val > 0.5)
188  val = 0.5;
189  if (val < -0.5)
190  val = -0.5;
191  }
192  }
193  if (mode_alt == ALT_FIXED)
194  {
195  float wanted_alt = (float)alt * 0.3048;
196  float current_alt = (ref_l->AbsPosition.y + ref_r->AbsPosition.y) / 2.0;
197  if (wanted_vs < 0)
198  wanted_vs = -wanted_vs; //absolute value
199  float wanted_vs2 = (wanted_alt - current_alt) / 8.0;
200  if (wanted_vs2 < -wanted_vs)
201  wanted_vs2 = -wanted_vs;
202  if (wanted_vs2 > wanted_vs)
203  wanted_vs2 = wanted_vs;
204  val = (wanted_vs2 - current_vs) / 40.0 + pitch_var / 40.0;
205  if (val > 0.5)
206  val = 0.5;
207  if (val < -0.5)
208  val = -0.5;
209  }
210  }
211  return val;
212 }

219 float Autopilot::getThrottle(float thrtl, float dt)
220 {
221  if (!mode_ias) { return thrtl; };
222 
223  float val = thrtl;
224  if (ref_l && ref_r)
225  {
226  //tropospheric model valid up to 11.000m (33.000ft)
227  float altitude = ref_l->AbsPosition.y;
228  //float sea_level_temperature=273.15+15.0; //in Kelvin
229  float sea_level_pressure = 101325; //in Pa
230  //float airtemperature=sea_level_temperature-altitude*0.0065; //in Kelvin
231  float airpressure = sea_level_pressure * pow(1.0 - 0.0065 * altitude / 288.15, 5.24947); //in Pa
232  float airdensity = airpressure * 0.0000120896;//1.225 at sea level
233 
234  float gspd = 1.94384449 * ((ref_l->Velocity + ref_r->Velocity) / 2.0).length();
235 
236  float spd = gspd * sqrt(airdensity / 1.225); //KIAS
237 
238  if (spd > (float)ias)
239  val = val - dt / 15.0;
240  if (spd < (float)ias)
241  val = val + dt / 15.0;
242  if (val > 1.0)
243  val = 1.0;
244  if (val < 0.02)
245  val = 0.02;
246  }
247  return val;
248 }

316 void Autopilot::gpws_update(float spawnheight)
317 {
318 #ifdef USE_OPENAL
319  if (SoundScriptManager::getSingleton().isDisabled())
320  return;
321  if (mode_gpws && gEnv->terrainManager->getHeightFinder() && ref_b)
322  {
323  float groundalt = gEnv->terrainManager->getHeightFinder()->getHeightAt(ref_c->AbsPosition.x, ref_c->AbsPosition.z);
324  if (gEnv->terrainManager->getWater() && groundalt < gEnv->terrainManager->getWater()->getHeight())
325  groundalt = gEnv->terrainManager->getWater()->getHeight();
326  float height = (ref_c->AbsPosition.y - groundalt - spawnheight) * 3.28083f; //in feet!
327  //skip height warning sounds when the plane is slower then ~10 knots
328  if ((ref_c->Velocity.length() * 1.9685f) > 10.0f)
329  {
330  if (height < 10 && last_gpws_height > 10)
331  SoundScriptManager::getSingleton().trigOnce(trucknum, SS_TRIG_GPWS_10);
332  if (height < 20 && last_gpws_height > 20)
333  SoundScriptManager::getSingleton().trigOnce(trucknum, SS_TRIG_GPWS_20);
334  if (height < 30 && last_gpws_height > 30)
335  SoundScriptManager::getSingleton().trigOnce(trucknum, SS_TRIG_GPWS_30);
336  if (height < 40 && last_gpws_height > 40)
337  SoundScriptManager::getSingleton().trigOnce(trucknum, SS_TRIG_GPWS_40);
338  if (height < 50 && last_gpws_height > 50)
339  SoundScriptManager::getSingleton().trigOnce(trucknum, SS_TRIG_GPWS_50);
340  if (height < 100 && last_gpws_height > 100)
341  SoundScriptManager::getSingleton().trigOnce(trucknum, SS_TRIG_GPWS_100);
342  }
343  last_gpws_height = height;
344 
345  // pullup warning calculation
346  // height to meters
347  height *= 0.3048;
348  // get the y-velocity in meters/s
349  float yVel = ref_c->Velocity.y * 1.9685f;
350  // will trigger the pullup sound when vvi is high (avoid pullup warning when landing normal) and groundcontact will be in less then 10 seconds
351  if (yVel * 10.0f < -height && yVel < -10.0f)
352  SoundScriptManager::getSingleton().trigOnce(trucknum, SS_TRIG_GPWS_PULLUP);
353  }
354 #endif //OPENAL

357 void Autopilot::getRadioFix(TerrainObjectManager::localizer_t* localizers, int free_localizer, float* vdev, float* hdev)
358 {
359  if (!ref_l || !ref_r)
360  return;
361  Vector3 position = (ref_l->AbsPosition + ref_r->AbsPosition) / 2.0;
362  float closest_hdist = -1;
363  float closest_hangle = -90;
364  float closest_vdist = -1;
365  float closest_vangle = -90;
366  lastradiorwh = 0;
367  for (int i = 0; i < free_localizer; i++)
368  {
369  Plane hplane = Plane(Vector3::UNIT_Y, 0);
370  Vector3 plocd = hplane.projectVector(localizers[i].rotation * Vector3::UNIT_Z);
371  float loc = atan2(plocd.z, plocd.x);
372  Vector3 posd = hplane.projectVector(position - localizers[i].position);
373  float dir = atan2(posd.z, posd.x);
374  float diff = (dir - loc) * 57.295779513082;
375  if (diff > 180.0)
376  diff -= 360.0;
377  if (diff < -180.0)
378  diff += 360.0;
379  if (diff < 80 && diff > -80)
380  {
381  //we are in the visibility cone, we search the closest
382  float dist = (localizers[i].position - position).length();
383  //horizontal
384  if (localizers[i].type == Autopilot::LOCALIZER_HORIZONTAL)
385  {
386  if (closest_hdist < 0 || closest_hdist > dist)
387  {
388  closest_hdist = dist;
389  closest_hangle = diff;
390  lastradiorwh = loc * 57.295779513082 - 90.0; //dont ask me why
391  if (lastradiorwh < 0.0)
392  lastradiorwh += 360.0;
393  lastradiorwd = dist;
394  }
395  }
396  //vertical
397  if (localizers[i].type == Autopilot::LOCALIZER_VERTICAL)
398  {
399  if (closest_vdist < 0 || closest_vdist > dist)
400  {
401  Vector3 normv = (localizers[i].rotation * Vector3::UNIT_Z).crossProduct(Vector3::UNIT_Y);
402  Plane vplane = Plane(normv, 0);
403  float glideslope = 4; //4 degree is the norm
404  Vector3 posd2 = vplane.projectVector(position - localizers[i].position);
405  float d = posd2.length();
406  if (d < 0.01)
407  d = 0.01;
408  float dir2 = 90.0;
409  d = posd2.y / d;
410  if (d >= 1.0)
411  dir2 = 90.0;
412  else if (d <= -1.0)
413  dir2 = -90.0;
414  else
415  dir2 = asin(d) * 57.295779513082;
416  float diff2 = dir2 - glideslope;
417  closest_vdist = dist;
418  closest_vangle = diff2;
419  }
420  }
421  }
422  }
423  lastradioh = closest_hangle;
424  lastradiov = closest_vangle;
425  *hdev = closest_hangle;
426  *vdev = closest_vangle;
427 #ifdef USE_OPENAL
428  if (mode_heading == HEADING_NAV && mode_gpws && closest_hdist > 10.0 && closest_hdist < 350.0 && last_closest_hdist > 10.0 && last_closest_hdist > 350.0)
429  SoundScriptManager::getSingleton().trigOnce(trucknum, SS_TRIG_GPWS_MINIMUMS);
430 #endif //OPENAL
431  last_closest_hdist = closest_hdist;
432  if (mode_heading == HEADING_NAV && (closest_hdist < 20.0 || closest_vdist < 20.0))
433  wantsdisconnect = true;
434 }
