45 void MumbleIntegration::initMumble()
46 {
47 #ifdef _WIN32
48  HANDLE hMapObject = OpenFileMappingW(FILE_MAP_ALL_ACCESS, FALSE, L"MumbleLink");
49  if (hMapObject == NULL)
50  return;
51 
52  lm = (LinkedMem *) MapViewOfFile(hMapObject, FILE_MAP_ALL_ACCESS, 0, 0, sizeof(LinkedMem));
53  if (lm == NULL)
54  {
55  CloseHandle(hMapObject);
56  hMapObject = NULL;
57  return;
58  }
59 #else
60  char memname[256];
61  snprintf(memname, 256, "/MumbleLink.%d", getuid());
62 
63  int shmfd = shm_open(memname, O_RDWR, S_IRUSR | S_IWUSR);
64 
65  if (shmfd < 0)
66  {
67  return;
68  }
69 
70  lm = (LinkedMem *)(mmap(NULL, sizeof(struct LinkedMem), PROT_READ | PROT_WRITE, MAP_SHARED, shmfd,0));
71 
72  if (lm == (void *)(-1))
73  {
74  lm = NULL;
75  return;
76  }
77 #endif // _WIN32
78 }
