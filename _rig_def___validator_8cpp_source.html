163 bool Validator::CheckSection(RigDef::File::Keyword keyword, bool unique, bool required)
164 {
165  Ogre::String *containing_module_name = nullptr;
166 
167  std::list<std::shared_ptr<RigDef::File::Module>>::iterator module_itor = m_selected_modules.begin();
168  for (; module_itor != m_selected_modules.end(); module_itor++)
169  {
170  if (HasModuleKeyword(*module_itor, keyword))
171  {
172  if (containing_module_name == nullptr)
173  {
174  containing_module_name = & module_itor->get()->name;
175  }
176  else if (unique)
177  {
178  std::stringstream text;
179  text << "Duplicate section '" << RigDef::File::KeywordToString(keyword)
180  << "'; found in modules: '" << *containing_module_name
181  << "' & '" << module_itor->get()->name << "'";
182  AddMessage(Message::TYPE_FATAL_ERROR, text.str());
183  return false;
184  }
185  }
186  }
187 
188  if (containing_module_name == nullptr && required)
189  {
190  std::stringstream text;
191  text << "Missing required section '" << RigDef::File::KeywordToString(keyword) <<"'";
192  AddMessage(Message::TYPE_FATAL_ERROR, text.str());
193  return false;
194  }
195  return true;

198 bool Validator::HasModuleKeyword(std::shared_ptr<RigDef::File::Module> module, RigDef::File::Keyword keyword)
199 {
200  using namespace RigDef;
201 
202  switch (keyword)
203  {
204  /* Please maintain alphabetical order */
205 
206  case (File::KEYWORD_BEAMS):
207  return ! module->beams.empty();
208 
209  case (File::KEYWORD_ENGINE):
210  return (module->engine != nullptr);
211 
212  case (File::KEYWORD_ENGOPTION):
213  return (module->engoption != nullptr);
214 
215  case (File::KEYWORD_ENGTURBO) :
216  return (module->engturbo != nullptr);
217 
218  case (File::KEYWORD_EXTCAMERA):
219  return (module->ext_camera != nullptr);
220 
221  case (File::KEYWORD_FUSEDRAG):
222  return ! module->fusedrag.empty();
223 
224  case (File::KEYWORD_GLOBALS):
225  return (module->globals != nullptr);
226 
227  case (File::KEYWORD_GUISETTINGS):
228  return (module->gui_settings != nullptr);
229 
230  case (File::KEYWORD_MANAGEDMATERIALS):
231  return ! module->managed_materials.empty();
232 
233  case (File::KEYWORD_NODES):
234  return ! module->nodes.empty();
235 
236  case (File::KEYWORD_SPEEDLIMITER):
237  return (module->speed_limiter.is_enabled);
238 
239  case (File::KEYWORD_TORQUECURVE):
240  return (module->torque_curve != nullptr);
241 
242  /* TEMPLATE
243  case (File::SECTION_):
244  return (module->globals != nullptr);
245  */
246 
247  default:
248  return false;
249  };

420 bool Validator::CheckTrigger(RigDef::Trigger & def)
421 {
422  bool ok = true;
423 
424  bool hook_toggle =
425  BITMASK_IS_1(def.options, RigDef::Trigger::OPTION_H_LOCK_HOOKGROUPS_KEY)
426  || BITMASK_IS_1(def.options, RigDef::Trigger::OPTION_h_UNLOCK_HOOKGROUPS_KEY);
427 
428  bool trigger_blocker = BITMASK_IS_1(def.options, RigDef::Trigger::OPTION_B_BLOCK_TRIGGERS);
429  bool inv_trigger_blocker = BITMASK_IS_1(def.options, RigDef::Trigger::OPTION_A_INV_BLOCK_TRIGGERS);
430 
431  if (BITMASK_IS_0(def.options, RigDef::Trigger::OPTION_E_ENGINE_TRIGGER))
432  {
433  if (! trigger_blocker && ! inv_trigger_blocker && ! hook_toggle )
434  {
435  /* Make the full check */
436  if (def.shortbound_trigger_action < 1 || def.shortbound_trigger_action > MAX_COMMANDS)
437  {
438  std::stringstream msg;
439  msg << "Wrong parameter 'shortbound_trigger_action': " << def.shortbound_trigger_action;
440  msg << "; Alloved range is <0 - " << MAX_COMMANDS << ">. ";
441  msg << "Trigger deactivated.";
442  AddMessage(Message::TYPE_ERROR, msg.str());
443  ok = false;
444  }
445  }
446  else if (! hook_toggle)
447  {
448  /* This is a Trigger-Blocker, make special check */
449  if (def.shortbound_trigger_action < 0)
450  {
451  std::stringstream msg;
452  msg << "Wrong parameter 'shortbound_trigger_action': " << def.shortbound_trigger_action;
453  msg << "; Alloved range is <0 - " << MAX_COMMANDS << ">. ";
454  msg << "Trigger deactivated.";
455  AddMessage(Message::TYPE_ERROR, msg.str());
456  ok = false;
457  }
458  if (def.longbound_trigger_action < 0)
459  {
460  std::stringstream msg;
461  msg << "Wrong parameter 'longbound_trigger_action': " << def.longbound_trigger_action;
462  msg << "; Alloved range is <0 - " << MAX_COMMANDS << ">. ";
463  msg << "Trigger deactivated.";
464  AddMessage(Message::TYPE_ERROR, msg.str());
465  ok = false;
466  }
467  }
468  }
469  else
470  {
471  /* Engine trigger */
472  if (trigger_blocker || inv_trigger_blocker || hook_toggle || BITMASK_IS_1(def.options, RigDef::Trigger::OPTION_s_SWITCH_CMD_NUM))
473  {
474  AddMessage(Message::TYPE_ERROR, "Wrong command-eventnumber. Engine trigger deactivated.");
475  ok = false;
476  }
477  }
478 
479  return ok;
480 }
