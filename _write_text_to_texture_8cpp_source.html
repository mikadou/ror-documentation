50 void WriteToTexture(const String& str, TexturePtr destTexture, Image::Box destRectangle, Font* Reffont, const ColourValue& color, int fontSize, int fontDPI, char justify, bool wordwrap)
51 {
52  using namespace Ogre;
53 
54  if (destTexture->getHeight() < destRectangle.bottom)
55  destRectangle.bottom = destTexture->getHeight();
56  if (destTexture->getWidth() < destRectangle.right)
57  destRectangle.right = destTexture->getWidth();
58 
59  // Find our scaled up (or down) font.
60  std::string fontname = "WTTFont_" + TOSTRING(fontSize + fontDPI) + "_" + Reffont->getSource();
61  FontPtr font = FontManager::getSingleton().getByName(fontname);
62 
63  // Font not found, let's created it :)
64  if (font.isNull())
65  {
66  font = FontManager::getSingleton().create(fontname, "General");
67  font->setType(FT_TRUETYPE);
68  font->setSource(Reffont->getSource());
69 
70  font->setTrueTypeSize(fontSize);
71  font->setTrueTypeResolution(fontDPI);
72  font->load();
73  LOG("[WriteToTexture] Created font: " + fontname);
74  }
75 
76  TexturePtr fontTexture = (TexturePtr)TextureManager::getSingleton().getByName(font->getMaterial()->getTechnique(0)->getPass(0)->getTextureUnitState(0)->getTextureName());
77 
78  HardwarePixelBufferSharedPtr fontBuffer = fontTexture->getBuffer();
79  HardwarePixelBufferSharedPtr destBuffer = destTexture->getBuffer();
80 
81  PixelBox destPb = destBuffer->lock(destRectangle, HardwareBuffer::HBL_NORMAL);
82 
83  // The font texture buffer was created write only...so we cannot read it back :o). One solution is to copy the buffer instead of locking it. (Maybe there is a way to create a font texture which is not write_only ?)
84 
85  // create a buffer
86  size_t nBuffSize = fontBuffer->getSizeInBytes();
87  uint8* buffer = (uint8*)calloc(nBuffSize, sizeof(uint8));
88 
89  // create pixel box using the copy of the buffer
90  PixelBox fontPb(fontBuffer->getWidth(), fontBuffer->getHeight(), fontBuffer->getDepth(), fontBuffer->getFormat(), buffer);
91  fontBuffer->blitToMemory(fontPb);
92 
93  uint8* fontData = static_cast<uint8*>(fontPb.data);
94  uint8* destData = static_cast<uint8*>(destPb.data);
95 
96  const size_t fontPixelSize = PixelUtil::getNumElemBytes(fontPb.format);
97  const size_t destPixelSize = PixelUtil::getNumElemBytes(destPb.format);
98 
99  const size_t fontRowPitchBytes = fontPb.rowPitch * fontPixelSize;
100  const size_t destRowPitchBytes = destPb.rowPitch * destPixelSize;
101 
102  Box* GlyphTexCoords;
103  GlyphTexCoords = new Box[str.size()];
104 
105  Font::UVRect glypheTexRect;
106  size_t charheight = 0;
107  size_t charwidth = 0;
108 
109  for (unsigned int i = 0; i < str.size(); i++)
110  {
111  if ((str[i] != '\t') && (str[i] != '\n') && (str[i] != ' '))
112  {
113  glypheTexRect = font->getGlyphTexCoords(str[i]);
114  GlyphTexCoords[i].left = glypheTexRect.left * fontTexture->getSrcWidth();
115  GlyphTexCoords[i].top = glypheTexRect.top * fontTexture->getSrcHeight();
116  GlyphTexCoords[i].right = glypheTexRect.right * fontTexture->getSrcWidth();
117  GlyphTexCoords[i].bottom = glypheTexRect.bottom * fontTexture->getSrcHeight();
118 
119  if (GlyphTexCoords[i].getHeight() > charheight)
120  charheight = GlyphTexCoords[i].getHeight();
121  if (GlyphTexCoords[i].getWidth() > charwidth)
122  charwidth = GlyphTexCoords[i].getWidth();
123  }
124  }
125 
126  size_t cursorX = 0;
127  size_t cursorY = 0;
128  size_t lineend = destRectangle.getWidth();
129  bool carriagreturn = true;
130  for (unsigned int strindex = 0; strindex < str.size(); strindex++)
131  {
132  switch (str[strindex])
133  {
134  case ' ': cursorX += charwidth;
135  break;
136  case '\t': cursorX += charwidth * 3;
137  break;
138  case '\n': cursorY += charheight;
139  carriagreturn = true;
140  break;
141  default:
142  {
143  //wrapping
144  if ((cursorX + GlyphTexCoords[strindex].getWidth() > lineend) && !carriagreturn)
145  {
146  cursorY += charheight;
147  carriagreturn = true;
148  }
149 
150  //justify
151  if (carriagreturn)
152  {
153  size_t l = strindex;
154  size_t textwidth = 0;
155  size_t wordwidth = 0;
156 
157  while (l < str.size() && str[l] != '\n')
158  {
159  wordwidth = 0;
160 
161  switch (str[l])
162  {
163  case ' ': wordwidth = charwidth;
164  ++l;
165  break;
166  case '\t': wordwidth = charwidth * 3;
167  ++l;
168  break;
169  case '\n': l = str.size();
170  }
171 
172  if (wordwrap)
173  while (l < str.size() && str[l] != ' ' && str[l] != '\t' && str[l] != '\n')
174  {
175  wordwidth += GlyphTexCoords[l].getWidth();
176  ++l;
177  }
178  else
179  {
180  wordwidth += GlyphTexCoords[l].getWidth();
181  l++;
182  }
183 
184  if ((textwidth + wordwidth) <= destRectangle.getWidth())
185  textwidth += (wordwidth);
186  else
187  break;
188  }
189 
190  if ((textwidth == 0) && (wordwidth > destRectangle.getWidth()))
191  textwidth = destRectangle.getWidth();
192 
193  switch (justify)
194  {
195  case 'c': cursorX = (destRectangle.getWidth() - textwidth) / 2;
196  lineend = destRectangle.getWidth() - cursorX;
197  break;
198 
199  case 'r': cursorX = (destRectangle.getWidth() - textwidth);
200  lineend = destRectangle.getWidth();
201  break;
202 
203  default: cursorX = 0;
204  lineend = textwidth;
205  break;
206  }
207 
208  carriagreturn = false;
209  }
210 
211  //abort - net enough space to draw
212  if ((cursorY + charheight) > destRectangle.getHeight())
213  goto stop;
214 
215  //draw pixel by pixel
216  for (size_t i = 0; i < GlyphTexCoords[strindex].getHeight(); i++)
217  for (size_t j = 0; j < GlyphTexCoords[strindex].getWidth(); j++)
218  {
219  float alpha = color.a * (fontData[(i + GlyphTexCoords[strindex].top) * fontRowPitchBytes + (j + GlyphTexCoords[strindex].left) * fontPixelSize + 1] / 255.0);
220  float invalpha = 1.0 - alpha;
221  size_t offset = (i + cursorY) * destRowPitchBytes + (j + cursorX) * destPixelSize;
222  ColourValue pix;
223  PixelUtil::unpackColour(&pix, destPb.format, &destData[offset]);
224  pix = (pix * invalpha) + (color * alpha);
225  PixelUtil::packColour(pix, destPb.format, &destData[offset]);
226  }
227 
228  cursorX += GlyphTexCoords[strindex].getWidth();
229  }//default
230  }//switch
231  }//for
232 
233 stop:
234  delete[] GlyphTexCoords;
235 
236  destBuffer->unlock();
237 
238  // Free the memory allocated for the buffer
239  free(buffer);
240  buffer = 0;
241 }
