297 void MpClientList::update()
298 {
299 #ifdef USE_SOCKETW
300  int slotid = 0;
301 
302  MyGUI::IntSize gui_area = MyGUI::RenderManager::getInstance().getViewSize();
303  int x = gui_area.width - sidebarWidth, y = 30;
304  mpPanel->setPosition(x, y);
305 
306  // add local player to first slot always
307  RoRnet::UserInfo lu = RoR::Networking::GetLocalUserData();
308  updateSlot(&player_rows[slotid], lu, true);
309  slotid++;
310 
311  // add remote players
312  std::vector<RoRnet::UserInfo> users = RoR::Networking::GetUserInfos();
313  for (RoRnet::UserInfo user : users)
314  {
315  player_row_t* row = &player_rows[slotid];
316  slotid++;
317  try
318  {
319  updateSlot(row, user, false);
320  }
321  catch (...)
322  {
323  }
324  }
325  for (int i = slotid; i < RORNET_MAX_PEERS; i++)
326  {
327  player_row_t* row = &player_rows[i];
328  // not used, hide everything
329  row->flagimg->setVisible(false);
330  row->playername->setVisible(false);
331  row->statimg->setVisible(false);
332  row->usergoimg->setVisible(false);
333  row->userTruckOKImg->setVisible(false);
334  row->userTruckOKRemoteImg->setVisible(false);
335  }
336 
337  netmsgwin->setVisible(RoR::Networking::GetNetQuality() != 0);
338 
339  int height = lineheight * (slotid + 1);
340  mpPanel->setSize(sidebarWidth, height);
341 #endif
342 }
