72 void RoR::GfxActor::SetMaterialFlareOn(int flare_index, bool state_on)
73 {
74  for (FlareMaterial& entry: m_flare_materials)
75  {
76  if (entry.flare_index != flare_index)
77  {
78  continue;
79  }
80 
81  const int num_techniques = static_cast<int>(entry.mat_instance->getNumTechniques());
82  for (int i = 0; i < num_techniques; i++)
83  {
84  Ogre::Technique* tech = entry.mat_instance->getTechnique(i);
85  if (!tech)
86  continue;
87 
88  if (tech->getSchemeName() == "glow")
89  {
90  // glowing technique
91  // set the ambient value as glow amount
92  Ogre::Pass* p = tech->getPass(0);
93  if (!p)
94  continue;
95 
96  if (state_on)
97  {
98  p->setSelfIllumination(entry.emissive_color);
99  p->setAmbient(Ogre::ColourValue::White);
100  p->setDiffuse(Ogre::ColourValue::White);
101  }
102  else
103  {
104  p->setSelfIllumination(Ogre::ColourValue::ZERO);
105  p->setAmbient(Ogre::ColourValue::Black);
106  p->setDiffuse(Ogre::ColourValue::Black);
107  }
108  }
109  else
110  {
111  // normal technique
112  Ogre::Pass* p = tech->getPass(0);
113  if (!p)
114  continue;
115 
116  Ogre::TextureUnitState* tus = p->getTextureUnitState(0);
117  if (!tus)
118  continue;
119 
120  if (tus->getNumFrames() < 2)
121  continue;
122 
123  int frame = state_on ? 1 : 0;
124 
125  tus->setCurrentFrame(frame);
126 
127  if (state_on)
128  p->setSelfIllumination(entry.emissive_color);
129  else
130  p->setSelfIllumination(Ogre::ColourValue::ZERO);
131  }
132  } // for each technique
133  }
134 }

152 void RoR::GfxActor::SetCabLightsActive(bool state_on)
153 {
154  if (m_cab_mat_template_emissive.isNull()) // Both this and '_plain' are only set when emissive pass is present.
155  return;
156 
157  // NOTE: Updating material in-place like this is probably inefficient,
158  // but in order to maintain all the existing material features working together,
159  // we need to avoid any material swapping on runtime. ~ only_a_ptr, 05/2017
160  Ogre::MaterialPtr template_mat = (state_on) ? m_cab_mat_template_emissive : m_cab_mat_template_plain;
161  Ogre::Technique* dest_tech = m_cab_mat_visual->getTechnique(0);
162  Ogre::Technique* templ_tech = template_mat->getTechnique(0);
163  dest_tech->removeAllPasses();
164  for (unsigned short i = 0; i < templ_tech->getNumPasses(); ++i)
165  {
166  Ogre::Pass* templ_pass = templ_tech->getPass(i);
167  Ogre::Pass* dest_pass = dest_tech->createPass();
168  *dest_pass = *templ_pass; // Copy the pass! Reference: http://www.ogre3d.org/forums/viewtopic.php?f=2&t=83453
169  }
170  m_cab_mat_visual->compile();
171 }

195  vidcam.vcam_render_window->setActive(enable);
196  }
197  }

216 void RoR::GfxActor::UpdateVideoCameras(float dt_sec)
217 {
218  if (m_vidcam_state != VideoCamState::VCSTATE_ENABLED_ONLINE)
219  return;
220 
221  for (GfxActor::VideoCamera vidcam: m_videocameras)
222  {
223 #ifdef USE_CAELUM
224  // caelum needs to know that we changed the cameras
225  if (gEnv->sky && RoR::App::GetActiveAppState() == RoR::App::APP_STATE_SIMULATION)
226  {
227  gEnv->sky->notifyCameraChanged(vidcam.vcam_ogre_camera);
228  }
229 #endif // USE_CAELUM
230 
231  if ((vidcam.vcam_type == VideoCamType::VCTYPE_MIRROR_PROP_LEFT)
232  || (vidcam.vcam_type == VideoCamType::VCTYPE_MIRROR_PROP_RIGHT))
233  {
234  // Mirror prop - special processing.
235  float mirror_angle = 0.f;
236  Ogre::Vector3 offset(Ogre::Vector3::ZERO);
237  if (vidcam.vcam_type == VideoCamType::VCTYPE_MIRROR_PROP_LEFT)
238  {
239  mirror_angle = m_actor->leftMirrorAngle;
240  offset = Ogre::Vector3(0.07f, -0.22f, 0);
241  }
242  else
243  {
244  mirror_angle = m_actor->rightMirrorAngle;
245  offset = Ogre::Vector3(0.07f, +0.22f, 0);
246  }
247 
248  Ogre::Vector3 normal = vidcam.vcam_prop_scenenode->getOrientation()
249  * Ogre::Vector3(cos(mirror_angle), sin(mirror_angle), 0.0f);
250  Ogre::Vector3 center = vidcam.vcam_prop_scenenode->getPosition()
251  + vidcam.vcam_prop_scenenode->getOrientation() * offset;
252  Ogre::Radian roll = Ogre::Degree(360)
253  - Ogre::Radian(asin(m_actor->getDirection().dotProduct(Ogre::Vector3::UNIT_Y)));
254 
255  Ogre::Plane plane = Ogre::Plane(normal, center);
256  Ogre::Vector3 project = plane.projectVector(gEnv->mainCamera->getPosition() - center);
257 
258  vidcam.vcam_ogre_camera->setPosition(center);
259  vidcam.vcam_ogre_camera->lookAt(gEnv->mainCamera->getPosition() - 2.0f * project);
260  vidcam.vcam_ogre_camera->roll(roll);
261 
262  continue; // Done processing mirror prop.
263  }
264 
265  // update the texture now, otherwise shuttering
266  if (vidcam.vcam_render_target != nullptr)
267  vidcam.vcam_render_target->update();
268 
269  if (vidcam.vcam_render_window != nullptr)
270  vidcam.vcam_render_window->update();
271 
272  // get the normal of the camera plane now
273  const Ogre::Vector3 abs_pos_center = vidcam.vcam_node_center->AbsPosition;
274  const Ogre::Vector3 abs_pos_z = vidcam.vcam_node_dir_z->AbsPosition;
275  const Ogre::Vector3 abs_pos_y = vidcam.vcam_node_dir_y->AbsPosition;
276  Ogre::Vector3 normal = (-(abs_pos_center - abs_pos_z)).crossProduct(-(abs_pos_center - abs_pos_y));
277  normal.normalise();
278 
279  // add user set offset
280  Ogre::Vector3 pos = vidcam.vcam_node_alt_pos->AbsPosition +
281  (vidcam.vcam_pos_offset.x * normal) +
282  (vidcam.vcam_pos_offset.y * (abs_pos_center - abs_pos_y)) +
283  (vidcam.vcam_pos_offset.z * (abs_pos_center - abs_pos_z));
284 
285  //avoid the camera roll
286  // camup orientates to frustrum of world by default -> rotating the cam related to trucks yaw, lets bind cam rotation videocamera base (nref,ny,nz) as frustum
287  // could this be done faster&better with a plane setFrustumExtents ?
288  Ogre::Vector3 frustumUP = abs_pos_center - abs_pos_y;
289  frustumUP.normalise();
290  vidcam.vcam_ogre_camera->setFixedYawAxis(true, frustumUP);
291 
292  if (vidcam.vcam_type == GfxActor::VideoCamType::VCTYPE_MIRROR)
293  {
294  //rotate the normal of the mirror by user rotation setting so it reflects correct
295  normal = vidcam.vcam_rotation * normal;
296  // merge camera direction and reflect it on our plane
297  vidcam.vcam_ogre_camera->setDirection((pos - gEnv->mainCamera->getPosition()).reflect(normal));
298  }
299  else if (vidcam.vcam_type == GfxActor::VideoCamType::VCTYPE_VIDEOCAM)
300  {
301  // rotate the camera according to the nodes orientation and user rotation
302  Ogre::Vector3 refx = abs_pos_z - abs_pos_center;
303  refx.normalise();
304  Ogre::Vector3 refy = abs_pos_center - abs_pos_y;
305  refy.normalise();
306  Ogre::Quaternion rot = Ogre::Quaternion(-refx, -refy, -normal);
307  vidcam.vcam_ogre_camera->setOrientation(rot * vidcam.vcam_rotation); // rotate the camera orientation towards the calculated cam direction plus user rotation
308  }
309  else if (vidcam.vcam_type == GfxActor::VideoCamType::VCTYPE_TRACKING_VIDEOCAM)
310  {
311  normal = vidcam.vcam_node_lookat->AbsPosition - pos;
312  normal.normalise();
313  Ogre::Vector3 refx = abs_pos_z - abs_pos_center;
314  refx.normalise();
315  // why does this flip ~2-3 around zero orientation and only with trackercam. back to slower crossproduct calc, a bit slower but better .. sigh
316  // Ogre::Vector3 refy = abs_pos_center - abs_pos_y;
317  Ogre::Vector3 refy = refx.crossProduct(normal);
318  refy.normalise();
319  Ogre::Quaternion rot = Ogre::Quaternion(-refx, -refy, -normal);
320  vidcam.vcam_ogre_camera->setOrientation(rot * vidcam.vcam_rotation); // rotate the camera orientation towards the calculated cam direction plus user rotation
321  }
322 
323  if (vidcam.vcam_debug_node != nullptr)
324  {
325  vidcam.vcam_debug_node->setPosition(pos);
326  vidcam.vcam_debug_node->setOrientation(vidcam.vcam_ogre_camera->getOrientation());
327  }
328 
329  // set the new position
330  vidcam.vcam_ogre_camera->setPosition(pos);
331  }
332 }
