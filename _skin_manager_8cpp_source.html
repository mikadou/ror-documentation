48 void RoR::SkinManager::parseScript(Ogre::DataStreamPtr& stream, const Ogre::String& groupName)
49 {
50  SkinDef* skin_def = nullptr;
51  bool skin_is_new = true;
52  try
53  {
54  while(!stream->eof())
55  {
56  std::string line = RoR::Utils::SanitizeUtf8String(stream->getLine());
57 
58  // Ignore blanks & comments
59  if (!line.length() || line.substr(0, 2) == "//")
60  {
61  continue;
62  }
63 
64  if (!skin_def)
65  {
66  // No current skin -- So first valid data should be skin name
67  Ogre::StringUtil::trim(line);
68  auto search = m_skins.find(line);
69  if (search != m_skins.end())
70  {
71  skin_def = search->second;
72  skin_is_new = false;
73  }
74  else
75  {
76  skin_def = new SkinDef;
77  skin_def->name = line;
78  skin_is_new = true;
79  }
80  stream->skipLine("{");
81  }
82  else
83  {
84  // Already in skin
85  if (line == "}")
86  {
87  if (skin_is_new)
88  m_skins.insert(std::make_pair(skin_def->name, skin_def));
89  skin_def = nullptr;// Finished
90  }
91  else
92  {
93  this->ParseSkinAttribute(line, skin_def);
94  }
95  }
96  }
97  }
98  catch (Ogre::ItemIdentityException)
99  {
100  // this catches duplicates -> to be ignored
101  // this happens since we load the full skin data off the cache, so we don't need
102  // to re-add it to the SkinManager
103  return;
104  }
105 }

107 void RoR::SkinManager::ParseSkinAttribute(const std::string& line, SkinDef* skin_def)
108 {
109  Ogre::StringVector params = Ogre::StringUtil::split(line, "\t=,;\n");
110  for (unsigned int i=0; i < params.size(); i++)
111  {
112  Ogre::StringUtil::trim(params[i]);
113  }
114  Ogre::String& attrib = params[0];
115  Ogre::StringUtil::toLowerCase(attrib);
116 
117  if (attrib == "replacetexture" && params.size() == 3) { skin_def->replace_textures.insert(std::make_pair(params[1], params[2])); return; }
118  if (attrib == "replacematerial" && params.size() == 3) { skin_def->replace_materials.insert(std::make_pair(params[1], params[2])); return; }
119  if (attrib == "preview" && params.size() >= 2) { skin_def->thumbnail = params[1]; return; }
120  if (attrib == "description" && params.size() >= 2) { skin_def->description = params[1]; return; }
121  if (attrib == "authorname" && params.size() >= 2) { skin_def->author_name = params[1]; return; }
122  if (attrib == "authorid" && params.size() == 2) { skin_def->author_id = PARSEINT(params[1]); return; }
123  if (attrib == "guid" && params.size() >= 2) { skin_def->guid = params[1]; Ogre::StringUtil::trim(skin_def->guid); Ogre::StringUtil::toLowerCase(skin_def->guid); return; }
124  if (attrib == "name" && params.size() >= 2) { skin_def->name = params[1]; Ogre::StringUtil::trim(skin_def->name); return; }
125 }

137 void RoR::SkinManager::ReplaceMaterialTextures(SkinDef* skin_def, std::string materialName) // Static
138 {
139  const auto not_found = skin_def->replace_textures.end();
140  Ogre::MaterialPtr mat = RoR::OgreSubsystem::GetMaterialByName(materialName);
141  if (!mat.isNull())
142  {
143  for (int t = 0; t < mat->getNumTechniques(); t++)
144  {
145  Ogre::Technique* tech = mat->getTechnique(0);
146  if (!tech)
147  continue;
148  for (int p = 0; p < tech->getNumPasses(); p++)
149  {
150  Ogre::Pass* pass = tech->getPass(p);
151  if (!pass)
152  continue;
153  for (int tu = 0; tu < pass->getNumTextureUnitStates(); tu++)
154  {
155  Ogre::TextureUnitState* tus = pass->getTextureUnitState(tu);
156  if (!tus)
157  continue;
158 
159  //if (tus->getTextureType() != TEX_TYPE_2D) continue; // only replace 2d images
160  // walk the frames, usually there is only one
161  for (unsigned int fr = 0; fr < tus->getNumFrames(); fr++)
162  {
163  Ogre::String textureName = tus->getFrameTextureName(fr);
164  std::map<Ogre::String, Ogre::String>::iterator it = skin_def->replace_textures.find(textureName);
165  if (it != not_found)
166  {
167  textureName = it->second; //getReplacementForTexture(textureName);
168  tus->setFrameTextureName(textureName, fr);
169  }
170  }
171  }
172  }
173  }
174  }
