66 FlexBody* FlexFactory::CreateFlexBody(
67  RigDef::Flexbody* def,
68  const int ref_node,
69  const int x_node,
70  const int y_node,
71  Ogre::Quaternion const & rot,
72  std::vector<unsigned int> & node_indices)
73 {
74  const std::string resource_group_name
75  = Ogre::ResourceGroupManager::getSingleton().findGroupContainingResource(def->mesh_name);
76  if (resource_group_name.empty())
77  {
78  m_rig_spawner->AddMessage(RigSpawner::Message::TYPE_ERROR, "Failed to create flexbody, mesh not found: " + def->mesh_name);
79  return nullptr;
80  }
81  Ogre::MeshPtr common_mesh = Ogre::MeshManager::getSingleton().load(def->mesh_name, resource_group_name);
82  const std::string mesh_unique_name = m_rig_spawner->ComposeName("FlexbodyMesh", m_rig_spawner->GetRig()->free_flexbody);
83  Ogre::MeshPtr mesh = common_mesh->clone(mesh_unique_name);
84  if (BSETTING("Flexbody_EnableLODs", false))
85  {
86  this->ResolveFlexbodyLOD(def->mesh_name, mesh);
87  }
88  const std::string flexbody_name = m_rig_spawner->ComposeName("Flexbody", m_rig_spawner->GetRig()->free_flexbody);
89  Ogre::Entity* entity = gEnv->sceneManager->createEntity(flexbody_name, mesh_unique_name);
90  m_rig_spawner->SetupNewEntity(entity, Ogre::ColourValue(0.5, 0.5, 1));
91 
92  FLEX_DEBUG_LOG(__FUNCTION__);
93  FlexBodyCacheData* from_cache = nullptr;
94  if (m_is_flexbody_cache_loaded)
95  {
96  FLEX_DEBUG_LOG(__FUNCTION__ " >> Get entry from cache ");
97  from_cache = m_flexbody_cache.GetLoadedItem(m_flexbody_cache_next_index);
98  m_flexbody_cache_next_index++;
99  }
100 
101  FlexBody* new_flexbody = new FlexBody(
102  def,
103  from_cache,
104  m_rig_spawner->GetRig()->nodes,
105  m_rig_spawner->GetRig()->free_node,
106  entity,
107  ref_node,
108  x_node,
109  y_node,
110  rot,
111  node_indices);
112 
113  if (m_is_flexbody_cache_enabled)
114  {
115  m_flexbody_cache.AddItemToSave(new_flexbody);
116  }
117  return new_flexbody;
118 }

318 FlexBodyFileIO::ResultCode FlexBodyFileIO::SaveFile()
319 {
320  FLEX_DEBUG_LOG(__FUNCTION__);
321  if (m_items_to_save.size() == 0)
322  {
323  FLEX_DEBUG_LOG(__FUNCTION__ " >> No flexbodies to save >> EXIT");
324  return RESULT_CODE_OK;
325  }
326  try
327  {
328  this->OpenFile("wb");
329 
330  this->WriteSignature();
331  this->WriteMetadata();
332 
333  auto itor = m_items_to_save.begin();
334  auto end = m_items_to_save.end();
335  for (; itor != end; ++itor)
336  {
337  FlexBody* flexbody = *itor;
338  this->WriteFlexbodyHeader(flexbody);
339 
340  this->WriteFlexbodyLocatorList (flexbody);
341  this->WriteFlexbodyPositionsBuffer(flexbody);
342  this->WriteFlexbodyNormalsBuffer (flexbody);
343  this->WriteFlexbodyColorsBuffer (flexbody);
344  }
345  this->CloseFile();
346  FLEX_DEBUG_LOG(__FUNCTION__ " >> OK ");
347  return RESULT_CODE_OK;
348  }
349  catch (ResultCode result)
350  {
351  this->CloseFile();
352  FLEX_DEBUG_LOG(__FUNCTION__ " >> EXCEPTION!! ");
353  return result;
354  }
355 }

357 FlexBodyFileIO::ResultCode FlexBodyFileIO::LoadFile()
358 {
359  FLEX_DEBUG_LOG(__FUNCTION__);
360  try
361  {
362  this->OpenFile("rb");
363  this->ReadAndCheckSignature();
364 
365  FlexBodyFileMetadata meta;
366  this->ReadMetadata(&meta);
367  m_fileformat_version = meta.file_format_version;
368  if (m_fileformat_version != FILE_FORMAT_VERSION)
369  {
370  throw RESULT_CODE_ERR_VERSION_MISMATCH;
371  }
372  m_loaded_items.resize(meta.num_flexbodies);
373 
374  for (unsigned int i = 0; i < meta.num_flexbodies; ++i)
375  {
376  FlexBodyCacheData* data = & m_loaded_items[i];
377  this->ReadFlexbodyHeader(data);
378  if (!data->header.IsFaulty() && data->header.IsEnabled())
379  {
380  this->ReadFlexbodyLocatorList (data);
381  this->ReadFlexbodyPositionsBuffer(data);
382  this->ReadFlexbodyNormalsBuffer (data);
383  this->ReadFlexbodyColorsBuffer (data);
384  }
385  }
386 
387  this->CloseFile();
388  FLEX_DEBUG_LOG(__FUNCTION__ " >> OK ");
389  return RESULT_CODE_OK;
390  }
391  catch (ResultCode ret)
392  {
393  this->CloseFile();
394  FLEX_DEBUG_LOG(__FUNCTION__ " >> EXCEPTION!! ");
395  return ret;
396  }
397 }
