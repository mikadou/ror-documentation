289 const wxLanguageInfo *getLanguageInfoByName(const wxString name)
290 {
291  if(name.size() == 2)
292  {
293  for(int i=0;i<wxLANGUAGE_USER_DEFINED;i++)
294  {
295  const wxLanguageInfo *info = wxLocale::GetLanguageInfo(i);
296  if(!info)
297  continue;

299  if(cn == name)

317 int getAvLang(wxString dir, std::vector<wxLanguageInfo*> &files)
318 {
319  if(!wxDir::Exists(dir)) return 1;
320 
321  //WX portable version
322  wxDir dp(dir);
323  if (!dp.IsOpened())
324  {
325  wxLogStatus(wxT("error opening ") + dir);
326  return -1;
327  }
328  wxString name;
329  if (dp.GetFirst(&name, wxEmptyString, wxDIR_DIRS))
330  {
331  do
332  {
333  if (name.StartsWith(wxT(".")))
334  {
335  // do not add dot directories
336  continue;
337  }
338  if(name.Len() != 2)
339  {
340  // do not add other than language 2 letter code style files
341  continue;
342  }
343  wxLanguageInfo *li = const_cast<wxLanguageInfo *>(getLanguageInfoByName(name.substr(0,2)));
344  if(li)
345  {
346  files.push_back(li);
347  }
348  else
349  {
350  wxLogStatus(wxT("failed to get Language: "+name));
351  }
352  } while (dp.GetNext(&name));
353  }
354  return 0;
355 }

357 void initLanguage(wxString languagePath, wxString userpath)
358 {
359  // add language stuff
360 
361  // Initialize the catalogs we'll be using
362  wxString dirsep = wxT("/");
363  wxString basedir = languagePath;
364 #if OGRE_PLATFORM == OGRE_PLATFORM_WIN32
365  dirsep = wxT("\\");
366 #endif
367 
368  wxLocale::AddCatalogLookupPathPrefix(languagePath);
369 
370  // get all available languages
371  getAvLang(languagePath, avLanguages);
372  wxLogStatus(wxT("searching languages in ")+basedir);
373  if(avLanguages.size() > 0)
374  {
375  wxLogStatus(wxT("Available Languages:"));
376  std::vector<wxLanguageInfo *>::iterator it;
377  for(it = avLanguages.begin(); it!=avLanguages.end(); it++)
378  {
379  if(*it)
380  {
381  wxLogStatus(wxT(" * ") + (*it)->Description + wxT("(") + (*it)->CanonicalName + wxT(")"));
382  }
383  }
384  }
385  else
386  {
387  wxLogStatus(wxT("no Languages found!"));
388  }
389 
390  try
391  {
392  wxString rorcfg=userpath + dirsep + wxT("config") + dirsep + wxT("RoR.cfg");
393  Ogre::ConfigFile cfg;
394  // Don't trim whitespace
395  cfg.load((const char*)rorcfg.mb_str(wxConvUTF8), "=:\t", false);
396  wxString langSavedName = conv(cfg.getSetting("Language"));
397 
398  if(langSavedName.size() > 0)
399  language = const_cast<wxLanguageInfo *>(getLanguageInfoByName(langSavedName));
400  }
401  catch (...)
402  {
403  wxLogError(wxT(" unable to load RoR.cfg"));
404  }
405 
406  if(language == 0)
407  {
408  wxLogStatus(wxT("asking system for default language."));
409  language = const_cast<wxLanguageInfo *>(wxLocale::GetLanguageInfo(lang_locale.GetSystemLanguage()));
410  if(language)
411  {
412  wxLogStatus(wxT(" system returned: ") + language->Description + wxT("(") + language->CanonicalName + wxT(")"));
413  }
414  else
415  {
416  wxLogStatus(wxT("Failed to detect system language."));
417  return;
418  }
419  }
420  wxLogStatus(wxT("preferred language: ")+language->Description);
421  wxString lshort = language->CanonicalName.substr(0, 2);
422  wxString tmp = basedir + dirsep + lshort + dirsep + wxT("LC_MESSAGES") + dirsep + wxT("ror.mo");
423  wxLogStatus(wxT("lang file: ") + tmp);
424  if(wxFileName::FileExists(tmp))
425  {
426  wxLogStatus(wxT("language existing, using it!"));
427 #if wxCHECK_VERSION(2, 8, 0)
428  if(!lang_locale.IsAvailable((wxLanguage)language->Language))
429  {
430  wxLogStatus(wxT("language file existing, but not found via wxLocale!"));
431  wxLogStatus(wxT("is the language installed on your system?"));
432  }
433 #endif // version 2.8
434  bool res = lang_locale.Init((wxLanguage)language->Language);
435  if(!res)
436  {
437  wxLogStatus(wxT("error while initializing language!"));
438  }
439  res = lang_locale.AddCatalog(wxT("ror.mo"));
440  if(!res)
441  {
442  wxLogStatus(wxT("error while loading language!"));
443  }
444  }
445  else
446  {
447  lang_locale.Init(wxLANGUAGE_DEFAULT);
448  wxLogStatus(wxT("language not existing, no lang_locale support!"));
449  }
