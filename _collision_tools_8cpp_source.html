118 bool CollisionTools::collidesWithEntity(const Ogre::Vector3& fromPoint, const Ogre::Vector3& toPoint, const float collisionRadius, const float rayHeightLevel, const Ogre::uint32 queryMask)
119 {
120  Ogre::Vector3 fromPointAdj(fromPoint.x, fromPoint.y + rayHeightLevel, fromPoint.z);
121  Ogre::Vector3 toPointAdj(toPoint.x, toPoint.y + rayHeightLevel, toPoint.z);
122  Ogre::Vector3 normal = toPointAdj - fromPointAdj;
123  float distToDest = normal.normalise();
124  if (distToDest > 10)
125  return false;
126 
127  Ogre::Vector3 myResult(0, 0, 0);
128  Ogre::MovableObject* myObject = NULL;
129  float distToColl = 0.0f;
130 
131  if (raycastFromPoint(fromPointAdj, normal, myResult, myObject, distToColl, queryMask))
132  {
133  distToColl -= collisionRadius;
134  return (distToColl <= distToDest);
135  }
136  else
137  {
138  return false;
139  }

162 void CollisionTools::calculateY(Ogre::SceneNode* n, const bool doTerrainCheck, const bool doGridCheck, const float gridWidth, const Ogre::uint32 queryMask)
163 {
164  Ogre::Vector3 pos = n->getPosition();
165 
166  float x = pos.x;
167  float z = pos.z;
168  float y = pos.y;
169 
170  Ogre::Vector3 myResult(0, 0, 0);
171  Ogre::MovableObject* myObject = NULL;
172  float distToColl = 0.0f;
173 
174  float terrY = 0, colY = 0, colY2 = 0;
175 
176  if (raycastFromPoint(Ogre::Vector3(x, y, z), Ogre::Vector3::NEGATIVE_UNIT_Y, myResult, myObject, distToColl, queryMask))
177  {
178  if (myObject != NULL)
179  {
180  colY = myResult.y;
181  }
182  else
183  {
184  colY = -99999;
185  }
186  }
187 
188  //if doGridCheck is on, repeat not to fall through small holes for example when crossing a hangbridge
189  if (doGridCheck)
190  {
191  if (raycastFromPoint(Ogre::Vector3(x, y, z) + (n->getOrientation() * Ogre::Vector3(0, 0, gridWidth)), Ogre::Vector3::NEGATIVE_UNIT_Y, myResult, myObject, distToColl, queryMask))
192  {
193  if (myObject != NULL)
194  {
195  colY = myResult.y;
196  }
197  else
198  {
199  colY = -99999;
200  }
201  }
202  if (colY < colY2)
203  colY = colY2;
204  }
205 
206  // set the parameter to false if you are not using ETM or TSM
207  if (doTerrainCheck)
208  {
209 #ifdef ETM_TERRAIN
210  // ETM height value
211  terrY = mTerrainInfo->getHeightAt(x,z);
212 #else
213  // TSM height value
214  terrY = getTSMHeightAt(x, z);
215 #endif
216 
217  if (terrY < colY)
218  {
219  n->setPosition(x, colY + _heightAdjust, z);
220  }
221  else
222  {
223  n->setPosition(x, terrY + _heightAdjust, z);
224  }
225  }
226  else
227  {
228  if (!doTerrainCheck && colY == -99999)
229  colY = y;
230  n->setPosition(x, colY + _heightAdjust, z);
231  }
232 }

265 bool CollisionTools::raycast(const Ogre::Ray& ray, Ogre::Vector3& result, Ogre::MovableObject* & target, float& closest_distance, const Ogre::uint32 queryMask)
266 {
267  target = NULL;
268 
269  // check we are initialised
270  if (mRaySceneQuery != NULL)
271  {
272  // create a query object
273  mRaySceneQuery->setRay(ray);
274  mRaySceneQuery->setSortByDistance(true);
275  mRaySceneQuery->setQueryMask(queryMask);
276  // execute the query, returns a vector of hits
277  if (mRaySceneQuery->execute().size() <= 0)
278  {
279  // raycast did not hit an objects bounding box
280  return (false);
281  }
282  }
283  else
284  {
285  //LOG_ERROR << "Cannot raycast without RaySceneQuery instance" << ENDLOG;
286  return (false);
287  }
288 
289  // at this point we have raycast to a series of different objects bounding boxes.
290  // we need to test these different objects to see which is the first polygon hit.
291  // there are some minor optimizations (distance based) that mean we wont have to
292  // check all of the objects most of the time, but the worst case scenario is that
293  // we need to test every triangle of every object.
294  //Ogre::Ogre::Real closest_distance = -1.0f;
295  closest_distance = -1.0f;
296  Ogre::Vector3 closest_result = Ogre::Vector3::ZERO;
297  Ogre::RaySceneQueryResult& query_result = mRaySceneQuery->getLastResults();
298  for (size_t qr_idx = 0; qr_idx < query_result.size(); qr_idx++)
299  {
300  // stop checking if we have found a raycast hit that is closer
301  // than all remaining entities
302  if ((closest_distance >= 0.0f) &&
303  (closest_distance < query_result[qr_idx].distance))
304  {
305  break;
306  }
307 #if 0
308  if (query_result[qr_idx].movable != NULL)
309  LOG(TOSTRING(qr_idx) + ": " + query_result[qr_idx].movable->getMovableType() + " : " + query_result[qr_idx].movable->getName());
310 #endif
311  // mesh data to retrieve
312  bool valid = false;
313  mesh_info_t meshInfo;
314 
315  if ((query_result[qr_idx].movable != NULL) && !query_result[qr_idx].movable->getMovableType().compare("Entity"))
316  {
317  // movables
318  // get the entity to check
319  Ogre::MovableObject* pentity = static_cast<Ogre::MovableObject*>(query_result[qr_idx].movable);
320  // get the mesh information
321  GetMeshInformation(((Ogre::Entity*)pentity)->getMesh(), meshInfo.vertex_count, meshInfo.vertices, meshInfo.index_count, meshInfo.indices,
322  pentity->getParentNode()->_getDerivedPosition(),
323  pentity->getParentNode()->_getDerivedOrientation(),
324  pentity->getParentNode()->_getDerivedScale());
325  meshInfo.store = false;
326  valid = true;
327  }
328  else if ((query_result[qr_idx].movable != NULL) && !query_result[qr_idx].movable->getMovableType().compare("StaticGeometry"))
329  {
330  // static geometry
331  Ogre::MovableObject* pentity = static_cast<Ogre::MovableObject*>(query_result[qr_idx].movable);
332  Ogre::StaticGeometry::Region* rg = static_cast<Ogre::StaticGeometry::Region*>(query_result[qr_idx].movable);
333 
334  // this is a quick hack to prevent that we allocate unlimited amount of memory
335  // it clears the memory if we have more than 4 regions saved
336  if (meshInfoStorage.size() > 4)
337  {
338  // free everything
339  for (std::map<Ogre::String, mesh_info_t>::iterator it = meshInfoStorage.begin(); it != meshInfoStorage.end(); it++)
340  {
341  // free memory before clearing map
342  free(it->second.indices);
343  free(it->second.vertices);
344  }
345  // clear map
346  meshInfoStorage.clear();
347  }
348 
349  if (meshInfoStorage.find(rg->getName()) == meshInfoStorage.end())
350  {
351  // get mesh and store it
352  getStaticGeometry(rg->getParent(), rg, meshInfo.vertex_count, meshInfo.vertices, meshInfo.index_count, meshInfo.indices,
353 #if OGRE_VERSION>0x010602
354  rg->getParentNode()->_getDerivedPosition(),
355  rg->getParentNode()->_getDerivedOrientation(),
356 #else
357  rg->getParentNode()->getWorldPosition(),
358  rg->getParentNode()->getWorldOrientation(),
359 #endif //OGRE_VERSION
360  rg->getParentNode()->getScale());
361  meshInfo.store = true;
362  meshInfoStorage[rg->getName()] = meshInfo;
363  }
364  else
365  // existing, get it
366  meshInfo = meshInfoStorage[rg->getName()];
367  valid = true;
368  }
369 
370  if (valid)
371  {
372  // test for hitting individual triangles on the mesh
373  bool new_closest_found = false;
374  for (size_t i = 0; i < meshInfo.index_count; i += 3)
375  {
376  // check for a hit against this triangle
377  std::pair<bool, Ogre::Real> hit = Ogre::Math::intersects(ray, meshInfo.vertices[meshInfo.indices[i]],
378  meshInfo.vertices[meshInfo.indices[i + 1]], meshInfo.vertices[meshInfo.indices[i + 2]], true, false);
379 
380  // if it was a hit check if its the closest
381  if (hit.first)
382  {
383  if ((closest_distance < 0.0f) ||
384  (hit.second < closest_distance))
385  {
386  // this is the closest so far, save it off
387  closest_distance = hit.second;
388  new_closest_found = true;
389  }
390  }
391  }
392 
393  // free the verticies and indicies memory if we are not storing the data
394  if (!meshInfo.store)
395  {
396  delete[] meshInfo.vertices;
397  delete[] meshInfo.indices;
398  }
399 
400  // if we found a new closest raycast for this object, update the
401  // closest_result before moving on to the next object.
402  if (new_closest_found)
403  {
404  //target = pentity;
405  closest_result = ray.getPoint(closest_distance);
406  }
407  }
408  }
409 
410  // return the result
411  if (closest_distance >= 0.0f)
412  {
413  // raycast success
414  result = closest_result;
415  return (true);
416  }
417  else
418  {
419  // raycast failed
420  return (false);
421  }
422 }

425 // Code found on this forum link: http://www.ogre3d.org/wiki/index.php/RetrieveVertexData
426 void CollisionTools::GetMeshInformation(const Ogre::MeshPtr mesh,
427  size_t& vertex_count,
428  Ogre::Vector3* & vertices,
429  size_t& index_count,
430  Ogre::uint32* & indices,
431  const Ogre::Vector3& position,
432  const Ogre::Quaternion& orient,
433  const Ogre::Vector3& scale)
434 {
435  bool added_shared = false;
436  size_t current_offset = 0;
437  size_t shared_offset = 0;
438  size_t next_offset = 0;
439  size_t index_offset = 0;
440 
441  vertex_count = index_count = 0;
442 
443  // Calculate how many vertices and indices we're going to need
444  for (unsigned short i = 0; i < mesh->getNumSubMeshes(); ++i)
445  {
446  Ogre::SubMesh* submesh = mesh->getSubMesh(i);
447 
448  // We only need to add the shared vertices once
449  if (submesh->useSharedVertices)
450  {
451  if (!added_shared)
452  {
453  vertex_count += mesh->sharedVertexData->vertexCount;
454  added_shared = true;
455  }
456  }
457  else
458  {
459  vertex_count += submesh->vertexData->vertexCount;
460  }
461 
462  // Add the indices
463  index_count += submesh->indexData->indexCount;
464  }
465 
466  // Allocate space for the vertices and indices
467  vertices = new Ogre::Vector3[vertex_count];
468  indices = new Ogre::uint32[index_count];
469 
470  added_shared = false;
471 
472  // Run through the submeshes again, adding the data into the arrays
473  for (unsigned short i = 0; i < mesh->getNumSubMeshes(); ++i)
474  {
475  Ogre::SubMesh* submesh = mesh->getSubMesh(i);
476 
477  Ogre::VertexData* vertex_data = submesh->useSharedVertices ? mesh->sharedVertexData : submesh->vertexData;
478 
479  if ((!submesh->useSharedVertices) || (submesh->useSharedVertices && !added_shared))
480  {
481  if (submesh->useSharedVertices)
482  {
483  added_shared = true;
484  shared_offset = current_offset;
485  }
486 
487  const Ogre::VertexElement* posElem =
488  vertex_data->vertexDeclaration->findElementBySemantic(Ogre::VES_POSITION);
489 
490  Ogre::HardwareVertexBufferSharedPtr vbuf =
491  vertex_data->vertexBufferBinding->getBuffer(posElem->getSource());
492 
493  unsigned char* vertex =
494  static_cast<unsigned char*>(vbuf->lock(Ogre::HardwareBuffer::HBL_READ_ONLY));
495 
496  // There is _no_ baseVertexPointerToElement() which takes an Ogre::Ogre::Real or a double
497  // as second argument. So make it float, to avoid trouble when Ogre::Ogre::Real will
498  // be comiled/typedefed as double:
499  // Ogre::Ogre::Real* pOgre::Real;
500  float* pReal;
501 
502  for (size_t j = 0; j < vertex_data->vertexCount; ++j , vertex += vbuf->getVertexSize())
503  {
504  posElem->baseVertexPointerToElement(vertex, &pReal);
505 
506  Ogre::Vector3 pt(pReal[0], pReal[1], pReal[2]);
507 
508  vertices[current_offset + j] = (orient * (pt * scale)) + position;
509  }
510 
511  vbuf->unlock();
512  next_offset += vertex_data->vertexCount;
513  }
514 
515  Ogre::IndexData* index_data = submesh->indexData;
516  size_t numTris = index_data->indexCount / 3;
517  Ogre::HardwareIndexBufferSharedPtr ibuf = index_data->indexBuffer;
518 
519  bool use32bitindexes = (ibuf->getType() == Ogre::HardwareIndexBuffer::IT_32BIT);
520 
521  Ogre::uint32* pLong = static_cast<Ogre::uint32*>(ibuf->lock(Ogre::HardwareBuffer::HBL_READ_ONLY));
522  unsigned short* pShort = reinterpret_cast<unsigned short*>(pLong);
523 
524  size_t offset = (submesh->useSharedVertices) ? shared_offset : current_offset;
525 
526  if (use32bitindexes)
527  {
528  for (size_t k = 0; k < numTris * 3; ++k)
529  {
530  indices[index_offset++] = pLong[k] + static_cast<Ogre::uint32>(offset);
531  }
532  }
533  else
534  {
535  for (size_t k = 0; k < numTris * 3; ++k)
536  {
537  indices[index_offset++] = static_cast<Ogre::uint32>(pShort[k]) +
538  static_cast<Ogre::uint32>(offset);
539  }
540  }
541 
542  ibuf->unlock();
543  current_offset = next_offset;
544  }
545 }
