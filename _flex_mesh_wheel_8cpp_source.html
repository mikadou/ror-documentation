166 Vector3 FlexMeshWheel::updateVertices()
167 {
168  Vector3 center = (m_all_nodes[m_axis_node0_idx].AbsPosition + m_all_nodes[m_axis_node1_idx].AbsPosition) / 2.0;
169  Vector3 ray = m_all_nodes[m_start_node_idx].AbsPosition - m_all_nodes[m_axis_node0_idx].AbsPosition;
170  Vector3 axis = m_all_nodes[m_axis_node0_idx].AbsPosition - m_all_nodes[m_axis_node1_idx].AbsPosition;
171 
172  axis.normalise();
173 
174  for (size_t i=0; i<m_num_rays; i++)
175  {
176  Plane pl=Plane(axis, m_all_nodes[m_axis_node0_idx].AbsPosition);
177  ray=m_all_nodes[m_start_node_idx+i*2].AbsPosition-m_all_nodes[m_axis_node0_idx].AbsPosition;
178  ray=pl.projectVector(ray);
179  ray.normalise();
180  m_vertices[i*6 ].position=m_all_nodes[m_axis_node0_idx].AbsPosition+m_rim_radius*ray-center;
181 
182  m_vertices[i*6+1].position=m_all_nodes[m_start_node_idx+i*2].AbsPosition-0.05 *(m_all_nodes[m_start_node_idx+i*2].AbsPosition-m_all_nodes[m_axis_node0_idx].AbsPosition)-center;
183  m_vertices[i*6+2].position=m_all_nodes[m_start_node_idx+i*2].AbsPosition-0.1 *(m_all_nodes[m_start_node_idx+i*2].AbsPosition-m_all_nodes[m_start_node_idx+i*2+1].AbsPosition)-center;
184  m_vertices[i*6+3].position=m_all_nodes[m_start_node_idx+i*2+1].AbsPosition-0.1 *(m_all_nodes[m_start_node_idx+i*2+1].AbsPosition-m_all_nodes[m_start_node_idx+i*2].AbsPosition)-center;
185  m_vertices[i*6+4].position=m_all_nodes[m_start_node_idx+i*2+1].AbsPosition-0.05*(m_all_nodes[m_start_node_idx+i*2+1].AbsPosition-m_all_nodes[m_axis_node1_idx].AbsPosition)-center;
186 
187  pl=Plane(-axis, m_all_nodes[m_axis_node1_idx].AbsPosition);
188  ray=m_all_nodes[m_start_node_idx+i*2+1].AbsPosition-m_all_nodes[m_axis_node1_idx].AbsPosition;
189  ray=pl.projectVector(ray);
190  ray.normalise();
191  m_vertices[i*6+5].position=m_all_nodes[m_axis_node1_idx].AbsPosition+m_rim_radius*ray-center;
192 
193  //normals
194  m_vertices[i*6 ].normal=axis;
195  m_vertices[i*6+1].normal=(m_vertices[i*6].position-m_vertices[i*6+1].position).crossProduct(m_vertices[i*6].position-m_vertices[((i+1)%m_num_rays)*6+1].position)/m_norm_y;
196  m_vertices[i*6+2].normal=ray;
197  m_vertices[i*6+3].normal=ray;
198  m_vertices[i*6+4].normal=(m_vertices[i*6+4].position-m_vertices[i*6+5].position).crossProduct(m_vertices[i*6+4].position-m_vertices[((i+1)%m_num_rays)*6+4].position)/m_norm_y;
199  m_vertices[i*6+5].normal=-axis;
200  }
201  for (int i=0; i<6; i++)
202  {
203  m_vertices[m_num_rays*6+i].position=m_vertices[i].position;
204  m_vertices[m_num_rays*6+i].normal=m_vertices[i].normal;
205  }
206 
207  return center;
208 }

215 bool FlexMeshWheel::flexitPrepare()
216 {
217  Vector3 center = (m_all_nodes[m_axis_node0_idx].AbsPosition + m_all_nodes[m_axis_node1_idx].AbsPosition) / 2.0;
218  m_rim_scene_node->setPosition(center);
219 
220  Vector3 axis = m_all_nodes[m_axis_node0_idx].AbsPosition - m_all_nodes[m_axis_node1_idx].AbsPosition;
221  axis.normalise();
222 
223  if (m_is_rim_reverse) axis = -axis;
224  Vector3 ray = m_all_nodes[m_start_node_idx].AbsPosition - m_all_nodes[m_axis_node0_idx].AbsPosition;
225  Vector3 onormal = axis.crossProduct(ray);
226  onormal.normalise();
227  ray = axis.crossProduct(onormal);
228  m_rim_scene_node->setOrientation(Quaternion(axis, onormal, ray));
229 
230  return true;
231 }
