111 void Lens::recalculateDepthOfField(Real& _nearDepth, Real& _focalDepth, Real& _farDepth)
112 {
113  // Set focalDepth to the current focalDistance
114  _focalDepth = mFocalDistance;
115 
116  // Recalculate the hyperfocal length
117  recalculateHyperfocalLength();
118 
119  // Calculate the numerator of the optics equations
120  Real numerator = (mFocalDistance * (mHyperfocalLength - mFocalLength));
121 
122  Real nearClear = numerator / (mHyperfocalLength + mFocalDistance - (2.0 * mFocalLength));
123 
124  // Adjust the nearDepth relative to the aperture. This is an approximation.
125  _nearDepth = std::min(nearClear - nearClear * mFStop, 0.0f);
126 
127  if (mFocalDistance < mHyperfocalLength)
128  {
129  // Calculate the far clear plane
130  Real farClear = numerator / (mHyperfocalLength - mFocalDistance);
131 
132  // Adjust the farDepth relative to the aperture. This is an approximation.
133  _farDepth = farClear + farClear * mFStop;
134  }
135  else
136  {
137  // Far depth of field should be infinite
138  _farDepth = Math::POS_INFINITY;
139  }
140 }
