117  char line[1024] = {};

120  if (fd)
121  {

411 void Water::showWave(Vector3 refpos)
412 {
413  if (!wbuffer)
414  return;
415 
416  float xScaled = mapSize.x * mScale;
417  float zScaled = mapSize.z * mScale;
418 
419  for (int pz = 0; pz < WAVEREZ + 1; pz++)
420  {
421  for (int px = 0; px < WAVEREZ + 1; px++)
422  {
423  wbuffer[(pz * (WAVEREZ + 1) + px) * 8 + 1] = getHeightWaves(refpos + Vector3(xScaled * 0.5 - (float)px * xScaled / WAVEREZ, 0, (float)pz * zScaled / WAVEREZ - zScaled * 0.5)) - wHeight;
424  }
425  }
426 
427  //normals
428  for (int pz = 0; pz < WAVEREZ + 1; pz++)
429  {
430  for (int px = 0; px < WAVEREZ + 1; px++)
431  {
432  int left = std::max(0, px - 1);
433  int right = std::min(px + 1, WAVEREZ);
434  int up = std::max(0, pz - 1);
435  int down = std::min(pz + 1, WAVEREZ);
436 
437  Vector3 normal = (Vector3(wbuffer + ((pz * (WAVEREZ + 1) + left) * 8)) - Vector3(wbuffer + ((pz * (WAVEREZ + 1) + right) * 8))).crossProduct(Vector3(wbuffer + ((up * (WAVEREZ + 1) + px) * 8)) - Vector3(wbuffer + ((down * (WAVEREZ + 1) + px) * 8)));
438  normal.normalise();
439 
440  wbuffer[(pz * (WAVEREZ + 1) + px) * 8 + 3] = normal.x;
441  wbuffer[(pz * (WAVEREZ + 1) + px) * 8 + 4] = normal.y;
442  wbuffer[(pz * (WAVEREZ + 1) + px) * 8 + 5] = normal.z;
443  }
444  }
445 
446  wbuf->writeData(0, (WAVEREZ + 1) * (WAVEREZ + 1) * 32, wbuffer, true);
447 }

458 void Water::update()
459 {
460  if (!visible || !mRenderCamera)
461  return;
462 
463  if (pWaterNode)
464  {
465  Vector3 cameraPos(mRenderCamera->getPosition().x, wHeight, mRenderCamera->getPosition().z);
466  Vector3 sightPos(cameraPos);
467 
468  Ray lineOfSight(mRenderCamera->getPosition(), mRenderCamera->getDirection());
469  Plane waterPlane(Vector3::UNIT_Y, Vector3::UNIT_Y * wHeight);
470 
471  std::pair<bool, Real> intersection = lineOfSight.intersects(waterPlane);
472 
473  if (intersection.first && intersection.second > 0.0f)
474  sightPos = lineOfSight.getPoint(intersection.second);
475 
476  Real offset = std::min(cameraPos.distance(sightPos), std::min(mapSize.x, mapSize.z) * 0.5f);
477 
478  Vector3 waterPos = cameraPos + (sightPos - cameraPos).normalisedCopy() * offset;
479  Vector3 bottomPos = Vector3(waterPos.x, wbHeight, waterPos.z);
480 
481  if (waterPos.distance(pWaterNode->getPosition()) > 200.0f || ForceUpdate)
482  {
483  pWaterNode->setPosition(Vector3(waterPos.x, wHeight, waterPos.z));
484  pBottomNode->setPosition(bottomPos);
485  ForceUpdate = false; //Happens only once
486  }
487  if (RoR::App::GetGfxWaterUseWaves() && RoR::App::GetActiveMpState() == RoR::App::MP_STATE_DISABLED)
488  showWave(pWaterNode->getPosition());
489  }
490 
491  bool underwater = isCameraUnderWater();
492  static bool lastWaterMode = false;
493  bool underWaterModeChanged = false;
494  if (underwater != lastWaterMode)
495  {
496  underWaterModeChanged = true;

498  }
499 
500  framecounter++;

558 float Water::getHeightWaves(Vector3 pos)
559 {
560  // no waves?
561  if (!RoR::App::GetGfxWaterUseWaves() || RoR::App::GetActiveMpState() == RoR::App::MP_STATE_CONNECTED)
562  {
563  // constant height, sea is flat as pancake
564  return wHeight;
565  }
566 
567  // uh, some upper limit?!
568  if (pos.y > wHeight + maxampl)
569  return wHeight;
570 
571  float waveheight = getWaveHeight(pos);
572  // we will store the result in this variable, init it with the default height
573  float result = wHeight;
574  // now walk through all the wave trains. One 'train' is one sin/cos set that will generate once wave. All the trains together will sum up, so that they generate a 'rough' sea
575  for (int i = 0; i < free_wavetrain; i++)
576  {
577  // calculate the amplitude that this wave will have. wavetrains[i].amplitude is read from the config
578  // upper limit: prevent too big waves by setting an upper limit
579  float amp = std::min(wavetrains[i].amplitude * waveheight, wavetrains[i].maxheight);
580  // now the main thing:
581  // calculate the sinus with the values of the config file and add it to the result
582  result += amp * sin(Math::TWO_PI * ((gEnv->mrTime * wavetrains[i].wavespeed + wavetrains[i].dir_sin * pos.x + wavetrains[i].dir_cos * pos.z) / wavetrains[i].wavelength));
583  }
584  // return the summed up waves
585  return result;
586 }

588 bool Water::isUnderWater(Vector3 pos)
589 {
590  float waterheight = wHeight;
591 
592  if (RoR::App::GetGfxWaterUseWaves() && RoR::App::GetActiveMpState() == RoR::App::MP_STATE_DISABLED)
593  {
594  float waveheight = getWaveHeight(pos);
595 
596  if (pos.y > wHeight + maxampl * waveheight || pos.y > wHeight + maxampl)
597  return false;
598 
599  waterheight = getHeightWaves(pos);
600  }
601

605 Vector3 Water::getVelocity(Vector3 pos)
606 {
607  if (!RoR::App::GetGfxWaterUseWaves() || RoR::App::GetActiveMpState() == RoR::App::MP_STATE_CONNECTED)
608  return Vector3::ZERO;

611 
612  if (pos.y > wHeight + maxampl)
