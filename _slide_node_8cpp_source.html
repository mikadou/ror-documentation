212 void SlideNode::UpdateForces(float dt)
213 {
214  // only do calcs if we have a beam to slide on
215  if (!mSlidingBeam || mSlidingBeam->broken || getFlag(MASK_SLIDE_BROKEN))
216  {
217  return;
218  }
219 
220  // the threshold changes when attaching a node to a beam, if it's more than
221  // the initial amount then then we're still attaching.
222  if (mCurThreshold > mInitThreshold)
223  {
224  mCurThreshold -= (mAttachRate * dt);
225  }
226 
227  Ogre::Vector3 perpForces = getCorrectiveForces();
228  // perpendicular Forces are distributed according to the position along the Beam
229  if (perpForces.length() > mBreakForce)
230  setFlag(MASK_SLIDE_BROKEN);
231  mSlidingNode->Forces += -perpForces;
232  mSlidingBeam->p1->Forces += perpForces * (1 - mRatio);
233  mSlidingBeam->p2->Forces += perpForces * mRatio;
234 }

236 Rail* SlideNode::getClosestRailAll(RailGroup* railGroup, const Ogre::Vector3& point)
237 {
238  if (!railGroup)
239  return NULL;
240 
241  Rail* closestRail = (Rail*) railGroup->getStartRail();
242  Rail* curRail = (Rail*) railGroup->getStartRail()->next;
243 
244  Ogre::Real lenToClosest = getLenTo(closestRail, point);
245  Ogre::Real lenToCurrent = std::numeric_limits<Ogre::Real>::infinity();
246 
247  while (curRail &&
248  // loop detection, if both id's are the same on the start and the current
249  // then it has detected a loop. Hmm however this won't work if we loop
250  // back on the start but don't loop.
251  ((curRail->curBeam->p1->id != railGroup->getStartRail()->curBeam->p1->id) ||
252  (curRail->curBeam->p2->id != railGroup->getStartRail()->curBeam->p2->id)))
253  {
254  lenToCurrent = getLenTo(curRail, point);
255 
256  if (lenToCurrent < lenToClosest)
257  {
258  closestRail = curRail;
259  lenToClosest = getLenTo(closestRail, point);
260  }
261  curRail = curRail->next;
262  };
263 
264  return closestRail;

268 {

286 void SlideNode::UpdatePosition()
287 {
288  // only do calcs if we have a beam to slide on
289  if (!mSlidingBeam || mSlidingBeam->broken)
290  {
291  mIdealPosition = mSlidingNode->AbsPosition;
292  return;
293  }
294 
295  // find which beam to use
296  mSlidingRail = getClosestRail(mSlidingRail, mSlidingNode->AbsPosition);
297  mSlidingBeam = mSlidingRail->curBeam;
298 
299  // Get vector for beam
300  Ogre::Vector3 b = mSlidingBeam->p2->AbsPosition;
301  b -= mSlidingBeam->p1->AbsPosition;
302 
303  // pre-compute normal
304  const Ogre::Real bLen = b.length();
305 
306  // normalize b instead of using another variable
307  b.normalise();
308 
309  // Get dot product along the b beam
310  const Ogre::Real aDotBUnit = (mSlidingNode->AbsPosition - mSlidingBeam->p1->AbsPosition).dotProduct(b);
311 
312  // constrain Value between the two end points
313  const Ogre::Real len = std::max(0.0f, std::min(aDotBUnit, bLen));
314  mIdealPosition = b;
315  mIdealPosition *= len;
316  mIdealPosition += mSlidingBeam->p1->AbsPosition;
317 
318  // calculate(cache) the ratio between the the two end points,
319  // if bLen = 0.0f it means the beam is zero length so pick an end point
320  mRatio = (bLen > 0.0f) ? len / bLen : 0.0f;
321 }
