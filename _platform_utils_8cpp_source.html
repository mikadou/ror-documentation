115 void InstallCrashRpt()

122  info.pszAppVersion = ROR_VERSION_STRING;
123  info.pszEmailSubject = "Error Report for Rigs of Rods";
124  info.pszEmailTo = "error-report@rigsofrods.org";
125  info.pszUrl = "http://crashfix.rigsofrods.org/index.php/crashReport/uploadExternal";
126  info.uPriorities[CR_HTTP] = 1; // Use HTTP.
127  info.uPriorities[CR_SMTP] = CR_NEGATIVE_PRIORITY; // Not user SMTP.
128  info.uPriorities[CR_SMAPI] = CR_NEGATIVE_PRIORITY; // Not use Simple MAPI.
129  info.dwFlags = CR_INST_AUTO_THREAD_HANDLERS; // Install the per-thread exception handlers automatically
130  info.dwFlags |= CR_INST_ALL_POSSIBLE_HANDLERS; // Install all available exception handlers
131  info.dwFlags |= CR_INST_SHOW_ADDITIONAL_INFO_FIELDS; // Makes "Your E-mail" and "Describe what you were doing when the problem occurred" fields of Error Report dialog always visible.
132  info.pszPrivacyPolicyURL = "http://docs.rigsofrods.org/legal/crash-report-privacy-policy/"; // URL for the Privacy Policy link
133 
134  crSetCrashCallback(CrashRptCallback, NULL);
135
