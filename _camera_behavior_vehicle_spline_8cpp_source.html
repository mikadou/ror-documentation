194 void CameraBehaviorVehicleSpline::createSpline(const CameraManager::CameraContext& ctx)
195 {
196  splineClosed = false;
197  splineLength = 1.0f;
198 
199  spline->clear();
200  splineNodes.clear();
201 
202  for (int i = 0; i < ctx.mCurrTruck->free_camerarail; i++)
203  {
204  splineNodes.push_back(&ctx.mCurrTruck->nodes[ctx.mCurrTruck->cameraRail[i]]);
205  }
206 
207  std::list<Beam*> linkedBeams = ctx.mCurrTruck->getAllLinkedBeams();
208 
209  numLinkedBeams = linkedBeams.size();
210 
211  if (numLinkedBeams > 0)
212  {
213  for (std::list<Beam*>::iterator it = linkedBeams.begin(); it != linkedBeams.end(); ++it)
214  {
215  if ((*it)->free_camerarail <= 0)
216  continue;
217 
218  Vector3 curSplineFront = splineNodes.front()->AbsPosition;
219  Vector3 curSplineBack = splineNodes.back()->AbsPosition;
220 
221  Vector3 linkedSplineFront = (*it)->nodes[(*it)->cameraRail[0]].AbsPosition;
222  Vector3 linkedSplineBack = (*it)->nodes[(*it)->cameraRail[(*it)->free_camerarail - 1]].AbsPosition;
223 
224  if (curSplineBack.distance(linkedSplineFront) < 5.0f)
225  {
226  for (int i = 1; i < (*it)->free_camerarail; i++)
227  {
228  splineNodes.push_back(&(*it)->nodes[(*it)->cameraRail[i]]);
229  }
230  }
231  else if (curSplineFront.distance(linkedSplineFront) < 5.0f)
232  {
233  for (int i = 1; i < (*it)->free_camerarail; i++)
234  {
235  splineNodes.push_front(&(*it)->nodes[(*it)->cameraRail[i]]);
236  }
237  }
238  else if (curSplineBack.distance(linkedSplineBack) < 5.0f)
239  {
240  for (int i = (*it)->free_camerarail - 2; i >= 0; i--)
241  {
242  splineNodes.push_back(&(*it)->nodes[(*it)->cameraRail[i]]);
243  }
244  }
245  else if (curSplineFront.distance(linkedSplineBack) < 5.0f)
246  {
247  for (int i = (*it)->free_camerarail - 2; i >= 0; i--)
248  {
249  splineNodes.push_front(&(*it)->nodes[(*it)->cameraRail[i]]);
250  }
251  }
252  }
253  }
254 
255  for (unsigned int i = 0; i < splineNodes.size(); i++)
256  {
257  spline->addPoint(splineNodes[i]->AbsPosition);
258  }
259 
260  Vector3 firstPoint = spline->getPoint(0);
261  Vector3 lastPoint = spline->getPoint(spline->getNumPoints() - 1);
262 
263  if (firstPoint.distance(lastPoint) < 1.0f)
264  {
265  splineClosed = true;
266  }
267 
268  for (int i = 1; i < spline->getNumPoints(); i++)
269  {
270  splineLength += spline->getPoint(i - 1).distance(spline->getPoint(i));
271  }
272 
273  splineLength /= 2.0f;
274 
275  if (!splineObject && ctx.mDebug)
276  {
277  splineObject = gEnv->sceneManager->createManualObject();
278  SceneNode* splineNode = gEnv->sceneManager->getRootSceneNode()->createChildSceneNode();
279 
280  splineObject->begin("tracks/transred", Ogre::RenderOperation::OT_LINE_STRIP);
281  for (int i = 0; i < splineDrawResolution; i++)
282  {
283  splineObject->position(0, 0, 0);
284  }
285  splineObject->end();
286 
287  splineNode->attachObject(splineObject);
288  }
289 }
