168  void Mesh::_createGeometry()
169  {
170  int& Complexity = mOptions.MeshComplexity;
171 
172  int numVertices = Complexity*Complexity;
173  int numEle = 6 * (Complexity-1)*(Complexity-1);
174 
175  // Vertex buffers
176  mSubMesh->vertexData = new Ogre::VertexData();
177  mSubMesh->vertexData->vertexStart = 0;
178  mSubMesh->vertexData->vertexCount = numVertices;
179 
180  Ogre::VertexDeclaration* vdecl = mSubMesh->vertexData->vertexDeclaration;
181  Ogre::VertexBufferBinding* vbind = mSubMesh->vertexData->vertexBufferBinding;
182 
183  size_t offset = 0;
184 
185  switch (mOptions.MeshVertexType)
186  {
187  case VT_POS_NORM_UV:
188  {
189  vdecl->addElement(0, 0, Ogre::VET_FLOAT3, Ogre::VES_POSITION);
190  offset += Ogre::VertexElement::getTypeSize(Ogre::VET_FLOAT3);
191  vdecl->addElement(0, offset, Ogre::VET_FLOAT3, Ogre::VES_NORMAL);
192  offset += Ogre::VertexElement::getTypeSize(Ogre::VET_FLOAT3);
193  vdecl->addElement(0, offset, Ogre::VET_FLOAT2, Ogre::VES_TEXTURE_COORDINATES);
194 
195  mVertexBuffer = Ogre::HardwareBufferManager::getSingleton().
196  createVertexBuffer(sizeof(POS_NORM_UV_VERTEX),
197  numVertices,
198  Ogre::HardwareBuffer::HBU_DYNAMIC_WRITE_ONLY);
199  }
200  break;
201 
202  case VT_POS_NORM:
203  {
204  vdecl->addElement(0, 0, Ogre::VET_FLOAT3, Ogre::VES_POSITION);
205  offset += Ogre::VertexElement::getTypeSize(Ogre::VET_FLOAT3);
206  vdecl->addElement(0, offset, Ogre::VET_FLOAT3, Ogre::VES_NORMAL);
207 
208  mVertexBuffer = Ogre::HardwareBufferManager::getSingleton().
209  createVertexBuffer(sizeof(POS_NORM_VERTEX),
210  numVertices,
211  Ogre::HardwareBuffer::HBU_DYNAMIC_WRITE_ONLY);
212  }
213  break;
214 
215  case VT_POS_UV:
216  {
217  vdecl->addElement(0, 0, Ogre::VET_FLOAT3, Ogre::VES_POSITION);
218  offset += Ogre::VertexElement::getTypeSize(Ogre::VET_FLOAT3);
219  vdecl->addElement(0, offset, Ogre::VET_FLOAT2, Ogre::VES_TEXTURE_COORDINATES);
220 
221  mVertexBuffer = Ogre::HardwareBufferManager::getSingleton().
222  createVertexBuffer(sizeof(POS_UV_VERTEX),
223  numVertices,
224  Ogre::HardwareBuffer::HBU_DYNAMIC_WRITE_ONLY);
225  }
226  break;
227 
228  case VT_POS:
229  {
230  vdecl->addElement(0, 0, Ogre::VET_FLOAT3, Ogre::VES_POSITION);
231 
232  mVertexBuffer = Ogre::HardwareBufferManager::getSingleton().
233  createVertexBuffer(sizeof(POS_VERTEX),
234  numVertices,
235  Ogre::HardwareBuffer::HBU_DYNAMIC_WRITE_ONLY);
236  }
237  break;
238  }
239 
240  vbind->setBinding(0, mVertexBuffer);
241 
242  unsigned int *indexbuffer = new unsigned int[numEle];
243 
244  int i = 0;
245  for(int v=0; v<Complexity-1; v++){
246  for(int u=0; u<Complexity-1; u++){
247  // face 1 |/
248  indexbuffer[i++] = v*Complexity + u;
249  indexbuffer[i++] = v*Complexity + u + 1;
250  indexbuffer[i++] = (v+1)*Complexity + u;
251 
252  // face 2 /|
253  indexbuffer[i++] = (v+1)*Complexity + u;
254  indexbuffer[i++] = v*Complexity + u + 1;
255  indexbuffer[i++] = (v+1)*Complexity + u + 1;
256  }
257  }
258 
259  // Prepare buffer for indices
260  mIndexBuffer =
261  Ogre::HardwareBufferManager::getSingleton().createIndexBuffer(
262  Ogre::HardwareIndexBuffer::IT_32BIT,
263  numEle,
264  Ogre::HardwareBuffer::HBU_STATIC, true);
265 
266  mIndexBuffer->
267  writeData(0,
268  mIndexBuffer->getSizeInBytes(),
269  indexbuffer,
270  true);
271 
272  delete []indexbuffer;
273 
274  // Set index buffer for this submesh
275  mSubMesh->indexData->indexBuffer = mIndexBuffer;
276  mSubMesh->indexData->indexStart = 0;
277  mSubMesh->indexData->indexCount = numEle;
278  }

299  bool Mesh::isPointInGrid(const Ogre::Vector2 &Position)
300  {
301  Ogre::AxisAlignedBox WordMeshBox = mEntity->getWorldBoundingBox();
302 
303  // Get our mesh grid rectangle:
304  // c-----------d
305  // | |
306  // | |
307  // | |
308  // a-----------b
309  Ogre::Vector3
310  a = WordMeshBox.getCorner(Ogre::AxisAlignedBox::FAR_LEFT_BOTTOM),
311  b = WordMeshBox.getCorner(Ogre::AxisAlignedBox::FAR_RIGHT_BOTTOM),
312  c = WordMeshBox.getCorner(Ogre::AxisAlignedBox::NEAR_RIGHT_BOTTOM),
313  d = WordMeshBox.getCorner(Ogre::AxisAlignedBox::NEAR_LEFT_BOTTOM);
314 
315  // Transform all corners to Ogre::Vector2 array
316  Ogre::Vector2 Corners2D[4] =
317  {Ogre::Vector2(a.x, a.z),
318  Ogre::Vector2(b.x, b.z),
319  Ogre::Vector2(c.x, c.z),
320  Ogre::Vector2(d.x, d.z)};
321 
322  // Determinate if Position is into our rectangle, we use a line intersection detection
323  // because our mesh rectangle can be rotated, if the number of collisions with the four
324  // segments AB, BC, CD, DA is one, the Position point is into the rectangle, else(if number
325  // of collisions are 0 or 2, the Position point is outside the rectangle.
326  int NumberOfCollisions = 0;
327  // Find a point wich isn't be inside the rectangle
328  Ogre::Vector2 DestPoint = Corners2D[0] + (Corners2D[1]-Corners2D[0])*2;
329  for (int k = 0; k < 3; k++)
330  {
331  if (Math::intersectionOfTwoLines(Corners2D[k], Corners2D[k+1], Position, DestPoint) != Ogre::Vector2::ZERO)
332  {
333  NumberOfCollisions ++;
334  }
335 
336  if (k == 2)
337  {
338  if (Math::intersectionOfTwoLines(Corners2D[3], Corners2D[0], Position, DestPoint) != Ogre::Vector2::ZERO)
339  {
340  NumberOfCollisions ++;
341  }
342  }
343  }
344  if (NumberOfCollisions == 1)
345  {
346  return true;
347  }
348 
349  return false;
350  }

352  Ogre::Vector2 Mesh::getGridPosition(const Ogre::Vector2 &Position)
353  {
354  if (mOptions.MeshSize.Width == 0 && mOptions.MeshSize.Height == 0)
355  {
356  return Position;
357  }
358 
359  if (!isPointInGrid(Position))
360  {
361  return Ogre::Vector2(-1,-1);
362  }
363 
364  Ogre::AxisAlignedBox WordMeshBox = mEntity->getWorldBoundingBox();
365 
366  // Get our mesh grid rectangle: (Only a,b,c corners)
367  // c
368  // |
369  // |
370  // |
371  // a-----------b
372  Ogre::Vector3
373  a = WordMeshBox.getCorner(Ogre::AxisAlignedBox::FAR_LEFT_BOTTOM),
374  b = WordMeshBox.getCorner(Ogre::AxisAlignedBox::FAR_RIGHT_BOTTOM),
375  c = WordMeshBox.getCorner(Ogre::AxisAlignedBox::NEAR_LEFT_BOTTOM);
376 
377  // Transform all corners to Ogre::Vector2 array
378  Ogre::Vector2 Corners2D[3] =
379  {Ogre::Vector2(a.x, a.z),
380  Ogre::Vector2(b.x, b.z),
381  Ogre::Vector2(c.x, c.z)};
382 
383  // Get segments AB and AC
384  Ogre::Vector2 AB = Corners2D[1]-Corners2D[0],
385  AC = Corners2D[2]-Corners2D[0];
386 
387  // Find the X/Y position projecting the Position point to AB and AC segments.
388  Ogre::Vector2 XProjectedPoint = Position-AC,
389  YProjectedPoint = Position-AB;
390 
391  // Fint the intersections points
392  Ogre::Vector2 XPoint = Math::intersectionOfTwoLines(Corners2D[0],Corners2D[1],Position,XProjectedPoint),
393  YPoint = Math::intersectionOfTwoLines(Corners2D[0],Corners2D[2],Position,YProjectedPoint);
394 
395  // Find lengths
396  Ogre::Real ABLength = AB.length(),
397  ACLength = AC.length(),
398  XLength = (XPoint-Corners2D[0]).length(),
399  YLength = (YPoint-Corners2D[0]).length();
400 
401  // Find final x/y grid positions in [0,1] range
402  Ogre::Real XFinal = XLength / ABLength,
403  YFinal = YLength / ACLength;
404 
405  return Ogre::Vector2(XFinal,YFinal);
406  }
