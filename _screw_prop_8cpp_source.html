45 void Screwprop::updateForces(int update)
46 {
47  if (!gEnv->terrainManager->getWater())
48  return;
49  float depth = gEnv->terrainManager->getWater()->getHeightWaves(nodes[noderef].AbsPosition) - nodes[noderef].AbsPosition.y;
50  if (depth < 0)
51  return; //out of water!
52  Vector3 dir = nodes[nodeback].RelPosition - nodes[noderef].RelPosition;
53  Vector3 rudaxis = nodes[noderef].RelPosition - nodes[nodeup].RelPosition;
54  dir.normalise();
55  if (reverse)
56  dir = -dir;
57  rudaxis.normalise();
58  dir = (throtle * fullpower) * (Quaternion(Degree(rudder), rudaxis) * dir);
59  nodes[noderef].Forces += dir;
60 
61  if (update && splashp && throtle > 0.1)
62  {
63  if (depth < 0.2)
64  splashp->allocSplash(nodes[noderef].AbsPosition, 10.0 * dir / fullpower);
65  else
66  splashp->allocSplash(nodes[noderef].AbsPosition, 5.0 * dir / fullpower);
67  ripplep->allocRipple(nodes[noderef].AbsPosition, 10.0 * dir / fullpower);
68  }
69 }

71 void Screwprop::setThrottle(float val)
