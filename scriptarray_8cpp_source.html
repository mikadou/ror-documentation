125 // This optional callback is called when the template type is first used by the compiler.
126 // It allows the application to validate if the template can be instantiated for the requested
127 // subtype at compile time, instead of at runtime. The output argument dontGarbageCollect
128 // allow the callback to tell the engine if the template instance type shouldn't be garbage collected,
129 // i.e. no asOBJ_GC flag.
130 static bool ScriptArrayTemplateCallback(asITypeInfo *ti, bool &dontGarbageCollect)
131 {
132  // Make sure the subtype can be instantiated with a default factory/constructor,
133  // otherwise we won't be able to instantiate the elements.
134  int typeId = ti->GetSubTypeId();
135  if( typeId == asTYPEID_VOID )
136  return false;
137  if( (typeId & asTYPEID_MASK_OBJECT) && !(typeId & asTYPEID_OBJHANDLE) )
138  {
139  asITypeInfo *subtype = ti->GetEngine()->GetTypeInfoById(typeId);
140  asDWORD flags = subtype->GetFlags();
141  if( (flags & asOBJ_VALUE) && !(flags & asOBJ_POD) )
142  {
143  // Verify that there is a default constructor
144  bool found = false;
145  for( asUINT n = 0; n < subtype->GetBehaviourCount(); n++ )
146  {
147  asEBehaviours beh;
148  asIScriptFunction *func = subtype->GetBehaviourByIndex(n, &beh);
149  if( beh != asBEHAVE_CONSTRUCT ) continue;
150 
151  if( func->GetParamCount() == 0 )
152  {
153  // Found the default constructor
154  found = true;
155  break;
156  }
157  }
158 
159  if( !found )
160  {
161  // There is no default constructor
162  // TODO: Should format the message to give the name of the subtype for better understanding
163  ti->GetEngine()->WriteMessage("array", 0, 0, asMSGTYPE_ERROR, "The subtype has no default constructor");
164  return false;
165  }
166  }
167  else if( (flags & asOBJ_REF) )
168  {
169  bool found = false;
170 
171  // If value assignment for ref type has been disabled then the array
172  // can be created if the type has a default factory function
173  if( !ti->GetEngine()->GetEngineProperty(asEP_DISALLOW_VALUE_ASSIGN_FOR_REF_TYPE) )
174  {
175  // Verify that there is a default factory
176  for( asUINT n = 0; n < subtype->GetFactoryCount(); n++ )
177  {
178  asIScriptFunction *func = subtype->GetFactoryByIndex(n);
179  if( func->GetParamCount() == 0 )
180  {
181  // Found the default factory
182  found = true;
183  break;
184  }
185  }
186  }
187 
188  if( !found )
189  {
190  // No default factory
191  // TODO: Should format the message to give the name of the subtype for better understanding
192  ti->GetEngine()->WriteMessage("array", 0, 0, asMSGTYPE_ERROR, "The subtype has no default factory");
193  return false;
194  }
195  }
196 
197  // If the object type is not garbage collected then the array also doesn't need to be
198  if( !(flags & asOBJ_GC) )
199  dontGarbageCollect = true;
200  }
201  else if( !(typeId & asTYPEID_OBJHANDLE) )
202  {
203  // Arrays with primitives cannot form circular references,
204  // thus there is no need to garbage collect them
205  dontGarbageCollect = true;
206  }
207  else
208  {
209  assert( typeId & asTYPEID_OBJHANDLE );
210 
211  // It is not necessary to set the array as garbage collected for all handle types.
212  // If it is possible to determine that the handle cannot refer to an object type
213  // that can potentially form a circular reference with the array then it is not
214  // necessary to make the array garbage collected.
215  asITypeInfo *subtype = ti->GetEngine()->GetTypeInfoById(typeId);
216  asDWORD flags = subtype->GetFlags();
217  if( !(flags & asOBJ_GC) )
218  {
219  if( (flags & asOBJ_SCRIPT_OBJECT) )
220  {
221  // Even if a script class is by itself not garbage collected, it is possible
222  // that classes that derive from it may be, so it is not possible to know
223  // that no circular reference can occur.
224  if( (flags & asOBJ_NOINHERIT) )
225  {
226  // A script class declared as final cannot be inherited from, thus
227  // we can be certain that the object cannot be garbage collected.
228  dontGarbageCollect = true;
229  }
230  }
231  else
232  {
233  // For application registered classes we assume the application knows
234  // what it is doing and don't mark the array as garbage collected unless
235  // the type is also garbage collected.
236  dontGarbageCollect = true;
237  }
238  }
239  }

241  // The type is ok
242  return true;

364 {
365  // The object type should be the template instance of the array
366  assert( ti && string(ti->GetName()) == "array" );
367 
368  refCount = 1;
369  gcFlag = false;
370  objType = ti;
371  objType->AddRef();
372  buffer = 0;
373 
374  Precache();
375 
376  asIScriptEngine *engine = ti->GetEngine();
377 
378  // Determine element size
379  if( subTypeId & asTYPEID_MASK_OBJECT )
380  elementSize = sizeof(asPWORD);
381  else
382  elementSize = engine->GetSizeOfPrimitiveType(subTypeId);
383 
384  // Determine the initial size from the buffer
385  asUINT length = *(asUINT*)buf;
386 
387  // Make sure the array size isn't too large for us to handle
388  if( !CheckMaxSize(length) )
389  {
390  // Don't continue with the initialization
391  return;
392  }
393 
394  // Copy the values of the array elements from the buffer
395  if( (ti->GetSubTypeId() & asTYPEID_MASK_OBJECT) == 0 )
396  {
397  CreateBuffer(&buffer, length);
398 
399  // Copy the values of the primitive type into the internal buffer
400  if( length > 0 )
401  memcpy(At(0), (((asUINT*)buf)+1), length * elementSize);
402  }
403  else if( ti->GetSubTypeId() & asTYPEID_OBJHANDLE )
404  {
405  CreateBuffer(&buffer, length);
406 
407  // Copy the handles into the internal buffer
408  if( length > 0 )
409  memcpy(At(0), (((asUINT*)buf)+1), length * elementSize);
410 
411  // With object handles it is safe to clear the memory in the received buffer
412  // instead of increasing the ref count. It will save time both by avoiding the
413  // call the increase ref, and also relieve the engine from having to release
414  // its references too
415  memset((((asUINT*)buf)+1), 0, length * elementSize);
416  }
417  else if( ti->GetSubType()->GetFlags() & asOBJ_REF )
418  {
419  // Only allocate the buffer, but not the objects
420  subTypeId |= asTYPEID_OBJHANDLE;
421  CreateBuffer(&buffer, length);
422  subTypeId &= ~asTYPEID_OBJHANDLE;
423 
424  // Copy the handles into the internal buffer
425  if( length > 0 )
426  memcpy(buffer->data, (((asUINT*)buf)+1), length * elementSize);
427 
428  // For ref types we can do the same as for handles, as they are
429  // implicitly stored as handles.
430  memset((((asUINT*)buf)+1), 0, length * elementSize);
431  }
432  else
433  {
434  // TODO: Optimize by calling the copy constructor of the object instead of
435  // constructing with the default constructor and then assigning the value
436  // TODO: With C++11 ideally we should be calling the move constructor, instead
437  // of the copy constructor as the engine will just discard the objects in the
438  // buffer afterwards.
439  CreateBuffer(&buffer, length);
440 
441  // For value types we need to call the opAssign for each individual object
442  for( asUINT n = 0; n < length; n++ )
443  {
444  void *obj = At(n);
445  asBYTE *srcObj = (asBYTE*)buf;
446  srcObj += 4 + n*ti->GetSubType()->GetSize();
447  engine->AssignScriptObject(obj, srcObj, ti->GetSubType());
448  }
449  }
450 
451  // Notify the GC of the successful creation
452  if( objType->GetFlags() & asOBJ_GC )
453  objType->GetEngine()->NotifyGarbageCollectorOfNewObject(this, objType);
454 }

673 void CScriptArray::Resize(int delta, asUINT at)
674 {
675  if( delta < 0 )
676  {
677  if( -delta > (int)buffer->numElements )
678  delta = -(int)buffer->numElements;
679  if( at > buffer->numElements + delta )
680  at = buffer->numElements + delta;
681  }
682  else if( delta > 0 )
683  {

685  if( delta > 0 && !CheckMaxSize(buffer->numElements + delta) )
686  return;
687 
688  if( at > buffer->numElements )
689  at = buffer->numElements;
690  }
691 
692  if( delta == 0 ) return;
693 
694  if( buffer->maxElements < buffer->numElements + delta )
695  {

698  if( newBuffer )

701  newBuffer->maxElements = newBuffer->numElements;

704  {

707  if( ctx )
708  ctx->SetException("Out of memory");
709  return;
710  }
711

713  // since we're just copying the pointers to objects and not the actual objects.
714  memcpy(newBuffer->data, buffer->data, at*elementSize);
715  if( at < buffer->numElements )
716  memcpy(newBuffer->data + (at+delta)*elementSize, buffer->data + at*elementSize, (buffer->numElements-at)*elementSize);

719  Construct(newBuffer, at, at+delta);
720

722  userFree(buffer);
723 
724  buffer = newBuffer;
725  }
726  else if( delta < 0 )

730  // since we're just copying the pointers to objects and not the actual objects.
731  memmove(buffer->data + at*elementSize, buffer->data + (at-delta)*elementSize, (buffer->numElements - (at-delta))*elementSize);

1420

1422  if( ctx )

1425

1427 #if defined(_MSC_VER) && _MSC_VER >= 1500 && !defined(__S3E__)

1437 #endif

1440  }
1441 
1442  return;
1443  }
1444  }
1445 
1446  // No need to sort
1447  if( count < 2 )

1449  return;
1450  }
1451

1453  int end = startAt + count;

1458  asIScriptContext *ctx = asGetActiveContext();
1459

1461  if( ctx )
1462  {
1463  ctx->SetException("Index out of bounds");
1464  }
1465 
1466  return;
1467  }
1468 
1469  asBYTE tmp[16];
1470  asIScriptContext *cmpContext = 0;
1471  bool isNested = false;
1472 
1473  if( subTypeId & ~asTYPEID_MASK_SEQNBR )
1474  {
1475  // Try to reuse the active context
1476  cmpContext = asGetActiveContext();
1477  if( cmpContext )
1478  {
1479  if( cmpContext->GetEngine() == objType->GetEngine() && cmpContext->PushState() >= 0 )
1480  isNested = true;
1481  else
1482  cmpContext = 0;
1483  }
1484  if( cmpContext == 0 )
1485  {
1486  // TODO: Ideally this context would be retrieved from a pool, so we don't have to
1487  // create a new one everytime. We could keep a context with the array object
1488  // but that would consume a lot of resources as each context is quite heavy.
1489  cmpContext = objType->GetEngine()->CreateContext();
1490  }
1491  }
1492 
1493  // Insertion sort
1494  for( int i = start + 1; i < end; i++ )
1495  {
1496  Copy(tmp, GetArrayItemPointer(i));
1497 
1498  int j = i - 1;
1499

1501  {
1502  Copy(GetArrayItemPointer(j + 1), GetArrayItemPointer(j));
1503  j--;
1504  }
1505 
1506  Copy(GetArrayItemPointer(j + 1), tmp);
1507  }
1508 
1509  if( cmpContext )
1510  {
1511  if( isNested )
1512  {
1513  asEContextState state = cmpContext->GetState();
1514  cmpContext->PopState();
1515  if( state == asEXECUTION_ABORTED )

1518  else
1519  cmpContext->Release();

1523 // internal
1524 void CScriptArray::CopyBuffer(SArrayBuffer *dst, SArrayBuffer *src)
1525 {
1526  asIScriptEngine *engine = objType->GetEngine();
1527  if( subTypeId & asTYPEID_OBJHANDLE )
1528  {
1529  // Copy the references and increase the reference counters
1530  if( dst->numElements > 0 && src->numElements > 0 )
1531  {
1532  int count = dst->numElements > src->numElements ? src->numElements : dst->numElements;
1533 
1534  void **max = (void**)(dst->data + count * sizeof(void*));
1535  void **d = (void**)dst->data;
1536  void **s = (void**)src->data;
1537 
1538  for( ; d < max; d++, s++ )
1539  {
1540  void *tmp = *d;
1541  *d = *s;
1542  if( *d )
1543  engine->AddRefScriptObject(*d, objType->GetSubType());
1544  // Release the old ref after incrementing the new to avoid problem incase it is the same ref
1545  if( tmp )
1546  engine->ReleaseScriptObject(tmp, objType->GetSubType());
1547  }
1548  }
1549  }
1550  else
1551  {
1552  if( dst->numElements > 0 && src->numElements > 0 )
1553  {
1554  int count = dst->numElements > src->numElements ? src->numElements : dst->numElements;
1555  if( subTypeId & asTYPEID_MASK_OBJECT )
1556  {
1557  // Call the assignment operator on all of the objects
1558  void **max = (void**)(dst->data + count * sizeof(void*));
1559  void **d = (void**)dst->data;
1560  void **s = (void**)src->data;
1561 
1562  asITypeInfo *subType = objType->GetSubType();
1563  for( ; d < max; d++, s++ )
1564  engine->AssignScriptObject(*d, *s, subType);
1565  }
1566  else
1567  {
1568  // Primitives are copied byte for byte
1569  memcpy(dst->data, src->data, count*elementSize);
1570  }
1571  }
1572  }

1575 // internal
1576 // Precache some info
1577 void CScriptArray::Precache()
1578 {
1579  subTypeId = objType->GetSubTypeId();
1580 
1581  // Check if it is an array of objects. Only for these do we need to cache anything
1582  // Type ids for primitives and enums only has the sequence number part
1583  if( !(subTypeId & ~asTYPEID_MASK_SEQNBR) )
1584  return;
1585 
1586  // The opCmp and opEquals methods are cached because the searching for the
1587  // methods is quite time consuming if a lot of array objects are created.
1588 
1589  // First check if a cache already exists for this array type
1590  SArrayCache *cache = reinterpret_cast<SArrayCache*>(objType->GetUserData(ARRAY_CACHE));
1591  if( cache ) return;
1592 
1593  // We need to make sure the cache is created only once, even
1594  // if multiple threads reach the same point at the same time
1595  asAcquireExclusiveLock();
1596 
1597  // Now that we got the lock, we need to check again to make sure the
1598  // cache wasn't created while we were waiting for the lock
1599  cache = reinterpret_cast<SArrayCache*>(objType->GetUserData(ARRAY_CACHE));
1600  if( cache )
1601  {
1602  asReleaseExclusiveLock();
1603  return;
1604  }
1605 
1606  // Create the cache
1607  cache = reinterpret_cast<SArrayCache*>(userAlloc(sizeof(SArrayCache)));
1608  memset(cache, 0, sizeof(SArrayCache));
1609 
1610  // If the sub type is a handle to const, then the methods must be const too
1611  bool mustBeConst = (subTypeId & asTYPEID_HANDLETOCONST) ? true : false;
1612 
1613  asITypeInfo *subType = objType->GetEngine()->GetTypeInfoById(subTypeId);
1614  if( subType )
1615  {
1616  for( asUINT i = 0; i < subType->GetMethodCount(); i++ )
1617  {
1618  asIScriptFunction *func = subType->GetMethodByIndex(i);
1619 
1620  if( func->GetParamCount() == 1 && (!mustBeConst || func->IsReadOnly()) )
1621  {
1622  asDWORD flags = 0;
1623  int returnTypeId = func->GetReturnTypeId(&flags);
1624 
1625  // The method must not return a reference
1626  if( flags != asTM_NONE )
1627  continue;
1628 
1629  // opCmp returns an int and opEquals returns a bool
1630  bool isCmp = false, isEq = false;
1631  if( returnTypeId == asTYPEID_INT32 && strcmp(func->GetName(), "opCmp") == 0 )
1632  isCmp = true;
1633  if( returnTypeId == asTYPEID_BOOL && strcmp(func->GetName(), "opEquals") == 0 )
1634  isEq = true;
1635 
1636  if( !isCmp && !isEq )
1637  continue;
1638 
1639  // The parameter must either be a reference to the subtype or a handle to the subtype
1640  int paramTypeId;
1641  func->GetParam(0, &paramTypeId, &flags);
1642 
1643  if( (paramTypeId & ~(asTYPEID_OBJHANDLE|asTYPEID_HANDLETOCONST)) != (subTypeId & ~(asTYPEID_OBJHANDLE|asTYPEID_HANDLETOCONST)) )
1644  continue;
1645 
1646  if( (flags & asTM_INREF) )
1647  {
1648  if( (paramTypeId & asTYPEID_OBJHANDLE) || (mustBeConst && !(flags & asTM_CONST)) )
1649  continue;
1650  }
1651  else if( paramTypeId & asTYPEID_OBJHANDLE )
1652  {
1653  if( mustBeConst && !(paramTypeId & asTYPEID_HANDLETOCONST) )
1654  continue;
1655  }
1656  else
1657  continue;
1658 
1659  if( isCmp )
1660  {
1661  if( cache->cmpFunc || cache->cmpFuncReturnCode )
1662  {
1663  cache->cmpFunc = 0;
1664  cache->cmpFuncReturnCode = asMULTIPLE_FUNCTIONS;
1665  }
1666  else
1667  cache->cmpFunc = func;
1668  }
1669  else if( isEq )
1670  {
1671  if( cache->eqFunc || cache->eqFuncReturnCode )
1672  {
1673  cache->eqFunc = 0;
1674  cache->eqFuncReturnCode = asMULTIPLE_FUNCTIONS;
1675  }
1676  else
1677  cache->eqFunc = func;
1678  }
1679  }
1680  }
1681  }
1682 
1683  if( cache->eqFunc == 0 && cache->eqFuncReturnCode == 0 )
1684  cache->eqFuncReturnCode = asNO_FUNCTION;
1685  if( cache->cmpFunc == 0 && cache->cmpFuncReturnCode == 0 )
1686  cache->cmpFuncReturnCode = asNO_FUNCTION;
1687 
1688  // Set the user data only at the end so others that retrieve it will know it is complete
1689  objType->SetUserData(cache, ARRAY_CACHE);
1690 
1691  asReleaseExclusiveLock();
1692 }
