154 void Settings::loadSettings(String configFile, bool overwrite)
155 {
156  ConfigFile cfg;
157  cfg.load(configFile, "=:\t", false);
158 
159  // load all settings into a map!
160  ConfigFile::SettingsIterator i = cfg.getSettingsIterator();
161  String s_value, s_name;
162  while (i.hasMoreElements())
163  {
164  s_name = SanitizeUtf8String(i.peekNextKey());
165  s_value = SanitizeUtf8String(i.getNext());
166  if (!overwrite && !settings[s_name].empty())
167  {
168  continue;
169  }
170  settings[s_name] = s_value;
171  }
172 
173  // generate hash of the token
174  String usertoken = SSETTING("User Token", "");
175  char usertokensha1result[250];
176  memset(usertokensha1result, 0, 250);
177  if (usertoken.size()>0)
178  {
179  RoR::CSHA1 sha1;
180  sha1.UpdateHash((uint8_t *)usertoken.c_str(), (uint32_t)usertoken.size());
181  sha1.Final();
182  sha1.ReportHash(usertokensha1result, RoR::CSHA1::REPORT_HEX_SHORT);
183  }
184 
185  setSetting("User Token Hash", String(usertokensha1result));
186 }

188 bool Settings::get_system_paths(char *program_path, char *user_path)
189 {
190 #if OGRE_PLATFORM == OGRE_PLATFORM_WIN32
191  // note: we enforce usage of the non-UNICODE interfaces (since its easier to integrate here)
192  if (!GetModuleFileNameA(NULL, program_path, 512))
193  {
194  ErrorUtils::ShowError(_L("Startup error"), _L("Error while retrieving program space path"));
195  return false;
196  }
197  GetShortPathNameA(program_path, program_path, 512); //this is legal
198  path_descend(program_path);
199 
200  if (getSetting("userpath", "").empty())
201  {
202  if (SHGetFolderPathA(NULL, CSIDL_PERSONAL, NULL, SHGFP_TYPE_CURRENT, user_path)!=S_OK)
203  {
204  ErrorUtils::ShowError(_L("Startup error"), _L("Error while retrieving user space path"));
205  return false;
206  }
207  GetShortPathNameA(user_path, user_path, 512); //this is legal
208  sprintf(user_path, "%s\\Rigs of Rods %s\\", user_path, ROR_VERSION_STRING_SHORT); // do not use the full string, as same minor versions should share this directory
209  } else
210  {
211  strcpy(user_path, getSetting("userpath", "").c_str());
212  }
213 
214 #elif OGRE_PLATFORM == OGRE_PLATFORM_LINUX
215  //true program path is impossible to get from POSIX functions
216  //lets hack!
217  pid_t pid = getpid();
218  char procpath[256];
219  sprintf(procpath, "/proc/%d/exe", pid);
220  int ch = readlink(procpath,program_path,240);
221  if (ch != -1)
222  {
223  program_path[ch] = 0;
224  path_descend(program_path);
225  } else return false;
226  //user path is easy
227  char home_path[256];
228  strncpy(home_path, getenv ("HOME"), 240);
229  //sprintf(user_path, "%s/RigsOfRods/", home_path); // old version
230  sprintf(user_path, "%s/.rigsofrods/", home_path);
231 #elif OGRE_PLATFORM == OGRE_PLATFORM_APPLE
232  int cx;
233 
234  // program path
235  const int len = 256;
236  uint32_t lenb = (uint32_t) len;
237  char procpath[len];
238 
239  if (_NSGetExecutablePath(procpath, &lenb) == -1)
240  return false;
241 
242  /*
243  * _NSGetExecutablePath returns the absolute path to the binary so procpath
244  * has "./RoR" at its end. We only want the path to the directory so we
245  * cut off the last 5 characters of the string.
246  */
247  assert(strlen(procpath) > 5);
248  procpath[strlen(procpath) - 4] = '\0';
249 
250  cx = snprintf(program_path, len, "%s", procpath);
251  if ( cx < 0 || cx >= len)
252  return false;
253 
254  // user path
255  cx = snprintf(user_path, 256, "%s/RigsOfRods/", getenv("HOME"));
256  if ( cx < 0 || cx >= 256)
257  return false;
258 #endif
259  return true;
260 }

280  }

284  _chdir(program_path);
285 #endif // _WIN32
286

289

299 {

301 #if OGRE_PLATFORM == OGRE_PLATFORM_WIN32
302  dirsep='\\';
303 #endif
304  char* pt=path+strlen(path)-1;

307  if (pt>=path) *(pt+1)=0;
