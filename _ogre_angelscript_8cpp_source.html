161 void registerOgreObjects(AngelScript::asIScriptEngine* engine)
162 {
163  int r;
164 
165  // We start by registering some data types, so angelscript knows that they exist
166 
167  // Ogre::Degree
168  r = engine->RegisterObjectType("degree", sizeof(Degree), asOBJ_VALUE | asOBJ_POD | asOBJ_APP_CLASS_CA | asOBJ_APP_CLASS_ALLFLOATS);
169  MYASSERT( r >= 0 );
170 
171  // Ogre::Radian
172  r = engine->RegisterObjectType("radian", sizeof(Radian), asOBJ_VALUE | asOBJ_POD | asOBJ_APP_CLASS_CA | asOBJ_APP_CLASS_ALLFLOATS);
173  MYASSERT( r >= 0 );
174 
175  // Ogre::Vector3
176  r = engine->RegisterObjectType("vector3", sizeof(Ogre::Vector3), asOBJ_VALUE | asOBJ_POD | asOBJ_APP_CLASS_CA | asOBJ_APP_CLASS_ALLFLOATS);
177  MYASSERT( r >= 0 );
178 
179  // Ogre::Quaternion
180  r = engine->RegisterObjectType("quaternion", sizeof(Quaternion), asOBJ_VALUE | asOBJ_POD | asOBJ_APP_CLASS_CA | asOBJ_APP_CLASS_ALLFLOATS);
181  MYASSERT( r >= 0 );
182 
183  registerOgreRadian(engine);
184  registerOgreDegree(engine);
185  registerOgreVector3(engine);
186  registerOgreQuaternion(engine);
187 }
