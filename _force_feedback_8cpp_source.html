56 void ForceFeedback::SetForces(float roll, float pitch, float wspeed, float dircommand, float stress)
57 {
58  assert(m_device);
59 
60  //LOG(String("ForceFeedback: R=")+TOSTRING(roll)+" D="+TOSTRING(dir)+" S="+TOSTRING(wspeed)+" H="+TOSTRING(stress));
61  if (!m_hydro_effect)
62  {
63  //we create effect at the last moment, because it does not works otherwise
64  m_hydro_effect = new OIS::Effect(OIS::Effect::ConstantForce, OIS::Effect::Constant);
65  m_hydro_effect->direction = OIS::Effect::North;
66  m_hydro_effect->trigger_button = 0;
67  m_hydro_effect->trigger_interval = 0;
68  m_hydro_effect->replay_length = OIS::Effect::OIS_INFINITE; // Linux/Win32: Same behaviour as 0.
69  m_hydro_effect->replay_delay = 0;
70  m_hydro_effect->setNumAxes(1);
71  OIS::ConstantEffect* hydroConstForce = dynamic_cast<OIS::ConstantEffect*>(m_hydro_effect->getForceEffect());
72  hydroConstForce->level = 0; //-10K to +10k
73  hydroConstForce->envelope.attackLength = 0;
74  hydroConstForce->envelope.attackLevel = (unsigned short)hydroConstForce->level;
75  hydroConstForce->envelope.fadeLength = 0;
76  hydroConstForce->envelope.fadeLevel = (unsigned short)hydroConstForce->level;
77 
78  m_device->upload(m_hydro_effect);
79  }
80 
81  OIS::ConstantEffect* hydroConstForce = dynamic_cast<OIS::ConstantEffect*>(m_hydro_effect->getForceEffect());
82  float stress_gain = App::GetIoFFbackStressGain();
83  float centering_gain = App::GetIoFFbackCenterGain();
84  float ff = -stress * stress_gain + dircommand * 100.0 * centering_gain * wspeed * wspeed;
85  if (ff > 10000)
86  ff = 10000;
87  if (ff < -10000)
88  ff = -10000;
89  hydroConstForce->level = ff; //-10K to +10k
90  m_device->modify(m_hydro_effect);
91 }
