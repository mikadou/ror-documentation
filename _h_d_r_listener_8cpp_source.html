33 void HDRListener::notifyCompositor(Ogre::CompositorInstance* instance)
34 {
35  // Get some RTT dimensions for later calculations
36  Ogre::CompositionTechnique::TextureDefinitionIterator defIter =
37  instance->getTechnique()->getTextureDefinitionIterator();
38  while (defIter.hasMoreElements())
39  {
40  Ogre::CompositionTechnique::TextureDefinition* def =
41  defIter.getNext();
42  if (def->name == "rt_bloom0")
43  {
44  mBloomSize = (int)def->width; // should be square
45  // Calculate Gaussian texture offsets & weights
46  float deviation = 3.0f;
47  float texelSize = 1.0f / (float)mBloomSize;
48 
49  // central sample, no offset
50  mBloomTexOffsetsHorz[0][0] = 0.0f;
51  mBloomTexOffsetsHorz[0][1] = 0.0f;
52  mBloomTexOffsetsVert[0][0] = 0.0f;
53  mBloomTexOffsetsVert[0][1] = 0.0f;
54  mBloomTexWeights[0][0] = mBloomTexWeights[0][1] =
55  mBloomTexWeights[0][2] = Ogre::Math::gaussianDistribution(0, 0, deviation);
56  mBloomTexWeights[0][3] = 1.0f;
57 
58  // 'pre' samples
59  for (int i = 1; i < 8; ++i)
60  {
61  mBloomTexWeights[i][0] = mBloomTexWeights[i][1] =
62  mBloomTexWeights[i][2] = 1.25f * Ogre::Math::gaussianDistribution(i, 0, deviation);
63  mBloomTexWeights[i][3] = 1.0f;
64  mBloomTexOffsetsHorz[i][0] = i * texelSize;
65  mBloomTexOffsetsHorz[i][1] = 0.0f;
66  mBloomTexOffsetsVert[i][0] = 0.0f;
67  mBloomTexOffsetsVert[i][1] = i * texelSize;
68  }
69  // 'post' samples
70  for (int i = 8; i < 15; ++i)
71  {
72  mBloomTexWeights[i][0] = mBloomTexWeights[i][1] =
73  mBloomTexWeights[i][2] = mBloomTexWeights[i - 7][0];
74  mBloomTexWeights[i][3] = 1.0f;
75 
76  mBloomTexOffsetsHorz[i][0] = -mBloomTexOffsetsHorz[i - 7][0];
77  mBloomTexOffsetsHorz[i][1] = 0.0f;
78  mBloomTexOffsetsVert[i][0] = 0.0f;
79  mBloomTexOffsetsVert[i][1] = -mBloomTexOffsetsVert[i - 7][1];
80  }
81  }
82  }
83 }
