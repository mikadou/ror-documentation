254 // Run a range mode bandwidth test
256 void OpenCLTestBandwidth::testBandwidthRange(unsigned int start, unsigned int end, unsigned int increment,
257  memcpyKind kind, accessMode accMode, memoryMode memMode, int startDevice, int endDevice)
258 {
259  //count the number of copies we're going to run
260  unsigned int count = 1 + ((end - start) / increment);
261 
262  unsigned int * memSizes = (unsigned int *)malloc(count * sizeof( unsigned int ));
263  double* bandwidths = (double*)malloc(count * sizeof(double));
264 
265  // Before calculating the cumulative bandwidth, initialize bandwidths array to NULL
266  for (unsigned int i = 0; i < count; i++)
267  bandwidths[i] = 0.0;
268 
269  // Use the device asked by the user
270  for (int currentDevice = startDevice; currentDevice <= endDevice; currentDevice++)
271  {
272  // Allocate command queue for the device (dealloc first if already allocated)
273  createQueue(currentDevice);
274 
275  //run each of the copies
276  for(unsigned int i = 0; i < count; i++)
277  {
278  memSizes[i] = start + i * increment;
279  switch(kind)
280  {
281  case DEVICE_TO_HOST: bandwidths[i] += testDeviceToHostTransfer(memSizes[i], accMode, memMode);
282  break;
283  case HOST_TO_DEVICE: bandwidths[i] += testHostToDeviceTransfer(memSizes[i], accMode, memMode);
284  break;
285  case DEVICE_TO_DEVICE: bandwidths[i] += testDeviceToDeviceTransfer(memSizes[i]);
286  break;
287  }
288  }
289  } // Complete the bandwidth computation on all the devices
290 
291  //print results
292  printResultsReadable(memSizes, bandwidths, count, kind, accMode, memMode, (1 + endDevice - startDevice));
293 
294  //clean up
295  free(memSizes);
296  free(bandwidths);
297 }

300 // Intense shmoo mode - covers a large range of values with varying increments
302 void OpenCLTestBandwidth::testBandwidthShmoo(memcpyKind kind, accessMode accMode,
303  memoryMode memMode, int startDevice, int endDevice)
304 {
305  //count the number of copies to make
306  unsigned int count = 1 + (SHMOO_LIMIT_20KB / SHMOO_INCREMENT_1KB)
307  + ((SHMOO_LIMIT_50KB - SHMOO_LIMIT_20KB) / SHMOO_INCREMENT_2KB)
308  + ((SHMOO_LIMIT_100KB - SHMOO_LIMIT_50KB) / SHMOO_INCREMENT_10KB)
309  + ((SHMOO_LIMIT_1MB - SHMOO_LIMIT_100KB) / SHMOO_INCREMENT_100KB)
310  + ((SHMOO_LIMIT_16MB - SHMOO_LIMIT_1MB) / SHMOO_INCREMENT_1MB)
311  + ((SHMOO_LIMIT_32MB - SHMOO_LIMIT_16MB) / SHMOO_INCREMENT_2MB)
312  + ((SHMOO_MEMSIZE_MAX - SHMOO_LIMIT_32MB) / SHMOO_INCREMENT_4MB);
313 
314  unsigned int *memSizes = (unsigned int *)malloc(count * sizeof(unsigned int));
315  double* bandwidths = (double*)malloc(count * sizeof(double));
316 
317  // Before calculating the cumulative bandwidth, initialize bandwidths array to NULL
318  for (unsigned int i = 0; i < count; i++)
319  bandwidths[i] = 0.0;
320 
321  // Use the device asked by the user
322  for (int currentDevice = startDevice; currentDevice <= endDevice; currentDevice++)
323  {
324  // Allocate command queue for the device (dealloc first if already allocated)
325  createQueue(currentDevice);
326 
327  //Run the shmoo
328  int iteration = 0;
329  unsigned int memSize = 0;
330  while(memSize <= SHMOO_MEMSIZE_MAX )
331  {
332  if(memSize < SHMOO_LIMIT_20KB )
333  {
334  memSize += SHMOO_INCREMENT_1KB;
335  }
336  else if( memSize < SHMOO_LIMIT_50KB)
337  {
338  memSize += SHMOO_INCREMENT_2KB;
339  }
340  else if( memSize < SHMOO_LIMIT_100KB)
341  {
342  memSize += SHMOO_INCREMENT_10KB;
343  }
344  else if( memSize < SHMOO_LIMIT_1MB)
345  {
346  memSize += SHMOO_INCREMENT_100KB;
347  }
348  else if( memSize < SHMOO_LIMIT_16MB)
349  {
350  memSize += SHMOO_INCREMENT_1MB;
351  }
352  else if( memSize < SHMOO_LIMIT_32MB)
353  {
354  memSize += SHMOO_INCREMENT_2MB;
355  }
356  else
357  {
358  memSize += SHMOO_INCREMENT_4MB;
359  }
360 
361  memSizes[iteration] = memSize;
362  switch(kind)
363  {
364  case DEVICE_TO_HOST: bandwidths[iteration] += testDeviceToHostTransfer(memSizes[iteration], accMode, memMode);
365  break;
366  case HOST_TO_DEVICE: bandwidths[iteration] += testHostToDeviceTransfer(memSizes[iteration], accMode, memMode);
367  break;
368  case DEVICE_TO_DEVICE: bandwidths[iteration] += testDeviceToDeviceTransfer(memSizes[iteration]);
369  break;
370  }
371  iteration++;
372  stream << ".";
373  }
374  } // Complete the bandwidth computation on all the devices
375 
376  //print results

378  printResultsReadable(memSizes, bandwidths, count, kind, accMode, memMode, (endDevice - startDevice));
379 
380  //clean up

388 double OpenCLTestBandwidth::testDeviceToHostTransfer(unsigned int memSize, accessMode accMode, memoryMode memMode)
389 {
390  unsigned int timer = 0;
391  double elapsedTimeInSec = 0.0;
392  double bandwidthInMBs = 0.0;
393  unsigned char *h_data = NULL;
394  cl_mem cmPinnedData = NULL;
395  cl_mem cmDevData = NULL;
396  cl_int ciErrNum = CL_SUCCESS;
397 
398  //allocate and init host memory, pinned or conventional
399  if(memMode == PINNED)
400  {
401  // Create a host buffer
402  cmPinnedData = clCreateBuffer(cxGPUContext, CL_MEM_READ_WRITE | CL_MEM_ALLOC_HOST_PTR, memSize, NULL, &ciErrNum);
403  ownCheckError(ciErrNum, CL_SUCCESS);
404 
405  // Get a mapped pointer
406  h_data = (unsigned char*)clEnqueueMapBuffer(cqCommandQueue, cmPinnedData, CL_TRUE, CL_MAP_WRITE, 0, memSize, 0, NULL, NULL, &ciErrNum);
407  ownCheckError(ciErrNum, CL_SUCCESS);
408 
409  //initialize
410  for(unsigned int i = 0; i < memSize/sizeof(unsigned char); i++)
411  {
412  h_data[i] = (unsigned char)(i & 0xff);
413  }
414 
415  // unmap and make data in the host buffer valid
416  ciErrNum = clEnqueueUnmapMemObject(cqCommandQueue, cmPinnedData, (void*)h_data, 0, NULL, NULL);
417  ownCheckError(ciErrNum, CL_SUCCESS);
418  }
419  else
420  {
421  // standard host alloc
422  h_data = (unsigned char *)malloc(memSize);
423 
424  //initialize
425  for(unsigned int i = 0; i < memSize/sizeof(unsigned char); i++)
426  {
427  h_data[i] = (unsigned char)(i & 0xff);
428  }
429  }
430 
431  // allocate device memory
432  cmDevData = clCreateBuffer(cxGPUContext, CL_MEM_READ_WRITE, memSize, NULL, &ciErrNum);
433  ownCheckError(ciErrNum, CL_SUCCESS);
434 
435  // initialize device memory
436  if(memMode == PINNED)
437  {
438  // Get a mapped pointer
439  h_data = (unsigned char*)clEnqueueMapBuffer(cqCommandQueue, cmPinnedData, CL_TRUE, CL_MAP_WRITE, 0, memSize, 0, NULL, NULL, &ciErrNum);
440 
441  ciErrNum = clEnqueueWriteBuffer(cqCommandQueue, cmDevData, CL_FALSE, 0, memSize, h_data, 0, NULL, NULL);
442  ownCheckError(ciErrNum, CL_SUCCESS);
443  }
444  else
445  {
446  ciErrNum = clEnqueueWriteBuffer(cqCommandQueue, cmDevData, CL_FALSE, 0, memSize, h_data, 0, NULL, NULL);
447  ownCheckError(ciErrNum, CL_SUCCESS);
448  }
449  ownCheckError(ciErrNum, CL_SUCCESS);
450 
451  // Sync queue to host, start timer 0, and copy data from GPU to Host
452  ciErrNum = clFinish(cqCommandQueue);
453  shrDeltaT(0);
454  if(accMode == DIRECT)
455  {
456  // DIRECT: API access to device buffer
457  for(unsigned int i = 0; i < MEMCOPY_ITERATIONS; i++)
458  {
459  ciErrNum = clEnqueueReadBuffer(cqCommandQueue, cmDevData, CL_FALSE, 0, memSize, h_data, 0, NULL, NULL);
460  ownCheckError(ciErrNum, CL_SUCCESS);
461  }
462  ciErrNum = clFinish(cqCommandQueue);
463  ownCheckError(ciErrNum, CL_SUCCESS);
464  }
465  else
466  {
467  // MAPPED: mapped pointers to device buffer for conventional pointer access
468  void* dm_idata = clEnqueueMapBuffer(cqCommandQueue, cmDevData, CL_TRUE, CL_MAP_WRITE, 0, memSize, 0, NULL, NULL, &ciErrNum);
469  ownCheckError(ciErrNum, CL_SUCCESS);
470  for(unsigned int i = 0; i < MEMCOPY_ITERATIONS; i++)
471  {
472  memcpy(h_data, dm_idata, memSize);
473  }
474  ciErrNum = clEnqueueUnmapMemObject(cqCommandQueue, cmDevData, dm_idata, 0, NULL, NULL);
475  ownCheckError(ciErrNum, CL_SUCCESS);
476  }
477

479  elapsedTimeInSec = shrDeltaT(0);
480 
481  //calculate bandwidth in MB/s
482  bandwidthInMBs = ((double)memSize * (double)MEMCOPY_ITERATIONS) / (elapsedTimeInSec * (double)(1 << 20));
483 
484  //clean up memory
485  if(cmDevData)clReleaseMemObject(cmDevData);
486  if(cmPinnedData)
487  {
488  clEnqueueUnmapMemObject(cqCommandQueue, cmPinnedData, (void*)h_data, 0, NULL, NULL);
489  clReleaseMemObject(cmPinnedData);
490  }
491  h_data = NULL;
492 
493  return bandwidthInMBs;
494 }

620  ciErrNum = clEnqueueWriteBuffer(cqCommandQueue, d_idata, CL_TRUE, 0, memSize, h_idata, 0, NULL, NULL);
621  ownCheckError(ciErrNum, CL_SUCCESS);
622 
623  // Sync queue to host, start timer 0, and copy data from one GPU buffer to another GPU bufffer
624  clFinish(cqCommandQueue);
625  shrDeltaT(0);
626  for(unsigned int i = 0; i < MEMCOPY_ITERATIONS; i++)
627  {
628  ciErrNum = clEnqueueCopyBuffer(cqCommandQueue, d_idata, d_odata, 0, 0, memSize, 0, NULL, NULL);
629  ownCheckError(ciErrNum, CL_SUCCESS);
630  }
631 
632  // Sync with GPU
633  clFinish(cqCommandQueue);
634 
635  //get the the elapsed time in seconds
636  elapsedTimeInSec = shrDeltaT(0);
637 
638  // Calculate bandwidth in MB/s
639  // This is for kernels that read and write GMEM simultaneously
640  // Obtained Throughput for unidirectional block copies will be 1/2 of this #
641  bandwidthInMBs = 2.0 * ((double)memSize * (double)MEMCOPY_ITERATIONS)/(elapsedTimeInSec * (double)(1 << 20));
642 
643  //clean up memory on host and device
644  free(h_idata);
645  clReleaseMemObject(d_idata);
646  clReleaseMemObject(d_odata);
647 
648  return bandwidthInMBs;
