41 class Task
42 {
43  friend class ThreadPool;
44  public:
46  void join() const
47  {
48  // Wait until the tasks is_finished property is set to true by the ThreadPool instance.
49  // Three possible scenarios:
50  // 1) Execution of task has not started yet
51  // - locks task_mutex
52  // - is_finished will be false
53  // - therefore unlocks task_mutex again and waits until signaled by thread pool
54  // - then is_finished will be true (except in case of spurious wakeups for which the waiting continues)
55  // 2) Task is being executed
56  // - will try to lock task_mutex, but will fail
57  // - task_mutex is locked while the task is still running
58  // - is_finished will be true after aquiring the lock
59  // 3) Task has already finished execution
60  // - locks task_mutex
61  // - is_finished will be true
62  std::unique_lock<std::mutex> lock(m_task_mutex);
63  m_finish_cv.wait(lock, [this]{ return m_is_finished; });
64  }
65 
66  private:
67  // Only constructable by friend class ThreadPool
68  Task(std::function<void()> task_func) : m_task_func(task_func) {}
69  Task(Task &) = delete;
70  Task & operator=(Task &) = delete;
71 
72  bool m_is_finished = false;
73  mutable std::condition_variable m_finish_cv;
74  mutable std::mutex m_task_mutex;
75  const std::function<void()> m_task_func;
76 };

102 class ThreadPool {
103 public:
108  ThreadPool(int num_threads)
109  {
110  if (num_threads < 1) { throw std::invalid_argument("Number of threads is zero or negative."); }
111 
112  // Generic function (to be run on a separate thread) within which submitted tasks
113  // are executed. It implements an endless loop (only returning when the ThreadPool
114  // instance itself is destructed) which constantly checks the task queue, grabbing
115  // and executing the frontmost task while the queue is not empty.
116  auto thread_body = [this]{
117  while (true) {
118  // Get next task from queue (synchronized access via taskqueue_mutex).
119  // If the queue is empty wait until either
120  // - being signaled about an available task.
121  // - the terminate flag is true (i.e. the ThreadPool instance is being destructed).
122  // In this case return from the running thread.
123  std::unique_lock<std::mutex> queue_lock(m_taskqueue_mutex);
124  while (m_taskqueue.empty()) {
125  if (m_terminate.load()) { return; }
126  m_task_available_cv.wait(queue_lock);
127  }
128  const auto current_task = m_taskqueue.front();
129  m_taskqueue.pop();
130  queue_lock.unlock();
131 
132  // Execute the actual task and signal the associated Task instance when finished.
133  {
134  std::lock_guard<std::mutex> task_lock(current_task->m_task_mutex);
135  current_task->m_task_func();
136  current_task->m_is_finished = true;
137  }
138  current_task->m_finish_cv.notify_all();
139  }
140  };
141 
142  // Launch the specified number of threads
143  for (int i = 0; i < num_threads; ++i) {
144  m_threads.emplace_back(thread_body);
145  }
146  }
147 
148  ~ThreadPool() {
149  // Indicate termination and signal potential waiting threads to wake up.
150  // Then wait for all threads to finish their work and return properly.
151  m_terminate = true;
152  m_task_available_cv.notify_all();
153  for (auto &t : m_threads) { t.join(); }
154  }

157  std::shared_ptr<Task> RunTask(const std::function<void()> &task_func) {
158  // Wrap provided task callable object in task handle. Then append it to the task queue and
159  // notify a waiting worker thread (if any) about the newly available task
160  auto task = std::shared_ptr<Task>(new Task(task_func));
161  {
162  std::lock_guard<std::mutex> lock(m_taskqueue_mutex);
163  m_taskqueue.push(task);
164  }
165  m_task_available_cv.notify_one();
166 
167  // Return task handle for later synchronization
168  return task;
169  }

172  void Parallelize(const std::vector<std::function<void()>> &task_funcs)
173  {
174  if (task_funcs.empty()) return;
175 
176  // Launch all provided tasks (except for the first) in parallel and store the associated handles
177  auto it = begin(task_funcs);
178  const auto first_task = it++;
179  std::vector<std::shared_ptr<Task>> handles;
180  for(; it != end(task_funcs); ++it)
181  {
182  handles.push_back(RunTask(*it));
183  }
184 
185  // Run the first task locally on the current thread
186  (*first_task)();
187 
188  // Synchronize, i.e. wait for all parallelized tasks to complete
189  for(const auto &h : handles) { h->join(); }
190  }
