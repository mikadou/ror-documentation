244  // Audio Devices
245  int audio_selection = 0;

258  // #### Init OGRE UI entries ####
259 
260  Ogre::RenderSystem* rs_active = Ogre::Root::getSingleton().getRenderSystem();
261  const std::string rs_name = rs_active->getName();
262  const auto rs_list = Ogre::Root::getSingleton().getAvailableRenderers();
263 
264  int rs_index = 0;
265  for (auto itor = rs_list.begin(); itor != rs_list.end(); ++itor)
266  {
267  auto itor_name = (*itor)->getName();
268  m_render_sys->addItem(itor_name);
269  if (itor_name == rs_name)
270  {
271  m_render_sys->setIndexSelected(rs_index);
272  }
273  ++rs_index;
274  }
275 
276  auto render_itor = rs_active->getConfigOptions().begin();
277  auto render_end = rs_active->getConfigOptions().end();
278  for (; render_itor != render_end; ++render_itor)
279  {
280  if (render_itor->first == OGRECFG_VIDEO_MODE)
281  {
282  // Fill 'Video mode' listbox & select active entry
283  auto vid_modes = render_itor->second;
284  int sel_index = 0;
285  for (auto itor = vid_modes.possibleValues.begin(); itor != vid_modes.possibleValues.end(); ++itor)
286  {
287  m_resolution->addItem(*itor);
288  if (*itor == vid_modes.currentValue)
289  {
290  m_resolution->setIndexSelected(sel_index);
291  }
292  ++sel_index;
293  }
294  }
295  else if (render_itor->first == OGRECFG_FSAA)
296  {
297  // Fill 'FSAA' listbox & select active entry
298  auto fsaa_modes = render_itor->second;
299  int sel_index = 0;
300  for (auto itor = fsaa_modes.possibleValues.begin(); itor != fsaa_modes.possibleValues.end(); ++itor)
301  {
302  m_fsaa->addItem(*itor);
303  if (*itor == fsaa_modes.currentValue)
304  {
305  m_fsaa->setIndexSelected(sel_index);
306  }
307  ++sel_index;
308  }
309  }
310  }
311 
312  // #### Final tasks ####
313  m_ogre_cfg.load(App::GetSysConfigDir() + PATH_SLASH + "ogre.cfg");
314  m_is_initialized = true;
315 }
