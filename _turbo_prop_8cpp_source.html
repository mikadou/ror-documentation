247 void Turboprop::updateForces(float dt, int doUpdate)
248 {
249  if (doUpdate)
250  {
251  //tropospheric model valid up to 11.000m (33.000ft)
252  float altitude = nodes[noderef].AbsPosition.y;
253  //float sea_level_temperature=273.15+15.0; //in Kelvin
254  float sea_level_pressure = 101325; //in Pa
255  //float airtemperature=sea_level_temperature-altitude*0.0065; //in Kelvin
256  float airpressure = sea_level_pressure * pow(1.0 - 0.0065 * altitude / 288.15, 5.24947); //in Pa
257  airdensity = airpressure * 0.0000120896;//1.225 at sea level
258 #ifdef USE_OPENAL
259  //sound update
260  SoundScriptManager::getSingleton().modulate(trucknum, mod_id, rpm);
261 #endif //OPENAL
262  }
263 
264  timer += dt;
265  //evaluate the rotation speed
266  float velacc = 0;
267  for (int i = 0; i < numblades; i++)
268  {
269  velacc += (nodes[nodep[i]].Velocity - nodes[noderef].Velocity).length();
270  }
271  rpm = (velacc / numblades) * RAD_PER_SEC_TO_RPM / radius;
272  //check for broken prop
273  Vector3 avg = Vector3::ZERO;
274  for (int i = 0; i < numblades; i++)
275  {
276  avg += nodes[nodep[i]].RelPosition;
277  }
278  avg = avg / numblades;
279  if ((avg - nodes[noderef].RelPosition).length() > 0.4)
280  {
281  failed = true;
282  }
283  //evaluate engine power
284  float enginepower = 0; //in kilo-Watt
285  float warmupfactor = 1.0;
286  if (warmup)
287  {
288  warmupfactor = (timer - warmupstart) / warmuptime;
289  if (warmupfactor >= 1.0)
290  warmup = false;
291  }
292  float revpenalty = 1.0;
293  if (reverse)
294  revpenalty = 0.5;
295  if (!failed && ignition)
296  enginepower = (0.0575 + throtle * revpenalty * 0.9425) * fullpower * warmupfactor;
297  //the magic formula
298  float enginecouple = 0.0; //in N.m
299  if (rpm > 10.0)
300  enginecouple = 9549.3 * enginepower / rpm;
301  else
302  enginecouple = 9549.3 * enginepower / 10.0;
303  indicated_torque = enginecouple;
304 
305  if (torquenode != -1)
306  {
307  Vector3 along = nodes[noderef].RelPosition - nodes[nodeback].RelPosition;
308  Plane ppl = Plane(along, 0);
309  Vector3 orth = ppl.projectVector(nodes[noderef].RelPosition) - ppl.projectVector(nodes[torquenode].RelPosition);
310  Vector3 cdir = orth.crossProduct(along);
311  cdir.normalise();
312  nodes[torquenode].Forces += (enginecouple / torquedist) * cdir;
313  }
314 
315  float tipforce = (enginecouple / radius) / numblades;
316  //okay, now we know the contribution from the engine
317 
318  //pitch
319  if (fixed_pitch > 0)
320  pitch = fixed_pitch;
321  else
322  {
323  if (!reverse)
324  {
325  if (throtle < 0.01)
326  {
327  //beta range
328  if (pitch > 0 && rpm < regspeed * 1.4)
329  pitch -= pitchspeed * dt;
330  if (rpm > regspeed * 1.4)
331  pitch += pitchspeed * dt;
332  }
333  else
334  {
335  float dpitch = rpm - regspeed;
336  if (dpitch > pitchspeed)
337  dpitch = pitchspeed;
338  if (dpitch < -pitchspeed)
339  dpitch = -pitchspeed;
340  if (!(dpitch < 0 && pitch < 0) && !(dpitch > 0 && pitch > 45))
341  pitch += dpitch * dt;
342  }
343  }
344  else
345  {
346  if (rpm < regspeed * 1.1)
347  {
348  if (pitch < -4.0)
349  pitch += pitchspeed * dt;
350  else
351  pitch -= pitchspeed * dt;
352  }
353  if (rpm > regspeed * 1.11)
354  {
355  pitch -= pitchspeed * dt;
356  }
357  }
358  }
359  if (!failed)
360  {
361  axis = nodes[noderef].RelPosition - nodes[nodeback].RelPosition;
362  axis.normalise();
363  }
364  //estimate amount of energy
365  float estrotenergy = 0.5 * numblades * nodes[nodep[0]].mass * radius * radius * (rpm / RAD_PER_SEC_TO_RPM) * (rpm / RAD_PER_SEC_TO_RPM);
366  //for each blade
367  float totthrust = 0;
368  float tottorque = 0;
369  for (int i = 0; i < numblades; i++)
370  {
371  if (!failed && ignition)
372  {
373  Vector3 totaltipforce = Vector3::ZERO;
374  //span vector, left to right
375  Vector3 spanv = (nodes[nodep[i]].RelPosition - nodes[noderef].RelPosition);
376  spanv.normalise();
377  //chord vector, front to back
378  Vector3 refchordv = -axis.crossProduct(spanv);
379  //grab this for propulsive forces
380  Vector3 tipf = -refchordv;
381  totaltipforce += (tipforce - rpm / 10.0) * tipf; //add a bit of mechanical friction
382  //for each blade segment (there are 6 elements)
383  for (int j = 0; j < 5; j++) //outer to inner, the 6th blade element is ignored
384  {
385  //proportion
386  float proport = ((float)j + 0.5) / 6.0;
387  //evaluate wind direction
388  Vector3 wind = -(nodes[nodep[i]].Velocity * (1.0 - proport) + nodes[noderef].Velocity * proport);
389  float wspeed = wind.length();
390 
391  Vector3 liftv = spanv.crossProduct(-wind);
392  liftv.normalise();
393  //rotate according to pitch
394  Vector3 chordv = Quaternion(Degree(pitch + twistmap[j] - 7.0), spanv) * refchordv;
395  //wing normal
396  Vector3 normv = chordv.crossProduct(spanv);
397  //calculate angle of attack
398  Vector3 pwind = Plane(Vector3::ZERO, normv, chordv).projectVector(wind);
399  Vector3 dumb;
400  Degree daoa;
401  chordv.getRotationTo(pwind).ToAngleAxis(daoa, dumb);
402  float aoa = daoa.valueDegrees();
403  float raoa = daoa.valueRadians();
404  if (dumb.dotProduct(spanv) > 0)
405  {
406  aoa = -aoa;
407  raoa = -raoa;
408  };
409  //get airfoil data
410  float cz, cx, cm;
411  airfoil->getparams(aoa, 1.0, 0.0, &cz, &cx, &cm);
412  //surface computation
413  float s = radius * bladewidth / 6.0;
414 
415  //drag
416  //wforce=(8.0*cx+cx*cx/(3.14159*radius/0.4))*0.5*airdensity*wspeed*s*wind;
417  Vector3 eforce = (4.0 * cx + cx * cx / (3.14159 * radius / bladewidth)) * 0.5 * airdensity * wspeed * s * wind;
418  //lift
419  float lift = cz * 0.5 * airdensity * wspeed * wspeed * s;
420  eforce += lift * liftv;
421  totthrust += eforce.dotProduct(axis);
422 
423  //apply forces
424  nodes[noderef].Forces += eforce * proport;
425  totaltipforce += eforce * (1.0 - proport);
426  }
427  tottorque += tipf.dotProduct(totaltipforce) * radius;
428  //correct amount of energy
429  float correctfactor = 0;
430  if (rpm > 100)
431  correctfactor = (rotenergy - estrotenergy) / (numblades * radius * dt * rpm / RAD_PER_SEC_TO_RPM);
432  if (correctfactor > 1000.0)
433  correctfactor = 1000.0;
434  if (correctfactor < -1000.0)
435  correctfactor = -1000.0;
436  nodes[nodep[i]].Forces += totaltipforce + correctfactor * tipf;
437  }
438  else
439  {
440  //failed case
441  //add drag
442  Vector3 wind = -nodes[nodep[i]].Velocity;
443  // determine nodes speed and divide by engines speed (with some magic numbers for tuning) to keep it rotating longer when shutoff in flight and stop after a while when plane is stopped (on the ground)
444  float wspeed = (wind.length() / 15.0f) / (nodes[noderef].Velocity.length() / 2.0f);
445  nodes[nodep[i]].Forces += airdensity * wspeed * wind;
446  }
447  }
448  //compute the next energy level
449  rotenergy += (double)tottorque * dt * rpm / RAD_PER_SEC_TO_RPM;
450  // sprintf(debug, "pitch %i thrust %i totenergy=%i apparentenergy=%i", (int)pitch, (int)totthrust, (int)rotenergy, (int)estrotenergy);
451  //prop wash
452  float speed = nodes[noderef].Velocity.length();
453  float thrsign = 1.0;
454  if (totthrust < 0)
455  {
456  thrsign = -0.1;
457  totthrust = -totthrust;
458  };
459  if (!failed)
460  propwash = thrsign * sqrt(totthrust / (0.5 * airdensity * proparea) + speed * speed) - speed;
461  else
462  propwash = 0;
463  if (propwash < 0)
464  propwash = 0;
465 }
