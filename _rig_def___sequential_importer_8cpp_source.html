55 void SequentialImporter::GenerateNodesForWheel(File::Keyword generated_from, int num_rays, bool has_rigidity_node)
56 {
57  // Old parser logic:
58  // Note: Axis nodes are A/B, where B has bigger Z coordinate.
59  //
60  // Section | Function() | Generated nodes
61  // -------------------------------------------------------------------------------------------
62  // wheels | addWheel() | Tyre nodes: A, B, A, B... (num_rays*2)
63  // wheels2 | addWheel2() | Rim nodes: A, B, A, B... (num_rays*2), Tyre nodes: A, B, A, B... (num_rays*2)
64  // meshwheels | addWheel() | Tyre nodes: A, B, A, B... (num_rays*2)
65  // meshwheels2 | addWheel() | Tyre nodes: A, B, A, B... (num_rays*2)
66  // flexbodywheels | addWheel3() | Rim nodes: A, B, A, B... (num_rays*2), Tyre nodes: A, B, A, B... (num_rays*2)
67 
68  NodeMapEntry::OriginDetail detail;
69  if (generated_from == File::KEYWORD_FLEXBODYWHEELS || generated_from == File::KEYWORD_WHEELS2)
70  {
71  // Rim nodes
72  for (int i = 0; i < num_rays*2; ++i)
73  {
74  detail = (i % 2 == 0) ? NodeMapEntry::DETAIL_WHEEL_RIM_A : NodeMapEntry::DETAIL_WHEEL_RIM_B;
75  this->AddGeneratedNode(generated_from, detail);
76  }
77  }
78  // Tyre nodes
79  for (int i = 0; i < num_rays*2; ++i)
80  {
81  detail = (i % 2 == 0) ? NodeMapEntry::DETAIL_WHEEL_TYRE_A : NodeMapEntry::DETAIL_WHEEL_TYRE_B;
82  this->AddGeneratedNode(generated_from, detail);
83  }
84 }

125 unsigned SequentialImporter::GetNodeArrayOffset(File::Keyword keyword)
126 {
127  // Reverse order of elements in nodes-array
128  // "break" keyword is omitted - control falls through!
129  unsigned out_offset = 0;
130  switch (keyword)
131  {
132  case File::KEYWORD_FLEXBODYWHEELS: out_offset += m_num_meshwheels2_nodes; // NO break!
133  case File::KEYWORD_MESHWHEELS2: out_offset += m_num_meshwheels_nodes; // NO break!
134  case File::KEYWORD_MESHWHEELS: out_offset += m_num_wheels2_nodes; // NO break!
135  case File::KEYWORD_WHEELS2: out_offset += m_num_wheels_nodes; // NO break!
136  case File::KEYWORD_WHEELS: out_offset += m_num_cinecam_nodes; // NO break!
137  case File::KEYWORD_CINECAM: out_offset += m_num_named_nodes; // NO break!
138  case File::KEYWORD_NODES2: out_offset += m_num_numbered_nodes; // NO break!
139  case File::KEYWORD_NODES: break; // Starts at 0
140  default: break;
141  }
142  return out_offset;
143 }

145 Node::Ref SequentialImporter::ResolveNodeByIndex(unsigned int index_in, unsigned int def_line_number)
146 {
147  if (index_in >= m_all_nodes.size())
148  {
149  std::stringstream msg;
150  msg << "Cannot resolve node by index [" << index_in << "], node is not defined, highest available number is: " << m_all_nodes.size() - 1;
151  this->AddMessage(Message::TYPE_ERROR, msg.str());
152  return Node::Ref(); // Invalid reference
153  }
154  NodeMapEntry node_entry = m_all_nodes[index_in];
155  unsigned index_out = this->GetNodeArrayOffset(node_entry.origin_keyword) + node_entry.node_sub_index;
156  if (index_out != index_in)
157  {
158  std::stringstream msg;
159  msg << "Node resolved by index.\n\tSource: [" << index_in << "]\n\tResult: [" << index_out << "]";
160  if (node_entry.origin_keyword != File::KEYWORD_NODES && node_entry.origin_keyword != File::KEYWORD_NODES2)
161  {
162  msg << " (generated by: " << File::KeywordToString(node_entry.origin_keyword)
163  << ", sub-index: " << node_entry.node_sub_index << ")";
164  }
165  this->AddMessage(Message::TYPE_INFO, msg.str());
166  }
167  else
168  {
169  ++m_num_resolved_to_self; // The happy statistic -> nodes which resolved to the same index.
170  }
171  return Node::Ref(TOSTRING(index_out), index_out, Node::Ref::IMPORT_STATE_IS_VALID | Node::Ref::IMPORT_STATE_IS_RESOLVED_NUMBERED, def_line_number);
172 }

174 Node::Ref SequentialImporter::ResolveNode(Node::Ref const & noderef_in)
175 {
176  ++m_total_resolved;
177  if (!noderef_in.GetImportState_IsValid())
178  {
179  return noderef_in;
180  }
181 
182  // Work like legacy function SerializedRig::parse_node_number()
183  // If node may be named, try resolving it as named first
184  // If that fails, try numbered
185 
186  if (noderef_in.GetImportState_MustCheckNamedFirst())
187  {
188  auto result = m_named_nodes.find(noderef_in.Str());
189  if (result != m_named_nodes.end())
190  {
191  return Node::Ref(noderef_in.Str(), 0, Node::Ref::IMPORT_STATE_IS_VALID | Node::Ref::IMPORT_STATE_IS_RESOLVED_NAMED, noderef_in.GetLineNumber());
192  }
193  }
194  if (noderef_in.Num() >= m_all_nodes.size())
195  {
196  // Return exactly what SerializedRig::parse_node_number() would return on this error, for compatibility.
197  // This definitely isn't valid, but it behaves exactly as RoR 0.38, so it's perfectly IMPORT_VALID! :D
198  Node::Ref out_ref("0", 0, Node::Ref::IMPORT_STATE_IS_VALID | Node::Ref::IMPORT_STATE_IS_RESOLVED_NUMBERED, noderef_in.GetLineNumber());
199  std::stringstream msg;
200  msg << "Cannot resolve " << noderef_in.ToString() << " - not a named node, and index is not defined (highest is: "
201  << m_all_nodes.size() - 1 << "). For backwards compatibility, converting to: " << out_ref.ToString();
202  this->AddMessage(Message::TYPE_ERROR, msg.str());
203  return out_ref; // Invalid
204  }
205  auto entry = m_all_nodes[noderef_in.Num()];
206  if (entry.node_id.IsTypeNamed())
207  {
208  Node::Ref out_ref(entry.node_id.Str(), 0, Node::Ref::IMPORT_STATE_IS_VALID | Node::Ref::IMPORT_STATE_IS_RESOLVED_NAMED, noderef_in.GetLineNumber());
209  std::stringstream msg;
210  msg << "Node resolved\n\tSource: " << noderef_in.ToString() << "\n\tResult: " << out_ref.ToString();
211  this->AddMessage(Message::TYPE_INFO, msg.str());
212  return out_ref;
213  }
214  else if (entry.node_id.IsTypeNumbered())
215  {
216  unsigned out_index = this->GetNodeArrayOffset(entry.origin_keyword) + entry.node_sub_index;
217  Node::Ref out_ref(TOSTRING(out_index), out_index, Node::Ref::IMPORT_STATE_IS_VALID | Node::Ref::IMPORT_STATE_IS_RESOLVED_NUMBERED, noderef_in.GetLineNumber());
218  std::stringstream msg;
219  msg << "Node resolved\n\tSource: " << noderef_in.ToString() << "\n\tResult: " << out_ref.ToString()
220  << "\n\tOrigin: " << RigDef::File::KeywordToString(entry.origin_keyword) << " SubIndex: " << entry.node_sub_index;
221  this->AddMessage(Message::TYPE_INFO, msg.str());
222  return out_ref;
223  }
224  else
225  {
226  std::stringstream msg;
227  msg << "Cannot resolve " << noderef_in.ToString() << " - found node is not valid";
228  this->AddMessage(Message::TYPE_ERROR, msg.str());
229  return Node::Ref(); // Invalid
230  }

233 void SequentialImporter::ResolveFlexbodyForset(std::vector<Node::Range>& in_ranges, std::vector<Node::Ref>& out_nodes)
234 {
235  auto range_itor = in_ranges.begin();
236  auto range_end = in_ranges.end();
237  for ( ; range_itor != range_end; ++range_itor)
238  {
239  Node::Range& range = *range_itor;
240  if (!range.IsRange())
241  {
242  if (!range.start.GetImportState_IsValid())
243  {
244  std::stringstream msg;
245  msg << "ResolveFlexbodyForset(): Skipping node because it's marked INVALID:" << range.start.ToString();
246  this->AddMessage(Message::TYPE_ERROR, msg.str());
247  }
248  else
249  {
250  Node::Ref node_ref = this->ResolveNodeByIndex(range.start.Num(), range.start.GetLineNumber()); // Forset nodes are numbered-only
251  // Invalid nodes are simply thrown away, for backwards compatibility
252  // (FlexBody loops through existing nodes first, and then evaluates if they are in the SET)
253  if (node_ref.IsValidAnyState())
254  {
255  out_nodes.push_back(node_ref);
256  }
257  else
258  {
259  std::stringstream msg;
260  msg << "ResolveFlexbodyForset(): Stand-alone node [" << range.start.ToString() << "] resolved invalid, removing from FORSET";
261  this->AddMessage(Message::TYPE_WARNING, msg.str());
262  }
263  }
264  }
265  else // It's a range
266  {
267  if (!range.start.GetImportState_IsValid() || !range.end.GetImportState_IsValid())
268  {
269  std::stringstream msg;
270  msg << "ResolveFlexbodyForset(): Skipping range because some nodes are marked INVALID, start: [" << range.start.ToString() << "], end: [" << range.end.ToString() << "]";
271  this->AddMessage(Message::TYPE_ERROR, msg.str());
272  }
273  else if (range.start.GetImportState_IsResolvedNamed() || range.end.GetImportState_IsResolvedNamed())
274  {
275  std::stringstream msg;
276  msg << "ResolveFlexbodyForset(): Some nodes in range are already resolved as named, unable to process, start: [" << range.start.ToString() << "], end: [" << range.end.ToString() << "]";
277  this->AddMessage(Message::TYPE_ERROR, msg.str());
278  }
279  else
280  {
281  unsigned int end_index = range.end.Num();
282  unsigned int line_num = range.start.GetLineNumber();
283  for (unsigned int i = range.start.Num(); i <= end_index; ++i)
284  {
285  Node::Ref node_ref = this->ResolveNodeByIndex(i, line_num); // Forset nodes are numbered-only
286  // Invalid nodes are simply thrown away, for backwards compatibility
287  // (FlexBody loops through existing nodes first, and then evaluates if they are in the SET -> invalid nodes are silently ignored)
288  if (node_ref.IsValidAnyState())
289  {
290  out_nodes.push_back(node_ref);
291  }
292  else
293  {
294  std::stringstream msg;
295  msg << "ResolveFlexbodyForset(): Node ["<<i<<"] from range [" << range.start.ToString() << " - " << range.end.ToString() << "] resolved invalid, removing from FORSET";
296  this->AddMessage(Message::TYPE_WARNING, msg.str());
297  }
298  }
299  }
300  }
301  }

304 void SequentialImporter::ResolveNodeRanges(std::vector<Node::Range>& ranges)
305 {
306  auto in_ranges = ranges; // Copy vector
307  ranges.clear();
308 
309  auto range_itor = in_ranges.begin();
310  auto range_end = in_ranges.end();
311  for ( ; range_itor != range_end; ++range_itor)
312  {
313  Node::Range& range = *range_itor;
314  if (!range.IsRange())
315  {
316  ranges.push_back(Node::Range(this->ResolveNode(range.start)));
317  }
318  else if (!range.start.GetImportState_IsValid() || !range.end.GetImportState_IsValid())
319  {
320  std::stringstream msg;
321  msg << "Some nodes in range are invalid, start: [" << range.start.ToString() << "], end: [" << range.end.ToString() << "]";
322  this->AddMessage(Message::TYPE_ERROR, msg.str());
323  }
324  else if (range.start.GetImportState_IsResolvedNamed() || range.end.GetImportState_IsResolvedNamed())
325  {
326  std::stringstream msg;
327  msg << "Some nodes in range are already resolved as named, unable to process, start: [" << range.start.ToString() << "], end: [" << range.end.ToString() << "]";
328  this->AddMessage(Message::TYPE_ERROR, msg.str());
329  }
330  else
331  {
332  unsigned int end_index = range.end.Num();
333  for (unsigned int i = range.start.Num(); i < end_index; ++i)
334  {
335  ranges.push_back(Node::Range(this->ResolveNodeByIndex(i, range.start.GetLineNumber())));
336  }
337  }
338  }
