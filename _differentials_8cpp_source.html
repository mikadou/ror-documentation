117 void Axle::calcOpenDiff(differential_data_t& diff_data)
118 {
119  /* Open differential calculations *************************
120  * These calculation are surprisingly tricky
121  * the power ratio is based on normalizing the speed of the
122  * wheel. Normalizing is when the sum of all values equals
123  * one. more detail provided below
124  */
125 
126  // velocity at which open diff calculations are used 100%
127  // this value is twice the speed of the actual threshold
128  // the factor of 2 comes from finding the average speed and
129  // is done so to remove unessecary calculations
130  const Ogre::Real threshold_vel = 10.0f;
131 
132  // combined total velocity
133  const Ogre::Real sum_of_vel =
134  fabs(diff_data.speed[0]) + fabs(diff_data.speed[1]);
135 
136  // normalizing the wheel speeds the wheels to sputter at near 0
137  // speeds, to over come this we gradually transition from even
138  // power distribution to splitting power based on speed. The
139  // Transition ratio defines how much of each formula we take from,
140  // Until a threshold has been reached
141  const Ogre::Real transition_ratio = (sum_of_vel / threshold_vel < 1) ?
142  sum_of_vel / threshold_vel : 1;
143 
144  // normalize the wheel speed, at a speed of 0 power is split evenly
145  const Ogre::Real power_ratio = (fabs(sum_of_vel) > 0.0) ?
146  fabs(diff_data.speed[0]) / sum_of_vel : 0.5;
147 
148  diff_data.out_torque[0] = diff_data.out_torque[1] = diff_data.in_torque;
149 
150  // Diff model taken from Torcs, ror needs to model reaction torque for this to work.
151  //const Ogre::Real spider_acc = (diff_data.speed[0] - diff_data.speed[1])/diff_data.dt;
152  //DrTq0 = DrTq*0.5f + spiderTq;
153  //DrTq1 = DrTq*0.5f - spiderTq;
154 
155  // get the final ratio based on the speed of the wheels
156  diff_data.out_torque[0] *= 2 * (transition_ratio * power_ratio + (1 - transition_ratio) * 0.5f);
157  diff_data.out_torque[1] *= 2 * (transition_ratio * (1 - power_ratio) + (1 - transition_ratio) * 0.5f);
158 
159  diff_data.delta_rotation = 0.0f;
160 }

162 void Axle::calcLockedDiff(differential_data_t& diff_data)
163 {
164  /* Locked axle calculation ********************************
165  * This is straight forward, two wheels are joined together
166  * by a torsion spring. the torsion spring keeps the two
167  * wheels in the same orientation as when they were first
168  * locked.
169  */
170 
171  // Torsion spring rate that holds axles together when locked
172  // keep as variable for now since this value will be user configurable
173  const Ogre::Real torsion_rate = 1000000.0f;
174  const Ogre::Real torsion_damp = torsion_rate / 100.0f;
175  const Ogre::Real delta_speed = diff_data.speed[0] - diff_data.speed[1];
176 
177  diff_data.out_torque[0] = diff_data.out_torque[1] = diff_data.in_torque;
178 
179  // derive how far wheels traveled relative to each during the last time step
180  diff_data.delta_rotation += (delta_speed) * diff_data.dt;
181 
182  // torque cause by axle shafts
183  diff_data.out_torque[0] -= diff_data.delta_rotation * torsion_rate;
184  // damping
185  diff_data.out_torque[0] -= (delta_speed) * torsion_damp;
186 
187  // torque cause by axle shafts
188  diff_data.out_torque[1] += diff_data.delta_rotation * torsion_rate;
189  // damping
190  diff_data.out_torque[1] -= -(delta_speed) * torsion_damp;
191 }
