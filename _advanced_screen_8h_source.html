68  void write()

79  long isize = mWidth * mHeight * (long)Ogre::PixelUtil::getNumElemBytes(pf);
80  Ogre::uchar* data = OGRE_ALLOC_T(Ogre::uchar, isize, Ogre::MEMCATEGORY_RENDERSYS);

84  // now do the fancy stuff ;)
85 
86  // 0. allocate enough buffer
87  Ogre::uchar* databuf = OGRE_ALLOC_T(Ogre::uchar, 32768, Ogre::MEMCATEGORY_RENDERSYS);
88  char* ptr = (char *)databuf;
89  // header
90  long dsize = 0;
91  int w = sprintf(ptr, "RORED\n");
92  ptr += w;
93  dsize += w;
94 
95  // 1. convert the std::map to something properly
96  for (std::map<Ogre::String, Ogre::String>::iterator it = map.begin(); it != map.end(); it++)
97  {
98  int w2 = sprintf(ptr, "%s:%s\n", it->first.c_str(), it->second.c_str());
99  ptr += w2;
100  dsize += w2;
101  }
102 
103  // now buffer ready, put it into the image
104  Ogre::uchar* ptri = data;
105  // set data pointer to the start again
106  ptr = (char *)databuf;
107  int bc = 7;
108  for (long b = 0; b < isize && b < dsize * 8 + 40; b++ , ptri++) // 40 = 5 zero bytes
109  {
110  if (CHECK_BIT(*ptr, bc))
111  SET_LSB(*ptri);
112  else
113  CLEAR_LSB(*ptri);
114  bc--;
115  if (bc < 0)
116  {
117  bc = 7;
118  ptr++;
119  }
120  }
121 
122  std::thread(save, data, databuf, mWidth, mHeight, pf, filename).detach();
123  }
