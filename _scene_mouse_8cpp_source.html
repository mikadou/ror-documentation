112 bool SceneMouse::mouseMoved(const OIS::MouseEvent& _arg)
113 {
114  const OIS::MouseState ms = _arg.state;
115 
116  // check if handled by the camera
117  if (!gEnv->cameraManager || gEnv->cameraManager->mouseMoved(_arg))
118  return true;
119 
120  // experimental mouse hack
121  if (ms.buttonDown(OIS::MB_Left) && mouseGrabState == 0)
122  {
123  lastMouseY = ms.Y.abs;
124  lastMouseX = ms.X.abs;
125 
126  Ray mouseRay = getMouseRay();
127 
128  // walk all trucks
129  Beam** trucks = m_sim_controller->GetBeamFactory()->getTrucks();
130  int trucksnum = m_sim_controller->GetBeamFactory()->getTruckCount();
131  minnode = -1;
132  grab_truck = NULL;
133  for (int i = 0; i < trucksnum; i++)
134  {
135  if (trucks[i] && trucks[i]->state == SIMULATED)
136  {
137  // check if our ray intersects with the bounding box of the truck
138  std::pair<bool, Real> pair = mouseRay.intersects(trucks[i]->boundingBox);
139  if (!pair.first)
140  continue;
141 
142  for (int j = 0; j < trucks[i]->free_node; j++)
143  {
144  if (trucks[i]->nodes[j].no_mouse_grab)
145  continue;
146 
147  // check if our ray intersects with the node
148  std::pair<bool, Real> pair = mouseRay.intersects(Sphere(trucks[i]->nodes[j].AbsPosition, 0.1f));
149  if (pair.first)
150  {
151  // we hit it, check if its the nearest node
152  if (pair.second < mindist)
153  {
154  mindist = pair.second;
155  minnode = j;
156  grab_truck = trucks[i];
157  break;
158  }
159  }
160  }
161  }
162 
163  if (grab_truck)
164  break;
165  }
166 
167  // check if we hit a node
168  if (grab_truck && minnode != -1)
169  {
170  mouseGrabState = 1;
171  pickLineNode->setVisible(true);
172 
173  for (std::vector<hook_t>::iterator it = grab_truck->hooks.begin(); it != grab_truck->hooks.end(); it++)
174  {
175  if (it->hookNode->id == minnode)
176  {
177  grab_truck->hookToggle(it->group, MOUSE_HOOK_TOGGLE, minnode);
178  }
179  }
180  }
181  }
182  else if (ms.buttonDown(OIS::MB_Left) && mouseGrabState == 1)
183  {
184  // force applying and so forth happens in update()
185  lastMouseY = ms.Y.abs;
186  lastMouseX = ms.X.abs;
187  // not fixed
188  return false;
189  }
190  else if (!ms.buttonDown(OIS::MB_Left) && mouseGrabState == 1)
191  {
192  releaseMousePick();
193  // not fixed
194  return false;
195  }
196 
197  return false;
198 }

221 bool SceneMouse::mousePressed(const OIS::MouseEvent& _arg, OIS::MouseButtonID _id)
222 {
223  if (App::GetActiveSimState() == App::SIM_STATE_PAUSED) { return true; } // Do nothing when paused
224 
225  const OIS::MouseState ms = _arg.state;
226 
227  if (ms.buttonDown(OIS::MB_Middle))
228  {
229  if (gEnv->cameraManager && gEnv->cameraManager->getCurrentBehavior() == CameraManager::CAMERA_BEHAVIOR_VEHICLE)
230  {
231  Beam* truck = m_sim_controller->GetBeamFactory()->getCurrentTruck();
232 
233  if (truck)
234  {
235  lastMouseY = ms.Y.abs;
236  lastMouseX = ms.X.abs;
237  Ray mouseRay = getMouseRay();
238 
239  Real nearest_camera_distance = std::numeric_limits<float>::max();
240  Real nearest_ray_distance = std::numeric_limits<float>::max();
241  int nearest_node_index = -1;
242 
243  for (int i = 0; i < truck->free_node; i++)
244  {
245  std::pair<bool, Real> pair = mouseRay.intersects(Sphere(truck->nodes[i].AbsPosition, 0.25f));
246  if (pair.first)
247  {
248  Real ray_distance = mouseRay.getDirection().crossProduct(truck->nodes[i].AbsPosition - mouseRay.getOrigin()).length();
249  if (ray_distance < nearest_ray_distance || (ray_distance == nearest_ray_distance && pair.second < nearest_camera_distance))
250  {
251  nearest_camera_distance = pair.second;
252  nearest_ray_distance = ray_distance;
253  nearest_node_index = i;
254  }
255  }
256  }
257  if (truck->m_custom_camera_node != nearest_node_index)
258  {
259  truck->m_custom_camera_node = nearest_node_index;
260  truck->calculateAveragePosition();
261  gEnv->cameraManager->NotifyContextChange();
262  }
263  }
264  }
265  }
266 
267  if (gEnv->cameraManager)
268  {
269  gEnv->cameraManager->mousePressed(_arg, _id);
270  }
271 
272  return true;
273 }
