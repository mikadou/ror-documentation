137 void CSHA1::UpdateHash(uint8_t *data, uint32_t len)
138 {
139  uint32_t i, j;
140 
141  j = (m_count[0] >> 3) & 63;
142 
143  if ((m_count[0] += len << 3) < (len << 3)) m_count[1]++;
144 
145  m_count[1] += (len >> 29);
146 
147  if ((j + len) > 63)
148  {
149  i = 64 - j;
150  memcpy(&m_buffer[j], data, i);
151  Transform(m_state, m_buffer);
152 
153  for ( ; i + 63 < len; i += 64) Transform(m_state, &data[i]);
154 
155  j = 0;
156  }
157  else i = 0;
158 
159  memcpy(&m_buffer[j], &data[i], len - i);
160 }

208 void CSHA1::Final()
209 {
210  uint32_t i;
211  uint8_t finalcount[8];
212 
213  for (i = 0; i < 8; i++)
214  finalcount[i] = (uint8_t)((m_count[((i >= 4) ? 0 : 1)]
215  >> ((3 - (i & 3)) * 8) ) & 255); // Endian independent
216 
217  UpdateHash((uint8_t *)"\200", 1);
218 
219  while ((m_count[0] & 504) != 448)
220  UpdateHash((uint8_t *)"\0", 1);
221 
222  UpdateHash(finalcount, 8); // Cause a SHA1Transform()
223 
224  for (i = 0; i < 20; i++)
225  {
226  m_digest[i] = (uint8_t)((m_state[i >> 2] >> ((3 - (i & 3)) * 8) ) & 255);
227  }
228 
229  // Wipe variables for security reasons
230 #ifdef SHA1_WIPE_VARIABLES
231  i = 0;
232  memset(m_buffer, 0, 64);
233  memset(m_state, 0, 20);
234  memset(m_count, 0, 8);
235  memset(finalcount, 0, 8);
236  Transform(m_state, m_buffer);
237 #endif
238 }
