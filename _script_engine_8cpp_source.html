825  int result=0;
826

828  // search for #include directives, and load any included files as

830  OgreScriptBuilder builder;
831 
832  AngelScript::asIScriptModule *mod = 0;
833  // try to load bytecode
834  bool cached = false;
835

837  {

843  return result;
844  }
845

847 
848  result = builder.AddSectionFromFile(scriptName.c_str());
849  if ( result < 0 )

852  SLOG("Failed to add script file");

854  }

856  if ( result < 0 )
857  {
858  SLOG("Failed to build the module");
859  return result;
860  }
861

863  scriptHash = builder.getHash();
864  {
865  String filepath = App::GetSysCacheDir() + PATH_SLASH + "script" + scriptHash + "_" + scriptName + "c";
866  SLOG("saving script bytecode to file " + filepath);
867  CBytecodeStream bstream(filepath);
868  mod->SaveByteCode(&bstream);
869  }
870  }
871 
872  // get some other optional functions
873  frameStepFunctionPtr = mod->GetFunctionByDecl("void frameStep(float)");
874  if (frameStepFunctionPtr != nullptr)
875  callbacks["frameStep"].push_back(frameStepFunctionPtr);

898  // expected function in the script.
