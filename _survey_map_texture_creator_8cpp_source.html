88 void SurveyMapTextureCreator::update()
89 {
90  if (!mRttTex)
91  return;
92 
93  mMapSize = Vector3::ZERO;
94  mMapCenter = Vector2::ZERO;

96

99 
100  if (gEnv->surveyMap)

102  mMapCenter = gEnv->surveyMap->getMapCenter();
103  mMapZoom = gEnv->surveyMap->getMapZoom();
104  }
105 
106  float orthoWindowWidth = mMapSize.x - (mMapSize.x - 20.0f) * mMapZoom;
107  float orthoWindowHeight = mMapSize.z - (mMapSize.z - 20.0f) * mMapZoom;
108 
109  mCamera->setFarClipDistance(mMapSize.y + 3.0f);
110  mCamera->setOrthoWindow(orthoWindowWidth, orthoWindowHeight);
111  mCamera->setPosition(Vector3(mMapCenter.x, mMapSize.y + 2.0f, mMapCenter.y));
112  mCamera->lookAt(Vector3(mMapCenter.x, 0.0f, mMapCenter.y));
113 
114  preRenderTargetUpdate();
115 
116  mRttTex->update();
117 
118  postRenderTargetUpdate();
119 }
