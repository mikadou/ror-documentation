35 bool Terrn2Parser::LoadTerrn2(Terrn2Def& def, Ogre::DataStreamPtr &ds)
36 {
37  RoR::ConfigFile file;
38  file.load(ds, "\t:=", true);
39 
40  // read in the settings
41  def.name = file.GetStringEx("Name", "General");
42  if (def.name.empty())
43  {
44  this->AddMessage("FATAL: Terrain name is empty");
45  return false;
46  }
47 
48  def.ogre_ter_conf_filename = file.GetStringEx("GeometryConfig", "General");
49  // otc = ogre terrain config
50  if (def.ogre_ter_conf_filename.find(".otc") == String::npos)
51  {
52  this->AddMessage("FATAL: Invalid geometry config file; only '.otc' is supported");
53  return false;
54  }
55 
56  def.ambient_color = file.GetColourValue("AmbientColor", "General", ColourValue::White);
57  def.category_id = file.GetInt ("CategoryID", "General", 129);
58  def.guid = file.GetStringEx ("GUID", "General");
59  def.version = file.GetInt ("Version", "General", 1);
60  def.gravity = file.GetFloat ("Gravity", "General", -9.81);
61  def.caelum_config = file.GetStringEx ("CaelumConfigFile", "General");
62  def.cubemap_config = file.GetStringEx ("SandStormCubeMap", "General");
63  def.caelum_fog_start = file.GetInt ("CaelumFogStart", "General", -1);
64  def.caelum_fog_end = file.GetInt ("CaelumFogEnd", "General", -1);
65  def.has_water = file.GetBool ("Water", "General", false);
66  def.hydrax_conf_file = file.GetStringEx ("HydraxConfigFile", "General");
67  def.traction_map_file = file.GetStringEx ("TractionMap", "General");
68  def.water_height = file.GetFloat ("WaterLine", "General");
69  def.water_bottom_height = file.GetFloat ("WaterBottomLine", "General");
70 
71  def.start_position = StringConverter::parseVector3(file.GetStringEx("StartPosition", "General"), Vector3(512.0f, 0.0f, 512.0f));
72 
73  try
74  {
75  auto itor = file.getSettingsIterator("Authors");
76  while (itor.hasMoreElements())
77  {
78  String type = RoR::Utils::SanitizeUtf8String(itor.peekNextKey()); // e.g. terrain
79  String name = RoR::Utils::SanitizeUtf8String(itor.peekNextValue()); // e.g. john doe
80 
81  if (!name.empty())
82  {
83  Terrn2Author author;
84  author.type = type;
85  author.name = name;
86  def.authors.push_back(author);
87  }
88 
89  itor.moveNext();
90  }
91  }
92  catch (Ogre::Exception) {} // Thrown if section 'Authors' doesn't exist
93 
94  try
95  {
96  auto itor = file.getSettingsIterator("Objects");
97  while (itor.hasMoreElements())
98  {
99  Ogre::String tobj_filename = itor.peekNextKey();
100  Ogre::StringUtil::trim(tobj_filename);
101  def.tobj_files.push_back(tobj_filename);
102  itor.moveNext();
103  }
104  }
105  catch (Ogre::Exception) {} // Thrown if section 'Objects' doesn't exist
106 
107  try
108  {
109  auto itor = file.getSettingsIterator("Scripts");
110  while (itor.hasMoreElements())
111  {
112  Ogre::String as_filename = itor.peekNextKey();
113  Ogre::StringUtil::trim(as_filename);
114  def.as_files.push_back(as_filename);
115  itor.moveNext();
116  }
117  }
118  catch (Ogre::Exception) {} // Thrown if section 'Scripts' doesn't exist
119 
120  this->ProcessTeleport(def, &file);
121 
122  return true;
123 }

125 void Terrn2Parser::ProcessTeleport(Terrn2Def& def, RoR::ConfigFile* file)
126 {
127  size_t telepoint_number = 1;
128  for (;;)
129  {
130  char key_position [50];
131  char key_name [50];
132 
133  snprintf(key_position, 50, "Telepoint%d/Position" , telepoint_number);
134  snprintf(key_name, 50, "Telepoint%d/Name" , telepoint_number);
135 
136  std::string pos_str = file->GetStringEx(key_position, "Teleport", VALUE_NOT_FOUND);
137  if (pos_str == VALUE_NOT_FOUND)
138  {
139  break; // No more telepoints
140  }
141  Terrn2Telepoint t_point;
142  if (sscanf(pos_str.c_str(), "%f, %f, %f", &t_point.position.x, &t_point.position.y, &t_point.position.z) != 3)
143  {
144  char msg_buf[500];
145  snprintf(msg_buf, 500,
146  "ERROR: Field '[Teleport]/%s' ('%s') is not valid XYZ position. Skipping telepoint %d.",
147  key_position, pos_str.c_str(), telepoint_number);
148  this->AddMessage(msg_buf);
149  }
150  else
151  {
152  t_point.name = file->GetStringEx(key_name, "Teleport"); // Optional field
153  def.telepoints.push_back(t_point); // Persist the entry
154  }
155 
156  ++telepoint_number;
157  }
158 }
