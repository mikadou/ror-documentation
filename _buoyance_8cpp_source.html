50 //compute pressure and drag force on a submerged triangle
51 Vector3 Buoyance::computePressureForceSub(Vector3 a, Vector3 b, Vector3 c, Vector3 vel, int type)
52 {
53  //compute normal vector
54  Vector3 normal = (b - a).crossProduct(c - a);
55  float surf = normal.length();
56  if (surf < 0.00001)
57  return Vector3::ZERO;
58  normal = normal / surf; //normalize
59  surf = surf / 2.0; //surface
60  float vol = 0.0;
61  if (type != BUOY_DRAGONLY)
62  {
63  //compute pression prism points
64  Vector3 ap = a + (gEnv->terrainManager->getWater()->getHeightWaves(a) - a.y) * 9810 * normal;
65  Vector3 bp = b + (gEnv->terrainManager->getWater()->getHeightWaves(b) - b.y) * 9810 * normal;
66  Vector3 cp = c + (gEnv->terrainManager->getWater()->getHeightWaves(c) - c.y) * 9810 * normal;
67  //find centroid
68  Vector3 ctd = (a + b + c + ap + bp + cp) / 6.0;
69  //compute volume
70  vol += computeVolume(ctd, a, b, c);
71  vol += computeVolume(ctd, a, ap, bp);
72  vol += computeVolume(ctd, a, bp, b);
73  vol += computeVolume(ctd, b, bp, cp);
74  vol += computeVolume(ctd, b, cp, c);
75  vol += computeVolume(ctd, c, cp, ap);
76  vol += computeVolume(ctd, c, ap, a);
77  vol += computeVolume(ctd, ap, cp, bp);
78  };
79  Vector3 drg = Vector3::ZERO;
80  if (type != BUOY_DRAGLESS)
81  {
82  //now, the drag
83  //take in account the wave speed
84  //compute center
85  Vector3 tc = (a + b + c) / 3.0;
86  vel = vel - gEnv->terrainManager->getWater()->getVelocity(tc);
87  float vell = vel.length();
88  if (vell > 0.01)
89  {
90  float cosaoa = fabs(normal.dotProduct(vel / vell));
91  // drg=(-500.0*surf*vell*cosaoa)*vel;
92  drg = (-500.0 * surf * vell * vell * cosaoa) * normal;
93  if (normal.dotProduct(vel / vell) < 0)
94  drg = -drg;
95  if (update && splashp)
96  {
97  float fxl = vell * cosaoa * surf;
98  if (fxl > 1.5) //if enough pushing drag
99  {
100  Vector3 fxdir = fxl * normal;
101  if (fxdir.y < 0)
102  fxdir.y = -fxdir.y;
103  if (gEnv->terrainManager->getWater()->getHeightWaves(a) - a.y < 0.1)
104  splashp->malloc(a, fxdir);
105  else if (gEnv->terrainManager->getWater()->getHeightWaves(b) - b.y < 0.1)
106  splashp->malloc(b, fxdir);
107  else if (gEnv->terrainManager->getWater()->getHeightWaves(c) - c.y < 0.1)
108  splashp->malloc(c, fxdir);
109  }
110  }
111  }
112  }
113  //okay
114  if (sink)
115  return drg;
116  return vol * normal + drg;
117 }

119 //compute pressure and drag forces on a random triangle
120 Vector3 Buoyance::computePressureForce(Vector3 a, Vector3 b, Vector3 c, Vector3 vel, int type)
121 {
122  float wha = gEnv->terrainManager->getWater()->getHeightWaves((a + b + c) / 3.0);
123  //check if fully emerged
124  if (a.y > wha && b.y > wha && c.y > wha)
125  return Vector3::ZERO;
126  //check if semi emerged
127  if (a.y > wha || b.y > wha || c.y > wha)
128  {
129  //okay, several cases
130  //one dip
131  if (a.y < wha && b.y > wha && c.y > wha)
132  {
133  return computePressureForceSub(a, a + (wha - a.y) / (b.y - a.y) * (b - a), a + (wha - a.y) / (c.y - a.y) * (c - a), vel, type);
134  }
135  if (b.y < wha && c.y > wha && a.y > wha)
136  {
137  return computePressureForceSub(b, b + (wha - b.y) / (c.y - b.y) * (c - b), b + (wha - b.y) / (a.y - b.y) * (a - b), vel, type);
138  }
139  if (c.y < wha && a.y > wha && b.y > wha)
140  {
141  return computePressureForceSub(c, c + (wha - c.y) / (a.y - c.y) * (a - c), c + (wha - c.y) / (b.y - c.y) * (b - c), vel, type);
142  }
143  //two dips
144  if (a.y > wha && b.y < wha && c.y < wha)
145  {
146  Vector3 tb = a + (wha - a.y) / (b.y - a.y) * (b - a);
147  Vector3 tc = a + (wha - a.y) / (c.y - a.y) * (c - a);
148  Vector3 f = computePressureForceSub(tb, b, tc, vel, type);
149  return f + computePressureForceSub(tc, b, c, vel, type);
150  }
151  if (b.y > wha && c.y < wha && a.y < wha)
152  {
153  Vector3 tc = b + (wha - b.y) / (c.y - b.y) * (c - b);
154  Vector3 ta = b + (wha - b.y) / (a.y - b.y) * (a - b);
155  Vector3 f = computePressureForceSub(tc, c, ta, vel, type);
156  return f + computePressureForceSub(ta, c, a, vel, type);
157  }
158  if (c.y > wha && a.y < wha && b.y < wha)
159  {
160  Vector3 ta = c + (wha - c.y) / (a.y - c.y) * (a - c);
161  Vector3 tb = c + (wha - c.y) / (b.y - c.y) * (b - c);
162  Vector3 f = computePressureForceSub(ta, a, tb, vel, type);
163  return f + computePressureForceSub(tb, a, b, vel, type);
164  }
165  return Vector3::ZERO;
166  }
167  else
168  {
169  //fully submerged case
170  return computePressureForceSub(a, b, c, vel, type);
171  }
172 }
