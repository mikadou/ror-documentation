38  // Create submeshes
39  m_submeshes.reserve(submesh_defs.size());
40  for (size_t j=0; j<submesh_defs.size(); j++)
41  {
42  Ogre::SubMesh* submesh = m_mesh->createSubMesh();
43  switch (submesh_defs[j].backmesh_type)
44  {
45  case CabSubmesh::BACKMESH_OPAQUE: submesh->setMaterialName(backtexname); break;
46  case CabSubmesh::BACKMESH_TRANSPARENT: submesh->setMaterialName(transtexname); break;
47  default: submesh->setMaterialName(texname);
48  }
49  m_submeshes.push_back(submesh);

52  // Define the m_vertices_raw (8 vertices, each consisting of 3 groups of 3 floats
53  m_vertex_count = texcoords.size();
54  m_vertices_raw=(float*)malloc(((2*3+2)*m_vertex_count)*sizeof(float));
55  m_vertex_nodes=(int*)malloc(m_vertex_count*sizeof(int));

58  {
59  m_vertex_nodes[i] = texcoords[i].node_id; //define node ids
60  m_vertices[i].texcoord=Vector2(texcoords[i].texcoord_u, texcoords[i].texcoord_v); //textures coordinates
61  }

63  // Define triangles
64  // The values in this table refer to vertices in the above table
65  m_index_count = 3*numtriangles;
66  m_indices=(unsigned short*)malloc(m_index_count*sizeof(unsigned short));
67  for (size_t i=0; i<m_index_count; i++)
68  {
69  m_indices[i]=ComputeVertexPos(i/3, triangles[i], submesh_defs);
70  }
71 
72  m_s_ref=(float*)malloc(numtriangles*sizeof(float));
73 
74  for (size_t i=0; i<(unsigned int)numtriangles;i++)
75  {
76  Ogre::Vector3 base_pos = m_all_nodes[m_vertex_nodes[m_indices[i*3]]].RelPosition;
77  Ogre::Vector3 v1 = m_all_nodes[m_vertex_nodes[m_indices[i*3+1]]].RelPosition - base_pos;
78  Ogre::Vector3 v2 = m_all_nodes[m_vertex_nodes[m_indices[i*3+2]]].RelPosition - base_pos;

112  {

115  index_count = 3*submesh_defs[j].cabs_pos;

134  m_submeshes[j]->indexData->indexCount = index_count;
135  m_submeshes[j]->indexData->indexStart = 0;
136  }
137 
138  // Set bounding information (for culling)
139  m_mesh->_setBounds(AxisAlignedBox(-100,-100,-100,100,100,100), true);
140 
141  // Notify Mesh object that it has been loaded
142  m_mesh->load();
143 }

154 int FlexObj::ComputeVertexPos(int tidx, int v, std::vector<CabSubmesh>& submeshes)
155 {
156  // --- NOTE: The following logic is arcane, I'm keeping it mostly like I found it for stability ~ only_a_ptr, 08/2017
157 
158  // Find the 'context' - a submesh (and respective span of vertices) which the vertex belong to.
159  int context;
160  int num_submeshes = static_cast<int>(submeshes.size());
161  for (context = 0; context < num_submeshes; ++context)
162  {
163  if (tidx < submeshes[context].cabs_pos)
164  {
165  --context;
166  break;
167  }
168  }
169 
170  // Find the vertex itself
171  int i_min = (context < 0) ? 0 : submeshes[context].texcoords_pos; // Fix for single-submesh case... It seems to have worked with a negative index until now ~ only_a_ptr, 08/2017
172  int i_max = submeshes[context + 1].texcoords_pos;
173  for (int i = i_min; i < i_max; ++i)
174  {
175  if (m_vertex_nodes[i] == v)
176  return i;
177  }
178 
179  return 0;
180 }

182 Vector3 FlexObj::UpdateMesh()
183 {
184  Ogre::Vector3 center=(m_all_nodes[m_vertex_nodes[0]].AbsPosition+m_all_nodes[m_vertex_nodes[1]].AbsPosition)/2.0;
185  for (size_t i=0; i<m_vertex_count; i++)
186  {
187  //set position
188  m_vertices[i].position=m_all_nodes[m_vertex_nodes[i]].AbsPosition-center;
189  //reset normals
190  m_vertices[i].normal=Vector3::ZERO;
191  }
192  //accumulate normals per triangle
193  for (size_t i=0; i<m_index_count/3; i++)
194  {
195  Vector3 v1, v2;
196  v1=m_all_nodes[m_vertex_nodes[m_indices[i*3+1]]].AbsPosition-m_all_nodes[m_vertex_nodes[m_indices[i*3]]].AbsPosition;
197  v2=m_all_nodes[m_vertex_nodes[m_indices[i*3+2]]].AbsPosition-m_all_nodes[m_vertex_nodes[m_indices[i*3]]].AbsPosition;
198  v1=v1.crossProduct(v2);
199  float s=v1.length();
200 
201  //avoid large tris
202  if (s>m_s_ref[i])
203  {
204  m_vertices[m_indices[i*3+1]].position=m_vertices[m_indices[i*3]].position+Vector3(0.1,0,0);
205  m_vertices[m_indices[i*3+2]].position=m_vertices[m_indices[i*3]].position+Vector3(0,0,0.1);
206  }
207 
208  if (s == 0)
209  continue;
210 
211  v1=v1/s;
212  m_vertices[m_indices[i*3]].normal+=v1;
213  m_vertices[m_indices[i*3+1]].normal+=v1;
214  m_vertices[m_indices[i*3+2]].normal+=v1;
215  }
216  //normalize
217  for (size_t i=0; i<m_vertex_count; i++)
218  {
219  m_vertices[i].normal = approx_normalise(m_vertices[i].normal);
220  }
221 
222  return center;
223 }
