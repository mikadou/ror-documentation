68 void OutProtocol::startup()
69 {
70 #if defined(_WIN32) && defined(USE_SOCKETW)
71  SWBaseSocket::SWBaseError error;
72 
73  // startup winsock
74  WSADATA wsd;
75  if (WSAStartup(MAKEWORD(2, 2), &wsd) != 0)
76  {
77  LOG("[RoR|OutGauge] Error starting up winsock. OutGauge disabled.");
78  return;
79  }
80 
81  // open a new socket
82  if ((sockfd = socket(AF_INET, SOCK_DGRAM, 0)) < 0)
83  {
84  LOG(String("[RoR|OutGauge] Error creating socket for OutGauge: ").append(strerror(errno)));
85  return;
86  }
87 
88  // get the IP of the remote side, this function is compatible with windows 2000
89  hostent* remoteHost = gethostbyname(App::GetIoOutGaugeIp().c_str());
90  char* ip = inet_ntoa(*(struct in_addr *)*remoteHost->h_addr_list);
91 
92  // init socket data
93  struct sockaddr_in sendaddr;
94  memset(&sendaddr, 0, sizeof(sendaddr));
95  sendaddr.sin_family = AF_INET;
96  sendaddr.sin_addr.s_addr = inet_addr(ip);
97  sendaddr.sin_port = htons(App::GetIoOutGaugePort());
98 
99  // connect
100  if (connect(sockfd, (struct sockaddr *) &sendaddr, sizeof(sendaddr)) == SOCKET_ERROR)
101  {
102  LOG(String("[RoR|OutGauge] Error connecting socket for OutGauge: ").append(strerror(errno)));
103  return;
104  }
105 
106  LOG("[RoR|OutGauge] Connected successfully");
107  working = true;
108 #else
109  // TODO: fix linux
110 #endif // _WIN32
111 }

197  gd.Clutch = 1 - truck->engine->getClutch(); // 0-1
198
