98 unsigned int getNumberOfCPUCores()
99 {
100 #if defined(_WIN32) || defined(__x86_64__) || defined(__i386)
101  unsigned regs[4];
102 
103  // Get CPU vendor
104  char vendor[12];
105  cpuID(0, regs);
106  memcpy((void *)vendor, (void *)&regs[1], 4); // EBX
107  memcpy((void *)&vendor[4], (void *)&regs[3], 4); // EDX
108  memcpy((void *)&vendor[8], (void *)&regs[2], 4); // ECX
109  std::string cpuVendor = std::string(vendor, 12);
110 
111  // Get CPU features
112  cpuID(1, regs);
113  unsigned cpuFeatures = regs[3]; // EDX
114 
115  // Logical core count per CPU
116  cpuID(1, regs);
117  unsigned logical = (regs[1] >> 16) & 0xff; // EBX[23:16]
118  unsigned cores = logical;
119 
120  if (cpuVendor == "GenuineIntel")
121  {
122  // Get DCP cache info
123  cpuID(4, regs);
124  cores = ((regs[0] >> 26) & 0x3f) + 1; // EAX[31:26] + 1
125  }
126  else if (cpuVendor == "AuthenticAMD")
127  {
128  // Get NC: Number of CPU cores - 1
129  cpuID(0x80000008, regs);
130  cores = ((unsigned)(regs[2] & 0xff)) + 1; // ECX[7:0] + 1
131  }
132 
133  // Detect hyper-threads
134  bool hyperThreads = cpuFeatures & (1 << 28) && cores < logical;
135 
136  LOG("BEAMFACTORY: " + TOSTRING(logical) + " logical CPU cores" + " found");
137  LOG("BEAMFACTORY: " + TOSTRING(cores) + " CPU cores" + " found");
138  LOG("BEAMFACTORY: Hyper-Threading " + TOSTRING(hyperThreads));
139 #else
140  unsigned cores = hardware_concurrency();
141  LOG("BEAMFACTORY: " + TOSTRING(cores) + " CPU cores" + " found");
142 #endif
143  return cores;
144 }

654  {
655  current_truck->sleeptime = 0.0f;
656  this->RecursiveActivation(m_current_truck, visited);
657  }
658  // Snowball effect (activate all trucks which might soon get hit by a moving truck)
659  for (int t = 0; t < m_free_truck; t++)

1023 {

1025 
1026  // do not allow dt > 1/20
1027  dt = std::min(dt, 1.0f / 20.0f);
1028 
1029  dt *= m_simulation_speed;
1030 
1031  dt += m_dt_remainder;
1032  m_physics_steps = dt / PHYSICS_DT;
1033  m_dt_remainder = dt - (m_physics_steps * PHYSICS_DT);
1034  dt = PHYSICS_DT * m_physics_steps;
1035 
1036  gEnv->mrTime += dt;
1037 
1038  this->SyncWithSimThread();
1039 
1040  this->UpdateSleepingState(dt);
1041 
1042  for (int t = 0; t < m_free_truck; t++)
1043  {
1044  if (!m_trucks[t])
1045  continue;
1046 
1047  m_trucks[t]->handleResetRequests(dt);
1048  m_trucks[t]->updateAngelScriptEvents(dt);
1049 
1050 #ifdef USE_ANGELSCRIPT
1051  if (m_trucks[t]->vehicle_ai && (m_trucks[t]->vehicle_ai->IsActive()))
1052  m_trucks[t]->vehicle_ai->update(dt, 0);
1053 #endif // USE_ANGELSCRIPT
1054 
1055  switch (m_trucks[t]->state)
1056  {
1057  case NETWORKED:

1060 
1061  case INVALID:
1062  break;
1063 
1064  default:
1065  if (m_trucks[t]->state != SIMULATED && m_trucks[t]->engine)
1066  m_trucks[t]->engine->update(dt, 1);
1067  if (m_trucks[t]->state < SLEEPING)
1068  m_trucks[t]->UpdatePropAnimations(dt);
1069  if (m_trucks[t]->networking)
1070  {
1071  if (m_trucks[t]->state == SIMULATED)
1072  m_trucks[t]->sendStreamData();
1073  else if (m_trucks[t]->state == SLEEPING && m_trucks[t]->netTimer.getMilliseconds() < 10000)
1074  // Also send update messages for 'SLEEPING' trucks during the first 10 seconds of lifetime
1075  m_trucks[t]->sendStreamData();
1076  else if (m_trucks[t]->state == SLEEPING && m_trucks[t]->netTimer.getMilliseconds() - m_trucks[t]->lastNetUpdateTime > 5000)
1077  // Also send update messages for 'SLEEPING' trucks periodically every 5 seconds
1078  m_trucks[t]->sendStreamData();
1079  }
1080  break;
1081  }
1082  }
1083 
1084  m_simulated_truck = m_current_truck;
1085 
1086  if (m_simulated_truck == -1)
1087  {
1088  for (int t = 0; t < m_free_truck; t++)
1089  {
1090  if (m_trucks[t] && m_trucks[t]->state == SIMULATED)
1091  {
1092  m_simulated_truck = t;
1093  break;
1094  }
1095  }
1096  }

1099  {
1100  if (m_simulated_truck == m_current_truck)
1101  {
1102 
1103  m_trucks[m_simulated_truck]->updateDashBoards(dt);
1104 
1105 #ifdef FEAT_TIMING
1106  if (m_trucks[m_simulated_truck]->statistics) m_trucks[m_simulated_truck]->statistics->frameStep(dt);
