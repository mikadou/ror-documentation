940 {

942  int nBestMatchLen = 0; // matching characters of best match
943  int nLastMatchLen = 0; // matching characters of last best match
944 
945  for (int n = 0; m_rgOptions[n].nId >= 0; ++n)
946  {
947  // the option table must use hyphens as the option character,
948  // the slash character is converted to a hyphen for testing.
949  SO_ASSERT(m_rgOptions[n].pszArg[0] != (SOCHAR)'/');

952  if (nMatchLen == -1)

954  return n;
955  }
956  if (nMatchLen > 0 && nMatchLen >= nBestMatchLen)
957  {
958  nLastMatchLen = nBestMatchLen;
959  nBestMatchLen = nMatchLen;
960  nBestMatch = n;
961  }
962  }

965  // don't return a partial match unless it is a clear winner
966  if (HasFlag(SO_O_EXACT) || nBestMatch == -1)

975 template <class SOCHAR>
976 int
977 CSimpleOptTempl<SOCHAR>::CalcMatch(
978  const SOCHAR* a_pszSource,
979  const SOCHAR* a_pszTest
980 ) const
981 {
982  if (!a_pszSource || !a_pszTest)
983  {
984  return 0;
985  }
986 
987  // determine the argument type
988  int nArgType = SO_O_ICASE_LONG;
989  if (a_pszSource[0] != '-')
990  {
991  nArgType = SO_O_ICASE_WORD;
992  }
993  else if (a_pszSource[1] != '-' && !a_pszSource[2])
994  {
995  nArgType = SO_O_ICASE_SHORT;
996  }
997 
998  // match and skip leading hyphens
999  while (*a_pszSource == (SOCHAR)'-' && *a_pszSource == *a_pszTest)
1000  {
1001  ++a_pszSource;
1002  ++a_pszTest;
1003  }
1004  if (*a_pszSource == (SOCHAR)'-' || *a_pszTest == (SOCHAR)'-')
1005  {
1006  return 0;
1007  }
1008 
1009  // find matching number of characters in the strings
1010  int nLen = 0;
1011  while (*a_pszSource && IsEqual(*a_pszSource, *a_pszTest, nArgType))
1012  {
1013  ++a_pszSource;
1014  ++a_pszTest;
1015  ++nLen;
1016  }
1017 
1018  // if we have exhausted the source...
1019  if (!*a_pszSource)
1020  {
1021  // and the test strings, then it's a perfect match
1022  if (!*a_pszTest)
1023  {
1024  return -1;
1025  }
1026 
1027  // otherwise the match failed as the test is longer than
1028  // the source. i.e. "--mant" will not match the option "--man".
1029  return 0;
1030  }
1031 
1032  // if we haven't exhausted the test string then it is not a match
1033  // i.e. "--mantle" will not best-fit match to "--mandate" at all.
1034  if (*a_pszTest)
1035  {
1036  return 0;
1037  }
1038 
1039  // partial match to the current length of the test string
1040  return nLen;
1041 }
