745 template <class SOCHAR>
746 int
747 CSimpleGlobTempl<SOCHAR>::Add(
748  const SOCHAR* a_pszFileSpec
749 )
750 {
751 #ifdef _WIN32
752  // Windows FindFirst/FindNext recognizes forward slash as the same as
753  // backward slash and follows the directories. We need to do the same
754  // when calculating the prefix and when we have no wildcards.
755  SOCHAR szFileSpec[MAX_PATH];
756  SimpleGlobUtil::strcpy_s(szFileSpec, MAX_PATH, a_pszFileSpec);
757  const SOCHAR* pszPath = SimpleGlobUtil::strchr(szFileSpec, '/');
758  while (pszPath)
759  {
760  szFileSpec[pszPath - szFileSpec] = SG_PATH_CHAR;
761  pszPath = SimpleGlobUtil::strchr(pszPath + 1, '/');
762  }
763  a_pszFileSpec = szFileSpec;
764 #endif
765 
766  // if this doesn't contain wildcards then we can just add it directly
767  m_szPathPrefix[0] = 0;
768  if (!SimpleGlobUtil::strchr(a_pszFileSpec, '*') &&
769  !SimpleGlobUtil::strchr(a_pszFileSpec, '?'))
770  {
771  SG_FileType nType = GetFileTypeS(a_pszFileSpec);
772  if (nType == SG_FILETYPE_INVALID)
773  {
774  if (m_uiFlags & SG_GLOB_NOCHECK)
775  {
776  return AppendName(a_pszFileSpec, false);
777  }
778  return SG_ERR_NOMATCH;
779  }
780  return AppendName(a_pszFileSpec, nType == SG_FILETYPE_DIR);
781  }
782 
783 #ifdef _WIN32
784  // Windows doesn't return the directory with the filename, so we need to
785  // extract the path from the search string ourselves and prefix it to the
786  // filename we get back.
787  const SOCHAR* pszFilename =
788  SimpleGlobUtil::strrchr(a_pszFileSpec, SG_PATH_CHAR);
789  if (pszFilename)
790  {
791  SimpleGlobUtil::strcpy_s(m_szPathPrefix, MAX_PATH, a_pszFileSpec);
792  m_szPathPrefix[pszFilename - a_pszFileSpec + 1] = 0;
793  }
794 #endif
795 
796  // search for the first match on the file
797  int rc = FindFirstFileS(a_pszFileSpec, m_uiFlags);
798  if (rc != SG_SUCCESS)
799  {
800  if (rc == SG_ERR_NOMATCH && (m_uiFlags & SG_GLOB_NOCHECK))
801  {
802  int ok = AppendName(a_pszFileSpec, false);
803  if (ok != SG_SUCCESS)
804  rc = ok;
805  }
806  return rc;
807  }
808 
809  // add it and find all subsequent matches
810  int nError, nStartLen = m_nArgsLen;
811  bool bSuccess;
812  do
813  {
814  nError = AppendName(GetFileNameS((SOCHAR)0), IsDirS((SOCHAR)0));
815  bSuccess = FindNextFileS((SOCHAR)0);
816  }
817  while (nError == SG_SUCCESS && bSuccess);
818  SimpleGlobBase<SOCHAR>::FindDone();
819 
820  // sort these files if required
821  if (m_nArgsLen > nStartLen && !(m_uiFlags & SG_GLOB_NOSORT))
822  {
823  if (m_uiFlags & SG_GLOB_FULLSORT)
824  {
825  nStartLen = m_nReservedSlots;
826  }
827  SetArgvArrayType(POINTERS);
828  qsort(
829  m_rgpArgs + nStartLen,
830  m_nArgsLen - nStartLen,
831  sizeof(m_rgpArgs[0]), fileSortCompare);
832  }
833 
834  return nError;
835 }

856 template <class SOCHAR>
857 int
858 CSimpleGlobTempl<SOCHAR>::AppendName(
859  const SOCHAR* a_pszFileName,
860  bool a_bIsDir
861 )
862 {
863  // we need the argv array as offsets in case we resize it
864  SetArgvArrayType(OFFSETS);
865 
866  // check for special cases which cause us to ignore this entry
867  if ((m_uiFlags & SG_GLOB_ONLYDIR) && !a_bIsDir)
868  {
869  return SG_SUCCESS;
870  }
871  if ((m_uiFlags & SG_GLOB_ONLYFILE) && a_bIsDir)
872  {
873  return SG_SUCCESS;
874  }
875  if ((m_uiFlags & SG_GLOB_NODOT) && a_bIsDir)
876  {
877  if (a_pszFileName[0] == '.')
878  {
879  if (a_pszFileName[1] == '\0')
880  {
881  return SG_SUCCESS;
882  }
883  if (a_pszFileName[1] == '.' && a_pszFileName[2] == '\0')
884  {
885  return SG_SUCCESS;
886  }
887  }
888  }
889 
890  // ensure that we have enough room in the argv array
891  if (!GrowArgvArray(m_nArgsLen + 1))
892  {
893  return SG_ERR_MEMORY;
894  }
895 
896  // ensure that we have enough room in the string buffer (+1 for null)
