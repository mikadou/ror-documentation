278 float calculate_collision_depth(Vector3 pos)
279 {
280  Vector3 query = pos + Vector3::UNIT_Y;
281  while (query.y > pos.y)
282  {
283  if (gEnv->collisions->collisionCorrect(&query, false))
284  break;
285  query.y -= 0.01f;
286  }
287  return query.y - pos.y;

290 void Character::update(float dt)
291 {
292  if (physicsEnabled && !remote)
293  {
294  // disable character movement when using the free camera mode or when the menu is opened
295  // TODO: check for menu being opened
296  if (gEnv->cameraManager && gEnv->cameraManager->gameControlsLocked())
297  return;
298 
299  Vector3 position = mCharacterNode->getPosition();
300 
301  // gravity force is always on
302  position.y += characterVSpeed * dt;
303  characterVSpeed += dt * -9.8f;
304 
305  // Auto compensate minor height differences
306  float depth = calculate_collision_depth(position);
307  if (depth > 0.0f)
308  {
309  characterVSpeed = std::max(0.0f, characterVSpeed);
310  canJump = true;
311  if (depth < 0.3f)
312  {
313  position.y += depth;
314  if (depth > 0.01f && gEnv->cameraManager)
315  {
316  gEnv->cameraManager->NotifyContextChange();
317  }
318  }
319  }
320 
321  // Trigger script events and handle mesh (ground) collision
322  {
323  Vector3 query = position;
324  gEnv->collisions->collisionCorrect(&query);
325  if (std::abs(position.y - query.y) > 0.1f && gEnv->cameraManager)
326  {
327  gEnv->cameraManager->NotifyContextChange();
328  }
329  position.y = query.y;
330  }
331 
332  // Obstacle detection
333  if (mLastPosition.size() > 0)
334  {
335  Vector3 lastPosition = mLastPosition.front();
336  Vector3 diff = mCharacterNode->getPosition() - lastPosition;
337  Vector3 h_diff = Vector3(diff.x, 0.0f, diff.z);
338  if (depth <= 0.0f || h_diff.squaredLength() > 0.0f)
339  {
340  const int numstep = 100;
341  Vector3 base = lastPosition + Vector3::UNIT_Y * 0.5f;
342  for (int i = 1; i < numstep; i++)
343  {
344  Vector3 query = base + diff * ((float)i / numstep);
345  if (gEnv->collisions->collisionCorrect(&query, false))
346  {
347  position = lastPosition + diff * ((float)(i - 1) / numstep);;
348  break;
349  }
350  }
351  }
352  }
353 
354  mLastPosition.push_front(position);
355 
356  if (mLastPosition.size() > 10)
357  {
358  mLastPosition.pop_back();
359  }
360 
361  // ground contact
362  float pheight = gEnv->terrainManager->getHeightFinder()->getHeightAt(position.x, position.z);
363 
364  if (position.y < pheight)
365  {
366  position.y = pheight;
367  characterVSpeed = 0.0f;
368  canJump = true;
369  }
370 
371  // water stuff
372  bool isswimming = false;
373  float wheight = -99999;
374 
375  if (gEnv->terrainManager->getWater())
376  {
377  wheight = gEnv->terrainManager->getWater()->getHeightWaves(position);
378  if (position.y < wheight - 1.8f)
379  {
380  position.y = wheight - 1.8f;
381  characterVSpeed = 0.0f;
382  }
383  }
384 
385  // 0.1 due to 'jumping' from waves -> not nice looking
386  if (gEnv->terrainManager->getWater() && (wheight - pheight > 1.8f) && (position.y + 0.1f <= wheight))
387  {
388  isswimming = true;
389  }
390 
391  float tmpJoy = 0.0f;
392  if (canJump)
393  {
394  if (RoR::App::GetInputEngine()->getEventBoolValue(EV_CHARACTER_JUMP))
395  {
396  characterVSpeed = 2.0f;
397  canJump = false;
398  }
399  }
400 
401  bool idleanim = true;
402 
403  tmpJoy = RoR::App::GetInputEngine()->getEventValue(EV_CHARACTER_RIGHT);
404  if (tmpJoy > 0.0f)
405  {
406  float scale = RoR::App::GetInputEngine()->isKeyDown(OIS::KC_LMENU) ? 0.1f : 1.0f;
407  setRotation(characterRotation + dt * 2.0f * scale * Radian(tmpJoy));
408  if (!isswimming)
409  {
410  setAnimationMode("Turn", -dt);
411  idleanim = false;
412  }
413  }
414 
415  tmpJoy = RoR::App::GetInputEngine()->getEventValue(EV_CHARACTER_LEFT);
416  if (tmpJoy > 0.0f)
417  {
418  float scale = RoR::App::GetInputEngine()->isKeyDown(OIS::KC_LMENU) ? 0.1f : 1.0f;
419  setRotation(characterRotation - dt * scale * 2.0f * Radian(tmpJoy));
420  if (!isswimming)
421  {
422  setAnimationMode("Turn", dt);
423  idleanim = false;
424  }
425  }
426 
427  float tmpRun = RoR::App::GetInputEngine()->getEventValue(EV_CHARACTER_RUN);
428  float accel = 1.0f;
429 
430  tmpJoy = accel = RoR::App::GetInputEngine()->getEventValue(EV_CHARACTER_SIDESTEP_LEFT);
431  if (tmpJoy > 0.0f)
432  {
433  if (tmpRun > 0.0f)
434  accel = 3.0f * tmpRun;
435  // animation missing for that
436  position += dt * characterSpeed * 0.5f * accel * Vector3(cos(characterRotation.valueRadians() - Math::HALF_PI), 0.0f, sin(characterRotation.valueRadians() - Math::HALF_PI));
437  if (!isswimming)
438  {
439  setAnimationMode("Side_step", -dt);
440  idleanim = false;
441  }
442  }
443 
444  tmpJoy = accel = RoR::App::GetInputEngine()->getEventValue(EV_CHARACTER_SIDESTEP_RIGHT);
445  if (tmpJoy > 0.0f)
446  {
447  if (tmpRun > 0.0f)
448  accel = 3.0f * tmpRun;
449  // animation missing for that
450  position += dt * characterSpeed * 0.5f * accel * Vector3(cos(characterRotation.valueRadians() + Math::HALF_PI), 0.0f, sin(characterRotation.valueRadians() + Math::HALF_PI));
451  if (!isswimming)
452  {
453  setAnimationMode("Side_step", dt);
454  idleanim = false;
455  }
456  }
457 
458  tmpJoy = accel = RoR::App::GetInputEngine()->getEventValue(EV_CHARACTER_FORWARD) + RoR::App::GetInputEngine()->getEventValue(EV_CHARACTER_ROT_UP);
459  float tmpBack = RoR::App::GetInputEngine()->getEventValue(EV_CHARACTER_BACKWARDS) + RoR::App::GetInputEngine()->getEventValue(EV_CHARACTER_ROT_DOWN);
460 
461  tmpJoy = std::min(tmpJoy, 1.0f);
462  tmpBack = std::min(tmpBack, 1.0f);
463 
464  if (tmpJoy > 0.0f || tmpRun > 0.0f)
465  {
466  if (tmpRun > 0.0f)
467  accel = 3.0f * tmpRun;
468 
469  float time = dt * tmpJoy * characterSpeed;
470 
471  if (isswimming)
472  {
473  setAnimationMode("Swim_loop", time);
474  idleanim = false;
475  }
476  else
477  {
478  if (tmpRun > 0.0f)
479  {
480  setAnimationMode("Run", time);
481  idleanim = false;
482  }
483  else
484  {
485  setAnimationMode("Walk", time);
486  idleanim = false;
487  }
488  }
489  // 0.005f fixes character getting stuck on meshes
490  position += dt * characterSpeed * 1.5f * accel * Vector3(cos(characterRotation.valueRadians()), 0.01f, sin(characterRotation.valueRadians()));
491  }
492  else if (tmpBack > 0.0f)
493  {
494  float time = -dt * characterSpeed;
495  if (isswimming)
496  {
497  setAnimationMode("Spot_swim", time);
498  idleanim = false;
499  }
500  else
501  {
502  setAnimationMode("Walk", time);
503  idleanim = false;
504  }
505  // 0.005f fixes character getting stuck on meshes
506  position -= dt * characterSpeed * tmpBack * Vector3(cos(characterRotation.valueRadians()), 0.01f, sin(characterRotation.valueRadians()));
507  }
508 
509  if (idleanim)
510  {
511  if (isswimming)
512  {
513  setAnimationMode("Spot_swim", dt * 2.0f);
514  }
515  else
516  {
517  setAnimationMode("Idle_sway", dt * 1.0f);
518  }
519  }
520 
521  mCharacterNode->setPosition(position);
522  updateMapIcon();
523  }
524  else if (beamCoupling && beamCoupling->hasDriverSeat()) // beamCoupling = The vehicle or machine which the character occupies
525  {
526  Vector3 pos;
527  Quaternion rot;
528  beamCoupling->calculateDriverPos(pos, rot);
529  float angle = beamCoupling->hydrodirwheeldisplay * -1.0f; // not getSteeringAngle(), but this, as its smoothed
530  mCharacterNode->setOrientation(rot);
531  setPosition(pos + (rot * Vector3(0.f, -0.6f, 0.f))); // hack to position the character right perfect on the default seat
532 
533  // Animation
534  this->setAnimationMode("Driving");
535  Real anim_length = mAnimState->getAnimationState("Driving")->getLength();
536  float anim_time_pos = ((angle + 1.0f) * 0.5f) * anim_length;
537  // prevent animation flickering on the borders:
538  if (anim_time_pos < 0.01f)
539  {
540  anim_time_pos = 0.01f;
541  }
542  if (anim_time_pos > anim_length - 0.01f)
543  {
544  anim_time_pos = anim_length - 0.01f;
545  }
546  mAnimState->getAnimationState("Driving")->setTimePosition(anim_time_pos);
547  }
548 
549 #ifdef USE_SOCKETW
550  if ((App::GetActiveMpState() == App::MP_STATE_CONNECTED) && !remote)
551  {
552  sendStreamData();
553  }
554 #endif // USE_SOCKETW
555 }

657 void Character::receiveStreamData(unsigned int& type, int& source, unsigned int& streamid, char* buffer)
658 {
659 #ifdef USE_SOCKETW
660  if (type == RoRnet::MSG2_STREAM_DATA && m_source_id == source && m_stream_id == streamid)
661  {
662  auto* msg = reinterpret_cast<Networking::CharacterMsgGeneric*>(buffer);
663  if (msg->command == Networking::CHARACTER_CMD_POSITION)
664  {
665  auto* pos_msg = reinterpret_cast<Networking::CharacterMsgPos*>(buffer);
666  this->setPosition(Ogre::Vector3(pos_msg->pos_x, pos_msg->pos_y, pos_msg->pos_z));
667  this->setRotation(Ogre::Radian(pos_msg->rot_angle));
668  this->setAnimationMode(Utils::SanitizeUtf8String(pos_msg->anim_name), pos_msg->anim_time);
669  }
670  else if (msg->command == Networking::CHARACTER_CMD_DETACH)
671  {
672  if (beamCoupling != nullptr)
673  this->setBeamCoupling(false);
674  else
675  this->ReportError("Received command `DETACH`, but not currently attached to a vehicle. Ignoring command.");
676  }
677  else if (msg->command == Networking::CHARACTER_CMD_ATTACH)
678  {
679  auto* attach_msg = reinterpret_cast<Networking::CharacterMsgAttach*>(buffer);
680  Beam* beam = m_sim_controller->GetBeamFactory()->getBeam(attach_msg->source_id, attach_msg->stream_id);
681  if (beam != nullptr)
682  {
683  this->setBeamCoupling(true, beam);
684  }
685  else
686  {
687  char err_buf[200];
688  snprintf(err_buf, 200, "Received command `ATTACH` with target{SourceID: %d, StreamID: %d}, "
689  "but corresponding vehicle doesn't exist. Ignoring command.",
690  attach_msg->source_id, attach_msg->stream_id);
691  this->ReportError(err_buf);
692  }
693  }
694  else
695  {
696  char err_buf[100];
697  snprintf(err_buf, 100, "Received invalid command: %d. Cannot process.", msg->command);
698  this->ReportError(err_buf);
699  }
700  }
701 #endif
702 }
