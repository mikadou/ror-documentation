36 void LandVehicleSimulation::UpdateCruiseControl(Beam* curr_truck, float dt)
37 {
38  if ((curr_truck->engine->getGear() > 0 && RoR::App::GetInputEngine()->getEventValue(EV_TRUCK_BRAKE) > 0.05f) ||
39  (curr_truck->engine->getGear() > 0 && RoR::App::GetInputEngine()->getEventValue(EV_TRUCK_MANUAL_CLUTCH) > 0.05f) ||
40  (curr_truck->engine->getGear() > 0 && curr_truck->parkingbrake) ||
41  (curr_truck->engine->getGear() < 0) ||
42  (curr_truck->cc_target_speed < curr_truck->cc_target_speed_lower_limit) ||
43  !curr_truck->engine->isRunning() ||
44  !curr_truck->engine->hasContact())
45  {
46  curr_truck->cruisecontrolToggle();
47  return;
48  }
49 
50  float acc = curr_truck->engine->getAccToHoldRPM(curr_truck->engine->getRPM());
51 
52  if (curr_truck->engine->getGear() > 0)
53  {
54  // Try to maintain the target speed
55  float torque = (curr_truck->engine->getEngineTorque() + curr_truck->engine->getBrakingTorque()) * 0.8f;
56  float forceRatio = curr_truck->getTotalMass(true) / torque;
57  acc += (curr_truck->cc_target_speed - curr_truck->WheelSpeed) * forceRatio * 0.25;
58  acc = std::max(-2.0f, acc);
59  acc = std::min(acc, +2.0f);
60  }
61  else if (curr_truck->engine->getGear() == 0) // out of gear
62  {
63  // Try to maintain the target rpm
64  float rpmRatio = 1.0f / (curr_truck->engine->getMaxRPM() - curr_truck->engine->getMinRPM());
65  acc += (curr_truck->cc_target_rpm - curr_truck->engine->getRPM()) * rpmRatio * 2.0f;
66  }
67 
68  curr_truck->cc_accs.push_front(acc);
69 
70  float avgAcc = 0.0f;
71 
72  for (unsigned int i = 0; i < curr_truck->cc_accs.size(); i++)
73  {
74  avgAcc += curr_truck->cc_accs[i];
75  }
76 
77  if (curr_truck->cc_accs.size() > 10)
78  {
79  curr_truck->cc_accs.pop_back();
80  }
81 
82  avgAcc /= curr_truck->cc_accs.size();
83 
84  float accl = avgAcc;
85  accl = std::max(curr_truck->engine->getAcc(), accl);
86  accl = std::min(accl, 1.0f);
87  curr_truck->engine->autoSetAcc(accl);
88 
89  if (RoR::App::GetInputEngine()->getEventBoolValue(EV_TRUCK_CRUISE_CONTROL_ACCL))
90  {
91  if (curr_truck->engine->getGear() > 0)
92  {
93  curr_truck->cc_target_speed += 2.5f * dt;
94  curr_truck->cc_target_speed = std::max(curr_truck->cc_target_speed_lower_limit, curr_truck->cc_target_speed);
95  if (curr_truck->sl_enabled)
96  {
97  curr_truck->cc_target_speed = std::min(curr_truck->cc_target_speed, curr_truck->sl_speed_limit);
98  }
99  }
100  else if (curr_truck->engine->getGear() == 0) // out of gear
101  {
102  curr_truck->cc_target_rpm += 1000.0f * dt;
103  curr_truck->cc_target_rpm = std::min(curr_truck->cc_target_rpm, curr_truck->engine->getMaxRPM());
104  }
105  }
106  if (RoR::App::GetInputEngine()->getEventBoolValue(EV_TRUCK_CRUISE_CONTROL_DECL))
107  {
108  if (curr_truck->engine->getGear() > 0)
109  {
110  curr_truck->cc_target_speed -= 2.5f * dt;
111  curr_truck->cc_target_speed = std::max(curr_truck->cc_target_speed_lower_limit, curr_truck->cc_target_speed);
112  }
113  else if (curr_truck->engine->getGear() == 0) // out of gear
114  {
115  curr_truck->cc_target_rpm -= 1000.0f * dt;
116  curr_truck->cc_target_rpm = std::max(curr_truck->engine->getMinRPM(), curr_truck->cc_target_rpm);
117  }
118  }
119  if (RoR::App::GetInputEngine()->getEventBoolValue(EV_TRUCK_CRUISE_CONTROL_READJUST))
120  {
121  curr_truck->cc_target_speed = std::max(curr_truck->WheelSpeed, curr_truck->cc_target_speed);
122  if (curr_truck->sl_enabled)
123  {
124  curr_truck->cc_target_speed = std::min(curr_truck->cc_target_speed, curr_truck->sl_speed_limit);
125  }
126  curr_truck->cc_target_rpm = curr_truck->engine->getRPM();
127  }
128 
129  if (curr_truck->cc_can_brake)
130  {
131  if (curr_truck->WheelSpeed > curr_truck->cc_target_speed + 0.5f && !RoR::App::GetInputEngine()->getEventValue(EV_TRUCK_ACCELERATE))
132  {
133  float brake = (curr_truck->WheelSpeed - curr_truck->cc_target_speed) * 0.5f;
134  brake = std::min(brake, 1.0f);
135  curr_truck->brake = curr_truck->brakeforce * brake;
136  }
137  }

140 void LandVehicleSimulation::CheckSpeedLimit(Beam* curr_truck, float dt)
141 {
142  if (curr_truck->sl_enabled && curr_truck->engine->getGear() != 0)
143  {
144  float accl = (curr_truck->sl_speed_limit - std::abs(curr_truck->WheelSpeed)) * 2.0f;
145  accl = std::max(0.0f, accl);
146  accl = std::min(accl, curr_truck->engine->getAcc());
147  curr_truck->engine->setAcc(accl);
148  }
