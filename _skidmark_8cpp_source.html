40 void SkidmarkConfig::loadDefaultModels()
41 {
42  LOG("[RoR] Loading skidmarks.cfg...");
43  String group = "";
44  try
45  {
46  group = ResourceGroupManager::getSingleton().findGroupContainingResource("skidmarks.cfg");
47  if (group.empty())
48  {
49  LOG("[RoR] Failed to load skidmarks.cfg (file not found)");
50  return;
51  }
52 
53  DataStreamPtr ds = ResourceGroupManager::getSingleton().openResource("skidmarks.cfg", group);
54  String line = "";
55  String currentModel = "";
56 
57  while (!ds->eof())
58  {
59  line = RoR::Utils::SanitizeUtf8String(ds->getLine());
60  StringUtil::trim(line);
61 
62  if (line.empty() || line[0] == ';')
63  continue;
64 
65  StringVector args = StringUtil::split(line, ",");
66 
67  if (args.size() == 1)
68  {
69  currentModel = line;
70  continue;
71  }
72 
73  // process the line if we got a model
74  if (!currentModel.empty())
75  this->processLine(args, currentModel);
76  }
77  }
78  catch (...)
79  {
80  LOG("[RoR] Error loading skidmarks.cfg (unknown error)");
81  m_models.clear(); // Delete anything we might have loaded
82  return;
83  }
84  LOG("[RoR] skidmarks.cfg loaded OK");
85 }

87 int SkidmarkConfig::processLine(StringVector args, String modelName)
88 {
89  // we only accept 4 arguments
90  if (args.size() != 4)
91  return 1;
92 
93  // parse the data
94  SkidmarkDef cfg;
95  cfg.ground = args[0];
96  StringUtil::trim(cfg.ground);
97  cfg.texture = args[1];
98  StringUtil::trim(cfg.texture);
99 
100  cfg.slipFrom = StringConverter::parseReal(args[2]);
101  cfg.slipTo = StringConverter::parseReal(args[3]);
102 
103  if (!m_models.size() || m_models.find(modelName) == m_models.end())
104  m_models[modelName] = std::vector<SkidmarkDef>();
105 
106  m_models[modelName].push_back(cfg);
107  return 0;

163 {

171 
172  // new material
173  char bname[256] = "";
174  sprintf(bname, "mat-skidmark-%d", instanceCounter);
175  skid.material = MaterialManager::getSingleton().create(bname, ResourceGroupManager::DEFAULT_RESOURCE_GROUP_NAME);
176  Pass* p = skid.material->getTechnique(0)->getPass(0);
177 
178  TextureUnitState* tus = p->createTextureUnitState(texture);
179  p->setSceneBlending(SBT_TRANSPARENT_ALPHA);
180  p->setLightingEnabled(false);
181  p->setDepthWriteEnabled(false);
182  p->setDepthBias(3, 3);
183  p->setCullingMode(CULL_NONE);
184 
185  skid.points.resize(m_length);
186  skid.faceSizes.resize(m_length);
187  skid.groundTexture.resize(m_length);
188  skid.obj = gEnv->sceneManager->createManualObject("skidmark" + TOSTRING(instanceCounter++));
189  skid.obj->setDynamic(true);
190  skid.obj->setRenderingDistance(800); // 800m view distance
191  skid.obj->begin(bname, RenderOperation::OT_TRIANGLE_STRIP);
192  for (int i = 0; i < m_length; i++)
193  {
194  skid.points[i] = start;
195  skid.faceSizes[i] = 0;
196  skid.groundTexture[i] = "0.png";
197  skid.obj->position(start);
198  skid.obj->textureCoord(0, 0);
199  }
200  skid.obj->end();

202 
203  m_objects.push(skid);
204

206 }

236 void Skidmark::updatePoint()
237 {
238  Vector3 thisPoint = m_wheel->lastContactType ? m_wheel->lastContactOuter : m_wheel->lastContactInner;
239  Vector3 axis = m_wheel->lastContactType ? (m_wheel->refnode1->RelPosition - m_wheel->refnode0->RelPosition) : (m_wheel->refnode0->RelPosition - m_wheel->refnode1->RelPosition);
240  Vector3 thisPointAV = thisPoint + axis * 0.5f;
241  Real distance = 0;
242  Real maxDist = m_max_distance;
243  String texture = "none";
244  m_config->getTexture("default", m_wheel->lastGroundModel->name, m_wheel->lastSlip, texture);
245 
246  // dont add points with no texture
247  if (texture == "none")
248  return;
249 
250  if (m_wheel->speed > 1)
251  maxDist *= m_wheel->speed;
252 
253  if (!m_objects.size())
254  {
255  // add first bucket
256  this->addObject(thisPoint, texture);
257  }
258  else
259  {
260  // check existing buckets
261  SkidmarkSegment skid = m_objects.back();
262 
263  distance = skid.lastPointAv.distance(thisPointAV);
264  // too near to update?
265  if (distance < m_min_distance)
266  {
267  //LOG("E: too near for update");
268  return;
269  }
270 
271  // change ground texture if required
272  if (skid.pos > 0 && skid.groundTexture[0] != texture)
273  {
274  // new object with new texture
275  if (distance > maxDist)
276  {
277  // to far away for connection
278  this->addObject(thisPoint, texture);
279  }
280  else
281  {
282  // add new bucket with connection to last bucket
283  Vector3 lp1 = m_objects.back().points[m_objects.back().pos - 1];
284  Vector3 lp2 = m_objects.back().points[m_objects.back().pos - 2];
285  this->addObject(lp1, texture);
286  this->addPoint(lp2, distance, texture);
287  this->addPoint(lp1, distance, texture);
288  }
289  }
290  else
291  {
292  // no texture change required :D
293 
294  // far enough for new bucket?
295  if (skid.pos >= (int)skid.points.size())
296  {
297  if (distance > maxDist)
298  {
299  // to far away for connection
300  this->addObject(thisPoint, texture);
301  }
302  else
303  {
304  // add new bucket with connection to last bucket
305  Vector3 lp1 = m_objects.back().points[m_objects.back().pos - 1];
306  Vector3 lp2 = m_objects.back().points[m_objects.back().pos - 2];
307  this->addObject(lp1, texture);
308  this->addPoint(lp2, distance, texture);
309  this->addPoint(lp1, distance, texture);
310  }
311  }
312  else if (distance > m_max_distance)
313  {
314  // just new bucket, no connection to last bucket
315  this->addObject(thisPoint, texture);
316  }
317  }
318  }
319 
320  const float overaxis = 0.2f;
321  // tactics: we always choose the latest point and then create two points
322 
323  // choose node m_wheel by the latest added point
324  if (!m_wheel->lastContactType)
325  {
326  // choose inner
327  this->addPoint(m_wheel->lastContactInner - (axis * overaxis), distance, texture);
328  this->addPoint(m_wheel->lastContactInner + axis + (axis * overaxis), distance, texture);
329  }
330  else
331  {
332  // choose outer
333  this->addPoint(m_wheel->lastContactOuter + axis + (axis * overaxis), distance, texture);
334  this->addPoint(m_wheel->lastContactOuter - (axis * overaxis), distance, texture);
335  }
336 
337  // save as last point (in the middle of the m_wheel)
338  m_objects.back().lastPointAv = thisPointAV;
339 }

352 void Skidmark::update()
353 {
354  if (!m_objects.size())
355  return;
356  SkidmarkSegment skid = m_objects.back();
357  Vector3 vaabMin = skid.points[0];
358  Vector3 vaabMax = skid.points[0];
359  skid.obj->beginUpdate(0);
360  bool behindEnd = false;
361  Vector3 lastValid = Vector3::ZERO;
362  int to_counter = 0;
363  float tcox_counter = 0;
364 
365  for (int i = 0; i < m_length; i++ , to_counter++)
366  {
367  if (i >= skid.pos)
368  behindEnd = true;
369 
370  if (to_counter > 3)
371  to_counter = 0;
372 
373  if (!behindEnd)
374  tcox_counter += skid.faceSizes[i] / m_min_distance;
375 
376  while (tcox_counter > 1)
377  tcox_counter--;
378 
379  if (behindEnd)
380  {
381  skid.obj->position(lastValid);
382  skid.obj->textureCoord(0, 0);
383  }
384  else
385  {
386  skid.obj->position(skid.points[i]);
387 
388  Vector2 tco = m_tex_coords[to_counter];
389  tco.x *= skid.faceSizes[i] / m_min_distance; // scale texture according face size
390  skid.obj->textureCoord(tco);
391 
392  lastValid = skid.points[i];
393  }
394 
395  if (skid.points[i].x < vaabMin.x)
396  vaabMin.x = skid.points[i].x;
397  if (skid.points[i].y < vaabMin.y)
398  vaabMin.y = skid.points[i].y;
399  if (skid.points[i].z < vaabMin.z)
400  vaabMin.z = skid.points[i].z;
401  if (skid.points[i].x > vaabMax.x)
402  vaabMax.x = skid.points[i].x;
403  if (skid.points[i].y > vaabMax.y)
404  vaabMax.y = skid.points[i].y;
405  if (skid.points[i].z > vaabMax.z)
406  vaabMax.z = skid.points[i].z;
407  }
408  skid.obj->end();
409 
410  skid.obj->setBoundingBox(AxisAlignedBox(vaabMin, vaabMax));
411 
412  m_is_dirty = false;
413 }
