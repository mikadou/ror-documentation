41 String hexdump(void* pAddressIn, long lSize)
42 {
43  char szBuf[100];
44  long lIndent = 1;
45  long lOutLen, lIndex, lIndex2, lOutLen2;
46  long lRelPos;
47  struct
48  {
49  char* pData;
50  unsigned long lSize;
51  } buf;
52  unsigned char *pTmp, ucTmp;
53  unsigned char* pAddress = (unsigned char *)pAddressIn;
54 
55  buf.pData = (char *)pAddress;
56  buf.lSize = lSize;
57 
58  String result = "";
59 
60  while (buf.lSize > 0)
61  {
62  pTmp = (unsigned char *)buf.pData;
63  lOutLen = (int)buf.lSize;
64  if (lOutLen > 16)
65  lOutLen = 16;
66 
67  // create a 64-character formatted output line:
68  sprintf(szBuf, " > "
69  " "
70  " %08lX", (long unsigned int)(pTmp - pAddress));
71  lOutLen2 = lOutLen;
72 
73  for (lIndex = 1 + lIndent , lIndex2 = 53 - 15 + lIndent , lRelPos = 0;
74  lOutLen2;
75  lOutLen2-- , lIndex += 2 , lIndex2++
76  )
77  {
78  ucTmp = *pTmp++;
79 
80  sprintf(szBuf + lIndex, "%02X ", (unsigned short)ucTmp);
81  if (!isprint(ucTmp))
82  ucTmp = '.'; // nonprintable char
83  szBuf[lIndex2] = ucTmp;
84 
85  if (!(++lRelPos & 3)) // extra blank after 4 bytes
86  {
87  lIndex++;
88  szBuf[lIndex + 2] = ' ';
89  }
90  }
91 
92  if (!(lRelPos & 3))
93  lIndex--;
94 
95  szBuf[lIndex] = '<';
96  szBuf[lIndex + 1] = ' ';
97 
98  result += String(szBuf) + "\n";
99 
100  buf.pData += lOutLen;
101  buf.lSize -= lOutLen;
102  }
103  return result;
104 }

112 UTFString formatBytes(double bytes)
113 {
114  wchar_t tmp[128] = L"";
115  const wchar_t* si_prefix[] = {L"B", L"KB", L"MB", L"GB", L"TB", L"EB", L"ZB", L"YB"};
116  int base = 1024;
117  int c = std::min((int)(log(bytes) / log((float)base)), (int)sizeof(si_prefix) - 1);
118  swprintf(tmp, 128, L"%1.2f %ls", bytes / pow((float)base, c), si_prefix[c]);
119  return UTFString(tmp);

168  filename[i] = '_';
169  }

270 void generateHashFromDataStream(DataStreamPtr& ds, Ogre::String& hash)
271 {
272  size_t location = ds->tell();
273  // copy whole file into a buffer
274  uint8_t* buf = 0;
275  ds->seek(0); // from start
276  // alloc buffer
277  uint32_t bufSize = ds->size();
278  buf = (uint8_t *)malloc(bufSize + 1);
279  // read into buffer
280  ds->read(buf, bufSize);
281 
282  // and build the hash over it
283  char hash_result[250];
284  memset(hash_result, 0, 249);
285 
286  {
287  RoR::CSHA1 sha1;
288  sha1.UpdateHash(buf, bufSize);
289  sha1.Final();
290  sha1.ReportHash(hash_result, RoR::CSHA1::REPORT_HEX_SHORT);
291  }
292 
293  // revert DS to previous position
294  ds->seek(location);
295  // release memory
296  free(buf);
297  buf = 0;
298 
299  hash = String(hash_result);
300 }

312 std::string TrimBlanksAndLinebreaks(std::string const& input)
313 {
314  int substr_start = 0;
315  int substr_count = input.length();
316  while (substr_start < substr_count)
317  {
318  char c = input.at(substr_start);
319  if (c == ' ' || c == '\t' || c == '\n' || c == '\r')
320  {
321  ++substr_start;
322  --substr_count;
323  }
324  else
325  {
326  break;
327  }
328  }
329  while (substr_count > 0)
330  {
331  char c = input.at(substr_count - 1);
332  if (c == ' ' || c == '\t' || c == '\n' || c == '\r')
333  {
334  --substr_count;
335  }
336  else
337  {
338  break;
339  }
340  }
341  return input.substr(substr_start, substr_count);
342 }
